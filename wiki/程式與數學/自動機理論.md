# 自動機理論：形式語言與計算模型

## 概述

自動機理論研究抽象計算機器的數學模型，是形式語言、編譯器設計和計算複雜度的理論基礎。自動機從簡單的有限狀態機到複雜的圖靈機，構成了理解計算本質的層次結構。本頁介紹主要的自動機模型及其與正規語言、上下文無關語言的關係。

## 有限狀態自動機

```python
class FiniteAutomaton:
    """有限狀態自動機"""
    
    def __init__(self, states, alphabet, transitions, start_state, accept_states):
        self.states = set(states)
        self.alphabet = set(alphabet)
        self.transitions = transitions  # {(state, symbol): state}
        self.start_state = start_state
        self.accept_states = set(accept_states)
    
    def process(self, input_string):
        """處理輸入字符串"""
        current_state = self.start_state
        
        for symbol in input_string:
            if symbol not in self.alphabet:
                return False
            
            key = (current_state, symbol)
            if key in self.transitions:
                current_state = self.transitions[key]
            else:
                return False
        
        return current_state in self.accept_states

class NFA(FiniteAutomaton):
    """非確定性有限自動機"""
    
    def __init__(self, states, alphabet, transitions, start_state, accept_states):
        super().__init__(states, alphabet, transitions, start_state, accept_states)
    
    def process(self, input_string):
        """NFA 處理 - 探索所有可能的狀態"""
        current_states = {self.start_state}
        
        for symbol in input_string:
            next_states = set()
            for state in current_states:
                key = (state, symbol)
                if key in self.transitions:
                    next_states.update(self.transitions[key])
            current_states = next_states
        
        return bool(current_states & self.accept_states)

class DFA(FiniteAutomaton):
    """確定性有限自動機"""
    
    def __init__(self, states, alphabet, transitions, start_state, accept_states):
        super().__init__(states, alphabet, transitions, start_state, accept_states)
    
    def process(self, input_string):
        """DFA 處理"""
        return super().process(input_string)
    
    def to_dfa_from_nfa(self, nfa):
        """將 NFA 轉換為 DFA (子集構造)"""
        # 計算 epsilon 閉包
        def epsilon_closure(states):
            closure = set(states)
            stack = list(states)
            while stack:
                state = stack.pop()
                key = (state, 'ε')
                if key in nfa.transitions:
                    for s in nfa.transitions[key]:
                        if s not in closure:
                            closure.add(s)
                            stack.append(s)
            return closure
        
        # 構造 DFA 狀態
        dfa_states = []
        dfa_transitions = {}
        dfa_accept_states = []
        
        initial = epsilon_closure({nfa.start_state})
        unmarked = [frozenset(initial)]
        marked = set()
        
        while unmarked:
            current = unmarked.pop()
            if current in marked:
                continue
            marked.add(current)
            dfa_states.append(current)
            
            if current & nfa.accept_states:
                dfa_accept_states.append(current)
            
            for symbol in nfa.alphabet:
                next_set = set()
                for state in current:
                    key = (state, symbol)
                    if key in nfa.transitions:
                        next_set.update(nfa.transitions[key])
                next_set = epsilon_closure(next_set)
                
                if next_set:
                    dfa_transitions[(current, symbol)] = next_set
                    if next_set not in marked:
                        unmarked.append(next_set)
        
        return DFA(dfa_states, nfa.alphabet, dfa_transitions, 
                  frozenset(initial), dfa_accept_states)
```

## 正則表達式

```python
import re

class RegexToNFA:
    """正則表達式到 NFA 的轉換 (Thompson 構造)"""
    
    class Fragment:
        def __init__(self, start, end):
            self.start = start
            self.end = end
    
    def __init__(self):
        self.state_counter = 0
    
    def new_state(self):
        state = self.state_counter
        self.state_counter += 1
        return state
    
    def literal(self, char):
        """基本字符 a"""
        s0 = self.new_state()
        s1 = self.new_state()
        return self.Fragment(s0, s1), {(s0, char): {s1}}
    
    def union(self, frag1, frag2):
        """並聯 (a|b)"""
        s0 = self.new_state()
        s1 = self.new_state()
        
        transitions = {(s0, 'ε'): {frag1.start, frag2.start}}
        transitions[frag1.end] = {(frag1.end, 'ε'): {s1}}
        transitions[frag2.end] = {(frag2.end, 'ε'): {s1}}
        
        return self.Fragment(s0, s1), transitions
    
    def concatenation(self, frag1, frag2):
        """串聯 (ab)"""
        transitions = {}
        # frag1 到 frag2
        transitions[(frag1.end, 'ε')] = {frag2.start}
        
        return self.Fragment(frag1.start, frag2.end), transitions
    
    def kleene_star(self, frag):
        """克萊尼星 (a*)"""
        s0 = self.new_state()
        s1 = self.new_state()
        
        transitions = {
            (s0, 'ε'): {frag.start, s1},
            (frag.end, 'ε'): {frag.start, s1}
        }
        
        return self.Fragment(s0, s1), transitions
    
    def parse(self, regex):
        """解析正則表達式"""
        # 簡化版本：使用 Python 的 re 模組
        return re.compile(regex)

def build_email_automaton():
    """構建簡化的 email 自動機"""
    states = {'q0', 'q1', 'q2', 'q3', 'q4', 'q5', 'q6'}
    alphabet = set('abcdefghijklmnopqrstuvwxyz0123456789@.')
    
    transitions = {}
    
    # 局部部分
    for c in alphabet - {'@', '.'}:
        transitions[('q0', c)] = 'q1'
    
    transitions[('q1', '.')] = 'q2'
    for c in 'abcdefghijklmnopqrstuvwxyz':
        transitions[('q1', c)] = 'q1'
        transitions[('q2', c)] = 'q1'
    
    # @ 符號
    transitions[('q1', '@')] = 'q3'
    
    # 域名部分
    for c in 'abcdefghijklmnopqrstuvwxyz':
        transitions[('q3', c)] = 'q4'
    
    transitions[('q4', '.')] = 'q5'
    for c in 'abcdefghijklmnopqrstuvwxyz':
        transitions[('q4', c)] = 'q4'
        transitions[('q5', c)] = 'q6'
    
    accept_states = {'q6'}
    
    return DFA(states, alphabet, transitions, 'q0', accept_states)
```

## 圖靈機

```python
class TuringMachine:
    """圖靈機"""
    
    def __init__(self, states, alphabet, tape_alphabet, transitions, 
                 start_state, accept_state, reject_state, blank='B'):
        self.states = set(states)
        self.alphabet = set(alphabet)
        self.tape_alphabet = set(tape_alphabet)
        self.transitions = transitions
        self.start_state = start_state
        self.accept_state = accept_state
        self.reject_state = reject_state
        self.blank = blank
    
    def process(self, input_string):
        """處理輸入"""
        tape = list(self.blank + input_string + self.blank)
        head_pos = 1
        current_state = self.start_state
        
        steps = 0
        max_steps = 10000
        
        while current_state not in {self.accept_state, self.reject_state}:
            if steps > max_steps:
                return False
            
            symbol = tape[head_pos]
            key = (current_state, symbol)
            
            if key not in self.transitions:
                current_state = self.reject_state
                break
            
            new_state, new_symbol, direction = self.transitions[key]
            tape[head_pos] = new_symbol
            
            if direction == 'R':
                head_pos += 1
                if head_pos >= len(tape):
                    tape.append(self.blank)
            elif direction == 'L':
                head_pos -= 1
                if head_pos < 0:
                    tape.insert(0, self.blank)
                    head_pos = 0
            
            current_state = new_state
            steps += 1
        
        return current_state == self.accept_state

# 圖靈機示例：判斷回文數
def build_palindrome_tm():
    """構建判斷回文數的圖靈機"""
    states = {'q0', 'q1', 'q2', 'q3', 'q4', 'q5', 'accept'}
    alphabet = {'0', '1'}
    tape_alphabet = {'0', '1', 'X', 'Y', 'B'}
    
    transitions = {
        ('q0', '0'): ('q1', 'X', 'R'),  # 找到 0，標記為 X，向右
        ('q0', '1'): ('q3', 'Y', 'R'),  # 找到 1，標記為 Y，向右
        ('q0', 'Y'): ('q0', 'Y', 'R'),  # 跳過 Y
        ('q0', 'X'): ('q0', 'X', 'R'),  # 跳過 X
        # ... 更多轉換
    }
    
    return TuringMachine(states, alphabet, tape_alphabet, 
                        transitions, 'q0', 'accept', 'reject')
```

## 總結

自動機理論建立了語言和計算之間的橋樑。有限自動機用於識別正規語言，正規表達式與之等價；下推自動機用於識別上下文無關語言；圖靈機則可以識別所有可計算的語言。理解這些模型對於編譯器設計、形式驗證和計算複雜度分析至關重要。

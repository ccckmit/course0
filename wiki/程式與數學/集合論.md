# 集合論：離散數學的基礎

## 概述

集合論是離散數學的基礎，研究對象的集合及其運算。集合論的概念滲透到計算機科學的各個角落：從資料庫的關係模型到程式語言的類型系統，從演算法的複雜度分析到機器學習的特徵空間。本頁介紹集合論的核心概念、運算和Python實現。

## 集合基礎

```python
class Set:
    """集合的 Python 實現"""
    
    def __init__(self, elements=None):
        self.elements = set(elements) if elements else set()
    
    def add(self, element):
        self.elements.add(element)
    
    def remove(self, element):
        self.elements.discard(element)
    
    def union(self, other):
        return Set(self.elements | other.elements)
    
    def intersection(self, other):
        return Set(self.elements & other.elements)
    
    def difference(self, other):
        return Set(self.elements - other.elements)
    
    def symmetric_difference(self, other):
        return Set(self.elements ^ other.elements)
    
    def is_subset(self, other):
        return self.elements <= other.elements
    
    def is_superset(self, other):
        return self.elements >= other.elements
    
    def is_member(self, element):
        return element in self.elements
    
    def cardinality(self):
        return len(self.elements)
    
    def power_set(self):
        """冪集"""
        from itertools import combinations
        elements = list(self.elements)
        power = [Set()]
        for r in range(1, len(elements) + 1):
            for combo in combinations(elements, r):
                power.append(Set(combo))
        return power
    
    def cartesian_product(self, other):
        """笛卡爾積"""
        return Set([(a, b) for a in self.elements for b in other.elements])
    
    def __str__(self):
        return '{' + ', '.join(str(e) for e in self.elements) + '}'

# 集合恆等式示例
"""
德摩根定律:
- (A ∪ B)' = A' ∩ B'
- (A ∩ B)' = A' ∪ B'

分配律:
- A ∪ (B ∩ C) = (A ∪ B) ∩ (A ∪ C)
- A ∩ (B ∪ C) = (A ∩ B) ∪ (A ∩ C)

吸收律:
- A ∪ (A ∩ B) = A
- A ∩ (A ∪ B) = A

補餘律:
- A ∪ A' = U
- A ∩ A' = ∅
"""

def verify_morgans_law(A, B):
    """驗證德摩根定律"""
    left = (A.union(B)).difference(A.intersection(B))
    right = A.symmetric_difference(B)
    return left.is_subset(right) and right.is_subset(left)
```

## 關係與函數

```python
class Relation:
    """二元關係"""
    
    def __init__(self, domain, codomain, pairs):
        self.domain = Set(domain)
        self.codomain = Set(codomain)
        self.pairs = Set(pairs)
    
    def is_reflexive(self):
        """自反性: ∀a ∈ A, (a, a) ∈ R"""
        for a in self.domain.elements:
            if not self.pairs.is_member((a, a)):
                return False
        return True
    
    def is_symmetric(self):
        """對稱性: ∀(a,b) ∈ R, (b,a) ∈ R"""
        for (a, b) in self.pairs.elements:
            if not self.pairs.is_member((b, a)):
                return False
        return True
    
    def is_transitive(self):
        """傳遞性: ∀(a,b), (b,c) ∈ R, (a,c) ∈ R"""
        pairs_list = list(self.pairs.elements)
        for (a, b) in pairs_list:
            for (c, d) in pairs_list:
                if b == c and not self.pairs.is_member((a, d)):
                    return False
        return True
    
    def is_equivalence(self):
        """等價關係"""
        return (self.is_reflexive() and 
                self.is_symmetric() and 
                self.is_transitive())
    
    def is_partial_order(self):
        """偏序關係"""
        return (self.is_reflexive() and 
                self.is_antisymmetric() and 
                self.is_transitive())
    
    def is_antisymmetric(self):
        """反對稱性: ∀(a,b), (b,a) ∈ R, a = b"""
        for (a, b) in self.pairs.elements:
            if a != b and self.pairs.is_member((b, a)):
                return False
        return True

class Function:
    """函數"""
    
    def __init__(self, domain, codomain, mapping):
        self.domain = Set(domain)
        self.codomain = Set(codomain)
        self.mapping = mapping
    
    def is_well_defined(self):
        """定義良好性"""
        seen = {}
        for (a, b) in self.mapping:
            if a in seen and seen[a] != b:
                return False
            seen[a] = b
        return True
    
    def is_injective(self):
        """單射 (一一對應)"""
        seen = set()
        for (a, b) in self.mapping:
            if b in seen:
                return False
            seen.add(b)
        return True
    
    def is_surjective(self):
        """滿射 (映成)"""
        return Set([b for (a, b) in self.mapping]) == self.codomain
    
    def is_bijective(self):
        """雙射"""
        return self.is_injective() and self.is_surjective()
```

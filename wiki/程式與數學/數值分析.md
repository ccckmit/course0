# 數值分析：計算機上的數學方法

## 概述

數值分析研究在計算機上求解數學問題的近似方法。由於計算機使用有限精度表示數字（通常是 IEEE 754 浮點數），所有計算都會引入誤差。理解這些誤差的來源和傳播對於編寫正確的數值計算程式至關重要。本頁介紹浮點數表示、誤差分析和基本數值方法。

## 浮點數與誤差

```python
import math
import struct
import numpy as np

class FloatingPoint:
    """IEEE 754 浮點數表示"""
    
    def __init__(self, sign=0, exponent=0, mantissa=0):
        self.sign = sign
        self.exponent = exponent
        self.mantissa = mantissa
    
    @staticmethod
    def to_ieee754(value):
        """將 Python float 轉換為 IEEE 754 表示"""
        # 使用 struct 模組
        bits = struct.unpack('>Q', struct.pack('>d', value))[0]
        sign = (bits >> 63) & 1
        exponent = (bits >> 52) & 0x7FF
        mantissa = bits & 0xFFFFFFFFFFFFF
        return sign, exponent, mantissa
    
    @staticmethod
    def from_ieee754(sign, exponent, mantissa):
        """從 IEEE 754 表示恢復"""
        bits = (sign << 63) | (exponent << 52) | mantissa
        return struct.unpack('>d', struct.pack('>Q', bits))[0]

class ErrorAnalysis:
    """誤差分析"""
    
    @staticmethod
    def absolute_error(approximate, exact):
        """絕對誤差"""
        return abs(approximate - exact)
    
    @staticmethod
    def relative_error(approximate, exact):
        """相對誤差"""
        if exact == 0:
            return float('inf')
        return abs(approximate - exact) / abs(exact)
    
    @staticmethod
    def truncation_error(true_value, series_sum):
        """截斷誤差"""
        return abs(true_value - series_sum)
    
    @staticmethod
    def roundoff_error(actual, computed):
        """舍入誤差"""
        return abs(actual - computed)
    
    @staticmethod
    def forward_error_bound(cond_num, relative_input_error):
        """前向誤差界"""
        return cond_num * relative_input_error
    
    @staticmethod
    def condition_number(f, x, h=1e-8):
        """計算條件數"""
        # 使用數值微分
        f_prime = (f(x + h) - f(x - h)) / (2 * h)
        return abs(x * f_prime / f(x))

# 浮點數精度問題示例
def demonstrate_precision():
    """展示浮點數精度問題"""
    
    # 1. 浮點數相加失去精度
    a = 0.1
    b = 0.2
    c = 0.3
    print(f"0.1 + 0.2 == 0.3: {a + b == c}")  # False!
    print(f"(0.1 + 0.2) - 0.3: {(a + b) - c}")  # ≈ 5.55e-17
    
    # 2. 大數吃小數
    large = 1e15
    small = 1e-5
    result = large + small
    print(f"1e15 + 1e-5 = {result}")  # 等於 1e15，小數被忽略
    
    # 3. 取消問題
    x = 1e10
    y = 1e-10
    z = 1e10
    result = (x + y) - z
    print(f"(1e10 + 1e-10) - 1e10 = {result}")  # 應該是 1e-10，但可能不對
    
    # 4. 安全的比較
    def is_close(a, b, rel_tol=1e-9, abs_tol=1e-12):
        return abs(a - b) <= max(rel_tol * max(abs(a), abs(b)), abs_tol)
    
    print(f"is_close(0.1+0.2, 0.3): {is_close(0.1+0.2, 0.3)}")
```

## 數值線性代數

```python
class NumericalLinearAlgebra:
    """數值線性代數方法"""
    
    @staticmethod
    def lu_decomposition(A):
        """LU 分解"""
        n = len(A)
        L = [[0.0] * n for _ in range(n)]
        U = [[0.0] * n for _ in range(n)]
        
        for i in range(n):
            # U 的第 i 行
            for k in range(i, n):
                U[i][k] = A[i][k] - sum(L[i][j] * U[j][k] for j in range(i))
            
            # L 的第 i 列
            L[i][i] = 1
            for k in range(i + 1, n):
                if abs(U[i][i]) < 1e-12:
                    raise ValueError("Matrix is singular")
                L[k][i] = (A[k][i] - sum(L[k][j] * U[j][i] for j in range(i))) / U[i][i]
        
        return L, U
    
    @staticmethod
    def solve_lu(L, U, b):
        """使用 LU 分解求解 Ax = b"""
        n = len(b)
        
        # 前向替換: Ly = b
        y = [0.0] * n
        for i in range(n):
            y[i] = b[i] - sum(L[i][j] * y[j] for j in range(i))
        
        # 後向替換: Ux = y
        x = [0.0] * n
        for i in range(n - 1, -1, -1):
            x[i] = (y[i] - sum(U[i][j] * x[j] for j in range(i + 1, n))) / U[i][i]
        
        return x
    
    @staticmethod
    def jacobi_iteration(A, b, max_iter=100, tol=1e-10):
        """Jacobi 迭代法"""
        n = len(A)
        x = [0.0] * n
        
        for _ in range(max_iter):
            x_new = [0.0] * n
            for i in range(n):
                sigma = sum(A[i][j] * x[j] for j in range(n) if j != i)
                x_new[i] = (b[i] - sigma) / A[i][i]
            
            # 收斂檢查
            error = max(abs(x_new[i] - x[i]) for i in range(n))
            x = x_new
            
            if error < tol:
                break
        
        return x
    
    @staticmethod
    def gauss_seidel(A, b, max_iter=100, tol=1e-10):
        """Gauss-Seidel 迭代法"""
        n = len(A)
        x = [0.0] * n
        
        for _ in range(max_iter):
            x_old = x.copy()
            for i in range(n):
                x[i] = (b[i] - sum(A[i][j] * x[j] for j in range(i))
                       - sum(A[i][j] * x_old[j] for j in range(i + 1, n))) / A[i][i]
            
            error = max(abs(x[i] - x_old[i]) for i in range(n))
            if error < tol:
                break
        
        return x
```

## 數值微積分

```python
class NumericalIntegration:
    """數值積分方法"""
    
    @staticmethod
    def trapezoid(f, a, b, n=1000):
        """梯形法則"""
        h = (b - a) / n
        result = 0.5 * (f(a) + f(b))
        
        for i in range(1, n):
            result += f(a + i * h)
        
        return result * h
    
    @staticmethod
    def simpson(f, a, b, n=1000):
        """Simpson 法則"""
        if n % 2 == 1:
            n += 1
        
        h = (b - a) / n
        result = f(a) + f(b)
        
        for i in range(1, n):
            x = a + i * h
            result += f(x) * (4 if i % 2 == 1 else 2)
        
        return result * h / 3
    
    @staticmethod
    def adaptive_quadrature(f, a, b, tol=1e-10, max_depth=50):
        """自適應積分"""
        def recursive(f, a, b, tol, depth):
            c = (a + b) / 2
            left = NumericalIntegration.simpson(f, a, c, 2)
            right = NumericalIntegration.simpson(f, c, b, 2)
            whole = NumericalIntegration.simpson(f, a, b, 2)
            
            if depth >= max_depth:
                return left + right
            
            if abs(left + right - whole) < 15 * tol:
                return left + right + (left + right - whole) / 15
            
            return (recursive(f, a, c, tol/2, depth + 1) +
                   recursive(f, c, b, tol/2, depth + 1))
        
        return recursive(f, a, b, tol, 0)

class NumericalDifferentiation:
    """數值微分"""
    
    @staticmethod
    def forward_difference(f, x, h=1e-5):
        """前向差分"""
        return (f(x + h) - f(x)) / h
    
    @staticmethod
    def backward_difference(f, x, h=1e-5):
        """後向差分"""
        return (f(x) - f(x - h)) / h
    
    @staticmethod
    def central_difference(f, x, h=1e-5):
        """中心差分"""
        return (f(x + h) - f(x - h)) / (2 * h)
    
    @staticmethod
    def second_derivative(f, x, h=1e-5):
        """二階導數"""
        return (f(x + h) - 2 * f(x) + f(x - h)) / (h ** 2)
```

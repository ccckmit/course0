# 組合語言 (Assembly Language)

**標籤**: #組合語言 #低階程式設計 #RISC-V #x86 #ARM #機器碼
**日期**: 2024-01-15

組合語言是介於高階語言和機器碼之間的低階語言，每個組合語言指令對應一個機器指令，提供對硬體的直接控制能力。學習組合語言能夠幫助開發者深入理解電腦架構、記憶體管理、以及程式編譯過程，是電腦科學教育的重要基礎。

## 組合語言的發展歷程

### 早期發展 (1940s-1950s)

最早的電子計算機使用機器碼（二進制） programming directly in binary. The first assembly languages were developed to make programming easier, using mnemonics like "ADD" and "SUB" instead of binary codes.

### 標準化時代 (1960s-1980s)

隨著不同計算機架構的出現，出現了多種組合語言，如 x86、Motorola 68000、SPARC 等。這些架構各自有其獨特的指令集和語法。

### RISC 革命 (1980s-現在)

精簡指令集電腦 (RISC) 架構的出現帶來了更整齊的指令集，如 MIPS、ARM、RISC-V 等。這些架構的組合語言更適合教學和研究。

## 組合語言的價值與應用

### 為什麼學習組合語言

1. **理解電腦運作原理**: 組合語言直接操作硬�體，讓你了解 CPU、記憶體、暫存器如何協同工作
2. **效能優化**: 在效能關鍵的場景，組合語言可以實現比編譯器優化更好的結果
3. **安全研究**: 惡意軟體分析、漏洞挖掘需要理解組合語言
4. **編譯器開發**: 了解目標機器的指令集是設計編譯器的基礎
5. **嵌入式系統**: 資源受限的嵌入式系統經常需要組合語言

### 常見應用場景

- 作業系統核心 (bootloader、context switching)
- 驅動程式開發
- 效能關鍵的數學庫 (密碼學、訊號處理)
- 遊戲引擎優化
- BIOS/UEFI 開發
- 數位訊號處理器 (DSP) 程式設計

## 主流指令集架構

### x86/x64 (Intel/AMD)

x86 是最廣泛使用的處理器架構，從 1978 年的 Intel 8086 發展而來。x64 是其 64 位元擴展。

#### x86-64 基本語法 (Intel 語法)

```assembly
; Hello World - Windows x64
section .data
    msg db "Hello, World!", 0Dh, 0Ah, "$"

section .text
    global _main
    extern _printf

_main:
    push rbp
    mov rbp, rsp
    sub rsp, 32
    
    lea rcx, [msg]
    call _printf
    
    xor eax, eax
    add rsp, 32
    pop rbp
    ret
```

#### x86-64 基本指令

```assembly
; 算術運算
mov rax, 5          ; rax = 5
add rax, 3          ; rax = rax + 3 = 8
sub rax, 2          ; rax = rax - 2 = 6
imul rax, rbx       ; rax = rax * rbx
cqo                 ; 擴展被除數到 rdx:rax
idiv rbx            ; rax = rdx:rax / rbx, rdx = 餘數
inc rax             ; rax++
dec rax             ; rax--

; 邏輯運算
and rax, rbx        ; rax = rax AND rbx
or rax, rbx         ; rax = rax OR rbx
xor rax, rbx        ; rax = rax XOR rbx
not rax             ; rax = NOT rax
shl rax, 1          ; rax = rax << 1 (左移)
shr rax, 1          ; rax = rax >> 1 (邏輯右移)
sar rax, 1          ; rax = rax >> 1 (算術右移)

; 比較與分支
cmp rax, rbx        ; 比較 rax 和 rbx
je label            ; if rax == rbx, jump to label
jne label           ; if rax != rbx
jg label            ; if rax > rbx (signed)
jae label           ; if rax >= rbx (signed)
jl label            ; if rax < rbx (signed)
jle label           ; if rax <= rbx (signed)
ja label            ; if rax > rbx (unsigned)
jb label            ; if rax < rbx (unsigned)

; 暫存器移動
mov rax, rbx        ; rax = rbx
lea rax, [rbx+8]    ; rax = rbx + 8 (位址計算)
xchg rax, rbx       ; swap rax and rbx

; 記憶體存取
mov rax, [rbx]      ; rax = *rbx
mov [rax], rbx      ; *rax = rbx
mov rax, qword [rbx + rcx*8 + 16]  ; 複雜位址
```

#### x86-64 函式呼叫 (System V ABI)

```assembly
; 函式 prologue
push rbp
mov rbp, rsp
push rbx              ; 保存被呼叫者保存的暫存器
push r12
push r13
push r14
push r15
sub rsp, 32           ; 分配 shadow space 和對齊

; 函式 epilogue
add rsp, 32
pop r15
pop r14
pop r13
pop r12
pop rbx
pop rbp
ret

; 呼叫函式
; 引數: rdi, rsi, rdx, rcx, r8, r9
; 返回值: rax (或 rax:rdx for 128-bit)
call my_function
mov rbx, rax

; 內聯組合
void *memcpy_asm(void *dest, const void *src, size_t n) {
    __asm__ volatile (
        "mov rsi, rdi"           ; save dest
        "mov rcx, rdx"           ; count
        "rep movsb"              ; copy
        "mov rax, rsi"           ; return dest
    );
}
```

### ARM (AArch64)

ARM (Advanced RISC Machines) 廣泛用於行動裝置和嵌入式系統。ARM64 (AArch64) 是其 64 位元架構。

#### ARM64 基本語法

```assembly
; Hello World - ARM64 Linux
.section data
msg: .asciz "Hello, World!\n"
len = . - msg

.section text
.global _start

_start:
    // write(fd, msg, len)
    mov x0, 1              // fd = stdout
    ldr x1, =msg           // buf = msg
    mov x2, #len           // count = len
    mov x8, 64             // syscall: write
    svc 0                   // invoke syscall

    // exit(code)
    mov x0, 0              // code = 0
    mov x8, 93             // syscall: exit
    svc 0
```

#### ARM64 指令集

```assembly
// 算術運算
mov x0, 5              // x0 = 5
add x0, x0, 3          // x0 = x0 + 3
sub x0, x0, 2          // x0 = x0 - 2
mul x0, x1, x2         // x0 = x1 * x2
sdiv x0, x1, x2        // x0 = x1 / x2 (signed)
udiv x0, x1, x2        // x0 = x1 / x2 (unsigned)
lsl x0, x1, 2          // x0 = x1 << 2
lsr x0, x1, 2          // x0 = x1 >> 2 (logical)
asr x0, x1, 2          // x0 = x1 >> 2 (arithmetic)

// 邏輯運算
and x0, x1, x2         // x0 = x1 AND x2
orr x0, x1, x2         // x0 = x1 OR x2
eor x0, x1, x2         // x0 = x1 XOR x2
mvn x0, x1             // x0 = NOT x1
bic x0, x1, x2         // x0 = x1 AND NOT x2

// 比較
cmp x0, x1             // compare x0 - x1
cset x0, gt            // x0 = (x0 > x1) ? 1 : 0
csel x0, x1, x2, gt   // x0 = (gt) ? x1 : x2

// 分支
b label                // unconditional branch
bl label               // branch with link (call)
beq label              // branch if equal
bne label             // branch if not equal
bgt label              // branch if greater than
ret                   // return (ret to address in LR)

// 載入/儲存
ldr x0, [x1]           // x0 = *x1 (load)
str x0, [x1]           // *x1 = x0 (store)
ldp x0, x1, [x2]       // load pair: x0=*[x2], x1=*[x2+8]
stp x0, x1, [x2]       // store pair
ldr x0, [x1, #8]       // x0 = *(x1 + 8)
str x0, [x1, #8]       // *(x1 + 8) = x0
```

#### ARM64 函式呼叫 (AAPCS64)

```assembly
// 函式: int sum(int a, int b)
// a in w0, b in w1, return in w0
sum:
    add w0, w0, w1
    ret

// 呼叫函式
// 引數: x0-x7 (first 8 args)
// 返回值: x0 (or x0:x1 for 128-bit)
bl sum
mov w2, w0             // use result

// 完整函式範例: int factorial(int n)
// n in w0, return in w0
factorial:
    cmp w0, 1
    b.le .base_case
    
    sub sp, sp, 16     // allocate stack
    str lr, [sp, 8]    // save link register
    str w0, [sp]       // save n
    
    sub w0, w0, 1      // n - 1
    bl factorial       // factorial(n-1)
    
    ldr w1, [sp]       // load n
    mul w0, w0, w1     // n * factorial(n-1)
    
    ldr lr, [sp, 8]    // restore link register
    add sp, sp, 16     // deallocate stack
    ret
    
.base_case:
    mov w0, 1
    ret
```

### RISC-V

RISC-V 是開放原始碼的 RISC 指令集架構，近年來在學術界和產業界快速發展。

#### RISC-V 基本語法

```assembly
# Hello World - RISC-V Linux
    .data
msg:    .asciz "Hello, World!\n"
len = . - msg

    .text
    .globl _start

_start:
    # write(fd, msg, len)
    li a7, 64          # syscall number for write
    li a0, 1           # fd = stdout
    la a1, msg         # buf = msg
    li a2, len         # count = len
    ecall              # invoke syscall

    # exit(code)
    li a7, 93          # syscall number for exit
    li a0, 0           # code = 0
    ecall              # invoke syscall
```

#### RISC-V 指令集

```assembly
# 算術運算
li t0, 5           # t0 = 5 (load immediate)
add t0, t0, t1     # t0 = t0 + t1
sub t0, t0, t1     # t0 = t0 - t1
mul t0, t0, t1     # t0 = t0 * t1
div t0, t0, t1     # t0 = t0 / t1
rem t0, t0, t1     # t0 = t0 % t1

# 立即數算術
addi t0, t0, 10    # t0 = t0 + 10
addi t0, t0, -5    # t0 = t0 - 5
slli t0, t0, 2     # t0 = t0 << 2
srli t0, t0, 2     # t0 = t0 >> 2 (logical)
srai t0, t0, 2     # t0 = t0 >> 2 (arithmetic)

# 邏輯運算
and t0, t0, t1     # t0 = t0 & t1
or t0, t0, t1      # t0 = t0 | t1
xor t0, t0, t1     # t0 = t0 ^ t1
not t0, t0         # t0 = ~t0 (pseudo-instruction)

# 比較分支
beq t0, t1, label  # if t0 == t1, jump
bne t0, t1, label # if t0 != t1, jump
blt t0, t1, label # if t0 < t1 (signed)
bge t0, t1, label # if t0 >= t1 (signed)
bltu t0, t1, label # if t0 < t1 (unsigned)
bgeu t0, t1, label # if t0 >= t1 (unsigned)

# 立即數比較
li t0, 10
slt t1, t0, t2    # t1 = (t0 < t2) ? 1 : 0
slti t1, t0, 5    # t1 = (t0 < 5) ? 1 : 0

# 載入/儲存
lw t0, 0(sp)      # t0 = *(sp + 0) (load word)
lb t0, 0(sp)      # t0 = *(sp + 0) (load byte, signed)
lbu t0, 0(sp)     # t0 = *(sp + 0) (load byte, unsigned)
lui t0, 0x1000    # t0 = 0x1000 << 12 (load upper immediate)
auipc t0, 0x1000  # t0 = pc + (0x1000 << 12)

sw t0, 0(sp)       # *(sp + 0) = t0 (store word)
sb t0, 0(sp)       # *(sp + 0) = t0 (store byte)

# 跳躍
jal label          # jump and link: ra = pc+4, jump to label
jalr t0           # jump and register: pc = t0, ra = pc+4
ret               # return: jalr x0, 0, ra (pseudo-instruction)
```

#### RISC-V 函式呼叫約定

```assembly
# RISC-V 暫存器用途
# a0-a7 (x10-x17): 函式引數/返回值
# t0-t6 (x5-x7, x28-x31): 臨時暫存器
# s0-s11 (x8-x9, x18-x27): 保存的暫存器
# ra (x1): 返回位址
# sp (x2): 堆疊指標
# gp (x3): 全域指標
# tp (x4): 執行緒指標

# 函式範例: int sum_array(int* arr, int n)
# a0 = arr, a1 = n
# 返回值在 a0
sum_array:
    add t0, zero, zero    # i = 0
    add t1, zero, zero    # sum = 0
    
loop:
    bge t0, a1, end       # if i >= n, goto end
    
    # sum += arr[i]
    slli t2, t0, 2        # offset = i * 4
    add t2, a0, t2        # &arr[i]
    lw t3, 0(t2)          # arr[i]
    add t1, t1, t3        # sum += arr[i]
    
    addi t0, t0, 1        # i++
    jal zero, loop        # goto loop
    
end:
    add a0, t1, zero      # return sum
    jalr zero, ra, 0      # return

# 堆疊框架建立
addi sp, sp, -16          # allocate 16 bytes
sw ra, 0(sp)              # save return address
sw s0, 8(sp)              # save saved register

# ... 函式主體 ...

# 堆疊框架釋放
lw ra, 0(sp)              # restore return address
lw s0, 8(sp)              # restore saved register
addi sp, sp, 16           # deallocate
jalr zero, ra, 0          # return
```

## 組合語言程式設計技巧

### 效能優化

#### 指令級平行 (ILP)

```assembly
# x86-64: 交錯載入以隱藏延遲
; 不好: 依賴鏈太長
vpmovzxbd ymm0, [rsi]         ; load
vpmulld ymm1, ymm0, ymm0       ; multiply (依賴前面的 load)
vpmovzxbd ymm2, [rsi+16]      ; load (可以與 mul 平行)
vpmulld ymm3, ymm2, ymm2      ; multiply

; 好: 交錯指令
vpmovzxbd ymm0, [rsi]         ; load #1
vpmovzxbd ymm2, [rsi+16]      ; load #2 (獨立的 load)
vpmulld ymm1, ymm0, ymm0     ; mul #1
vpmulld ymm3, ymm2, ymm2     ; mul #2
```

#### 迴圈展開

```assembly
# ARM64: 展開 4 次以減少分支開銷
.loop:
    ldp q0, q1, [x1], #32     ; load 2 vectors
    ldp q2, q3, [x1], #32     ; load 2 more
    fmul v0, v0, v0           ; multiply
    fmul v1, v1, v1
    fmul v2, v2, v2
    fmul v3, v3, v3
    stp q0, q1, [x0], #32     ; store
    stp q2, q3, [x0], #32     ; store
    subs x2, x2, #4           ; counter -= 4
    b.ne .loop                 ; if counter != 0, loop
```

#### 軟體管線化

```assembly
# RISC-V: 軟體管線化迴圈
# 原始:
#   for (i = 0; i < n; i++) y[i] = a * x[i] + b;

# 軟體管線化版本:
    mv t0, a0                  # x 指標
    mv t1, a1                  # y 指標
    mv t2, a2                  # n
    
    li t3, 4                   # factor = a
    li t4, 2                   # offset = b

.preheader:
    addi t2, t2, -4            # 處理 4 個元素
    blez t2, .mainloop_done

.mainloop:
    # 每次迭代處理 4 個元素
    lw t5, 0(t0)
    lw t6, 4(t0)
    lw t7, 8(t0)
    lw t8, 12(t0)
    addi t0, t0, 16
    
    mul t5, t5, t3
    mul t6, t6, t3
    mul t7, t7, t3
    mul t8, t8, t3
    
    addi t5, t5, 4
    addi t6, t6, 4
    addi t7, t7, 4
    addi t8, t8, 4
    
    sw t5, 0(t1)
    sw t6, 4(t1)
    sw t7, 8(t1)
    sw t8, 12(t1)
    addi t1, t1, 16
    
    addi t2, t2, -4
    bgtz t2, .mainloop
```

### 分支優化

```assembly
# x86: 避免分支預測失敗
; 不好的版本
cmp eax, ebx
jge .L1
mov edx, 1
jmp .L2
.L1:
mov edx, 0
.L2:

; 好的版本: 使用條件移動
cmp eax, ebx
setge dl          ; dl = (eax >= ebx) ? 1 : 0
movzx edx, dl     ; zero-extend to 32-bit

; ARM64: 使用條件執行
cmp w0, w1
csel w2, wzr, w0, gt   # if (w0 > w1) w2 = w0 else w2 = 0
```

### 記憶體對齊

```assembly
; x86-64: 確保資料對齊
; 不對齊的存取較慢
movdqa xmm0, [rax]         ; 必須 16-byte 對齊
vmovups xmm0, [rax]         ; 不需要對齊但較慢

; ARM64: 對齊存取
ldp q0, q1, [x1], #32      ; 需要 16-byte 對齊
```

## 組合器指令與偽指令

### 常見偽指令

| 偽指令 | 說明 | 展開範例 |
|--------|------|----------|
| `li rd, imm` | 載入立即數 | lui + addi (RISC-V) |
| `la rd, label` | 載入位址 | auipc + addi (RISC-V) |
| `mv rd, rs` | 複製暫存器 | addi rd, rs, 0 |
| `neg rd, rs` | 取負 | sub rd, zero, rs |
| `not rd, rs` | 位元反轉 | xori rd, rs, -1 |
| `ret` | 返回 | jalr zero, ra, 0 |
| `nop` | 無操作 | addi zero, zero, 0 |
| `li rd, -1` | 載入 -1 | addi rd, zero, -1 |

### 章節指示

```assembly
# .data - 已初始化資料段
.section data
buffer: .space 4096          ; 分配 4096 bytes
msg: .asciz "Hello"          ; null-terminated string
arr: .word 1, 2, 3, 4        ; word 陣列

# .text - 程式碼段
.section text
.global main
main:
    # code here

# .bss - 未初始化資料段
.section bss
.comm buffer, 4096           ; common block

# .rodata - 唯讀資料段
.section rodata
format: .asciz "%d\n"
```

## 實際應用範例

### 字串處理

```assembly
# RISC-V: strlen
# a0 = str, 返回值在 a0
strlen:
    add t0, a0, zero      # save pointer
.loop:
    lbu t1, 0(t0)         # load byte
    addi t0, t0, 1        # next char
    bnez t1, .loop         # if not null, continue
    
    sub a0, t0, a0        # calculate length
    addi a0, a0, -1       # subtract null terminator
    jalr zero, ra, 0      # return

# RISC-V: strcpy
# a0 = dest, a1 = src
strcpy:
    add t0, a0, zero      # dest pointer
    add t1, a1, zero      # src pointer
.loop:
    lbu t2, 0(t1)         # load src byte
    sb t2, 0(t0)          # store to dest
    addi t0, t0, 1
    addi t1, t1, 1
    bnez t2, .loop        # if not null, continue
    jalr zero, ra, 0      # return
```

### 快速排序

```assembly
# x86-64 quicksort
# rdi = array, rsi = lo, rdx = hi
quicksort:
    push rbp
    mov rbp, rsp
    push rbx
    push r12
    push r13
    push r14
    
    cmp rsi, rdx
    jge .done
    
    mov rbx, rsi          ; pivot = lo
    mov r12, rsi          ; i = lo
    
    ; partition
.partition_loop:
    cmp r12, rdx
    jge .partition_end
    
    mov rax, [rdi + r12*8]
    cmp rax, [rdi + rdx*8]
    jge .skip_swap
    
    ; swap arr[i] and arr[j]
    mov r13, [rdi + r12*8]
    mov r14, [rdi + rbx*8]
    mov [rdi + rbx*8], r13
    mov [rdi + r12*8], r14
    add rbx, 1
    
.skip_swap:
    add r12, 1
    jmp .partition_loop
    
.partition_end:
    ; swap arr[pivot] and arr[hi]
    mov r13, [rdi + rbx*8]
    mov r14, [rdi + rdx*8]
    mov [rdi + rdx*8], r13
    mov [rdi + rbx*8], r14
    
    ; quicksort left part
    mov r8, rsi
    dec rbx
    call quicksort
    
    ; quicksort right part
    inc rbx
    mov rsi, rbx
    call quicksort
    
.done:
    pop r14
    pop r13
    pop r12
    pop rbx
    pop rbp
    ret
```

## 常見問題與解決方案

### 問題 1: 緩衝區溢位

```assembly
; 問題: 沒有邊界檢查
strcpy:
    lbu t0, 0(a1)
    sb t0, 0(a0)
    addi a0, a0, 1
    addi a1, a1, 1
    bnez t0, strcpy
    ret

; 解決: 加入邊界檢查
strncpy:
    mv t0, a0             ; save dest
    mv t2, a2             ; save n
.loop:
    beqz a2, .done        ; if n == 0, done
    lbu t1, 0(a1)         ; load src byte
    sb t1, 0(a0)          ; store to dest
    beqz t1, .done        ; if src == 0, done
    addi a0, a0, 1
    addi a1, a1, 1
    addi a2, a2, -1
    j .loop
.done:
    ret
```

### 問題 2: 暫存器衝突

```assembly
; 問題: 覆蓋了被呼叫者保存的暫存器
my_func:
    mov rax, rdi          ; 使用 rax (會被覆蓋)
    ; ... 做一些運算 ...
    mov rbx, rax          ; 保存結果 (但 rbx 不是這裡用的)
    mov rax, rbx          ; 恢復 (但這裡已經用了 rax)
    ret

; 解決: 正確使用暫存器
my_func:
    push rbx              ; 保存 rbx
    mov rax, rdi
    ; ... 做一些運算 ...
    mov rbx, rax          ; 保存結果到 rbx
    mov rax, rbx          ; 恢復到返回值暫存器
    pop rbx               ; 恢復 rbx
    ret
```

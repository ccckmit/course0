# 分支預測：提升流水線效率的關鍵技術

## 概述

分支預測是現代高性能處理器中最關鍵的技術之一。在流水線處理器中，每個時鐘週期都會執行一條指令，但如果遇到條件分支，處理器在分支條件計算完成之前無法確定下一條要執行的指令是什麼。這就是所謂的控制冒險（Control Hazard），如果處理器在每個分支處都停頓等待結果，大量的效能將被浪費。

分支預測器通過猜測分支的走向來解決這個問題。處理器可以立即開始執行猜測路徑上的指令（推測執行），如果猜測正確，效能損失為零；如果猜測錯誤，處理器需要丟棄錯誤執行的指令並從正確路徑重新開始，這就是分支錯誤預測代價。

分支預測的準確率直接影響處理器的實際效能。現代處理器的分支預測準確率通常在 95-99% 之間，每一次錯誤預測都會造成 10-20 個時鐘週期的效能損失（等於流水線深度）。因此，分支預測器的設計是處理器架構中最複雜的部分之一。

## 分支預測基礎

### 靜態分支預測

```c
// 靜態分支預測是最簡單的形式
// 處理器總是猜測分支向同一方向（通常是向後跳轉）

// 編譯器提示（x86）
__builtin_expect() // GCC/Clang
#if defined(__GNUC__)
    #define likely(x)   __builtin_expect(!!(x), 1)
    #define unlikely(x) __builtin_expect(!!(x), 0)
#else
    #define likely(x)   (x)
    #define unlikely(x) (x)
#endif

// 使用 likely/unlikely 優化分支
int binary_search(int *arr, int n, int target) {
    int lo = 0, hi = n;
    while (likely(lo < hi)) {  // 提示編譯器這個分支通常為真
        int mid = lo + (hi - lo) / 2;
        if (arr[mid] < target) {
            lo = mid + 1;
        } else {
            hi = mid;
        }
    }
    return lo;
}

// 產生的彙編可能類似:
/*
    cmp     edx, esi
    jge     .L3              ; 如果 lo >= hi，跳轉（可能跳回循環）
    ; ...
.L3:
    cmp     edx, esi
    jl      .L2              ; 如果 lo < hi，跳回
*/

// ARM64 的 likely/unlikely
#define likely(x)   __builtin_expect(!!(x), 1)
#define unlikely(x) __builtin_expect(!!(x), 0)

// 條件執行減少分支
// ARM 提供 CBNZ, CBZ, TBZ, TBNZ 等條件分支指令

// 避免分支的向量化
// 如果分支是高度可預測的，條件執行可能比向量化更慢
void process_array_conditional(int *arr, int n, int threshold) {
    for (int i = 0; i < n; i++) {
        if (arr[i] > threshold) {  // 這個分支可能大部分為假
            arr[i] = compute_expensive(arr[i]);
        }
    }
}

// 使用條件載入（x86 CMOV 或 ARM CSEL）
void process_array_cmov(int *arr, int n, int threshold) {
    for (int i = 0; i < n; i++) {
        int val = arr[i];
        int result = compute_expensive(val);
        // 如果 val > threshold，使用 result；否則保持 val
        arr[i] = (val > threshold) ? result : val;
    }
}

// ARM64 CSEL
void process_array_csel(int *arr, int n, int threshold) {
    for (int i = 0; i < n; i++) {
        int val = arr[i];
        int result = compute_expensive(val);
        // CSEL: 如果 N flag 設置（val > threshold），選擇 result
        __asm__ volatile(
            "cmp %[val], %[threshold]\n\t"
            "csel %[result], %[result], %[val], gt"
            : [result] "+r"(result)
            : [val] "r"(val), [threshold] "r"(threshold)
        );
        arr[i] = result;
    }
}
```

### 動態分支預測

```c
// 1 位元分支歷史表 (BHT) - 最簡單的動態預測
#define BHT_SIZE 4096

typedef struct {
    uint8_t predictions[BHT_SIZE];  // 每個條目 1 位元: 0=不跳轉, 1=跳轉
} bht_1bit_t;

bht_1bit_t global_bht;

uint8_t bht_1bit_predict(bht_1bit_t *bht, uint64_t pc) {
    uint64_t index = (pc ^ (pc >> 12)) % BHT_SIZE;
    return bht->predictions[index];
}

void bht_1bit_update(bht_1bit_t *bht, uint64_t pc, uint8_t taken) {
    uint64_t index = (pc ^ (pc >> 12)) % BHT_SIZE;
    bht->predictions[index] = taken;
}

// 2 位元分支歷史表 - 更穩定
typedef enum {
    STRONG_NOT_TAKEN = 0,  // 00
    WEAK_NOT_TAKEN = 1,     // 01
    WEAK_TAKEN = 2,        // 10
    STRONG_TAKEN = 3       // 11
} bht_2bit_state_t;

typedef struct {
    bht_2bit_state_t states[BHT_SIZE];
} bht_2bit_t;

uint8_t bht_2bit_predict(bht_2bit_t *bht, uint64_t pc) {
    uint64_t index = (pc ^ (pc >> 12)) % BHT_SIZE;
    return bht->states[index] >= WEAK_TAKEN ? 1 : 0;
}

void bht_2bit_update(bht_2bit_t *bht, uint64_t pc, uint8_t taken) {
    uint64_t index = (pc ^ (pc >> 12)) % BHT_SIZE;
    
    switch (bht->states[index]) {
        case STRONG_NOT_TAKEN:
            bht->states[index] = taken ? WEAK_NOT_TAKEN : STRONG_NOT_TAKEN;
            break;
        case WEAK_NOT_TAKEN:
            bht->states[index] = taken ? WEAK_TAKEN : STRONG_NOT_TAKEN;
            break;
        case WEAK_TAKEN:
            bht->states[index] = taken ? STRONG_TAKEN : WEAK_NOT_TAKEN;
            break;
        case STRONG_TAKEN:
            bht->states[index] = taken ? STRONG_TAKEN : WEAK_TAKEN;
            break;
    }
}

/*
2 位元狀態機:
        ┌──────────────────────────────────────────┐
        │                                          │
        ▼                                          │
   ┌─────────┐   taken ──► ┌─────────┐   taken ──► ┌────────────┐
   │ Strong  │            │  Weak   │             │   Strong   │
   │ Not     │ ◄── not    │  Not    │ ◄── not     │   Taken    │
   │ Taken   │   taken    │  Taken  │   taken     │            │
   │ (00)    │            │  (01)   │             │   (11)     │
   └─────────┘            └────┬────┘             └─────┬──────┘
                               │                          │
                               │ taken                    │ not taken
                               ▼                          ▼
                          ┌────────┐                ┌────────┐
                          │  Weak  │                │  Weak  │
                          │  Taken │                │  Taken │
                          │  (10)  │                │  (10)  │
                          └────────┘                └────────┘
*/
```

### 分支歷史寄存器 (BHR)

```c
// 全域分支歷史寄存器 (GHR)
// 記錄最近 N 個分支的走向

#define GHR_SIZE 16

typedef struct {
    uint16_t history;  // 16 位元歷史
} global_history_t;

uint8_t predict_global(global_history_t *ghr, bht_2bit_t *bht, uint64_t pc) {
    uint64_t index = (pc ^ (pc >> 12) ^ ghr->history) % BHT_SIZE;
    return bht->states[index] >= WEAK_TAKEN ? 1 : 0;
}

void update_global(global_history_t *ghr, uint8_t taken) {
    ghr->history = ((ghr->history << 1) | taken) & ((1 << GHR_SIZE) - 1);
}

// 模式歷史表 (PHT)
// 將分支歷史和分支地址結合起來

#define PHT_SIZE 4096
#define HISTORY_BITS 10

typedef struct {
    uint8_t counters[PHT_SIZE];  // 2 位元計數器
} pattern_history_table_t;

uint8_t pht_predict(pattern_history_table_t *pht, 
                   uint64_t pc, 
                   uint16_t history) {
    // 結合 PC 和歷史
    uint64_t index = ((pc ^ (pc >> 12)) ^ history) % PHT_SIZE;
    return pht->counters[index] >= 2 ? 1 : 0;  // 2 = WEAK_TAKEN
}

void pht_update(pattern_history_table_t *pht, 
                uint64_t pc, 
                uint16_t history,
                uint8_t taken) {
    uint64_t index = ((pc ^ (pc >> 12)) ^ history) % PHT_SIZE;
    
    if (taken && pht->counters[index] < 3) {
        pht->counters[index]++;
    } else if (!taken && pht->counters[index] > 0) {
        pht->counters[index]--;
    }
}
```

## 兩級分支預測器

### 分支歷史表組合

```c
// GAg: 全域歷史 + 全域分支表（所有分支共享一個歷史）
// Gshare: 簡單的全局歷史異或

typedef struct {
    uint8_t counters[8192];  // 13 位元索引
    uint16_t ghr;             // 12 位元全局歷史
} gshare_predictor_t;

uint8_t gshare_predict(gshare_predictor_t *pred, uint64_t pc) {
    uint64_t index = ((pc >> 2) ^ pred->ghr) & 0x1FFF;
    return pred->counters[index] >= 2;
}

void gshare_update(gshare_predictor_t *pred, uint64_t pc, uint8_t taken) {
    uint64_t index = ((pc >> 2) ^ pred->ghr) & 0x1FFF;
    
    if (taken && pred->counters[index] < 3)
        pred->counters[index]++;
    else if (!taken && pred->counters[index] > 0)
        pred->counters[index]--;
    
    pred->ghr = ((pred->ghr << 1) | taken) & 0xFFF;
}

// GAp: 全域歷史 + 每個分支的獨立分支表
typedef struct {
    uint8_t counters[256];  // 每個分支一個表
} per_branch_table_t;

typedef struct {
    per_branch_table_t tables[256];  // 256 個分支表
    uint8_t ghr;                     // 8 位元歷史
} gap_predictor_t;

uint8_t gap_predict(gap_predictor_t *pred, uint64_t pc) {
    uint8_t branch_id = pc & 0xFF;  // 分支 ID
    uint8_t index = pred->ghr;       // 使用歷史索引
    
    return pred->tables[branch_id].counters[index] >= 2;
}

// PAg: 分支本地歷史 + 全域分支表
// 每個分支有自己的本地歷史

#define LOCAL_HISTORY_BITS 8
#define NUM_LOCAL_HISTORIES 256

typedef struct {
    uint16_t local_history[NUM_LOCAL_HISTORIES];  // 每個分支的歷史
    uint8_t pht[NUM_LOCAL_HISTORIES][256];        // 每個分支的分支表
} pag_predictor_t;

uint8_t pag_predict(pag_predictor_t *pred, uint64_t pc) {
    uint8_t branch_id = pc & 0xFF;
    uint8_t history = pred->local_history[branch_id];
    
    return pred->pht[branch_id][history] >= 2;
}

// 混合分支預測器 - 結合多個預測器
typedef struct {
    gshare_predictor_t *gshare;
    bht_2bit_t *local;
    uint8_t chooser[1024];  // 選擇器表
} hybrid_predictor_t;

enum predictor_type { GSHARE, LOCAL, NUM_PREDICTORS };

uint8_t hybrid_predict(hybrid_predictor_t *pred, uint64_t pc) {
    uint8_t gshare_pred = gshare_predict(pred->gshare, pc);
    uint8_t local_pred = bht_2bit_predict(pred->local, pc);
    
    uint64_t index = ((pc >> 2) ^ pred->gshare->ghr) % 1024;
    uint8_t chooser = pred->chooser[index];
    
    // 選擇器: 0-1 選 local, 2-3 選 gshare
    return (chooser < 2) ? local_pred : gshare_pred;
}
```

###  tournament 預測器

```c
// tournament 預測器邏輯（Alpha 21264 使用）
// 使用選擇器在多個預測器之間切換

typedef struct {
    uint8_t gshare_pred;     // 0-1: 選 local, 2-3: 選 gshare
    uint8_t local_pred;
    uint8_t gshare_meta;     // 全局元預測器
} tournament_state_t;

uint8_t tournament_predict(tournament_state_t *state) {
    if (state->chooser < 2) {
        return state->local_pred;
    } else {
        return state->gshare_pred;
    }
}

void tournament_update(tournament_state_t *state,
                       uint8_t taken,
                       uint8_t local_correct,
                       uint8_t gshare_correct) {
    // 更新元預測器
    if (local_correct != gshare_correct) {
        if (local_correct) {
            // local 正確，傾向選擇 local
            if (state->chooser > 0) state->chooser--;
        } else {
            // gshare 正確，傾向選擇 gshare
            if (state->chooser < 3) state->chooser++;
        }
    }
    
    // 更新子預測器
    // ...
}

/*
tournament 選擇器狀態機:
chooser = 0: strongly prefer local
chooser = 1: weakly prefer local
chooser = 2: weakly prefer gshare
chooser = 3: strongly prefer gshare

local 正確且 chooser > 0: chooser--
gshare 正確且 chooser < 3: chooser++
*/
```

## 退回預測

### 函數返回堆疊

```c
// 函數返回是特殊的分支，可以高度預測

#define RETURN_STACK_SIZE 32

typedef struct {
    uint64_t addresses[RETURN_STACK_SIZE];
    uint8_t top;
    uint8_t size;
} return_stack_t;

return_stack_t rs;

void rs_push(return_stack_t *rs, uint64_t addr) {
    if (rs->size < RETURN_STACK_SIZE) {
        rs->addresses[rs->size++] = addr;
    }
}

uint64_t rs_pop(return_stack_t *rs) {
    if (rs->size > 0) {
        return rs->addresses[--rs->size];
    }
    return 0;  // 空棧
}

// 呼叫和返回配對
void handle_call(uint64_t return_addr) {
    rs_push(&rs, return_addr);
}

uint64_t handle_return() {
    return rs_pop(&rs);
}

// 預測返回地址
uint64_t predict_return(uint64_t current_pc) {
    if (rs.size > 0) {
        return rs.addresses[rs.size - 1];  // 棧頂元素
    }
    return current_pc + 4;  // 預設：下一條指令
}
```

### 間接分支預測

```c
// 間接分支（switch/虛擬函數調用）更難預測
// 需要目標快取

#define TARGET_BUFFER_SIZE 256

typedef struct {
    uint64_t pc;           // 分支地址
    uint64_t target;       // 目標地址
    uint8_t confidence;    // 置信度
} target_entry_t;

typedef struct {
    target_entry_t entries[TARGET_BUFFER_SIZE];
} target_buffer_t;

uint64_t predict_indirect(target_buffer_t *btb, uint64_t pc) {
    uint64_t index = (pc >> 2) % TARGET_BUFFER_SIZE;
    
    if (btb->entries[index].pc == pc && 
        btb->entries[index].confidence > 128) {
        return btb->entries[index].target;
    }
    
    // 沒有找到，預測返回地址或順序執行
    return pc + 4;
}

void update_target_buffer(target_buffer_t *btb, 
                         uint64_t pc, 
                         uint64_t actual_target,
                         uint8_t confidence) {
    uint64_t index = (pc >> 2) % TARGET_BUFFER_SIZE;
    
    // 使用置信度更新
    if (btb->entries[index].pc != pc || 
        confidence > btb->entries[index].confidence) {
        btb->entries[index].pc = pc;
        btb->entries[index].target = actual_target;
        btb->entries[index].confidence = confidence;
    }
}

// 最後一次值預測器
// 假設間接分支返回到上一次返回的地址

typedef struct {
    uint64_t last_return_target;
} last_value_predictor_t;

uint64_t predict_last_value(last_value_predictor_t *pred, 
                             uint64_t current_pc) {
    return pred->last_return_target;
}

void update_last_value(last_value_predictor_t *pred, 
                       uint64_t actual_target) {
    pred->last_return_target = actual_target;
}
```

## 程式設計優化

### 分支友好程式設計

```c
// 1. 將常見情況放在 if 分支中
// 改變前:
int process(char c) {
    if (c >= 'a' && c <= 'z') {
        return to_upper(c);
    } else if (c >= 'A' && c <= 'Z') {
        return to_lower(c);
    } else if (c >= '0' && c <= '9') {
        return c;
    } else {
        return -1;
    }
}

// 改變後: 如果大部分是小寫字母
int process_optimized(char c) {
    if (likely(c >= 'a' && c <= 'z')) {  // 常見情況放這裡
        return to_upper(c);
    }
    if (c >= 'A' && c <= 'Z') {
        return to_lower(c);
    }
    if (c >= '0' && c <= '9') {
        return c;
    }
    return -1;
}

// 2. 消除分支：使用查找表
static const int8_t transform_table[256] = { /* ... */ };

int process_lookup(char c) {
    return transform_table[(unsigned char)c];
}

// 3. 展開循環以減少分支
// 改變前:
void sum_array(int *arr, int n, int *result) {
    *result = 0;
    for (int i = 0; i < n; i++) {
        *result += arr[i];
    }
}

// 改變後 (8 路展開):
void sum_array_unrolled(int *arr, int n, int *result) {
    int sum0 = 0, sum1 = 0, sum2 = 0, sum3 = 0;
    int sum4 = 0, sum5 = 0, sum6 = 0, sum7 = 0;
    
    int i = 0;
    int limit = n - 7;
    
    for (; i < limit; i += 8) {
        sum0 += arr[i];
        sum1 += arr[i + 1];
        sum2 += arr[i + 2];
        sum3 += arr[i + 3];
        sum4 += arr[i + 4];
        sum5 += arr[i + 5];
        sum6 += arr[i + 6];
        sum7 += arr[i + 7];
    }
    
    // 處理尾部
    for (; i < n; i++) {
        sum0 += arr[i];
    }
    
    *result = sum0 + sum1 + sum2 + sum3 + sum4 + sum5 + sum6 + sum7;
}

// 4. 使用 likely/unlikely 提示編譯器
// 現代編譯器使用這些提示來:
void process_data(data_t *data) {
    if (unlikely(data == NULL)) {  // 罕見情況
        return NULL;
    }
    
    if (likely(data->valid)) {     // 常見情況
        return process(data);
    }
    
    return NULL;
}

// 5. 替換分支為算術運算
// 改變前:
int absolute(int x) {
    if (x < 0) return -x;
    return x;
}

// 改變後 (無分支):
int absolute_branchless(int x) {
    int mask = x >> 31;  // 全 0 或全 1
    return (x + mask) ^ mask;
}
```

## 總結

分支預測是現代處理器實現高效能的關鍵技術。從簡單的靜態預測到複雜的混合預測器，設計者不斷挑戰更高的準確率。對於程式設計者來說，理解分支預測的原理可以幫助編寫更高效的代碼：使用 likely/unlikely 提示、避免複雜的分支依賴、將常見情況放在快速路徑、以及在適當時用算術運算或查找表替換分支。

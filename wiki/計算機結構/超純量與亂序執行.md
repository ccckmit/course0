# 超純量與亂序執行：高效能 CPU 的核心技術

## 概述

超純量（Superscalar）和亂序執行（Out-of-Order Execution）是現代高效能處理器實現高指令吞吐量的兩大核心技術。超純量架構允許 CPU 在每個時鐘週期發射多條指令到多個執行單元，實現指令級並行（ILP）。亂序執行則放寬了指令必須按序執行的限制，讓 CPU 能夠根據執行單元的可用性和資料依賴關係動態調整指令順序，最大化執行效率。

想像一個廚房：順序執行就像廚師必須嚴格按照食譜順序操作（先切菜、再炒菜、最後調味）；而亂序執行則允許廚師在等待水燒開的同時開始切菜，只要最終能按時完成一道菜即可。這種靈活性是現代 CPU 達到每時鐘週期執行數十條指令的關鍵。

## 超純量架構

### 指令發射與執行

```c
// 超純量處理器的指令發射示意
/*
典型的 4 路超純量執行:
每個時鐘週期最多發射 4 條指令

時鐘週期 1:
┌─────────┬─────────┬─────────┬─────────┐
│ Inst 1  │ Inst 2  │ Inst 3  │ Inst 4  │
│ (ALU)   │ (ALU)   │ (LOAD)  │ (STORE) │
└─────────┴─────────┴─────────┴─────────┘
     │          │          │          │
     ▼          ▼          ▼          ▼
┌─────────┬─────────┬─────────┬─────────┐
│ Exec 1  │ Exec 2  │ Exec 3  │ Exec 4  │
│ (1 cyc) │ (1 cyc) │ (3 cyc) │ (2 cyc) │
└─────────┴─────────┴─────────┴─────────┘

時鐘週期 2:
┌─────────┬─────────┬─────────┬─────────┐
│ Inst 5  │ Inst 6  │ Inst 7  │ Inst 8  │
│ (ALU)   │ (ALU)   │ (FPMUL) │ (FPDIV) │
└─────────┴─────────┴─────────┴─────────┘
*/

// 執行單元延遲表
typedef struct {
    const char *name;
    int latency;      // 延遲週期數
    int reciprocal;    // 倒數吞吐率
} execution_unit_t;

execution_unit_t execution_units[] = {
    {"Integer ALU", 1, 1},      // 1 週期延遲，每週期 1 條
    {"Integer Multiply", 3, 1}, // 3 週期延遲
    {"Load/Store", 4, 1},        // 4 週期延遲
    {"FP Add", 4, 1},           // 4 週期延遲
    {"FP Multiply", 5, 2},       // 5 週期延遲，每 2 週期 1 條
    {"FP Divide", 15, 15},      // 15 週期延遲，每 15 週期 1 條
    {"Branch Unit", 1, 2},      // 1 週期延遲，每 2 週期 1 條
};
```

### 指令調度示意

```c
// 簡化的指令調度演算法
typedef enum {
    ISSUE_WIDTH = 4,              // 每週期最多發射 4 條
    NUM_REGISTERS = 32,
    NUM_RESERVATION_STATIONS = 36
} cpu_constants_t;

typedef enum {
    FU_NONE = 0,
    FU_INT,
    FU_FP,
    FU_LOAD,
    FU_STORE,
    FU_BRANCH
} functional_unit_t;

typedef struct {
    uint8_t opcode;
    uint8_t dest_reg;      // 目的暫存器
    uint8_t src1_reg;      // 源暫存器 1
    uint8_t src2_reg;      // 源暫存器 2
    bool src1_ready;       // 源 1 是否就緒
    bool src2_ready;       // 源 2 是否就緒
    int src1_value;        // 源 1 的值
    int src2_value;        // 源 2 的值
    functional_unit_t fu;  // 需要的功能單元
    int latency;           // 執行延遲
    bool issued;           // 是否已發射
} instruction_t;

// 預約站結構
typedef struct {
    instruction_t *inst;
    uint8_t rob_index;     // 在 reorder buffer 中的位置
    int remaining_cycles;  // 剩餘執行週期
    bool busy;
} reservation_station_t;

reservation_station_t stations[NUM_RESERVATION_STATIONS];

// 發射邏輯
int issue_instructions(instruction_t *inst_queue, int num_insts) {
    int issued = 0;
    
    for (int i = 0; i < num_insts && issued < ISSUE_WIDTH; i++) {
        instruction_t *inst = &inst_queue[i];
        
        if (inst->issued) continue;  // 已發射
        if (!inst->src1_ready || !inst->src2_ready) continue;  // 操作數未就緒
        
        // 檢查是否有可用的功能單元
        if (!is_functional_unit_available(inst->fu)) continue;
        
        // 檢查是否有可用的預約站
        int station = find_available_station();
        if (station < 0) continue;
        
        // 發射指令
        stations[station].inst = inst;
        stations[station].busy = true;
        stations[station].remaining_cycles = inst->latency;
        inst->issued = true;
        issued++;
    }
    
    return issued;
}

// 執行週期更新
void execute_cycle() {
    // 更新預約站
    for (int i = 0; i < NUM_RESERVATION_STATIONS; i++) {
        if (stations[i].busy && stations[i].remaining_cycles > 0) {
            stations[i].remaining_cycles--;
            
            if (stations[i].remaining_cycles == 0) {
                // 執行完成，發送到完成佇列
                complete_instruction(stations[i].inst);
                stations[i].busy = false;
            }
        }
    }
}
```

## 亂序執行引擎

### Tomasulo 演算法

```c
// Tomasulo 演算法的核心結構
/*
Tomasulo 演算法使用:
1. 預約站 (Reservation Stations)
2. 寄存器重命名
3. 公共資料匯流排 (CDB)

避免 WAR 和 WAW 衝突
*/

typedef struct {
    uint8_t op;           // 操作碼
    uint8_t qj, qk;      // 源操作數的預約站號（0 表示就緒）
    uint8_t vj, vk;       // 源操作數的值
    bool busy;            // 預約站是否忙碌
    uint8_t dest;         // 目的寄存器
} reservation_station_entry_t;

reservation_station_entry_t reservation_stations[ROB_SIZE];

// 寄存器重命名
typedef struct {
    uint8_t physical_reg;  // 物理寄存器號
    bool ready;             // 值是否就緒
    int value;              // 值（如果就緒）
} renamed_register_t;

renamed_register_t arch_regs[32];    // 結構寄存器
renamed_register_t phys_regs[128];   // 物理寄存器

// 分配物理寄存器
uint8_t allocate_physical_reg() {
    for (int i = 0; i < 128; i++) {
        if (!phys_regs[i].busy) {
            phys_regs[i].busy = true;
            return i;
        }
    }
    return -1;  // 分配失敗
}

// Tomasulo 發射階段
void issue Tomasulo(instruction_t inst) {
    // 查找空的預約站
    int station = find_empty_station(inst.op);
    if (station < 0) return;  // 預約站滿
    
    // 寄存器重命名 - 目的寄存器
    uint8_t phys_dest = allocate_physical_reg();
    rename_map[inst.arch_dest] = phys_dest;
    
    // 檢查源操作數
    if (phys_regs[rename_map[inst.src1]].ready) {
        vj = phys_regs[rename_map[inst.src1]].value;
        qj = 0;  // 就緒
    } else {
        vj = 0;
        qj = rename_map[inst.src1];  // 等待該預約站
    }
    
    // 發射到預約站
    reservation_stations[station] = (reservation_station_entry_t) {
        .op = inst.op,
        .qj = qj,
        .qk = qk,
        .vj = vj,
        .vk = vk,
        .dest = phys_dest,
        .busy = true
    };
}

// 執行階段
void execute() {
    for (int i = 0; i < NUM_STATIONS; i++) {
        if (!reservation_stations[i].busy) continue;
        
        // 檢查操作數是否就緒
        if (reservation_stations[i].qj == 0 && 
            reservation_stations[i].qk == 0) {
            
            // 執行指令
            int result = execute_operation(reservation_stations[i]);
            
            // 結果寫回 CDB
            write_to_cdb(reservation_stations[i].dest, result);
            
            // 標記預約站空閒
            reservation_stations[i].busy = false;
        }
    }
}

// CDB 廣播
void write_to_cdb(uint8_t phys_reg, int value) {
    // 更新所有等待此結果的預約站
    for (int i = 0; i < NUM_STATIONS; i++) {
        if (reservation_stations[i].qj == phys_reg) {
            reservation_stations[i].vj = value;
            reservation_stations[i].qj = 0;
        }
        if (reservation_stations[i].qk == phys_reg) {
            reservation_stations[i].vk = value;
            reservation_stations[i].qk = 0;
        }
    }
    
    // 更新寄存器
    phys_regs[phys_reg].value = value;
    phys_regs[phys_reg].ready = true;
}
```

### Reorder Buffer (ROB)

```c
// 順序提交確保精確定異常
typedef struct {
    uint8_t dest_phys_reg;    // 目的物理寄存器
    uint8_t dest_arch_reg;    // 目的結構寄存器
    int value;                // 執行結果
    bool ready;               // 是否就緒
    bool valid;               // 是否有有效條目
    instruction_type_t type;   // 指令類型
} rob_entry_t;

rob_entry_t reorder_buffer[ROB_SIZE];
uint8_t rob_head = 0;   // 最老的指令
uint8_t rob_tail = 0;  // 下一個空槽
uint8_t rob_count = 0;

// 添加到 ROB
int rob_add(instruction_t inst, uint8_t phys_dest) {
    if (rob_count >= ROB_SIZE) return -1;
    
    reorder_buffer[rob_tail] = (rob_entry_t) {
        .dest_phys_reg = phys_dest,
        .dest_arch_reg = inst.arch_dest,
        .ready = false,
        .valid = true,
        .type = inst.type
    };
    
    rob_tail = (rob_tail + 1) % ROB_SIZE;
    rob_count++;
    
    return (rob_tail - 1 + ROB_SIZE) % ROB_SIZE;  // 返回索引
}

// 提交階段 - 按順序提交已完成指令
void commit() {
    if (rob_count == 0) return;
    
    rob_entry_t *entry = &reorder_buffer[rob_head];
    
    // 只能提交已完成的指令
    if (!entry->ready || !entry->valid) return;
    
    // 根據指令類型提交
    switch (entry->type) {
        case INST_STORE:
            // 寫入記憶體
            memory[entry->address] = entry->value;
            break;
            
        case INST_BRANCH:
            // 更新 PC
            if (entry->value != 0) {
                // 分支taken
                flush_pipeline();
            }
            break;
            
        default:
            // 寄存器寫回
            commit_arch_reg(entry->dest_arch_reg, entry->dest_phys_reg);
            break;
    }
    
    // 釋放物理寄存器
    free_physical_reg(entry->dest_phys_reg);
    
    // 移動 head
    reorder_buffer[rob_head].valid = false;
    rob_head = (rob_head + 1) % ROB_SIZE;
    rob_count--;
}

// 處理分支錯誤預測
void handle_branch_misprediction(uint8_t rob_index) {
    // 刷新所有在錯誤分支之後的指令
    while (rob_tail != rob_index) {
        rob_tail = (rob_tail - 1 + ROB_SIZE) % ROB_SIZE;
        rob_count--;
        // 恢復寄存器...
    }
    
    // 恢復分支預測器狀態...
    flush_pipeline();
}
```

## 寄存器重命名

```c
// 寄存器重命名避免虛假依賴

/*
WAR (Write-After-Read) 衝突:
   add r1, r2, r3    ; r1 = r2 + r3
   sub r2, r4, r5    ; r2 = r4 - r5  <- 讀取被覆蓋

WAW (Write-After-Write) 衝突:
   add r1, r2, r3    ; r1 = r2 + r3
   add r1, r4, r5    ; r1 = r4 + r5  <- 寫入順序顛倒

寄存器重命名可以消除這些虛假依賴:
*/

typedef struct {
    uint8_t physical;   // 當前映射的物理寄存器
    uint8_t pending;     // 掛起的物理寄存器
} rename_entry_t;

rename_entry_t rename_table[32];
uint8_t free_list_head = 32;  // 物理寄存器分配
uint8_t free_list[96];       // 空閒列表

// 分配新物理寄存器
uint8_t allocate_reg() {
    if (free_list_head >= 128) return -1;  // 無可用
    return free_list[free_list_head++];
}

// 重命名指令
renamed_instruction_t rename(instruction_t inst) {
    renamed_instruction_t renamed;
    
    // 重命名目的寄存器
    renamed.dest_phys = allocate_reg();
    renamed.dest_arch = inst.dest;
    rename_table[inst.dest].physical = renamed.dest_phys;
    
    // 重命名源寄存器
    renamed.src1_phys = rename_table[inst.src1].physical;
    renamed.src2_phys = rename_table[inst.src2].physical;
    
    return renamed;
}

/*
示例 - WAR 消除:
指令 1: add r1, r2, r3  ->  p5 = p2 + p3, map[r1] = p5
指令 2: sub r2, r4, r5  ->  p6 = p4 + p5, map[r2] = p6
          原本會有 WAR 衝突（指令 2 寫 r2，但指令 1 讀 r2）
          重命名後，指令 1 讀 p2，指令 2 寫 p6，無衝突
*/
```

## 亂序執行限制

### 記憶體排序

```c
// 記憶體依賴更難處理
/*
load/store 指令的依賴關係:

1. Load-Load ordering:
   必須保證 program order 中的 load 在之後的 load 之前完成
   否則會讀取過時資料

2. Load-Store ordering:
   program order 中的 load 必須在 store 之前
   否則可能讀取到尚未寫入的值

3. Store-Load ordering (最困難):
   可能需要從 store buffer 中讀取（store-to-load forwarding）
   或等待 store 完成
*/

typedef struct {
    uint64_t address;
    uint64_t value;
    uint8_t size;
    bool valid;
    bool completed;        // 是否已完成
} store_buffer_entry_t;

store_buffer_entry_t store_buffer[STORE_BUFFER_SIZE];
int store_buffer_head = 0;
int store_buffer_tail = 0;

// Store-to-Load 轉發
int64_t load_with_forwarding(uint64_t addr, int size) {
    // 首先檢查 store buffer
    for (int i = store_buffer_head; i != store_buffer_tail; 
         i = (i + 1) % STORE_BUFFER_SIZE) {
        if (store_buffer[i].valid && 
            store_buffer[i].address == addr &&
            store_buffer[i].completed) {
            // 從 store buffer 轉發
            return extract_value(store_buffer[i].value, addr, size);
        }
    }
    
    // 否則從記憶體/l1 cache 加載
    return load_from_memory(addr, size);
}

// 記憶體屏障確保 ordering
void memory_barrier() {
    // 確保 store buffer 中的所有 store 都完成
    while (store_buffer_head != store_buffer_tail) {
        wait_for_store_complete();
    }
}
```

## 效能分析

```c
// IPC (Instructions Per Cycle) 計算
typedef struct {
    uint64_t total_instructions;
    uint64_t total_cycles;
    uint64_t branch_mispredicts;
    uint64_t cache_misses;
    uint64_t pipeline_stalls;
} performance_counters_t;

double calculate_ipc(performance_counters_t *perf) {
    return (double)perf->total_instructions / perf->total_cycles;
}

// 影響因素分析
void analyze_performance(performance_counters_t *perf) {
    double ipc = calculate_ipc(perf);
    
    // 每週期理論最大 IPC
    double issue_width = 4.0;
    double utilization = ipc / issue_width;
    
    printf("IPC: %.2f\n", ipc);
    printf("Issue width utilization: %.1f%%\n", utilization * 100);
    
    // 分支錯誤預測影響
    double mispredict_penalty = 15.0;  // 典型值
    double mispredict_stalls = 
        (double)perf->branch_mispredicts * mispredict_penalty;
    double stall_percentage = 
        mispredict_stalls / perf->total_cycles * 100;
    printf("Branch mispredict stall cycles: %.1f%%\n", stall_percentage);
    
    // 快取缺失影響
    double cache_miss_penalty = 100.0;  // 典型值
    double cache_stalls = 
        (double)perf->cache_misses * cache_miss_penalty;
    double cache_percentage = cache_stalls / perf->total_cycles * 100;
    printf("Cache miss stall cycles: %.1f%%\n", cache_percentage);
}

// Stall 類型分類
typedef enum {
    STALL_NONE = 0,
    STALL_BRANCH_MISPREDICT,
    STALL_CACHE_MISS,
    STALL_RESOURCE_CONFLICT,
    STALL_DEPENDENCY,
    STALL_RESERVATION_STATION_FULL
} stall_type_t;
```

## 總結

超純量和亂序執行是現代 CPU 實現高效能的兩大支柱。超純量通過在同一時鐘週期發射多條指令到多個執行單元來提高指令級並行度。亂序執行則通過寄存器重命名、預約站和 reorder buffer 等機制，在不違反程式碼語義的前提下最大化執行效率。

理解這些概念對於效能優化和並行程式設計至關重要。雖然硬體自動處理大多數複雜性，但程式設計者仍可通過減少分支錯誤預測、改善資料局部性和避免記憶體依賴等方式幫助硬體更好地工作。

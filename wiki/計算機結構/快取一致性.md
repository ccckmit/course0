# 快取一致性：多核處理器的資料同步

## 概述

在現代多核處理器中，每個核心通常擁有自己的私有快取層（L1、L2），而所有核心共享最後一級快取（LLC）和主記憶體。這種架構雖然提高了效能，卻帶來了快取一致性（Cache Coherence）的挑戰：當多個核心同時持有同一記憶體位置的不同副本時，如何確保它們看到一致的資料？

快取一致性是並行計算中的核心問題之一。想像一個簡單的場景：核心 A 和核心 B 同時讀取記憶體位置 X，兩者都將其快取到自己的 L1 中。現在如果核心 A 想要寫入 X，它會更新自己的快取副本，但如果核心 B 的快取副本沒有被通知，B 仍會看到舊值，這就造成了一致性問題。

處理器設計者開發了多種協定來解決這個問題，最著名的是 MESI 協定及其變體。本頁將深入探討快取一致性的原理、實現協定以及程式設計中的最佳實踐。

## 快取層次結構

### 多級快取架構

```c
// 典型多核處理器的快取層次結構
/*
       CPU 核心 0          CPU 核心 1          CPU 核心 2          CPU 核心 3
           |                   |                   |                   |
     +------+------+      +------+------+      +------+------+      +------+------+
     | L1d  | L1i  |      | L1d  | L1i  |      | L1d  | L1i  |      | L1d  | L1i  |
     | 32KB | 32KB |      | 32KB | 32KB |      | 32KB | 32KB |      | 32KB | 32KB |
     +------+------+      +------+------+      +------+------+      +------+------+
           |                   |                   |                   |
           +-------------------+-------------------+-------------------+
                               |
                         +------+------+
                         |    L2     |
                         |   256KB   |
                         +------------+
                               |
                         +------+------+
                         |    L3      |
                         |   8MB     |
                         +------------+
                               |
                         +------+------+
                         |  主記憶體   |
                         |   32GB     |
                         +------------+
*/

// 模擬快取行結構
#define CACHE_LINE_SIZE 64

typedef struct {
    uint64_t address;           // 標籤地址
    uint8_t data[CACHE_LINE_SIZE];  // 64 bytes 資料
    uint8_t state;             // MESI 狀態
    uint8_t core_id;           // 持有者的核心 ID
    uint64_t last_access;       // 上次訪問時間
} cache_line_t;

// MESI 狀態定義
typedef enum {
    MESI_INVALID = 0,           // I: 快取行無效
    MESI_SHARED = 1,            // S: 快取行乾淨但可能有多個副本
    MESI_EXCLUSIVE = 2,         // E: 快取行乾淨且只有一個副本
    MESI_MODIFIED = 3           // M: 快取行已修改，必須寫回
} mesi_state_t;
```

```c
    BUS_READ_EXCLUSIVE,      // 讀取並準備獨佔
    
    // 寫入相關
    BUS_WRITE,              // 寫入快取行
    BUS_UPGRADE,            // 從共享升級到獨佔
```

## MESI 協定

### 狀態轉換圖

```c
// MESI 協定狀態機實現

typedef struct {
    mesi_state_t state;
    bool waiting_for_response;
    bool pending_upgrade;
} core_cache_state_t;

// 狀態轉換表
/*
狀態轉換 (基於本地核心操作):
┌────────┬─────────────┬─────────────────────┬────────────────┐
│ 當前狀態 │  操作       │ 新狀態              │  總線事務       │
├────────┼─────────────┼─────────────────────┼────────────────┤
│ I      │ Read Miss   │ E (或 S 如果有共享) │ BusRd           │
│ I      │ Write Miss  │ M                   │ BusRdX          │
│ S      │ Read Hit    │ S                   │ -              │
│ S      │ Write Hit   │ M                   │ BusUpgr         │
│ E      │ Read Hit    │ E                   │ -              │
│ E      │ Write Hit   │ M                   │ -              │
│ M      │ Read Hit    │ M                   │ -              │
│ M      │ Write Hit   │ M                   │ -              │
└────────┴─────────────┴─────────────────────┴────────────────┘

狀態轉換 (基於其他核心操作 - 窺探):
┌────────┬─────────────┬─────────────────────┬────────────────┐
│ 當前狀態 │  總線事件    │ 新狀態              │  動作           │
├────────┼─────────────┼─────────────────────┼────────────────┤
│ S      │ BusRd        │ S                   │ -              │
│ S      │ BusRdX/BusUpg│ I                   │ -              │
│ E      │ BusRd        │ S                   │ 供應資料        │
│ E      │ BusRdX/BusUpg│ I                   │ 供應資料        │
│ M      │ BusRd        │ S                   │ 供應資料+寫回   │
│ M      │ BusRdX/BusUpg│ I                   │ 供應資料+寫回   │
└────────┴─────────────┴─────────────────────┴────────────────┘
*/

// 處理本地讀取
void handle_local_read(core_cache_state_t *state, uint64_t addr) {
    switch (state->state) {
        case MESI_INVALID:
            // 發送 BusRd 請求
            send_bus_request(BUS_READ, addr);
            // 等待回應
            state->waiting_for_response = true;
            wait_for_snoop_response(state);
            state->waiting_for_response = false;
            // 根據是否有其他共享者設置狀態
            if (other_cores_have_shared_copy(addr)) {
                state->state = MESI_SHARED;
            } else {
                state->state = MESI_EXCLUSIVE;
            }
            break;
            
        case MESI_SHARED:
        case MESI_EXCLUSIVE:
            // 快取命中
            state->state = (state->state == MESI_SHARED) ? MESI_SHARED : MESI_EXCLUSIVE;
            break;
            
        case MESI_MODIFIED:
            // 快取命中，無需操作
            break;
    }
}

// 處理本地寫入
void handle_local_write(core_cache_state_t *state, uint64_t addr) {
    switch (state->state) {
        case MESI_INVALID:
            // 發送 BusRdX 請求
            send_bus_request(BUS_READ_EXCLUSIVE, addr);
            wait_for_snoop_response(state);
            state->state = MESI_MODIFIED;
            break;
            
        case MESI_SHARED:
            // 發送升級請求
            send_bus_request(BUS_UPGRADE, addr);
            // 無需等待資料，但需要失效確認
            state->state = MESI_MODIFIED;
            break;
            
        case MESI_EXCLUSIVE:
            // 只需改變狀態
            state->state = MESI_MODIFIED;
            break;
            
        case MESI_MODIFIED:
            // 快取命中
            break;
    }
}

// 處理窺探到的總線事務
void handle_snoop_request(cache_line_t *line, coherence_cmd_t cmd, 
                         uint64_t addr, core_cache_state_t *state) {
    if (line->address != addr) return;
    
    switch (line->state) {
        case MESI_SHARED:
            if (cmd == BUS_READ) {
                // 保持共享，無需操作
            } else if (cmd == BUS_READ_EXCLUSIVE || cmd == BUS_UPGRADE) {
                // 需要失效
                line->state = MESI_INVALID;
                send_invalidate_ack();
            }
            break;
            
        case MESI_EXCLUSIVE:
            if (cmd == BUS_READ) {
                // 變為共享，並提供資料
                line->state = MESI_SHARED;
                provide_data_to_requestor();
            } else if (cmd == BUS_READ_EXCLUSIVE || cmd == BUS_UPGRADE) {
                // 變為失效，並提供資料
                line->state = MESI_INVALID;
                provide_data_to_requestor();
            }
            break;
            
        case MESI_MODIFIED:
            if (cmd == BUS_READ) {
                // 提供資料並寫回記憶體
                provide_data_and_writeback(line->data);
                line->state = MESI_SHARED;
            } else if (cmd == BUS_READ_EXCLUSIVE || cmd == BUS_UPGRADE) {
                // 提供資料並寫回記憶體
                provide_data_and_writeback(line->data);
                line->state = MESI_INVALID;
            }
            break;
            
        case MESI_INVALID:
            // 無需操作
            break;
    }
}
```

### MOESI 和 MESIF 擴展

```c
// MOESI 協定添加 Owned 狀態
typedef enum {
    MOESI_INVALID = 0,
    MOESI_SHARED = 1,
    MOESI_EXCLUSIVE = 2,
    MOESI_MODIFIED = 3,
    MOESI_OWNED = 4      // 所有者持有「髒」資料，但聲稱為乾淨
} moesi_state_t;

/*
MOESI 狀態轉換:
- O 狀態允許在 Modified 和 Shared 之間轉換而不立即寫回
- 當 Modified 資料被其他核心請求時，可以變為 Owned 並提供資料
- 避免了立即寫回記憶體的需求
*/

// MESIF 協定（Intel 使用）
typedef enum {
    MESIF_INVALID = 0,
    MESIF_SHARED = 1,
    MESIF_EXCLUSIVE = 2,
    MESIF_FORWARD = 3,   // Forward 狀態
    MESIF_MODIFIED = 4
} mesif_state_t;

/*
MESIF 特點:
- 只有一個核心可以處於 Forward 狀態
- Forward 狀態的核心負責響應其他核心的讀取請求
- 減少了總線流量，提高了效率
*/
```

```c
    uint8_t core_id;
    uint64_t timestamp;
    struct bus_request_queue *next;

// 窺探過濾器
// 大型系統中使用窺探過濾器減少不必要的總線流量
typedef struct snoop_filter {
    uint64_t *tags;           // 標籤陣列
    uint8_t *directory;       // 目錄訊息
    uint32_t num_entries;
    uint32_t associativity;
} snoop_filter_t;

// 窺探過濾器查詢
snoop_result_t snoop_filter_lookup(snoop_filter_t *filter, uint64_t addr) {
    uint32_t index = (addr / CACHE_LINE_SIZE) % filter->num_entries;
    
    for (int i = 0; i < filter->associativity; i++) {
        if (filter->tags[index + i] == addr) {
            return (snoop_result_t) {
                .hit = true,
                .state = filter->directory[index + i],
                .core_id = get_owner(filter->directory[index + i])
            };
        }
    }
    
    return (snoop_result_t) { .hit = false };
}
```

### 點對點協議

```c
// 基於目錄的一致性協議
typedef struct {
    uint64_t addr;                    // 記憶體地址
    uint8_t state;                    // 目錄狀態
    uint8_t sharers[64];              // 共享者位圖
    uint8_t owner;                    // 當前所有者
} directory_entry_t;

// 目錄狀態機
typedef enum {
    DIR_UNCACHED = 0,     // 記憶體是最新的
    DIR_SHARED = 1,       // 一個或多個核心持有乾淨副本
    DIR_MODIFIED = 2,     // 只有一個核心持有「髒」副本
    DIR_TRANSITIONAL = 3  // 等待響應中
} directory_state_t;

// 處理遠程讀取請求
void handle_remote_read_request(directory_entry_t *entry, uint8_t requester) {
    switch (entry->state) {
        case DIR_UNCACHED:
            // 記憶體直接提供資料
            send_data_to_core(requester, read_from_memory(entry->addr));
            add_to_sharers(entry, requester);
            entry->state = DIR_SHARED;
            break;
            
        case DIR_SHARED:
            // 請求所有共享者之一提供資料
            uint8_t provider = get_first_sharer(entry);
            send_data_request_to_core(provider, requester);
            add_to_sharers(entry, requester);
            break;
            
        case DIR_MODIFIED:
            // 請求所有者提供資料並寫回
            send_writeback_request(entry->owner, entry->addr);
            send_data_request_to_core(entry->owner, requester);
            add_to_sharers(entry, entry->owner); // 原所有者仍是共享者
            entry->state = DIR_SHARED;
            break;
    }
}

// 處理遠程寫入請求
void handle_remote_write_request(directory_entry_t *entry, uint8_t requester) {
    switch (entry->state) {
        case DIR_UNCACHED:
            entry->state = DIR_TRANSITIONAL;
            send_data_to_core(requester, read_from_memory(entry->addr));
            set_owner(entry, requester);
            entry->state = DIR_MODIFIED;
            break;
            
        case DIR_SHARED:
            // 需要使所有共享者失效
            entry->state = DIR_TRANSITIONAL;
            broadcast_invalidate(entry);
            set_owner(entry, requester);
            entry->state = DIR_MODIFIED;
            clear_sharers(entry);
            break;
            
        case DIR_MODIFIED:
            // 需要使所有者寫回並失效
            send_writeback_and_invalidate(entry->owner, entry->addr);
            set_owner(entry, requester);
            entry->state = DIR_MODIFIED;
            break;
    }
}
```

## 記憶體排序和柵欄

### 記憶體模型

```c
// x86-64 使用 TSO (Total Store Order) 模型
// 所有核心看到所有 store 的順序是一致的
// 但允許 store buffering 和 store forwarding

// 展示 x86 的記憶體順序
void x86_memory_ordering_example() {
    int x = 0, y = 0;
    int r1, r2;
    
    // Thread 1:          // Thread 2:
    // x = 1;            // y = 1;
    // r1 = y;           // r2 = x;
    
    // 在 x86 上，這個結果 r1=0 && r2=0 不會發生
    // 因為 store 必須最終對所有核心可見
}

// ARM64 使用更弱的記憶體模型 (ARMv8 Memory Model)
// 允許更廣泛的重排序

// 展示 ARM 的記憶體順序
void arm_memory_ordering_example() {
    int x = 0, y = 0;
    int r1, r2;
    
    // Thread 1:          // Thread 2:
    // STR x, #1         // STR y, #1
    // LDR r1, y         // LDR r2, x
    
    // 在 ARM 上，r1=0 && r2=0 是可能的！
}

// C11/C++11 記憶體順序抽象
#include <stdatomic.h>

// 順序一致（最強）
void sequential_consistent() {
    atomic_int x = ATOMIC_VAR_INIT(0);
    atomic_store(&x, 1, memory_order_seq_cst);
    int val = atomic_load(&x, memory_order_seq_cst);
}

// 釋放語義（允許 reorder）
void release_semantics() {
    atomic_int x = ATOMIC_VAR_INIT(0);
    atomic_store(&x, 1, memory_order_release);
    // 之前的所有記憶體操作都會在 store 之前完成
}

// 獲取語義
void acquire_semantics() {
    atomic_int x = ATOMIC_VAR_INIT(0);
    // 之後的所有記憶體操作都會在 load 之後開始
    int val = atomic_load(&x, memory_order_acquire);
}

// 生產者-消費者模式
atomic_int data = ATOMIC_VAR_INIT(0);
atomic_int ready = ATOMIC_VAR_INIT(0);

void producer() {
    atomic_store(&data, 42, memory_order_relaxed);
    atomic_store(&ready, 1, memory_order_release);  // 釋放屏障
}

void consumer() {
    while (atomic_load(&ready, memory_order_acquire) == 0) {}  // 獲取屏障
    // 此時 data 一定已經被設置
    printf("%d\n", atomic_load(&data, memory_order_relaxed));
}
```

## 程式設計最佳實踐

### False Sharing 避免

```c
// False sharing: 不同核心訪問不同變量，但這些變量在同一快取行
typedef struct {
    atomic<int> counter0;  // 核心 0 使用
    atomic<int> counter1;  // 核心 1 使用
    // 如果在同一快取行，會導致頻繁的快取一致性流量
} bad_padding_t;

// 解決方案 1: 填充
typedef struct {
    atomic<int> counter0;
    char pad[CACHE_LINE_SIZE - sizeof(atomic<int>)];  // 填充到快取行邊界
    atomic<int> counter1;
    char pad2[CACHE_LINE_SIZE - sizeof(atomic<int>)];
} good_padding_t;

// 解決方案 2: 使用 C11/C++11 alignas
struct aligned_counters {
    alignas(64) atomic<int> counter0;
    alignas(64) atomic<int> counter1;
};

// 解決方案 3: 陣列結構重排列 (SoA vs AoS)
struct particle_bad {  // Array of Structures
    float x, y, z;
    float vx, vy, vz;
    float mass;
    float density;
};

// 更好的佈局
struct particle_system {
    float x[1000000];
    float y[1000000];
    float z[1000000];
    float vx[1000000];
    // ...
};
```

```c
void *xmalloc(size_t size) {
    void *ptr;
    while (1) {
        if (_xbegin() == _XBEGIN_STARTED) {

                return ptr;

} stack_node_t;

// 簡單實現有 ABA 問題
stack_node_t* pop_stack(atomic_stack_ptr *head) {
    stack_node_t *node = atomic_load(head);
    while (node && !atomic_compare_exchange_weak(head, &node, 
               atomic_load(&node->next))) {
        // 可能在這裡有其他線程修改了棧
    }
    return node;
}

// 使用 Tagged Pointers 解決 ABA
typedef struct {
    atomic_uintptr_t head;
    atomic<uint64_t> tag;
} tagged_stack_t;

void push_tagged(tagged_stack_t *stack, uintptr_t value) {
    uint64_t old_tag = atomic_load(&stack->tag);
    uint64_t new_tag = old_tag + 1;
    
    stack_node_t *new_node = malloc(sizeof(stack_node_t));
    new_node->value = value;
    atomic_store(&new_node->next, atomic_load(&stack->head) & PTR_MASK);
    
    uintptr_t expected = old_tag << TAG_BITS | (old_head & PTR_MASK);
    atomic_store(&stack->head, (new_tag << TAG_BITS) | (uintptr_t)new_node);
}
```

## 總結

快取一致性是現代多核系統的核心挑戰。MESI 協定及其變種（MOESI、MESIF）提供了解決方案，但程式設計者仍需理解記憶體排序、柵欄指令和原子操作的正確用法。避免 false sharing、使用適當的同步原語、以及理解目標架構的記憶體模型，都是編寫高效多執行緒程式碼的關鍵。隨著核心數量的增加，更複雜的快取一致性協議和更好的程式設計抽象將變得越來越重要。

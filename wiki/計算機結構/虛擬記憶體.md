# 虛擬記憶體 (Virtual Memory)

## 概述

虛擬記憶體是一種記憶體管理技術，讓每個程式擁有獨立的虛擬位址空間，作業系統透過分頁機制將虛擬位址映射到實體記憶體。

## 分頁機制

### 虛擬位址格式 (x86-64)
```
┌─────────────────────────────────────────────────────────────────┐
│  63        48 │ 47      39 │ 38      30 │ 29      21 │ 20    12 │ 11     0 │
├────────────────┼─────────────────┼─────────────────┼──────────────┼──────────┤
│      PML4      │     PDPT         │      PD         │      PT      │  Offset │
│    Index       │    Index         │    Index        │    Index     │          │
│    (9 bits)    │    (9 bits)      │    (9 bits)     │    (9 bits)  │ (12 bits)│
└────────────────┴─────────────────┴─────────────────┴──────────────┴──────────┘
                       4-Level Page Table Structure
```

### 頁表條目 (PTE) 格式
```
┌────────────────────────────────────────────────────────┐
│ 63  │ 51 │ 50 │ 49 │ 48 │ 47 │ 12 │ 11 │ 9 │ 8 │ 7 │ 6 │ 5 │ 4 │ 3 │ 2 │ 1 │ 0 │
├─────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┴────┤
│  XTD│  -  │  -  │  -  │  -  │ XD │  -  │ A │ PCD│ PWT│ U/S│ R/W│ P │               │
│                     PFN                      │                                        │
└───────────────────────────────────────────────┴────────────────────────────────────────┘

P   = Present (頁面是否存在於記憶體)
R/W = Read/Write (讀寫權限)
U/S = User/Supervisor (使用者/核心模式存取)
PWT = Write-Through
PCD = Cache-Disable
A   = Accessed (是否被存取過)
XD  = Execute-Disable (禁止執行)
```

## 分頁硬體 (MMU)

### 位址轉換流程
```
虛擬位址 (VA)                              實體位址 (PA)
┌─────────────────┐                       ┌─────────────────┐
│  VPN (Virtual   │                       │  PFN            │
│  Page Number)   │                       │  (Physical      │
│                 │                       │   Frame Number)  │
├─────────────────┤                       │                 │
│  Offset         │                       ├─────────────────┤
└────────┬────────┘                       │  Offset         │
         │                                 └─────────────────┘
         │
         ▼
┌─────────────────┐
│     MMU         │
│                 │
│  1. 查詢 TLBs   │
│  2. 若未命中    │
│     查頁表      │
│  3. 更新 TLB    │
│  4. 產生實體位址│
└────────┬────────┘
         │
         ▼
   ┌───────────┐
   │  Page     │
   │  Fault?   │
   └─────┬─────┘
         │
         ▼
   ┌───────────┐
   │  Yes      │ ──→ 作業系統處理分頁錯誤
   │  (未映射)  │
   └───────────┘
```

### 翻譯後備緩衝器 (TLB)
```c
typedef struct {
    uint64_t vpn;        // 虛擬頁號
    uint64_t ppn;        // 實體頁號
    uint64_t asid;       // 位址空間識別符
    uint8_t valid;
    uint8_t dirty;
    uint8_t ref;         // 參考位元
} tlb_entry_t;

tlb_entry_t tlb[64];  // 64-entry TLB

uint64_t tlb_lookup(uint64_t vpn, uint64_t asid) {
    for (int i = 0; i < 64; i++) {
        if (tlb[i].valid && tlb[i].vpn == vpn && tlb[i].asid == asid) {
            tlb[i].ref = 1;  // 參考位元
            return tlb[i].ppn;
        }
    }
    return TLB_MISS;
}
```

## 頁面置換演算法

### LRU (Least Recently Used)
```c
typedef struct {
    uint64_t vpn;
    uint64_t last_used;
    int valid;
} page_entry_t;

page_entry_t page_table[PAGE_TABLE_SIZE];
uint64_t current_time = 0;

uint64_t lru_replace(uint64_t vpn) {
    uint64_t oldest_time = UINT64_MAX;
    int oldest_idx = -1;
    
    // 找到最久未使用的頁面
    for (int i = 0; i < PAGE_TABLE_SIZE; i++) {
        if (page_table[i].valid && page_table[i].last_used < oldest_time) {
            oldest_time = page_table[i].last_used;
            oldest_idx = i;
        }
    }
    
    // 寫回磁碟（如果髒了）
    if (page_table[oldest_idx].dirty) {
        write_to_disk(page_table[oldest_idx].vpn);
    }
    
    // 載入新頁面
    page_table[oldest_idx].vpn = vpn;
    page_table[oldest_idx].last_used = current_time++;
    page_table[oldest_idx].valid = 1;
    
    return oldest_idx;
}
```

### Clock (Second Chance) 演算法
```c
typedef struct {
    uint64_t vpn;
    int valid;
    int referenced;  // 參考位元
    int dirty;
} frame_entry_t;

frame_entry_t frames[NUM_FRAMES];
int hand = 0;  // 時鐘指標

int clock_replace(uint64_t vpn) {
    while (1) {
        if (!frames[hand].referenced) {
            // 如果沒有被參考，置換出去
            if (frames[hand].dirty) {
                write_to_disk(frames[hand].vpn);
            }
            frames[hand].vpn = vpn;
            frames[hand].referenced = 1;
            frames[hand].dirty = 0;
            return hand;
        }
        frames[hand].referenced = 0;  // 給予第二次機會
        hand = (hand + 1) % NUM_FRAMES;
    }
}
```

## 交換空間管理

### 交換空間配置
```c
typedef struct {
    int free_blocks;       // 空閒區塊數
    uint64_t swap_base;    // 交換空間基底位址
    bitmap_t block_map;    // 區塊使用圖
} swap_space_t;

int swap_out(uint64_t vpn, uint64_t *swap_offset) {
    int block = find_free_block(&swap_space.block_map);
    if (block < 0) return -1;  // 交換空間不足
    
    *swap_offset = swap_space.swap_base + block * PAGE_SIZE;
    copy_to_swap(vpn, *swap_offset);
    set_bit(&swap_space.block_map, block);
    
    return 0;
}

int swap_in(uint64_t swap_offset, uint64_t ppn) {
    int block = (swap_offset - swap_space.swap_base) / PAGE_SIZE;
    copy_from_swap(swap_offset, ppn);
    clear_bit(&swap_space.block_map, block);
    return 0;
}
```

## 記憶體映射檔案

```c
#include <sys/mman.h>

int main() {
    // 建立記憶體映射
    int fd = open("data.bin", O_RDWR);
    void *addr = mmap(NULL,            // 由系統選擇位址
                      4096,             // 映射大小
                      PROT_READ | PROT_WRITE,
                      MAP_SHARED,       // 共享映射
                      fd,               // 檔案描述符
                      0);               // 檔案偏移
    
    if (addr == MAP_FAILED) {
        perror("mmap");
        return 1;
    }
    
    // 使用映射記憶體
    char *data = (char *)addr;
    data[0] = 'H';
    data[1] = 'i';
    
    // 解除映射
    munmap(addr, 4096);
    close(fd);
    
    return 0;
}
```

## 分頁錯誤處理

```c
void page_fault_handler(uint64_t fault_addr, int error_code) {
    uint64_t vpn = fault_addr / PAGE_SIZE;
    
    // 檢查是否為合法存取
    if (!is_valid_address(fault_addr)) {
        kill_process(current_process, SIGSEGV);
        return;
    }
    
    // 分配實體頁框
    int frame = allocate_frame();
    if (frame < 0) {
        // 需要置換
        frame = lru_replace(vpn);
    }
    
    // 如果頁面在交換空間，讀回
    if (is_in_swap(vpn)) {
        swap_in(get_swap_offset(vpn), frame);
    } else if (is_file_backed(vpn)) {
        // 從檔案讀取
        read_from_file(vpn, frame);
    }
    
    // 更新頁表
    update_page_table(vpn, frame, PRESENT | READ_WRITE);
    
    // 刷新 TLB
    flush_tlb(vpn);
}
```

## 核心態 vs 使用者態

```c
// 頁表條目權限檢查
int check_access(uint64_t vaddr, int requested, pte_t *pte) {
    if (!pte->present) return ACCESS_FAULT;
    if (requested == READ && !pte->read) return ACCESS_FAULT;
    if (requested == WRITE && !pte->write) return ACCESS_FAULT;
    if (requested == EXECUTE && pte->xn) return ACCESS_FAULT;
    
    // 使用者態檢查
    if (current_mode == USER_MODE) {
        if (!pte->user_access) return ACCESS_FAULT;
    }
    
    return ACCESS_OK;
}
```

# 記憶體層次：高效能電腦系統的儲存架構

## 概述

記憶體層次（Memory Hierarchy）是電腦系統設計中最基本的原則之一。這個原則建立在一個簡單的觀察上：記憶體越快就越貴，越慢就越便宜。因此，我們使用多層不同速度和成本的儲存設備來構建一個整體上既快速又經濟的儲存系統。

典型的記憶體層次從上到下包括：CPU 寄存器（最快、最貴、容量最小）、一級快取（L1 Cache）、二級快取（L2 Cache）、三級快取（L3 Cache）、主記憶體（DRAM）、固態硬碟（SSD）、傳統硬碟（HDD）、最後是網路儲存或磁帶（最慢、最便宜、容量最大）。程式的局部性原理（時間局部性和空間局部性）是這個層次結構能夠有效工作的基礎：最近訪問的資料很可能很快再次被訪問，而相鄰地址的資料很可能也會被訪問。

理解記憶體層次的設計原理對於系統程式設計、效能優化和理解作業系統的記憶體管理至關重要。

## 暫存器與快取

### CPU 暫存器

```c
// x86-64 暫存器組織
/*
通用暫存器 (64 位元):
  rax, rbx, rcx, rdx, rsi, rdi, rbp, rsp, r8-r15

特殊用途:
  rsp: 堆疊指標 (Stack Pointer)
  rbp: 框架指標 (Frame Pointer)
  rflags: 標誌暫存器 (Zero, Carry, Sign, Overflow 等)

SIMD 暫存器 (x86):
  XMM0-XMM15 (128 位元)
  YMM0-YMM15 (256 位元, XMM 擴展)
  ZMM0-ZMM31 (512 位元, AVX-512)

SIMD 暫存器 (ARM64):
  V0-V31 (128 位元, 可作為 Q0-Q31 或 D0-D31 訪問)
*/

// 暫存器訪問延遲
/*
理論上，暫存器訪問是零延遲的（在同一時鐘週期內完成）
但實際上:
- 讀取: 0 週期
- 寫入: 0 週期（通常）
- 讀後寫: 1 週期（因為有依賴）
*/

// 暫存器溢位
// 當暫存器不夠時，編譯器會生成溢出到記憶體的代碼
// 這稱為 register spilling

// 示例: 8個臨時值的累加，但只有4個可用暫存器
long long sum8(long long *arr) {
    long long s0 = 0, s1 = 0, s2 = 0, s3 = 0;  // 4 個暫存器
    long long s4 = 0, s5 = 0, s6 = 0, s7 = 0;  // 溢出到棧
    
    for (int i = 0; i < 8; i++) {
        long long t = arr[i];
        s0 += t;
        s1 += arr[i + 8];
        s2 += arr[i + 16];
        s3 += arr[i + 24];
        s4 += arr[i + 32];
        s5 += arr[i + 40];
        s6 += arr[i + 48];
        s7 += arr[i + 56];
    }
    
    return s0 + s1 + s2 + s3 + s4 + s5 + s6 + s7;
}
```

### 快取組織

```c
// 快取結構模擬
#define CACHE_LINE_SIZE 64
#define L1_SIZE (32 * 1024)      // 32KB
#define L1_ASSOCIATIVITY 8       // 8 路組相聯
#define L1_NUM_SETS (L1_SIZE / (CACHE_LINE_SIZE * L1_ASSOCIATIVITY))

typedef struct {
    uint8_t valid;              // 是否有效
    uint8_t tag;                 // 標籤
    uint8_t data[CACHE_LINE_SIZE];  // 資料
    uint8_t lru_counter;         // LRU 計數器
} cache_line_t;

typedef struct {
    cache_line_t lines[L1_ASSOCIATIVITY];
} cache_set_t;

typedef struct {
    cache_set_t sets[L1_NUM_SETS];
} cache_t;

// 直接映射 vs 組相聯 vs 全相聯
/*
1. 直接映射 (Direct-Mapped):
   每個地址只能映射到一個快取行
   地址格式: [Tag][Index][Offset]
   優點: 硬體簡單，查找快
   缺點: 衝突未命中率高
   
2. 組相聯 (Set-Associative):
   每個地址可以映射到 N 個快取行（N 路）
   地址格式: [Tag][Set Index][Offset]
   8路組相聯是常見的設計
   
3. 全相聯 (Fully Associative):
   任何地址可以放在任何快取行
   地址格式: [Tag][Offset]
   優點: 衝突未命中率最低
   缺點: 需要比較所有標籤，硬體複雜
*/

// 快取查找
cache_line_t* cache_lookup(cache_t *cache, uint64_t addr) {
    uint64_t offset = addr & (CACHE_LINE_SIZE - 1);
    uint64_t set_index = (addr >> 6) % L1_NUM_SETS;
    uint64_t tag = addr >> (6 + 12);  // 假設 4096 sets
    
    cache_set_t *set = &cache->sets[set_index];
    
    for (int i = 0; i < L1_ASSOCIATIVITY; i++) {
        if (set->lines[i].valid && set->lines[i].tag == tag) {
            return &set->lines[i];
        }
    }
    
    return NULL;  // 快取未命中
}

// LRU 替換策略
int cache_find_victim(cache_set_t *set) {
    // 找到 LRU 計數器最小的行
    int lru_idx = 0;
    uint8_t min_counter = set->lines[0].lru_counter;
    
    for (int i = 1; i < L1_ASSOCIATIVITY; i++) {
        if (set->lines[i].lru_counter < min_counter) {
            min_counter = set->lines[i].lru_counter;
            lru_idx = i;
        }
    }
    
    return lru_idx;
}

void cache_update_lru(cache_set_t *set, int accessed_idx) {
    uint8_t accessed_counter = set->lines[accessed_idx].lru_counter;
    
    for (int i = 0; i < L1_ASSOCIATIVITY; i++) {
        if (set->lines[i].lru_counter < accessed_counter) {
            set->lines[i].lru_counter++;
        }
    }
    
    set->lines[accessed_idx].lru_counter = 0;
}
```

### 快取策略

```c
// 分配策略 (Allocation Policy)
// 寫入時是否分配快取行

// 寫分配 (Write-Allocate):
// 當寫入未命中時，先從記憶體加載快取行，再更新
void write_allocate(cache_t *cache, uint64_t addr, uint8_t value) {
    // 先讀取整個快取行
    cache_line_t *line = cache_load_line(cache, addr);
    
    if (line == NULL) {
        // 未命中，找到 victim
        int victim = cache_find_victim(cache);
        // 如果 victim 被修改，需要寫回
        if (cache->sets[...].lines[victim].dirty) {
            write_back_to_memory(...);
        }
        // 分配新行
        cache_fill_line(cache, addr);
        line = cache_lookup(cache, addr);
    }
    
    // 更新快取行
    line->data[offset] = value;
    line->dirty = true;
}

// 非寫分配 (Write-No-Allocate):
// 當寫入未命中時，直接寫入記憶體，不分配快取行
void write_no_allocate(cache_t *cache, uint64_t addr, uint8_t value) {
    if (cache_lookup(cache, addr) != NULL) {
        // 命中，直接更新快取
        cache->lines[...].data[offset] = value;
        cache->lines[...].dirty = true;
    } else {
        // 未命中，直接寫記憶體
        write_to_memory(addr, value);
    }
}

// 寫入策略 (Write Policy)

// 寫直達 (Write-Through):
// 同時寫入快取和記憶體
void write_through(cache_t *cache, uint64_t addr, uint8_t value) {
    // 更新快取
    cache->lines[...].data[offset] = value;
    // 同時寫記憶體
    write_to_memory(addr, value);
}

// 寫回 (Write-Back):
// 只寫入快取，標記為 dirty
// 只有當快取行被驅逐時才寫回記憶體
void write_back(cache_t *cache, uint64_t addr, uint8_t value) {
    // 更新快取
    cache->lines[...].data[offset] = value;
    cache->lines[...].dirty = true;
}

// 預取策略 (Prefetching)
typedef enum {
    PREFETCH_NONE = 0,
    PREFETCH_HW,    // 硬體預取
    PREFETCH_SW,    // 軟體預取
    PREFETCH_ADAPTIVE
} prefetch_strategy_t;

// 順序預取
void sequential_prefetch(cache_t *cache, uint64_t addr) {
    uint64_t next_line = addr + CACHE_LINE_SIZE;
    // 預取下一快取行
    prefetch_line(cache, next_line);
}

// 跨越預取 (Strided Prefetch)
void strided_prefetch(cache_t *cache, uint64_t addr, int stride) {
    // 預取 stride 大小的下一個地址
    uint64_t next_addr = addr + stride;
    prefetch_line(cache, next_addr);
}

// 軟體預取提示 (x86)
void software_prefetch_example(int *arr, int n) {
    for (int i = 0; i < n; i++) {
        // 預取 arr[i + 16]（16 個 int = 64 bytes = 1 cache line）
        __builtin_prefetch(&arr[i + 16], 0, 3);  // read prefetch
        __builtin_prefetch(&arr[i + 32], 1, 3);  // write prefetch
        
        process(arr[i]);
    }
}

// ARM64 軟體預取
void arm_prefetch_example(int *arr, int n) {
    for (int i = 0; i < n; i++) {
        // prfm 指令: PLDL1KEEP, addr
        __asm__ volatile(
            "prfm pldl1keep, [%[addr]]\n\t"
            :
            : [addr] "r" (&arr[i + 16])
        );
        
        process(arr[i]);
    }
}
```

## 虛擬記憶體

### 分頁機制

```c
// 虛擬地址到物理地址轉換
/*
x86-64 分頁:
  4 級頁表: PML4 -> PDPT -> PD -> PT
  每級 9 位元（512 項）
  每頁 4KB

虛擬地址格式:
  [PML4 (9)][PDPT (9)][PD (9)][PT (9)][Offset (12)]
  63-48     47-39    38-30   29-21    20-12    11-0

ARM64 分頁 (4KB 頁):
  4 級頁表: TTBR0 -> L0 -> L1 -> L2 -> L3
  虛擬地址: [L0 (9)][L1 (9)][L2 (9)][L3 (9)][Offset (12)]
*/

// 頁表項結構 (x86-64)
typedef struct {
    uint64_t present : 1;        // P: 是否在記憶體
    uint64_t writable : 1;       // R/W: 是否可寫
    uint64_t user : 1;           // U/S: 用戶/內核
    uint64_t pwt : 1;           // 頁級寫回
    uint64_t pcd : 1;           // 禁用快取
    uint64_t accessed : 1;       // A: 是否被訪問
    uint64_t dirty : 1;         // D: 是否被修改
    uint64_t pat : 1;           // PAT
    uint64_t global : 1;         // G: 全域頁（大於 4MB）
    uint64_t avail : 3;          // 可用於軟體
    uint64_t frame : 40;        // 物理頁框號
    uint64_t reserved : 11;
    uint64_t xd : 1;             // 禁止執行 (NX bit)
} __attribute__((packed)) x86_page_entry_t;

// 頁表遍歷
uint64_t translate_address(uint64_t virtual_addr) {
    // CR3 包含 PML4 的物理地址
    uint64_t cr3;
    __asm__ volatile("mov %%cr3, %0" : "=r"(cr3));
    
    // 提取索引
    int pml4_idx = (virtual_addr >> 39) & 0x1FF;
    int pdpt_idx = (virtual_addr >> 30) & 0x1FF;
    int pd_idx = (virtual_addr >> 21) & 0x1FF;
    int pt_idx = (virtual_addr >> 12) & 0x1FF;
    int offset = virtual_addr & 0xFFF;
    
    // 讀取 PML4 項
    x86_page_entry_t *pml4 = (x86_page_entry_t *)cr3;
    if (!pml4[pml4_idx].present) return 0;  // 頁錯誤
    
    // 讀取 PDPT 項
    x86_page_entry_t *pdpt = (x86_page_entry_t *)(pml4[pml4_idx].frame << 12);
    if (!pdpt[pdpt_idx].present) return 0;
    
    // 讀取 PD 項
    x86_page_entry_t *pd = (x86_page_entry_t *)(pdpt[pdpt_idx].frame << 12);
    if (!pd[pd_idx].present) return 0;
    
    // 讀取 PT 項
    x86_page_entry_t *pt = (x86_page_entry_t *)(pd[pd_idx].frame << 12);
    if (!pt[pt_idx].present) return 0;
    
    // 組合物理地址
    return (pt[pt_idx].frame << 12) | offset;
}
```

### 翻譯後備緩衝區 (TLB)

```c
// TLB 結構
/*
TLB 是用於快取虛擬地址到物理地址翻譯的硬體結構
通常有 32-128 項，高度相聯

TLB 查找比頁表遍歷快得多
*/
#define TLB_SIZE 64
#define TLB_ASSOCIATIVITY 4
#define TLB_SETS (TLB_SIZE / TLB_ASSOCIATIVITY)

typedef struct {
    uint64_t virtual_page;     // 虛擬頁號
    uint64_t physical_page;    // 物理頁框號
    uint8_t valid;              // 是否有效
    uint8_t global;             // 全域頁
    uint8_t asid;              // 地址空間 ID
    uint8_t protection;         // 權限
} tlb_entry_t;

tlb_entry_t tlb[TLB_SETS][TLB_ASSOCIATIVITY];

// TLB 查找
uint64_t tlb_lookup(uint64_t vaddr, uint8_t asid) {
    int set = (vaddr >> 12) % TLB_SETS;
    
    for (int i = 0; i < TLB_ASSOCIATIVITY; i++) {
        tlb_entry_t *entry = &tlb[set][i];
        
        if (entry->valid && 
            entry->virtual_page == (vaddr >> 12) &&
            (entry->global || entry->asid == asid)) {
            // TLB 命中
            return (entry->physical_page << 12) | (vaddr & 0xFFF);
        }
    }
    
    // TLB 未命中，需要遍歷頁表
    return 0;
}

// TLB 填充
void tlb_insert(uint64_t vaddr, uint64_t paddr, 
                 uint8_t asid, uint8_t global, uint8_t prot) {
    int set = (vaddr >> 12) % TLB_SETS;
    
    // 找到空槽或選擇 LRU victim
    int victim = -1;
    for (int i = 0; i < TLB_ASSOCIATIVITY; i++) {
        if (!tlb[set][i].valid) {
            victim = i;
            break;
        }
    }
    
    if (victim < 0) {
        victim = select_lru_victim(set);
    }
    
    tlb[set][victim] = (tlb_entry_t) {
        .virtual_page = vaddr >> 12,
        .physical_page = paddr >> 12,
        .valid = 1,
        .global = global,
        .asid = asid,
        .protection = prot
    };
}

// TLB 刷新
void tlb_flush_all() {
    for (int i = 0; i < TLB_SETS; i++) {
        for (int j = 0; j < TLB_ASSOCIATIVITY; j++) {
            tlb[i][j].valid = 0;
        }
    }
}

void tlb_flush_asid(uint8_t asid) {
    for (int i = 0; i < TLB_SETS; i++) {
        for (int j = 0; j < TLB_ASSOCIATIVITY; j++) {
            if (tlb[i][j].valid && tlb[i][j].asid == asid) {
                tlb[i][j].valid = 0;
            }
        }
    }
}

void tlb_flush_page(uint64_t vaddr) {
    int set = (vaddr >> 12) % TLB_SETS;
    
    for (int i = 0; i < TLB_ASSOCIATIVITY; i++) {
        if (tlb[set][i].valid && 
            tlb[set][i].virtual_page == (vaddr >> 12)) {
            tlb[set][i].valid = 0;
        }
    }
}
```

### 大頁面 (Huge Pages)

```c
// 大頁面可以減少 TLB 未命中和頁表記憶體開銷
/*
常見的大頁面大小:
x86-64:
  - 2MB (PDPE + PDE + Offset)
  - 1GB (PML4 + PDPT + PDE + Offset)

ARM64:
  - 2MB (L3 頁)
  - 1GB (L2 頁)
  - 512GB-16TB (L1 頁)
*/

// Linux 中使用大頁面
// 預分配大頁面池
// echo 256 > /sys/kernel/mm/hugepages/hugepages-2048kB/nr_hugepages

// 使用大頁面的範例 (Linux)
#include <sys/mman.h>

void *allocate_hugepage(size_t size) {
    // size 應該是大頁面的倍數
    void *addr = mmap(NULL, size, 
                      PROT_READ | PROT_WRITE,
                      MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB,
                      -1, 0);
    return addr;
}

// 使用 MADV_HUGEPAGE 提示
void *allocate_with_hint(size_t size) {
    void *addr = mmap(NULL, size,
                      PROT_READ | PROT_WRITE,
                      MAP_PRIVATE | MAP_ANONYMOUS,
                      -1, 0);
    
    madvise(addr, size, MADV_HUGEPAGE);
    return addr;
}

// Rust 中使用大頁面
#[repr(align(2097152))]  // 2MB 對齊
struct HugePageBuffer {
    data: [u8; 2097152],
}

fn allocate_huge_buffer() -> &'static mut [u8] {
    let mut buffer = Vec::with_capacity(2097152 / std::mem::size_of::<u8>());
    unsafe {
        let ptr = buffer.as_mut_ptr();
        std::mem::forget(buffer);
        std::slice::from_raw_parts_mut(ptr, 2097152)
    }
}
```

## 效能優化

### 記憶體訪問模式

```c
// 記憶體訪問模式對效能的影響巨大

// 1. 行優先 vs 列優先
#define N 1024
int matrix_a[N][N];
int matrix_b[N][N];
int result[N][N];

// 列優先遍歷（快取不友好）
void matrix_multiply_col_major() {
    for (int j = 0; j < N; j++) {
        for (int i = 0; i < N; i++) {
            for (int k = 0; k < N; k++) {
                result[i][j] += matrix_a[i][k] * matrix_b[k][j];
            }
        }
    }
}

// 行優先遍歷（快取友好）
void matrix_multiply_row_major() {
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            for (int k = 0; k < N; k++) {
                result[i][j] += matrix_a[i][k] * matrix_b[k][j];
            }
        }
    }
}

// 阻塞 (Blocking) 優化
#define BLOCK_SIZE 64
void matrix_multiply_blocked() {
    for (int ii = 0; ii < N; ii += BLOCK_SIZE) {
        for (int jj = 0; jj < N; jj += BLOCK_SIZE) {
            for (int kk = 0; kk < N; kk += BLOCK_SIZE) {
                // 處理區塊
                int i_max = min(ii + BLOCK_SIZE, N);
                int j_max = min(jj + BLOCK_SIZE, N);
                int k_max = min(kk + BLOCK_SIZE, N);
                
                for (int i = ii; i < i_max; i++) {
                    for (int k = kk; k < k_max; k++) {
                        int a_ik = matrix_a[i][k];
                        for (int j = jj; j < j_max; j++) {
                            result[i][j] += a_ik * matrix_b[k][j];
                        }
                    }
                }
            }
        }
    }
}

// 2. 預取內核數據結構
typedef struct {
    int x, y, z;
    float nx, ny, nz;
} Vertex;

void process_mesh(Vertex *vertices, int count) {
    for (int i = 0; i < count; i++) {
        // 預取下一個頂點
        __builtin_prefetch(&vertices[i + 1]);
        __builtin_prefetch(&vertices[i + 2]);
        
        process_vertex(&vertices[i]);
    }
}

// 3. 避免虛假分享 (False Sharing)
typedef struct {
    atomic<int> counter;  // 每核心獨立計數器
    char padding[64 - sizeof(atomic<int>)];  // 填充到快取行邊界
} per_core_counter_t;

// 確保不同核心的計數器在不同快取行
per_core_counter_t counters[16];

void update_counter(int core_id, int value) {
    atomic_fetch_add_explicit(&counters[core_id].counter, value, 
                              memory_order_relaxed);
}
```

### NUMA 感知

```c
// 非統一記憶體訪問 (NUMA)
// 在多插槽系統中，記憶體訪問延遲取決於 CPU 和記憶體的位置

// Linux NUMA API
#include <numa.h>
#include <numaif.h>

void numa_example() {
    numa_available();
    
    // 分配本地節點記憶體
    struct bitmask *nodemask = numa_allocate_nodemask();
    numa_bitmask_setbit(nodemask, 0);  // 節點 0
    
    void *local_mem = numa_alloc_onnode(sizeof(data_t) * N, 0);
    
    // 在指定節點執行
    struct bitmask *cpumask = numa_allocate_cpumask();
    numa_bitmask_setbit(cpumask, 0);  // CPU 0
    numa_run_on_node_mask(cpumask);
    
    // 親和性設置
    set_mempolicy(MPOL_BIND, nodemask, 0);
    
    // 清理
    numa_free(local_mem, sizeof(data_t) * N);
    numa_free_nodemask(nodemask);
    numa_free_cpumask(cpumask);
}

// 程式化 NUMA 感知
typedef struct {
    int *data;
    size_t size;
    int node;
} numa_data_t;

void distribute_data(numa_data_t *chunks, int num_chunks) {
    int num_nodes = numa_max_node() + 1;
    
    for (int i = 0; i < num_chunks; i++) {
        int node = i % num_nodes;
        chunks[i].node = node;
        
        // 在對應節點分配記憶體
        chunks[i].data = numa_alloc_onnode(
            sizeof(int) * CHUNK_SIZE, node);
        
        // 將進程綁定到節點
        struct bitmask *mask = numa_allocate_cpumask();
        numa_node_to_cpus(node, mask);
        numa_run_on_node(node);
        
        // 初始化數據
        for (size_t j = 0; j < CHUNK_SIZE; j++) {
            chunks[i].data[j] = j;
        }
    }
}
```

## 總結

記憶體層次是電腦系統設計中的核心概念。理解快取的工作原理、虛擬記憶體的轉換機制、以及如何編寫對記憶體層次友好的程式碼，對於效能優化至關重要。關鍵原則包括：利用時間局部性（將常用資料保持在快取中）、利用空間局部性（順序訪問資料）、避免虛假分享（確保不同核心的資料在不同快取行）、以及在 NUMA 系統中考慮記憶體位置。

# 字串演算法：高效文本處理

## 概述

字串演算法處理字串的匹配、搜索和處理問題，是電腦科學中最基礎也最實際的課題之一。從簡單的暴力搜索，到高效的 KMP 和 Boyer-Moore，再到現代的後綴陣列和 AC 自動機，字串演算法在文本編輯器、搜尋引擎、資料壓縮和生物資訊學等領域有廣泛應用。

## 暴力匹配

```python
def brute_force(text, pattern):
    """暴力字串匹配 - O(n*m)"""
    n, m = len(text), len(pattern)
    
    for i in range(n - m + 1):
        j = 0
        while j < m and text[i + j] == pattern[j]:
            j += 1
        if j == m:
            return i  # 找到匹配
    return -1  # 未找到
```

## KMP 演算法

```python
def compute_lps(pattern):
    """
    計算最長前綴後綴數組 (Longest Proper Prefix which is also Suffix)
    lps[i] = pattern[0:i+1] 的最長真前綴也是後綴的長度
    """
    m = len(pattern)
    lps = [0] * m
    length = 0  # 前一個字元的 lps 值
    i = 1
    
    while i < m:
        if pattern[i] == pattern[length]:
            length += 1
            lps[i] = length
            i += 1
        else:
            if length != 0:
                length = lps[length - 1]
            else:
                lps[i] = 0
                i += 1
    
    return lps

def kmp_search(text, pattern):
    """KMP 字串匹配 - O(n+m)"""
    n, m = len(text), len(pattern)
    lps = compute_lps(pattern)
    
    i = j = 0  # i = text index, j = pattern index
    while i < n:
        if pattern[j] == text[i]:
            i += 1
            j += 1
        
        if j == m:
            return i - j  # 找到匹配
        
        elif i < n and pattern[j] != text[i]:
            if j != 0:
                j = lps[j - 1]  # 使用 lps 跳轉
            else:
                i += 1
    
    return -1  # 未找到

def build_failure_function(pattern):
    """建立失敗函數（與 lps 相同）"""
    m = len(pattern)
    pi = [0] * m
    
    for i in range(1, m):
        j = pi[i - 1]
        while j > 0 and pattern[i] != pattern[j]:
            j = pi[j - 1]
        if pattern[i] == pattern[j]:
            j += 1
        pi[i] = j
    
    return pi
```

## Boyer-Moore 演算法

```python
def build_last_occurrence(pattern):
    """建立最後出現位置表"""
    last = {}
    for i, char in enumerate(pattern):
        last[char] = i
    return last

def boyer_moore(text, pattern):
    """Boyer-Moore 字串匹配 - 平均 O(n/m)，最壞 O(n*m)"""
    n, m = len(text), len(pattern)
    
    if m == 0:
        return 0
    
    last = build_last_occurrence(pattern)
    
    i = m - 1  # text 中的當前位置
    while i < n:
        j = m - 1  # pattern 中從後往前匹配
        
        while j >= 0 and text[i] == pattern[j]:
            i -= 1
            j -= 1
        
        if j < 0:
            return i + 1  # 找到匹配
        
        # 跳轉
        last_occurrence = last.get(text[i], -1)
        if last_occurrence < j:
            i += m - 1 - last_occurrence
        else:
            i += 1
    
    return -1  # 未找到

def boyer_moore_with_galil(text, pattern):
    """Boyer-Moore + Galil 優化 - 保證線性時間"""
    n, m = len(text), len(pattern)
    
    if m == 0:
        return 0
    
    last = build_last_occurrence(pattern)
    i = m - 1
    k = m - 1  # 記住上次匹配後應該開始的位置
    shift = 0
    
    while i < n:
        j = m - 1
        
        # 從後往前匹配
        while j >= k and text[i - shift] == pattern[j]:
            j -= 1
            i -= 1
        
        if j < k:
            return i + 1 + shift
        
        last_occ = last.get(text[i - shift], -1)
        shift = m - 1 - last_occ
        
        if last_occ < k:
            i += m - 1 - last_occ
        else:
            i += m - 1 - k + 1
            k = m - 1 - shift
    
    return -1
```

## Rabin-Karp 演算法

```python
def rabin_karp(text, pattern, base=256, mod=101):
    """Rabin-Karp 滾動雜湊 - O(n+m)，可能需要驗證"""
    n, m = len(text), len(pattern)
    
    if m == 0:
        return 0
    
    # 計算 base^m mod mod
    h_pattern = 0
    h_text = 0
    h_base = 1
    
    for i in range(m):
        h_pattern = (h_pattern * base + ord(pattern[i])) % mod
        h_text = (h_text * base + ord(text[i])) % mod
        if i < m - 1:
            h_base = (h_base * base) % mod
    
    for i in range(n - m + 1):
        if h_pattern == h_text:
            # 驗證（避免雜湊碰撞）
            if text[i:i + m] == pattern:
                return i
        
        # 滾動計算下一個雜湊值
        if i < n - m:
            h_text = (base * (h_text - ord(text[i]) * h_base) + ord(text[i + m])) % mod
            if h_text < 0:
                h_text += mod
    
    return -1

def rabin_karp_rolling(text, pattern, mod=10**9 + 7):
    """Rabin-Karp 的滾動雜湊實現"""
    n, m = len(text), len(pattern)
    power = pow(26, m, mod)
    hash_p = 0
    hash_t = 0
    
    for i in range(m):
        hash_p = (hash_p * 26 + ord(pattern[i]) - ord('a')) % mod
        hash_t = (hash_t * 26 + ord(text[i]) - ord('a')) % mod
    
    for i in range(n - m + 1):
        if hash_p == hash_t and text[i:i + m] == pattern:
            return i
        if i < n - m:
            hash_t = (hash_t * 26 + ord(text[i + m]) - ord('a') - power * (ord(text[i]) - ord('a'))) % mod
            hash_t = (hash_t + mod) % mod
    
    return -1
```

## Z 演算法

```python
def z_algorithm(s):
    """
    Z 演算法 - 計算 Z 陣列
    Z[i] = s[i:] 與 s 的最長公共前綴長度
    """
    n = len(s)
    Z = [0] * n
    Z[0] = n
    
    left = right = 0
    
    for i in range(1, n):
        if i <= right:
            Z[i] = min(right - i + 1, Z[i - left])
        
        while i + Z[i] < n and s[Z[i]] == s[i + Z[i]]:
            Z[i] += 1
        
        if i + Z[i] - 1 > right:
            left, right = i, i + Z[i] - 1
    
    return Z

def z_match(text, pattern):
    """使用 Z 演算法進行字串匹配"""
    combined = pattern + '$' + text
    Z = z_algorithm(combined)
    
    m = len(pattern)
    for i in range(m + 1, len(combined)):
        if Z[i] >= m:
            return i - m - 1
    return -1
```

## 編輯距離

```python
def levenshtein_distance(s1, s2):
    """Levenshtein 編輯距離 - O(n*m)"""
    m, n = len(s1), len(s2)
    
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(m + 1):
        dp[i][0] = i  # 刪除 i 個字元
    for j in range(n + 1):
        dp[0][j] = j  # 插入 j 個字元
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = 1 + min(
                    dp[i - 1][j],      # 刪除
                    dp[i][j - 1],      # 插入
                    dp[i - 1][j - 1]  # 替換
                )
    
    return dp[m][n]

def edit_distance_with_operations(s1, s2):
    """返回具體操作"""
    m, n = len(s1), len(s2))
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(m + 1):
        dp[i][0] = i
    for j in range(n + 1):
        dp[0][j] = j
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = 1 + min(
                    dp[i - 1][j],      # 刪除
                    dp[i][j - 1],      # 插入
                    dp[i - 1][j - 1]   # 替換
                )
    
    # 回溯找出操作
    operations = []
    i, j = m, n
    while i > 0 or j > 0:
        if i > 0 and j > 0 and s1[i - 1] == s2[j - 1]:
            i -= 1
            j -= 1
        elif i > 0 and dp[i][j] == dp[i - 1][j] + 1:
            operations.append(('DELETE', s1[i - 1], i - 1))
            i -= 1
        elif j > 0 and dp[i][j] == dp[i][j - 1] + 1:
            operations.append(('INSERT', s2[j - 1], i))
            j -= 1
        else:
            operations.append(('REPLACE', s2[j - 1], i - 1))
            i -= 1
            j -= 1
    
    return dp[m][n], list(reversed(operations))
```

## 字典樹 (Trie)

```python
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False
        self.frequency = 0

class Trie:
    """字典樹 - 字串集合的高效結構"""
    
    def __init__(self):
        self.root = TrieNode()
    
    def insert(self, word):
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end = True
        node.frequency += 1
    
    def search(self, word):
        node = self._find_node(word)
        return node is not None and node.is_end
    
    def starts_with(self, prefix):
        return self._find_node(prefix) is not None
    
    def _find_node(self, prefix):
        node = self.root
        for char in prefix:
            if char not in node.children:
                return None
            node = node.children[char]
        return node
    
    def autocomplete(self, prefix):
        """自動補全"""
        node = self._find_node(prefix)
        if not node:
            return []
        
        results = []
        self._dfs_collect(node, prefix, results)
        return results
    
    def _dfs_collect(self, node, path, results):
        if node.is_end:
            results.append(path)
        for char, child in node.children.items():
            self._dfs_collect(child, path + char, results)
```

## 總結

字串演算法是文本處理的基礎。KMP 實現線性時間匹配，Boyer-Moore 在實際應用中通常更快，Rabin-Karp 使用雜湊實現簡潔，Z 演算法則是多功能的工具。理解這些演算法的原理對於解決實際問題至關重要。

# 梯度下降法 (Gradient Descent)

## 概述

梯度下降法是最佳化問題的核心演算法，沿著函數梯度的負方向迭代更新參數，以找到函數的局部最小值。

## 基本原理

```
梯度指向函數上升最快的方向
沿負梯度方向移動可降低函數值
```

## 批量梯度下降

```python
def gradient_descent(f, grad_f, x0, learning_rate=0.1, max_iter=1000, tolerance=1e-6):
    """批量梯度下降"""
    x = x0
    
    for i in range(max_iter):
        gradient = grad_f(x)
        x_new = x - learning_rate * gradient
        
        if abs(f(x_new) - f(x)) < tolerance:
            return x_new, i + 1
        
        x = x_new
    
    return x, max_iter

def gradient_descent_multi(f, grad_f, x0, learning_rate=0.1, max_iter=1000, tolerance=1e-6):
    """多維梯度下降"""
    x = x0.copy()
    
    for i in range(max_iter):
        gradient = grad_f(x)
        x_new = x - learning_rate * gradient
        
        if abs(f(x_new) - f(x)) < tolerance:
            return x_new, i + 1
        
        x = x_new
    
    return x, max_iter
```

## 梯度下降變體

### 隨機梯度下降 (SGD)

```python
import random

def sgd(f, grad_f, x0, data, learning_rate=0.01, max_epochs=100, tolerance=1e-6):
    """隨機梯度下降"""
    x = x0
    
    for epoch in range(max_epochs):
        random.shuffle(data)
        prev_x = x.copy()
        
        for sample in data:
            gradient = grad_f(x, sample)
            x = x - learning_rate * gradient
        
        if abs(f(x) - f(prev_x)) < tolerance:
            return x, epoch + 1
        
        learning_rate *= 0.95  # 學習率衰減
    
    return x, max_epochs
```

### 小批量梯度下降 (Mini-batch GD)

```python
def mini_batch_gd(f, grad_f, x0, data, batch_size=32, learning_rate=0.01, max_epochs=100):
    """小批量梯度下降"""
    x = x0
    n = len(data)
    
    for epoch in range(max_epochs):
        random.shuffle(data)
        
        for i in range(0, n, batch_size):
            batch = data[i:i+batch_size]
            
            gradient = sum(grad_f(x, sample) for sample in batch) / len(batch)
            x = x - learning_rate * gradient
    
    return x
```

### 動量梯度下降

```python
def gradient_descent_momentum(f, grad_f, x0, learning_rate=0.1, momentum=0.9, max_iter=1000):
    """動量梯度下降"""
    x = x0
    velocity = 0
    
    for _ in range(max_iter):
        gradient = grad_f(x)
        velocity = momentum * velocity - learning_rate * gradient
        x = x + velocity
    
    return x
```

### Adam 優化器

```python
def adam(grad_f, x0, learning_rate=0.001, beta1=0.9, beta2=0.999, epsilon=1e-8, max_iter=1000):
    """Adam 優化器"""
    x = x0
    m = 0  # 第一動量
    v = 0  # 第二動量
    
    for t in range(1, max_iter + 1):
        gradient = grad_f(x)
        
        m = beta1 * m + (1 - beta1) * gradient
        v = beta2 * v + (1 - beta2) * gradient**2
        
        m_hat = m / (1 - beta1**t)
        v_hat = v / (1 - beta2**t)
        
        x = x - learning_rate * m_hat / (math.sqrt(v_hat) + epsilon)
    
    return x
```

## 學習率策略

```python
class LearningRateScheduler:
    """學習率調整策略"""
    
    @staticmethod
    def step_decay(x0, epoch, drop_every=10, drop_rate=0.5):
        """階梯衰減"""
        return x0 * (drop_rate ** (epoch // drop_every))
    
    @staticmethod
    def exponential_decay(x0, epoch, decay_rate=0.95):
        """指數衰減"""
        return x0 * (decay_rate ** epoch)
    
    @staticmethod
    def cosine_annealing(epoch, T_max, eta_min=0):
        """餘弦退火"""
        return eta_min + 0.5 * (x0 - eta_min) * (1 + math.cos(math.pi * epoch / T_max))
```

## 收斂性分析

```python
def check_convergence(grad_f, x, tolerance=1e-6):
    """檢查收斂"""
    gradient = grad_f(x)
    return sum(g**2 for g in gradient) < tolerance

def plot_convergence(history):
    """繪製收斂曲線"""
    import matplotlib.pyplot as plt
    plt.plot(history)
    plt.xlabel('Iteration')
    plt.ylabel('Objective Value')
    plt.title('Gradient Descent Convergence')
    plt.show()
```

## 應用實例

### 線性迴歸

```python
import numpy as np

def linear_regression_gradient_descent(X, y, learning_rate=0.01, max_iter=1000):
    """線性迴歸梯度下降"""
    m, n = X.shape
    theta = np.zeros(n)
    
    for _ in range(max_iter):
        predictions = X @ theta
        errors = predictions - y
        gradient = (2/m) * (X.T @ errors)
        theta = theta - learning_rate * gradient
    
    return theta

X = np.array([[1, 1], [1, 2], [2, 2], [2, 3]])
y = np.array([1, 2, 2, 3])
theta = linear_regression_gradient_descent(X, y)
print(f"Coefficients: {theta}")
```

### Logistic 迴歸

```python
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

def logistic_regression_gd(X, y, learning_rate=0.1, max_iter=1000):
    """Logistic 迴歸梯度下降"""
    m, n = X.shape
    theta = np.zeros(n)
    
    for _ in range(max_iter):
        h = sigmoid(X @ theta)
        gradient = (1/m) * (X.T @ (h - y))
        theta = theta - learning_rate * gradient
    
    return theta
```

## 梯度消失與爆炸

```python
def gradient_clipping(gradient, max_norm=5.0):
    """梯度裁剪"""
    norm = np.linalg.norm(gradient)
    if norm > max_norm:
        return gradient * max_norm / norm
    return gradient

def rmsprop(grad_f, x0, learning_rate=0.001, decay_rate=0.9, epsilon=1e-8, max_iter=1000):
    """RMSprop：自適應學習率"""
    x = x0
    cache = 0
    
    for t in range(1, max_iter + 1):
        gradient = grad_f(x)
        cache = decay_rate * cache + (1 - decay_rate) * gradient**2
        x = x - learning_rate * gradient / (np.sqrt(cache) + epsilon)
    
    return x
```

## 梯度下降比較

| 方法 | 學習率 | 收斂速度 | 適用場景 |
|------|--------|---------|----------|
| 批量 GD | 固定 | 慢 | 小數據集 |
| SGD | 需調整 | 快 | 大數據集 |
| Mini-batch | 需調整 | 中 | 一般場景 |
| Adam | 自適應 | 快 | 深度學習 |

## 何時收斂

1. **凸函數**：梯度下降保證收斂到全域最優
2. **非凸函數**：可能收斂到局部最優
3. **鞍點**：可被動量或自適應方法逃離

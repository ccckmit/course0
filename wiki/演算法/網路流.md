# 網路流：圖論中的流量優化

## 概述

網路流問題是圖論中的經典問題，研究如何在網路中有效地傳輸流量。典型應用包括：交通網路設計、資料傳輸路由、通訊網路容量規劃、航空公司調度等。最大流問題尋找從源點到匯點的最大可能流量，而最小割問題則找到容量最小的邊集合，分割源點和匯點。

根據最大流最小割定理（Max-Flow Min-Cut Theorem），任何網路中從源點到匯點的最大流量值等於最小割的容量。這個優美的對偶性是網路流理論的核心。

## 基礎概念

```python
from collections import defaultdict
import math

class Edge:
    """流的邊"""
    def __init__(self, to, cap, rev):
        self.to = to          # 目標節點
        self.cap = cap        # 剩餘容量
        self.rev = rev        # 反向邊的索引

class FlowNetwork:
    """網路流"""
    
    def __init__(self, n):
        self.n = n
        self.graph = [[] for _ in range(n)]
    
    def add_edge(self, fr, to, cap):
        """添加有向邊"""
        forward = Edge(to, cap, len(self.graph[to]))
        backward = Edge(fr, 0, len(self.graph[fr]))
        self.graph[fr].append(forward)
        self.graph[to].append(backward)
    
    def add_undirected_edge(self, fr, to, cap):
        """添加無向邊（雙向容量相同）"""
        forward = Edge(to, cap, len(self.graph[to]))
        backward = Edge(fr, cap, len(self.graph[fr]))
        self.graph[fr].append(forward)
        self.graph[to].append(backward)
```

## Ford-Fulkerson 方法

```python
def bfs(self, s, t, parent):
    """BFS 找增廣路徑"""
    visited = [False] * self.n
    queue = [s]
    visited[s] = True
    
    while queue:
        v = queue.pop(0)
        for edge in self.graph[v]:
            if not visited[edge.to] and edge.cap > 0:
                queue.append(edge.to)
                visited[edge.to] = True
                parent[edge.to] = (v, edge)
                if edge.to == t:
                    return True
    return False

def ford_fulkerson(self, s, t):
    """Ford-Fulkerson - O(E * max_flow)"""
    flow = 0
    parent = [None] * self.n
    
    while self.bfs(s, t, parent):
        # 找到瓶頸容量
        path_flow = math.inf
        v = t
        while v != s:
            u, edge = parent[v]
            path_flow = min(path_flow, edge.cap)
            v = u
        
        # 更新容量
        v = t
        while v != s:
            u, edge = parent[v]
            edge.cap -= path_flow
            # 找到反向邊
            self.graph[v][edge.rev].cap += path_flow
            v = u
        
        flow += path_flow
        parent = [None] * self.n
    
    return flow
```

## Edmonds-Karp 演算法

```python
def edmonds_karp(self, s, t):
    """Edmonds-Karp - BFS 版 Ford-Fulkerson，O(V*E²)"""
    flow = 0
    
    while True:
        parent = self.bfs_parent(t)
        if parent[t] is None:
            break
        
        # 找瓶頸
        path_flow = math.inf
        v = t
        while v != s:
            u, edge = parent[v]
            path_flow = min(path_flow, edge.cap)
            v = u
        
        # 更新
        v = t
        while v != s:
            u, edge = parent[v]
            edge.cap -= path_flow
            self.graph[v][edge.rev].cap += path_flow
            v = u
        
        flow += path_flow
    
    return flow

def bfs_parent(self, t):
    """返回 parent 數組"""
    parent = [None] * self.n
    queue = [0]
    parent[0] = (None, None)
    
    while queue:
        v = queue.pop(0)
        for edge in self.graph[v]:
            if parent[edge.to] is None and edge.cap > 0:
                parent[edge.to] = (v, edge)
                if edge.to == t:
                    return parent
                queue.append(edge.to)
    
    return parent
```

## Dinic 演算法

```python
def dinic(self, s, t):
    """Dinic - O(E * sqrt(V)) for unit capacity, O(V²*E) worst case"""
    flow = 0
    INF = math.inf
    
    while True:
        level = self.bfs_level(s, t)
        if level[t] < 0:
            break
        
        it = [0] * self.n  # 當前弧優化
        
        def dfs(v, upTo):
            if v == t:
                return upTo
            for i in range(it[v], len(self.graph[v])):
                edge = self.graph[v][i]
                if level[edge.to] == level[v] + 1 and edge.cap > 0:
                    d = dfs(edge.to, min(upTo, edge.cap))
                    if d > 0:
                        edge.cap -= d
                        self.graph[edge.to][edge.rev].cap += d
                        return d
                it[v] += 1
            return 0
        
        while True:
            f = dfs(s, INF)
            if f == 0:
                break
            flow += f
    
    return flow

def bfs_level(self, s, t):
    """構建層次圖"""
    level = [-1] * self.n
    level[s] = 0
    queue = [s]
    
    while queue:
        v = queue.pop(0)
        for edge in self.graph[v]:
            if level[edge.to] < 0 and edge.cap > 0:
                level[edge.to] = level[v] + 1
                queue.append(edge.to)
    
    return level
```

## 最小割

```python
def min_cut(self, s):
    """返回最小割的集合（從 s 可達的節點）"""
    visited = [False] * self.n
    queue = [s]
    visited[s] = True
    
    while queue:
        v = queue.pop(0)
        for edge in self.graph[v]:
            if not visited[edge.to] and edge.cap > 0:
                visited[edge.to] = True
                queue.append(edge.to)
    
    return visited

def all_cuts(self, s, t):
    """枚舉所有 s-t 割"""
    def dfs(v, cut):
        if v == t:
            yield cut.copy()
            return
        for edge in self.graph[v]:
            if edge.cap > 0 and edge not in cut:
                cut.add(edge)
                yield from dfs(edge.to, cut)
                cut.remove(edge)
    
    yield from dfs(s, set())
```

## 應用示例

```python
def bipartite_matching(graph, left, right):
    """
    二分圖最大匹配
    graph[u] = 與左側節點 u 相鄰的右側節點列表
    """
    n = len(left) + len(right) + 2
    s, t = 0, n - 1
    flow = FlowNetwork(n)
    
    # 源點到左側
    for i, u in enumerate(left):
        flow.add_edge(s, i + 1, 1)
    
    # 左側到右側
    for i, u in enumerate(left):
        for v in graph.get(u, []):
            j = len(left) + right.index(v) + 1 if v in right else -1
            if j > 0:
                flow.add_edge(i + 1, j, 1)
    
    # 右側到匯點
    for j, v in enumerate(right):
        flow.add_edge(len(left) + j + 1, t, 1)
    
    return flow.edmonds_karp(s, t)

def max_cardinality_matching(edges, n_left, n_right):
    """Hopcroft-Karp 的簡化版本"""
    adj = defaultdict(list)
    for u, v in edges:
        adj[u].append(v)
    
    INF = math.inf
    dist = [0] * n_left
    match_u = [-1] * n_left
    match_v = [-1] * n_right
    
    def bfs():
        queue = []
        for u in range(n_left):
            if match_u[u] == -1:
                dist[u] = 0
                queue.append(u)
            else:
                dist[u] = INF
        
        d = INF
        while queue:
            u = queue.pop(0)
            if dist[u] < d:
                for v in adj[u]:
                    if match_v[v] == -1:
                        d = dist[u] + 1
                    elif dist[match_v[v]] == INF:
                        dist[match_v[v]] = dist[u] + 1
                        queue.append(match_v[v])
        
        return d != INF
    
    def dfs(u):
        for v in adj[u]:
            if match_v[v] == -1 or (dist[match_v[v]] == dist[u] + 1 and dfs(match_v[v])):
                match_u[u] = v
                match_v[v] = u
                return True
        dist[u] = INF
        return False
    
    matching = 0
    while bfs():
        for u in range(n_left):
            if match_u[u] == -1 and dfs(u):
                matching += 1
    
    return matching, match_u, match_v
```

## 總結

網路流是圖論中的核心課題。Ford-Fulkerson 是基礎，Edmonds-Karp 使用 BFS 確保每次找到最短路徑，Dinic 的層次圖優化在實踐中效率最高。最大流最小割定理揭示了網路傳輸能力的根本限制，許多實際問題如二分匹配都可以歸約為網路流問題。

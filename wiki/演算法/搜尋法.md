# 搜尋法 (Search Algorithm)

## 概述

搜尋法在資料結構中尋找特定元素，是電腦科學最基本的運算之一。

## 線性搜尋

```python
def linear_search(arr, target):
    """線性搜尋：O(n)"""
    for i, x in enumerate(arr):
        if x == target:
            return i
    return -1

def linear_search_all(arr, target):
    """找出所有匹配位置"""
    return [i for i, x in enumerate(arr) if x == target]
```

## 二分搜尋

```python
def binary_search(arr, target):
    """二分搜尋：O(log n)，需有序陣列"""
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = (left + right) // 2
        
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1

def binary_search_leftmost(arr, target):
    """找最左邊的第一個"""
    left, right = 0, len(arr)
    
    while left < right:
        mid = (left + right) // 2
        if arr[mid] < target:
            left = mid + 1
        else:
            right = mid
    
    return left if left < len(arr) and arr[left] == target else -1

def binary_search_recursive(arr, target, left, right):
    """二分搜尋（遞迴版本）"""
    if left > right:
        return -1
    
    mid = (left + right) // 2
    
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return binary_search_recursive(arr, target, mid + 1, right)
    else:
        return binary_search_recursive(arr, target, left, mid - 1)
```

## 二叉搜尋樹

```python
class TreeNode:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None

def bst_search(root, target):
    """BST 搜尋：平均 O(log n)"""
    while root:
        if root.val == target:
            return root
        elif root.val < target:
            root = root.right
        else:
            root = root.left
    return None

def bst_insert(root, val):
    """BST 插入"""
    if not root:
        return TreeNode(val)
    
    if val < root.val:
        root.left = bst_insert(root.left, val)
    else:
        root.right = bst_insert(root.right, val)
    
    return root

def bst_delete(root, val):
    """BST 刪除"""
    if not root:
        return None
    
    if val < root.val:
        root.left = bst_delete(root.left, val)
    elif val > root.val:
        root.right = bst_delete(root.right, val)
    else:
        if not root.left:
            return root.right
        if not root.right:
            return root.left
        
        # 找右子樹最小值
        min_larger = root.right
        while min_larger.left:
            min_larger = min_larger.left
        root.val = min_larger.val
        root.right = bst_delete(root.right, min_larger.val)
    
    return root
```

## 深度優先搜尋 (DFS)

```python
def dfs_recursive(graph, node, visited=None):
    """DFS 遞迴版本"""
    if visited is None:
        visited = set()
    
    if node in visited:
        return
    
    visited.add(node)
    print(node, end=' ')
    
    for neighbor in graph[node]:
        dfs_recursive(graph, neighbor, visited)

def dfs_iterative(graph, start):
    """DFS 迭代版本（使用堆疊）"""
    visited = set()
    stack = [start]
    
    while stack:
        node = stack.pop()
        
        if node not in visited:
            visited.add(node)
            print(node, end=' ')
            
            for neighbor in reversed(graph[node]):
                if neighbor not in visited:
                    stack.append(neighbor)
```

## 廣度優先搜尋 (BFS)

```python
from collections import deque

def bfs(graph, start):
    """BFS：使用佇列"""
    visited = set([start])
    queue = deque([start])
    
    while queue:
        node = queue.popleft()
        print(node, end=' ')
        
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

def bfs_shortest_path(graph, start, end):
    """BFS 最短路徑"""
    visited = set([start])
    queue = deque([(start, [start])])
    
    while queue:
        node, path = queue.popleft()
        
        if node == end:
            return path
        
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))
    
    return None  # 無路徑
```

## 搜尋比較

| 方法 | 時間複雜度 | 空間複雜度 | 前提條件 |
|------|-----------|-----------|----------|
| 線性搜尋 | O(n) | O(1) | 無 |
| 二分搜尋 | O(log n) | O(1) | 有序 |
| BST | O(log n)~O(n) | O(h) | 二叉樹 |
| DFS | O(V+E) | O(V) | 圖 |
| BFS | O(V+E) | O(V) | 圖 |

## 應用場景

1. **資料庫索引**：B-Tree
2. **路徑規劃**：BFS/DFS
3. **AI 遊戲**： Minimax、Alpha-Beta
4. **最佳化問題**：分支定界

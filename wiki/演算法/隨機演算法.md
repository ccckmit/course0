# 隨機演算法：概率帶來的效率

## 概述

隨機演算法在執行過程中使用隨機數來做出決策。與確定性演算法不同，隨機演算法可能每次運行給出不同的結果，但通常能以更高的概率獲得正確或接近正確的答案。隨機演算法分為兩類：**拉斯維加斯演算法**（Las Vegas）總是返回正確答案，但運行時間隨機；**蒙特卡羅演算法**（Monte Carlo）運行時間確定，但可能返回錯誤答案。

```python
import math

class RandomizedAlgorithm:
    """隨機演算法的基類"""

        if seed:
            random.seed(seed)

        raise NotImplementedError
    
    def expected_time(self):
        """返回期望運行時間"""
        raise NotImplementedError
    
    def probability_of_success(self, *args):
        """返回成功概率"""
        raise NotImplementedError

def random_choice(n):
    """均勻隨機選擇"""
    return random.randint(0, n - 1)

def weighted_random_choice(weights):
    """加權隨機選擇"""
    total = sum(weights)
    r = random.uniform(0, total)
    cumsum = 0
    for i, w in enumerate(weights):
        cumsum += w
        if r < cumsum:
            return i
    return len(weights) - 1
```

## 快速排序（隨機化）

```python
def quicksort_randomized(arr):
    """
    隨機化快速排序 - 平均 O(n log n)
    隨機選擇 pivot 避免最壞情況
    """
    if len(arr) <= 1:
        return arr
    
    arr = arr.copy()
    _quicksort_randomized(arr, 0, len(arr) - 1)
    return arr

def _quicksort_randomized(arr, low, high):
    if low < high:
        pivot_idx = random.randint(low, high)
        arr[pivot_idx], arr[high] = arr[high], arr[pivot_idx]
        
        pivot = partition(arr, low, high)
        
        _quicksort_randomized(arr, low, pivot - 1)
        _quicksort_randomized(arr, pivot + 1, high)

def partition(arr, low, high):
    """分區"""
    pivot = arr[high]
    i = low - 1
    
    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1

def quicksort_3way(arr):
    """
    三路分區快速排序
    處理大量重複元素時更好
    """
    if len(arr) <= 1:
        return arr
    
    arr = arr.copy()
    _quicksort_3way(arr, 0, len(arr) - 1)
    return arr

def _quicksort_3way(arr, low, high):
    if low >= high:
        return
    
    pivot = arr[low]
    lt = low      # arr[low...lt-1] < pivot
    gt = high     # arr[gt+1...high] > pivot
    i = low + 1   # arr[lt...i-1] == pivot
    
    while i <= gt:
        if arr[i] < pivot:
            arr[lt], arr[i] = arr[i], arr[lt]
            lt += 1
            i += 1
        elif arr[i] > pivot:
            arr[gt], arr[i] = arr[i], arr[gt]
            gt -= 1
        else:
            i += 1
    
    _quicksort_3way(arr, low, lt - 1)
    _quicksort_3way(arr, gt + 1, high)
```

## 隨機化選擇

```python
def randomized_select(arr, k):
    """
    隨機化選擇第 k 小的元素 - 平均 O(n)
    拉斯維加斯：總是返回正確答案
    """
    arr = arr.copy()
    return _randomized_select(arr, 0, len(arr) - 1, k)

def _randomized_select(arr, low, high, k):
    if low == high:
        return arr[low]
    
    pivot_idx = random.randint(low, high)
    arr[pivot_idx], arr[high] = arr[high], arr[pivot_idx]
    pivot = arr[high]
    
    i = low
    for j in range(low, high):
        if arr[j] <= pivot:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    
    arr[i], arr[high] = arr[high], arr[i]
    
    if k == i:
        return arr[k]
    elif k < i:
        return _randomized_select(arr, low, i - 1, k)
    else:
        return _randomized_select(arr, i + 1, high, k)

def deterministic_select(arr, k):
    """
    確定性選擇 - 最壞情況 O(n)
    使用中位數的中位數作為 pivot
    """
    arr = arr.copy()
    return _deterministic_select(arr, 0, len(arr) - 1, k)

def _deterministic_select(arr, low, high, k):
    if low == high:
        return arr[low]
    
    # 將數組分成 5 個一組，找每組中位數
    groups = []
    i = low
    while i + 4 <= high:
        group = arr[i:i + 5]
        group.sort()
        groups.append(group[2])
        i += 5
    
    if i <= high:
        group = arr[i:high + 1]
        group.sort()
        groups.append(group[len(group) // 2])
    
    # 遞迴找中位數的中位數
    if len(groups) == 1:
        pivot = groups[0]
    else:
        pivot = _deterministic_select(groups, 0, len(groups) - 1, len(groups) // 2)
    
    # 分區
    pivot_idx = arr.index(pivot, low, high + 1)
    arr[pivot_idx], arr[high] = arr[high], arr[pivot_idx]
    
    i = low
    for j in range(low, high):
        if arr[j] <= pivot:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    
    arr[i], arr[high] = arr[high], arr[i]
    
    if k == i:
        return arr[k]
    elif k < i:
        return _deterministic_select(arr, low, i - 1, k)
    else:
        return _deterministic_select(arr, i + 1, high, k)
```

## 蒙特卡羅方法

```python
def monte_carlo_pi(n_points):
    """
    蒙特卡羅方法估計 π
    落在單位圓內的概率 = π/4
    """
    inside = 0
    
    for _ in range(n_points):
        x = random.random()
        y = random.random()
        
        if x**2 + y**2 <= 1:
            inside += 1
    
    return 4 * inside / n_points

def monte_carlo_integration(f, a, b, n_points):
    """
    蒙特卡羅數值積分
    """
    total = 0
    
    for _ in range(n_points):
        x = random.uniform(a, b)
        total += f(x)
    
    return total * (b - a) / n_points

def monte_carlo_primality(n, k=20):
    """
    Miller-Rabin 素數測試 - 蒙特卡羅
    錯誤概率 ≤ 4^(-k)
    """
    if n < 2:
        return False
    if n == 2 or n == 3:
        return True
    if n % 2 == 0:
        return False
    
    # 寫 n-1 = 2^r * d
    r, d = 0, n - 1
    while d % 2 == 0:
        r += 1
        d //= 2
    
    for _ in range(k):
        a = random.randint(2, n - 2)
        x = pow(a, d, n)
        
        if x == 1 or x == n - 1:
            continue
        
        for _ in range(r - 1):
            x = pow(x, 2, n)
            if x == n - 1:
                break
        else:
            return False
    
    return True

def random_hash(x, m, p=2**61 - 1):
    """隨機哈希函數"""
    a = random.randint(1, p - 1)
    b = random.randint(0, p - 1)
    return ((a * x + b) % p) % m
```

## Min-Cut

```python
def karger_min_cut(graph):
    """
    Karger 演算法 - 隨機化最小割
    O(n²) 時間，成功概率 ≥ 1/n²
    重複 O(n²) 次，成功概率 ≥ 常數
    """
    n = len(graph)
    
    def contract(graph):
        """收縮一條邊"""
        # 選擇一條隨機邊
        u = random.choice(list(graph.keys()))
        v = random.choice(list(graph[u]))
        
        # 合併 u 和 v
        for neighbor in graph[v]:
            if neighbor != u:
                graph[u].extend(graph[neighbor])
                graph[neighbor] = [u if x == v else x for x in graph[neighbor]]
        
        graph[u] = [x for x in graph[u] if x != v and x != u]
        del graph[v]
        
        # 自環
        graph[u] = [x for x in graph[u] if x != u]
        
        return graph, len(graph[v]) if v in graph else 0
    
    g = {i: list(neighbors) for i, neighbors in enumerate(graph.values())}
    
    while len(g) > 2:
        g = contract(g)[0]
    
    # 剩餘的兩邊之間的邊數
    u, v = list(g.keys())
    return len([x for x in g[u] if x == v])

def min_cut_repeated(graph, iterations=None):
    """
    重複 Karger 演算法
    成功概率隨迭代次數增加
    """
    if iterations is None:
        iterations = len(graph) ** 2
    
    best_cut = float('inf')
    
    for _ in range(iterations):
        cut = karger_min_cut(graph.copy())
        best_cut = min(best_cut, cut)
    
    return best_cut
```

```python
def random_walk_analysis(n):
    """
    分析隨機遊戲的期望時間
    """
    position = 0
    steps = 0
    
    while 0 < position < n:
        if random.random() < 0.5:
            position += 1
        else:
            position -= 1
        steps += 1
    
    return steps

def coupon_collector(n):
    """
    收集 coupon - 期望 O(n log n)
    """
    collected = set()
    draws = 0
    
    while len(collected) < n:
        coupon = random.randint(0, n - 1)
        collected.add(coupon)
        draws += 1
    
    return draws

def expected_coupon_collector(n):
    """
    收集 coupon 的期望次數分析
    E = n * H_n = n * (1 + 1/2 + ... + 1/n)
    """
    return n * sum(1 / i for i in range(1, n + 1))
```

## 隨機化負載均衡

```python
def random_load_balance(tasks, servers):
    """隨機分配任務到伺服器"""
    assignment = {s: [] for s in servers}
    
    for task in tasks:
        server = random.choice(servers)
        assignment[server].append(task)
    
    return assignment

def hash_load_balance(tasks, servers):
    """哈希分配 - 一致性哈希"""
    assignment = {s: [] for s in servers}
    
    for task in tasks:
        h = hash(task) % len(servers)
        assignment[servers[h]].append(task)
    
    return assignment

def power_of_two_choices(tasks, servers):
    """
    兩個選擇 - 將任務分配到負載最輕的兩個隨機伺服器中
    期望最大負載 = O(log n / log log n)
    """
    assignment = {s: [] for s in servers}
    loads = {s: 0 for s in servers}
    
    for task in tasks:
        # 選擇兩個隨機伺服器
        candidates = random.sample(servers, 2)
        
        # 選擇負載較輕的
        chosen = min(candidates, key=lambda s: loads[s])
        assignment[chosen].append(task)
        loads[chosen] += 1
    
    return assignment, max(loads.values())
```

## 總結

隨機演算法利用隨機性來簡化問題求解或提高效率。拉斯維加斯演算法（如隨機化快速排序）保證正確性，蒙特卡羅演算法（如 Miller-Rabin 素數測試）犧牲少量正確性換取效率。理解隨機演算法的概率分析對於評估其性能和可靠性至關重要。

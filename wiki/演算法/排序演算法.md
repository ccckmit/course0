# 排序演算法

排序演算法是電腦科學中最基礎也是最重要的演算法之一。排序不僅是解決實際問題的常用工具，也是學習演算法設計和複雜度分析的絕佳素材。從簡單的氣泡排序到高效的快速排序，每種排序演算法都有其獨特的設計理念和適用場景。理解這些演算法不僅能幫助我們解決實際問題，還能培養我們的程式設計思維和最佳化能力。

## 詳細演算法說明

### 1. 氣泡排序（Bubble Sort）

氣泡排序是最簡單的排序演算法，重複走過列表，比較相鄰元素並根據需要交換。

```python
def bubble_sort(arr):
    """
    氣泡排序
    時間複雜度: O(n²) - 最差/平均/最好都是 O(n²) 
    空間複雜度: O(1)
    穩定性: 穩定
    """
    n = len(arr)
    
    for i in range(n):
        swapped = False
        
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        
        # 如果沒有交換，陣列已經有序
        if not swapped:
            break
    
    return arr

def bubble_sort_optimized(arr):
    """優化版：記錄最後交換位置"""
    n = len(arr)
    last_swap = n - 1
    
    while last_swap > 0:
        new_last_swap = 0
        
        for j in range(last_swap):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                new_last_swap = j
        
        last_swap = new_last_swap
    
    return arr

# 範例
arr = [64, 34, 25, 12, 22, 11, 90]
print("原始:", arr)
print("排序後:", bubble_sort(arr.copy()))
```

**視覺化過程**：

```
[11, 12, 22, 25, 34, 64, 90]
[11, 12, 22, 25, 34, 64, 90]  # 完成
```

### 2. 選擇排序（Selection Sort）

在未排序的部分中找到最小元素，放到已排序部分的開頭。

```python
def selection_sort(arr):
    """
    選擇排序
    時間複雜度: O(n²) 
    空間複雜度: O(1)
    穩定性: 不穩定（因為交換可能改變相等元素的相對順序）
    """
    n = len(arr)
    
    for i in range(n):
        min_idx = i
        
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    
    return arr

def selection_sort_2way(arr):
    """雙向選擇排序（同時找最大和最小）"""
    n = len(arr)
    left, right = 0, n - 1
    
    while left < right:
        min_idx = max_idx = left
        
        for j in range(left, right + 1):
            if arr[j] < arr[min_idx]:
                min_idx = j
            if arr[j] > arr[max_idx]:
                max_idx = j
        
        arr[left], arr[min_idx] = arr[min_idx], arr[left]
        
        if max_idx == left:
            max_idx = min_idx
        
        arr[right], arr[max_idx] = arr[max_idx], arr[right]
        
        left += 1
        right -= 1
    
    return arr

# 範例
arr = [64, 25, 12, 22, 11]
print("選擇排序:", selection_sort(arr.copy()))
```

### 3. 插入排序（Insertion Sort）

將元素插入到已排序部分中的正確位置。

```python
def insertion_sort(arr):
    """
    插入排序
    時間複雜度: O(n²) 平均/最差, O(n) 最好（幾乎有序）
    空間複雜度: O(1)
    穩定性: 穩定
    """
    n = len(arr)
    
    for i in range(1, n):
        key = arr[i]
        j = i - 1
        
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        
        arr[j + 1] = key
    
    return arr

def binary_insertion_sort(arr):
    """二分插入排序：減少比較次數"""
    n = len(arr)
    
    for i in range(1, n):
        key = arr[i]
        
        # 二分搜尋插入位置
        left, right = 0, i - 1
        while left <= right:
            mid = (left + right) // 2
            if arr[mid] > key:
                right = mid - 1
            else:
                left = mid + 1
        
        # 移動元素
        for j in range(i, left, -1):
            arr[j] = arr[j - 1]
        
        arr[left] = key
    
    return arr

# 範例
arr = [12, 11, 13, 5, 6]
print("插入排序:", insertion_sort(arr.copy()))
```

### 4. 合併排序（Merge Sort）

分治法：將陣列分成兩半，分別排序後合併。

```python
def merge_sort(arr):
    """
    合併排序
    時間複雜度: O(n log n)
    空間複雜度: O(n)
    穩定性: 穩定
    """
    if len(arr) <= 1:
        return arr
    
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    
    return merge(left, right)

def merge(left, right):
    """合併兩個有序陣列"""
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    result.extend(left[i:])
    result.extend(right[j:])
    
    return result

def merge_sort_inplace(arr, left, right):
    """原地合併排序"""
    if left < right:
        mid = (left + right) // 2
        
        merge_sort_inplace(arr, left, mid)
        merge_sort_inplace(arr, mid + 1, right)
        merge_inplace(arr, left, mid, right)

def merge_inplace(arr, left, mid, right):
    """原地合併"""
    i = left
    j = mid + 1
    
    while i <= mid and j <= right:
        if arr[i] <= arr[j]:
            i += 1
        else:
            value = arr[j]
            # 將 arr[i: j] 向右移動一位
            k = j
            while k > i:
                arr[k] = arr[k - 1]
                k -= 1
            arr[i] = value
            
            i += 1
            mid += 1
            j += 1

# 範例
arr = [38, 27, 43, 3, 9, 82, 10]
print("合併排序:", merge_sort(arr))
```

### 5. 快速排序（Quick Sort）

分治法：選擇一個 pivot，將陣列分為兩個子陣列。

```python
def quick_sort(arr):
    """
    快速排序
    時間複雜度: O(n log n) 平均, O(n²) 最差
    空間複雜度: O(log n)
    穩定性: 不穩定
    """
    if len(arr) <= 1:
        return arr
    
    pivot = arr[0]
    less = [x for x in arr[1:] if x < pivot]
    equal = [x for x in arr[1:] if x == pivot]
    greater = [x for x in arr[1:] if x > pivot]
    
    return quick_sort(less) + [pivot] + equal + quick_sort(greater)

def quick_sort_inplace(arr, low, high):
    """原地快速排序"""
    if low < high:
        pi = partition(arr, low, high)
        quick_sort_inplace(arr, low, pi - 1)
        quick_sort_inplace(arr, pi + 1, high)

def partition(arr, low, high):
    """ Lomuto partition scheme """
    pivot = arr[high]
    i = low - 1
    
    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1

def quick_sort_3way(arr, low, high):
    """三分法快速排序（處理大量重複元素）"""
    if low >= high:
        return
    
    lt = low  # less than
    gt = high  # greater than
    i = low
    pivot = arr[low]
    
    while i <= gt:
        if arr[i] < pivot:
            arr[lt], arr[i] = arr[i], arr[lt]
            lt += 1
            i += 1
        elif arr[i] > pivot:
            arr[gt], arr[i] = arr[i], arr[gt]
            gt -= 1
        else:
            i += 1
    
    quick_sort_3way(arr, low, lt - 1)
    quick_sort_3way(arr, gt + 1, high)

# 範例
arr = [10, 7, 8, 9, 1, 5]
print("快速排序:", quick_sort(arr.copy()))
```

### 6. 堆排序（Heap Sort）

使用二元堆資料結構進行排序。

```python
def heap_sort(arr):
    """
    堆排序
    時間複雜度: O(n log n)
    空間複雜度: O(1)
    穩定性: 不穩定
    """
    n = len(arr)
    
    # 建立最大堆
    for i in range(n // 2 - 1, -1, -1):
        heapify(arr, n, i)
    
    # 從堆中取出元素
    for i in range(n - 1, 0, -1):
        arr[0], arr[i] = arr[i], arr[0]
        heapify(arr, i, 0)
    
    return arr

def heapify(arr, n, i):
    """維護最大堆性質"""
    largest = i
    left = 2 * i + 1
    right = 2 * i + 2
    
    if left < n and arr[left] > arr[largest]:
        largest = left
    
    if right < n and arr[right] > arr[largest]:
        largest = right
    
    if largest != i:
        arr[i], arr[largest] = arr[largest], arr[i]
        heapify(arr, n, largest)

# 範例
arr = [12, 11, 13, 5, 6, 7]
print("堆排序:", heap_sort(arr.copy()))
```

## 非比較排序

### 1. 計數排序（Counting Sort）

適用於範圍有限的整數排序。

```python
def counting_sort(arr):
    """
    計數排序
    時間複雜度: O(n + k), k 是元素範圍
    空間複雜度: O(n + k)
    穩定性: 穩定
    """
    if not arr:
        return arr
    
    # 找到範圍
    min_val = min(arr)
    max_val = max(arr)
    range_size = max_val - min_val + 1
    
    # 計數
    count = [0] * range_size
    for num in arr:
        count[num - min_val] += 1
    
    # 累加（穩定性的關鍵）
    for i in range(1, range_size):
        count[i] += count[i - 1]
    
    # 輸出
    result = [0] * len(arr)
    for num in reversed(arr):
        result[count[num - min_val] - 1] = num
        count[num - min_val] -= 1
    
    return result

def counting_sort_by_digit(arr, exp):
    """按某位數字進行計數排序（基數排序的子步驟）"""
    n = len(arr)
    output = [0] * n
    count = [0] * 10
    
    for num in arr:
        digit = (num // exp) % 10
        count[digit] += 1
    
    for i in range(1, 10):
        count[i] += count[i - 1]
    
    for i in range(n - 1, -1, -1):
        digit = (arr[i] // exp) % 10
        output[count[digit] - 1] = arr[i]
        count[digit] -= 1
    
    return output

# 範例
arr = [4, 2, 2, 8, 3, 3, 1]
print("計數排序:", counting_sort(arr))
```

### 2. 基數排序（Radix Sort）

按位元從低位到高位（或高位到低位）進行排序。

```python
def radix_sort(arr):
    """
    基數排序
    時間複雜度: O(nk), k 是位數
    空間複雜度: O(n + k)
    穩定性: 穩定
    """
    if not arr:
        return arr
    
    max_val = max(arr)
    exp = 1
    
    while max_val // exp > 0:
        arr = counting_sort_by_digit(arr, exp)
        exp *= 10
    
    return arr

# 範例
arr = [170, 45, 75, 90, 802, 24, 2, 66]
print("基數排序:", radix_sort(arr))
```

### 3. 桶排序（Bucket Sort）

將元素分配到多個桶中，在桶內使用其他排序演算法。

```python
def bucket_sort(arr):
    """
    桶排序
    時間複雜度: O(n + k) 平均
    空間複雜度: O(n + k)
    穩定性: 取決於桶內排序
    """
    if not arr:
        return arr
    
    n = len(arr)
    min_val = min(arr)
    max_val = max(arr)
    
    if min_val == max_val:
        return arr
    
    bucket_count = n
    buckets = [[] for _ in range(bucket_count)]
    
    # 分配到桶
    for num in arr:
        index = (num - min_val) * (bucket_count - 1) // (max_val - min_val)
        buckets[index].append(num)
    
    # 在桶內排序
    for bucket in buckets:
        bucket.sort()
    
    # 合併
    result = []
    for bucket in buckets:
        result.extend(bucket)
    
    return result

# 使用插入排序的桶排序
def bucket_sort_insertion(arr):
    n = len(arr)
    buckets = [[] for _ in range(n)]
    
    for num in arr:
        index = int(num * n)
        buckets[index].append(num)
    
    for bucket in buckets:
        insertion_sort(bucket)
    
    result = []
    for bucket in buckets:
        result.extend(bucket)
    
    return result

# 範例
arr = [0.78, 0.17, 0.39, 0.26, 0.72, 0.94, 0.21, 0.12, 0.23, 0.68]
print("桶排序:", bucket_sort(arr))
```

## 排序演算法選擇指南

```python
def choose_sort(arr_size, data_type="general"):
    """
    根據情況選擇合適的排序演算法
    """
    if data_type == "small":
        # 小資料量：插入排序常最快
        return "insertion_sort"
    
    elif data_type == "nearly_sorted":
        # 幾乎有序：插入排序
        return "insertion_sort"
    
    elif data_type == "large_range":
        # 大範圍整數：計數排序
        return "counting_sort"
    
    elif data_type == "decimal":
        # 小數：桶排序
        return "bucket_sort"
    
    elif data_type == "general":
        # 一般情況：快速排序或合併排序
        if arr_size < 1000:
            return "insertion_sort"
        else:
            return "quick_sort"
    
    elif data_type == "worst_case":
        # 需要保證最差效能：合併排序或堆排序
        return "merge_sort"
    
    elif data_type == "memory_limited":
        # 記憶體受限：原地排序
        return "heap_sort"
    
    return "quick_sort"
```

### 外部排序（大量資料）

```python
def external_sort(input_file, output_file, chunk_size=1000000):
    """
    外部排序：處理大於記憶體的資料
    """
    chunks = []
    
    # 1. 分割並排序每個塊
    with open(input_file, 'r') as f:
        chunk = []
        for line in f:
            chunk.append(int(line.strip()))
            if len(chunk) >= chunk_size:
                chunk.sort()
                chunks.append(chunk)
                chunk = []
        if chunk:
            chunk.sort()
            chunks.append(chunk)
    
    # 2. 合併有序塊（多路合併）
    import heapq
    
    with open(output_file, 'w') as f:
        # 使用堆進行有序合併
        heap = [(chunk[0], i, 0) for i, chunk in enumerate(chunks) if chunk]
        heapq.heapify(heap)
        
        while heap:
            val, chunk_idx, element_idx = heapq.heappop(heap)
            f.write(f"{val}\n")
            
            if element_idx + 1 < len(chunks[chunk_idx]):
                next_val = chunks[chunk_idx][element_idx + 1]
                heapq.heappush(heap, (next_val, chunk_idx, element_idx + 1))
```

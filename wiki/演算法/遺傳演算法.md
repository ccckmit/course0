# 遺傳演算法 (Genetic Algorithm)

## 概述

遺傳演算法是模擬自然選擇和遺傳機制的優化演算法，透過羣體進化搜索最優解。

## 基本流程

```
1. 初始化羣體
2. 評估適應度
3. 選擇父母
4. 交配產生後代
5. 突變
6. 取代舊羣體
7. 重複直到收斂
```

## 基本實現

```python
import random
import operator

class GeneticAlgorithm:
    def __init__(self, fitness_func, gene_space, population_size=100, 
                 generations=1000, mutation_rate=0.1, crossover_rate=0.7):
        self.fitness_func = fitness_func
        self.gene_space = gene_space
        self.pop_size = population_size
        self.generations = generations
        self.mutation_rate = mutation_rate
        self.crossover_rate = crossover_rate
        self.population = []
    
    def create_individual(self):
        """創建隨機個體"""
        return [random.choice(self.gene_space) for _ in range(len(self.gene_space))]
    
    def initialize_population(self):
        """初始化羣體"""
        self.population = [self.create_individual() for _ in range(self.pop_size)]
    
    def evaluate_population(self):
        """評估羣體"""
        fitness_scores = [(i, self.fitness_func(individual)) 
                        for i, individual in enumerate(self.population)]
        return sorted(fitness_scores, key=operator.itemgetter(1), reverse=True)
    
    def select_parents(self, fitness_scores):
        """選擇父母（錦標賽選擇）"""
        def tournament(fitness_scores, k=3):
            tournament = random.sample(fitness_scores, k)
            return max(tournament, key=operator.itemgetter(1))
        
        parents = []
        for _ in range(2):
            idx, score = tournament(fitness_scores)
            parents.append(self.population[idx])
        return parents
    
    def crossover(self, parent1, parent2):
        """單點交配"""
        if random.random() > self.crossover_rate:
            return parent1[:], parent2[:]
        
        point = random.randint(1, len(parent1) - 1)
        child1 = parent1[:point] + parent2[point:]
        child2 = parent2[:point] + parent1[point:]
        return child1, child2
    
    def mutate(self, individual):
        """翻轉突變"""
        for i in range(len(individual)):
            if random.random() < self.mutation_rate:
                individual[i] = random.choice(self.gene_space)
        return individual
    
    def evolve(self):
        """進化一代"""
        new_population = []
        
        while len(new_population) < self.pop_size:
            fitness_scores = self.evaluate_population()
            parent1, parent2 = self.select_parents(fitness_scores)
            child1, child2 = self.crossover(parent1, parent2)
            new_population.append(self.mutate(child1))
            if len(new_population) < self.pop_size:
                new_population.append(self.mutate(child2))
        
        self.population = new_population
    
    def run(self):
        """執行演化"""
        self.initialize_population()
        best_individual = None
        best_fitness = float('-inf')
        
        for gen in range(self.generations):
            fitness_scores = self.evaluate_population()
            top_individual, top_fitness = fitness_scores[0]
            
            if top_fitness > best_fitness:
                best_fitness = top_fitness
                best_individual = self.population[top_individual].copy()
            
            if gen % 100 == 0:
                print(f"Generation {gen}: Best fitness = {best_fitness}")
            
            self.evolve()
        
        return best_individual, best_fitness
```

## 經典問題

### 揹包問題

```python
def knapsack_ga():
    """遺傳演算法求解 0-1 揹包問題"""
    weights = [12, 7, 11, 8, 9]
    values = [24, 13, 23, 15, 16]
    capacity = 26
    
    def fitness(individual):
        total_weight = sum(w for w, i in zip(weights, individual) if i)
        total_value = sum(v for v, i in zip(values, individual) if i)
        if total_weight > capacity:
            return 0
        return total_value
    
    ga = GeneticAlgorithm(
        fitness_func=fitness,
        gene_space=[0, 1],
        population_size=100,
        generations=200,
        mutation_rate=0.05,
        crossover_rate=0.8
    )
    
    return ga.run()

best, score = knapsack_ga()
print(f"Best solution: {best}, Total value: {score}")
```

### 旅行推銷員問題

```python
def tsp_ga():
    """遺傳演算法求解 TSP"""
    n_cities = 10
    distances = [[random.randint(1, 100) for _ in range(n_cities)] 
                 for _ in range(n_cities)]
    
    def fitness(route):
        total = sum(distances[route[i]][route[i+1]] for i in range(len(route)-1))
        total += distances[route[-1]][route[0]]
        return 1 / total
    
    def pmx_crossover(parent1, parent2):
        """部分映射交配 (PMX)"""
        size = len(parent1)
        c1, c2 = [-1] * size, [-1] * size
        
        p1, p2 = random.randint(0, size-2), random.randint(p1+1, size-1)
        
        c1[p1:p2+1] = parent1[p1:p2+1]
        c2[p1:p2+1] = parent2[p1:p2+1]
        
        def fill_child(child, other, p1, p2):
            for i in range(p1, p2+1):
                if other[i] not in child:
                    pos = i
                    while p1 <= pos <= p2:
                        pos = other.index(child[pos])
                    child[pos] = other[i]
        
        fill_child(c1, parent2, p1, p2)
        fill_child(c2, parent1, p1, p2)
        return c1, c2
    
    def order_mutate(route):
        """順序交換突變"""
        i, j = random.sample(range(len(route)), 2)
        route[i], route[j] = route[j], route[i]
        return route
    
    # 簡化版 GA
    population = [random.sample(range(n_cities), n_cities) for _ in range(100)]
    
    for gen in range(500):
        fitness_scores = [(route, fitness(route)) for route in population]
        fitness_scores.sort(key=lambda x: x[1], reverse=True)
        
        new_population = [fitness_scores[0][0]]
        
        while len(new_population) < 100:
            p1, p2 = random.choices(
                [f[0] for f in fitness_scores[:50]], 
                weights=[f[1] for f in fitness_scores[:50]], 
                k=2
            )
            c1, c2 = pmx_crossover(p1, p2)
            new_population.extend([order_mutate(c1), order_mutate(c2)])
        
        population = new_population[:100]
    
    best = fitness_scores[0][0]
    return best, fitness(best)

best_route, best_dist = tsp_ga()
print(f"Best route: {best_route}, Distance: {1/best_dist:.2f}")
```

## 選擇方法

### 輪盤選擇

```python
def roulette_wheel_selection(population, fitness_func, n=2):
    """輪盤選擇"""
    fitness_scores = [fitness_func(ind) for ind in population]
    total_fitness = sum(fitness_scores)
    probabilities = [f / total_fitness for f in fitness_scores]
    
    selected = []
    for _ in range(n):
        r = random.random()
        cumulative = 0
        for i, p in enumerate(probabilities):
            cumulative += p
            if r <= cumulative:
                selected.append(population[i])
                break
    return selected
```

### 排名選擇

```python
def rank_selection(population, fitness_func, n=2):
    """排名選擇"""
    fitness_scores = [(ind, fitness_func(ind)) for ind in population]
    fitness_scores.sort(key=lambda x: x[1])
    
    ranks = list(range(1, len(fitness_scores) + 1))
    total_rank = sum(ranks)
    probabilities = [r / total_rank for r in ranks]
    
    indices = random.choices(range(len(population)), weights=probabilities, k=n)
    return [population[i] for i in indices]
```

```python
def single_point_crossover(p1, p2):
    """單點交配"""
    point = random.randint(1, len(p1) - 1)
    return p1[:point] + p2[point:], p2[:point] + p1[point:]
```

### 雙點交配

```python
def two_point_crossover(p1, p2):
    """雙點交配"""
    points = sorted(random.sample(range(1, len(p1)), 2))
    c1 = p1[:points[0]] + p2[points[0]:points[1]] + p1[points[1]:]
    c2 = p2[:points[0]] + p1[points[0]:points[1]] + p2[points[1]:]
    return c1, c2
```

## 突變方法

```python
def flip_bit_mutate(individual, mutation_rate=0.1):
    """翻轉位元突變"""
    return [1 - gene if random.random() < mutation_rate else gene 
            for gene in individual]

def swap_mutate(individual):
    """交換突變"""
    i, j = random.sample(range(len(individual)), 2)
    individual[i], individual[j] = individual[j], individual[i]
    return individual

def scramble_mutate(individual):
    """擾亂突變"""
    i, j = sorted(random.sample(range(len(individual)), 2))
    random.shuffle(individual[i:j+1])
    return individual

def inversion_mutate(individual):
    """反轉突變"""
    i, j = sorted(random.sample(range(len(individual)), 2))
    individual[i:j+1] = reversed(individual[i:j+1])
    return individual
```

## 參數設置

| 參數 | 典型值 | 說明 |
|------|--------|------|
| 羣體大小 | 50-200 | 越大越能找到好解 |
| 突變率 | 0.01-0.1 | 太高破壞好解 |
| 交配率 | 0.6-0.9 | 太低探索不足 |
| 代數 | 100-1000 | 視問題複雜度 |

| 特性 | 爬山法 | 遺傳演算法 |
|------|--------|-----------|

| 收斂速度 | 快 | 慢 |
| 參數敏感度 | 中 | 高 |

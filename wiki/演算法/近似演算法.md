# 近似演算法：處理 NP-困難問題

## 概述

近似演算法為 NP-困難優化問題提供接近最優的解。與精確演算法不同，近似演算法保證在多項式時間內找到與最優解相差一個可證明因子的解。近似比（approximation ratio）衡量解的質量：比值越小，解越好。

許多實際問題（如旅行商問題、集覆蓋問題、装箱問題）沒有已知的多項式時間精確演算法，近似演算法提供了一個實用的替代方案。

        self.time_complexity = None
    
    def approximate(self, instance):
        """返回近似解"""
        raise NotImplementedError
    
    def get_ratio_bound(self):
        """返回近似比上界"""
        return self.approximate_ratio

## 頂點覆蓋問題

```python
def vertex_cover_greedy(graph):
    """
    貪心頂點覆蓋 - 2-近似
    """
    cover = set()
    edges = set(graph.keys())
    
    while edges:
        # 選擇度最大的頂點
        degrees = {}
        for u, neighbors in graph.items():
            if u not in cover:
                count = sum(1 for v in neighbors if v not in cover)
                degrees[u] = count
        
        if not degrees:
            break
        
        u = max(degrees, key=degrees.get)
        cover.add(u)
        
        # 移除與 u 相鄰的邊
        edges = {(a, b) for a, b in edges if a != u and b != u}
    
    return cover

def vertex_cover_2_approx(graph):
    """
    2-近似頂點覆蓋
    使用最大匹配
    """
    # 構造最大匹配
    match = maximum_matching_2_approx(graph)
    
    # 匹配兩端的頂點構成 2-近似解
    cover = set()
    for u, v in match:
        cover.add(u)
        cover.add(v)
    
    return cover

def maximum_matching_2_approx(graph):
    """最大匹配（貪心，2-近似）"""
    matched = set()
    matching = []
    
    for u in graph:
        if u not in matched:
            for v in graph[u]:
                if v not in matched:
                    matching.append((u, v))
                    matched.add(u)
                    matched.add(v)
                    break
    
    return matching
```

## 集覆蓋問題

```python
def set_cover(universe, subsets):
    """
    貪心集覆蓋 - O(nm)
    近似比: H(d_max) 其中 H 是調和數，d_max 是最大集合大小
    
    理論保證: ln(n) + 1 近似
    """
    cover = set()
    uncovered = set(universe)
    result = []
    
    while uncovered:
        # 選擇覆蓋最多未覆蓋元素的集合
        best_set = None
        best_count = 0
        
        for subset in subsets:
            count = len(set(subset) & uncovered)
            if count > best_count:
                best_count = count
                best_set = subset
        
        if best_count == 0:
            break  # 沒有集合能覆蓋剩餘元素
        
        result.append(best_set)
        cover.add(best_set)
        uncovered -= set(best_set)
    
    return result, cover

def weighted_set_cover(universe, subsets, costs):
    """
    加權集覆蓋 - 貪心 ln(n)-近似
    """
    cover = set()
    uncovered = set(universe)
    result = []
    total_cost = 0
    
    costs = dict(costs)
    
    while uncovered:
        best_ratio = float('inf')
        best_set = None
        best_cost = None
        
        for i, subset in enumerate(subsets):
            subset_elements = set(subset) & uncovered
            if subset_elements:
                ratio = costs[i] / len(subset_elements)
                if ratio < best_ratio:
                    best_ratio = ratio
                    best_set = i
                    best_cost = costs[i]
        
        if best_set is None:
            break
        
        result.append(best_set)
        cover.add(best_set)
        uncovered -= set(subsets[best_set])
        total_cost += best_cost
    
    return result, cover, total_cost
```

## 旅行商問題

```python
import math

def tsp_2_approx(points):
    """
    度約束生成樹 - 2-近似 MST-TSP
    適用於三角不等式成立的情況
    """
    # 計算距離矩陣
    n = len(points)
    dist = [[0] * n for _ in range(n)]
    for i in range(n):
        for j in range(n):
            dist[i][j] = math.sqrt((points[i][0] - points[j][0])**2 + 
                                   (points[i][1] - points[j][1])**2)
    
    # 找 MST
    mst = prim_mst(dist)
    
    # 2-近似解：MST 的歐拉 tour 簡化
    tour = mst_dfs_tour(mst, n)
    
    return tour

def prim_mst(dist):
    """Prim 演算法求 MST"""
    n = len(dist)
    in_mst = [False] * n
    min_edge = [float('inf')] * n
    parent = [-1] * n
    
    min_edge[0] = 0
    mst = []
    
    for _ in range(n):
        u = min((i for i in range(n) if not in_mst[i]), key=lambda i: min_edge[i])
        in_mst[u] = True
        
        if parent[u] != -1:
            mst.append((parent[u], u))
        
        for v in range(n):
            if dist[u][v] < min_edge[v] and not in_mst[v]:
                min_edge[v] = dist[u][v]
                parent[v] = u
    
    return mst

def mst_dfs_tour(mst, n):
    """MST 的 DFS 遍歷形成 tour"""
    adj = [[] for _ in range(n)]
    for u, v in mst:
        adj[u].append(v)
        adj[v].append(u)
    
    visited = [False] * n
    tour = []
    
    def dfs(v):
        visited[v] = True
        tour.append(v)
        for u in adj[v]:
            if not visited[u]:
                dfs(u)
    
    dfs(0)
    tour.append(0)  # 返回起點
    
    return tour

def tsp_christofides(points):
    """
    Christofides 演算法 - 1.5-近似
    對於度量 TSP
    """
    # 1. 找 MST
    n = len(points)
    dist = [[math.sqrt((points[i][0] - points[j][0])**2 + 
                        (points[i][1] - points[j][1])**2)
             for j in range(n)] for i in range(n)]
    
    mst = prim_mst(dist)
    
    # 2. 找奇度頂點
    degree = [0] * n
    for u, v in mst:
        degree[u] += 1
        degree[v] += 1
    
    odd_vertices = [i for i in range(n) if degree[i] % 2 == 1]
    
    # 3. 在奇度頂點間找最小權匹配
    matching = minimum_weight_matching(odd_vertices, dist)
    
    # 4. 合併得到歐拉圖
    euler_graph = mst + matching
    
    # 5. 求歐拉 tour 並 shortcuts
    euler_tour = Hierholzer_eulerian(euler_graph)
    
    return shortcut_tour(euler_tour)  # 去除重複頂點
```

## 背包問題

```python
def knapsack_greedy(items, capacity):
    """
    分數背包 - 貪心 1-近似
    總能達到最優解
    """
    # 按單位價值排序
    items = sorted(items, key=lambda x: x[2]/x[1], reverse=True)
    
    total_value = 0
    total_weight = 0
    taken = []
    
    for name, weight, value in items:
        if total_weight + weight <= capacity:
            taken.append((name, weight, value))
            total_weight += weight
            total_value += value
        else:
            # 取部分物品
            remaining = capacity - total_weight
            fraction = remaining / weight
            taken.append((name, remaining, value * fraction))
            total_value += value * fraction
            break
    
    return total_value, taken

def knapsack_fptas(items, capacity, epsilon):
    """
    完全多項式時間近似方案 (FPTAS)
    (1-ε)-近似，時間 O(n³/ε)
    """
    n = len(items)
    max_value = max(item[2] for item in items)
    
    # 縮放參數
    K = epsilon * max_value / n
    
    # 縮放價值
    scaled_items = [(item[0], item[1], int(item[2] / K)) for item in items]
    
    # 動態規劃
    dp = [0] * (n * max(item[2] for item in items) + 1)
    
    for name, weight, scaled_value in scaled_items:
        for w in range(capacity, weight - 1, -1):
            dp[w] = max(dp[w], dp[w - weight] + scaled_value)
    
    # 反向縮放
    best_value = dp[capacity] * K
    return best_value
```

## 箱子裝載

```python
def first_fit_decreasing(items, bin_capacity):
    """
    First Fit Decreasing (FFD) - 箱子裝載
    近似比: 11/9 ≈ 1.22
    最差情況: 71/60 ≈ 1.183
    """
    items = sorted(items, reverse=True)
    bins = []
    
    for item in items:
        placed = False
        
        for bin_items in bins:
            if sum(bin_items) + item <= bin_capacity:
                bin_items.append(item)
                placed = True
                break
        
        if not placed:
            bins.append([item])
    
    return bins

def best_fit_decreasing(items, bin_capacity):
    """
    Best Fit Decreasing - 選擇最合適的箱子
    近似比: 1.7
    """
    items = sorted(items, reverse=True)
    bins = []
    bin_remaining = []
    
    for item in items:
        best_idx = -1
        best_remaining = bin_capacity + 1
        
        for i, remaining in enumerate(bin_remaining):
            if remaining >= item and remaining - item < best_remaining:
                best_remaining = remaining - item
                best_idx = i
        
        if best_idx == -1:
            bins.append([item])
            bin_remaining.append(bin_capacity - item)
        else:
            bins[best_idx].append(item)
            bin_remaining[best_idx] -= item
    
    return bins

def first_fit(items, bin_capacity):
    """First Fit - O(n log n)"""
    bins = []
    bin_remaining = []
    
    for item in items:
        best_idx = -1
        
        for i, remaining in enumerate(bin_remaining):
            if remaining >= item:
                best_idx = i
                break
        
        if best_idx == -1:
            bins.append([item])
            bin_remaining.append(bin_capacity - item)
        else:
            bins[best_idx].append(item)
            bin_remaining[best_idx] -= item
    
    return bins
```

## 總結

近似演算法為 NP-困難問題提供了實用的解決方案。不同的問題有不同的近似保證：頂點覆蓋有 2-近似，集覆蓋有 ln(n)-近似，度量 TSP 有 1.5-近似的 Christofides 演算法。選擇合適的近似演算法需要考慮問題特性、近似比和實際應用需求。

# 動態規劃法 (Dynamic Programming)

## 概述

動態規劃是一種將複雜問題分解為更小子問題的演算法策略，適用於具有最優子結構和重疊子問題的問題。

## 基本條件

1. **最優子結構**：問題的最優解包含子問題的最優解
2. **重疊子問題**：子問題會被多次計算

## 基本步驟

```
動態規劃流程：
1. 定義子問題
2. 建立遞迴關係
3. 計算並儲存子問題
4. 構造最優解
```

### 費波那契數列

```python
# 暴力法 - 指數時間
def fibonacci_naive(n):
    if n <= 1:
        return n
    return fibonacci_naive(n-1) + fibonacci_naive(n-2)

# 動態規劃 - 自頂向下（記憶化）
def fibonacci_memo(n, memo=None):
    if memo is None:
        memo = {}
    if n in memo:
        return memo[n]
    if n <= 1:
        return n
    memo[n] = fibonacci_memo(n-1, memo) + fibonacci_memo(n-2, memo)
    return memo[n]

# 動態規劃 - 自底向上（表格法）
def fibonacci_dp(n):
    if n <= 1:
        return n
    dp = [0] * (n + 1)
    dp[1] = 1
    for i in range(2, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    return dp[n]

# 空間優化
def fibonacci_optimized(n):
    if n <= 1:
        return n
    prev, curr = 0, 1
    for _ in range(2, n + 1):
        prev, curr = curr, prev + curr
    return curr
```

### 揹包問題

```python
# 0-1 揹包問題
def knapsack(weights, values, capacity):
    n = len(weights)
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]
    
    for i in range(1, n + 1):
        for w in range(capacity + 1):
            if weights[i-1] <= w:
                dp[i][w] = max(
                    dp[i-1][w],
                    dp[i-1][w-weights[i-1]] + values[i-1]
                )
            else:
                dp[i][w] = dp[i-1][w]
    
    return dp[n][capacity]

# 空間優化
def knapsack_optimized(weights, values, capacity):
    dp = [0] * (capacity + 1)
    
    for i in range(len(weights)):
        for w in range(capacity, weights[i] - 1, -1):
            dp[w] = max(dp[w], dp[w-weights[i]] + values[i])
    
    return dp[capacity]
```

### 最長公共子序列

```python
def lcs(s1, s2):
    """最長公共子序列"""
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    
    # 回溯構造 LCS
    lcs_str = []
    i, j = m, n
    while i > 0 and j > 0:
        if s1[i-1] == s2[j-1]:
            lcs_str.append(s1[i-1])
            i -= 1
            j -= 1
        elif dp[i-1][j] > dp[i][j-1]:
            i -= 1
        else:
            j -= 1
    
    return dp[m][n], ''.join(reversed(lcs_str))

print(lcs("ABCD", "ACBD"))  # (3, 'ABD' 或 'ACD')
```

### 編輯距離

```python
def edit_distance(s1, s2):
    """萊文斯坦距離（插入、刪除、替換）"""
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(m + 1):
        dp[i][0] = i
    for j in range(n + 1):
        dp[0][j] = j
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(
                    dp[i-1][j],
                    dp[i][j-1],
                    dp[i-1][j-1]
                )
    
    return dp[m][n]

print(edit_distance("kitten", "sitting"))  # 3
```

## 矩陣鏈乘法

```python
def matrix_chain_order(dimensions):
    """矩陣鏈乘法最小乘法次數"""
    n = len(dimensions) - 1
    dp = [[0] * n for _ in range(n)]
    
    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            dp[i][j] = float('inf')
            
            for k in range(i, j):
                cost = dp[i][k] + dp[k+1][j] + \
                       dimensions[i] * dimensions[k+1] * dimensions[j+1]
                dp[i][j] = min(dp[i][j], cost)
    
    return dp[0][n-1]

print(matrix_chain_order([10, 30, 5, 60]))  # 4500
```

## 最長遞增子序列

```python
def lis(nums):
    """最長遞增子序列"""
    if not nums:
        return 0
    
    dp = [1] * len(nums)
    
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[j] < nums[i]:
                dp[i] = max(dp[i], dp[j] + 1)
    
    return max(dp)

import bisect

def lis_optimized(nums):
    """最長遞增子序列 - O(n log n)"""
    tails = []
    
    for num in nums:
        pos = bisect.bisect_left(tails, num)
        if pos == len(tails):
            tails.append(num)
        else:
            tails[pos] = num
    
    return len(tails)

print(lis([10, 9, 2, 5, 3, 7, 101, 18]))  # 4
```

## 狀態壓縮 DP

```python
def traveling_salesman(dist):
    """旅行推銷員問題（狀態壓縮 DP）"""
    n = len(dist)
    dp = [[float('inf')] * n for _ in range(1 << n)]
    dp[1][0] = 0
    
    for mask in range(1 << n):
        for u in range(n):
            if dp[mask][u] == float('inf'):
                continue
            for v in range(n):
                if mask & (1 << v):
                    continue
                next_mask = mask | (1 << v)
                dp[next_mask][v] = min(
                    dp[next_mask][v],
                    dp[mask][u] + dist[u][v]
                )
    
    full_mask = (1 << n) - 1
    return min(dp[full_mask])
```

## DP 類型

| 類型 | 說明 | 範例 |
|------|------|------|
| 線性 DP | 一維狀態 | 費波那契、爬樓梯 |
| 區間 DP | 二維區間 | 矩陣鏈、分割 |
| 揹包 DP | 組合優化 | 0-1揹包、完全揹包 |
| 樹形 DP | 樹上問題 | 樹直徑 |
| 狀態壓縮 | 位元狀態 | TSP、圖著色 |

# 計算幾何：幾何問題的演算法

## 概述

計算幾何研究幾何問題的高效演算法，廣泛應用於電腦圖形學、機器人學、地理資訊系統、計算機視覺和遊戲開發等領域。基礎問題包括點的凸包、最近點對、線段相交和點定位等。

## 基礎工具

```python
import math

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __sub__(self, other):
        return Point(self.x - other.x, self.y - other.y)
    
    def __add__(self, other):
        return Point(self.x + other.x, self.y + other.y)
    
    def __mul__(self, scalar):
        return Point(self.x * scalar, self.y * scalar)
    
    def __repr__(self):
        return f"({self.x}, {self.y})"

def cross(o, a, b):
    """向量 OA × 向量 OB 的叉積"""
    return (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)

def distance(p1, p2):
    """兩點距離"""
    return math.sqrt((p1.x - p2.x) ** 2 + (p1.y - p2.y) ** 2)

def distance_sq(p1, p2):
    """兩點距離的平方"""
    return (p1.x - p2.x) ** 2 + (p1.y - p2.y) ** 2
```

## 凸包

```python
def convex_hull_graham(points):
    """
    Graham 掃描法 - O(n log n)
    構建點集的凸包
    """
    points = sorted(set(points))  # 去重並排序
    
    if len(points) <= 1:
        return points
    
    # 構造下半部分
    lower = []
    for p in points:
        while len(lower) >= 2 and cross(lower[-2], lower[-1], p) <= 0:
            lower.pop()
        lower.append(p)
    
    # 構造上半部分
    upper = []
    for p in reversed(points):
        while len(upper) >= 2 and cross(upper[-2], upper[-1], p) <= 0:
            upper.pop()
        upper.append(p)
    
    # 移除重複的端點
    return lower[:-1] + upper[:-1]

def convex_hull_jarvis(points):
    """
    Jarvis march (Gift wrapping) - O(nh)
    h 是凸包上的點數
    """
    if len(points) < 3:
        return points
    
    hull = []
    leftmost = min(points, key=lambda p: p.x)
    current = leftmost
    
    while True:
        hull.append(current)
        next_point = points[0]
        
        for p in points:
            if next_point == current or cross(current, next_point, p) < 0:
                next_point = p
        
        current = next_point
        if current == leftmost:
            break
    
    return hull

def convex_hull_andrew(points):
    """Andrew 凸包 - 雙向版本的 Graham scan"""
    points = sorted(points, key=lambda p: (p.x, p.y))
    
    if len(points) <= 1:
        return points
    
    lower = []
    for p in points:
        while len(lower) >= 2 and cross(lower[-2], lower[-1], p) <= 0:
            lower.pop()
        lower.append(p)
    
    upper = []
    for p in reversed(points):
        while len(upper) >= 2 and cross(upper[-2], upper[-1], p) <= 0:
            upper.pop()
        upper.append(p)
    
    return lower[:-1] + upper[:-1]

def convex_hull_area(hull):
    """計算凸包面積 (Shoelace formula)"""
    if len(hull) < 3:
        return 0
    
    area = 0
    n = len(hull)
    for i in range(n):
        j = (i + 1) % n
        area += hull[i].x * hull[j].y
        area -= hull[j].x * hull[i].y
    
    return abs(area) / 2
```

## 旋轉卡尺

```python
def rotating_calipers_diameter(hull):
    """
    旋轉卡尺法 - 找凸包的直徑（最遠點對）
    O(n)
    """
    if len(hull) < 2:
        return 0, hull[0], hull[0]
    if len(hull) == 2:
        return distance(hull[0], hull[1]), hull[0], hull[1]
    
    n = len(hull)
    max_dist = 0
    p1 = p2 = None
    
    j = 1
    for i in range(n):
        # 找對邊的法向量方向
        ni = (i + 1) % n
        while abs(cross(hull[ni] - hull[i], hull[(j + 1) % n] - hull[j])) > \
              abs(cross(hull[ni] - hull[i], hull[j] - hull[(j - 1) % n])):
            j = (j + 1) % n
        
        # 檢查當前對頂點
        for p in [hull[i], hull[ni]]:
            d = distance_sq(p, hull[j])
            if d > max_dist:
                max_dist = d
                p1, p2 = p, hull[j]
    
    return math.sqrt(max_dist), p1, p2

def rotating_calipers_width(hull):
    """旋轉卡尺法 - 找凸包的寬度（最近平行線距離）"""
    if len(hull) < 2:
        return 0
    if len(hull) == 2:
        return distance(hull[0], hull[1])
    
    n = len(hull)
    min_width = float('inf')
    
    j = 1
    for i in range(n):
        ni = (i + 1) % n
        edge = hull[ni] - hull[i]
        
        # 找垂直於邊的方向
        while cross(edge, hull[(j + 1) % n] - hull[j]) > 0:
            j = (j + 1) % n
        
        # 計算距離
        dist = abs(cross(edge, hull[i] - hull[j])) / math.sqrt(edge.x ** 2 + edge.y ** 2)
        min_width = min(min_width, dist)
    
    return min_width

def min_enclosing_triangle(hull):
    """最小面積外接三角形 - 旋轉卡尺"""
    if len(hull) == 3:
        return hull
    
    n = len(hull)
    hull = hull + hull
    
    min_area = float('inf')
    result = None
    
    i, j, k = 0, 1, 2
    for _ in range(n):
        while True:
            while True:
                while cross(hull[i], hull[j], hull[(k + 1) % n]) > \
                      cross(hull[i], hull[j], hull[k]):
                    k = (k + 1) % n
                if cross(hull[i], hull[j], hull[(k - 1) % n]) >= \
                   cross(hull[i], hull[j], hull[k]):
                    break
                j = (j + 1) % n
            
            area = abs(cross(hull[i], hull[j], hull[k])) / 2
            if area < min_area:
                min_area = area
                result = [hull[i], hull[j], hull[k]]
            
            while cross(hull[i], hull[(j + 1) % n], hull[k]) >= \
                  cross(hull[i], hull[j], hull[k]):
                j = (j + 1) % n
            
            i, j, k = j, k, i
    
    return result
```

## 線段相交

```python
def on_segment(p, q, r):
    """點 q 是否在線段 pr 上"""
    return min(p.x, r.x) <= q.x <= max(p.x, r.x) and \
           min(p.y, r.y) <= q.y <= max(p.y, r.y)

def segments_intersect(p1, q1, p2, q2):
    """判斷兩線段是否相交"""
    o1 = cross(p1, q1, p2)
    o2 = cross(p1, q1, q2)
    o3 = cross(p2, q2, p1)
    o4 = cross(p2, q2, q1)
    
    # 一般情況
    if o1 * o2 < 0 and o3 * o4 < 0:
        return True
    
    # 特殊情況
    if o1 == 0 and on_segment(p1, p2, q1):
        return True
    if o2 == 0 and on_segment(p1, q2, q1):
        return True
    if o3 == 0 and on_segment(p2, p1, q2):
        return True
    if o4 == 0 and on_segment(p2, q1, q2):
        return True
    
    return False

def line_intersection(p1, p2, p3, p4):
    """兩直線交點"""
    x1, y1 = p1.x, p1.y
    x2, y2 = p2.x, p2.y
    x3, y3 = p3.x, p3.y
    x4, y4 = p4.x, p4.y
    
    denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4)
    
    if abs(denom) < 1e-10:
        return None  # 平行線
    
    t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) / denom
    
    return Point(x1 + t * (x2 - x1), y1 + t * (y2 - y1))
```

## 最近點對

```python
def closest_pair(points):
    """
    分治法求最近點對 - O(n log n)
    """
    points = sorted(points, key=lambda p: p.x)
    return closest_pair_recursive(points)

def closest_pair_recursive(points):
    """分治遞迴"""
    n = len(points)
    if n <= 3:
        return brute_force_closest(points)
    
    mid = n // 2
    mid_x = points[mid].x
    
    left = points[:mid]
    right = points[mid:]
    
    dl, pairl = closest_pair_recursive(left)
    dr, pairr = closest_pair_recursive(right)
    
    d = min(dl, dr)
    pair = pairl if dl < dr else pairr
    
    strip = [p for p in points if abs(p.x - mid_x) < d]
    ds, pairs = closest_in_strip(strip, d, pair)
    
    if ds < d:
        return ds, pairs
    return d, pair

def brute_force_closest(points):
    """暴力法"""
    n = len(points)
    if n < 2:
        return float('inf'), (points[0], points[0])
    
    min_dist = float('inf')
    pair = (points[0], points[1])
    
    for i in range(n):
        for j in range(i + 1, n):
            d = distance(points[i], points[j])
            if d < min_dist:
                min_dist = d
                pair = (points[i], points[j])
    
    return min_dist, pair

def closest_in_strip(points, d, pair):
    """在條帶區域中找最近點對"""
    min_d = d
    result_pair = pair
    
    points = sorted(points, key=lambda p: p.y)
    n = len(points)
    
    for i in range(n):
        for j in range(i + 1, min(n, i + 16)):
            dy = points[j].y - points[i].y
            if dy * dy >= min_d:
                break
            d = distance_sq(points[i], points[j])
            if d < min_d:
                min_d = d
                result_pair = (points[i], points[j])
    
    return math.sqrt(min_d), result_pair
```

## 總結

計算幾何提供了處理幾何問題的基礎工具。凸包是許多幾何演算法的基礎，Graham 掃描和 Andrew 構造是高效的凸包演算法。旋轉卡尺利用凸包的對邊性質，在常數時間內處理每個點，可以用於找直徑、寬度等問題。最近點對問題的分治解法是經典的演算法設計範例。

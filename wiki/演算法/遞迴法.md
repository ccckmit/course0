# 遞迴法 (Recursion)

## 概述

遞迴是函式自我呼叫的程式技巧，將大問題分解為相似的子問題。

## 基本結構

```python
def recursion(input):
    if input <= base_case:  # 終止條件
        return result
    return process(recursion(smaller_input))
```

## 經典問題

### 費波那契數列

```python
def fibonacci(n):
    """基本遞迴：O(2^n)"""
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

def fibonacci_memo(n, memo=None):
    """記憶化遞迴：O(n)"""
    if memo is None:
        memo = {}
    if n in memo:
        return memo[n]
    if n <= 1:
        return n
    memo[n] = fibonacci_memo(n-1, memo) + fibonacci_memo(n-2, memo)
    return memo[n]
```

### 階乘

```python
def factorial(n):
    """n! = n × (n-1)!"""
    if n <= 1:
        return 1
    return n * factorial(n-1)

def factorial_tail(n, acc=1):
    """尾遞迴版本"""
    if n <= 1:
        return acc
    return factorial_tail(n-1, n * acc)
```

### 河內塔

```python
def hanoi(n, source='A', target='C', auxiliary='B'):
    """河內塔：將 n 個盤子從 source 移到 target"""
    if n == 1:
        print(f"Move disk 1 from {source} to {target}")
        return
    
    hanoi(n-1, source, auxiliary, target)
    print(f"Move disk {n} from {source} to {target}")
    hanoi(n-1, auxiliary, target, source)

# 3 個盤子需要 2³ - 1 = 7 步
hanoi(3)
```

### 二元搜尋樹遍歷

```python
class TreeNode:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None

def inorder_traversal(node):
    """中序遍歷：左、根、右"""
    if not node:
        return []
    return inorder_traversal(node.left) + [node.val] + inorder_traversal(node.right)

def preorder_traversal(node):
    """前序遍歷：根、左、右"""
    if not node:
        return []
    return [node.val] + preorder_traversal(node.left) + preorder_traversal(node.right)

def postorder_traversal(node):
    """後序遍歷：左、右、根"""
    if not node:
        return []
    return postorder_traversal(node.left) + postorder_traversal(node.right) + [node.val]
```

### 合併排序

```python
def merge_sort(arr):
    """合併排序：分而治之"""
    if len(arr) <= 1:
        return arr
    
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    
    return merge(left, right)

def merge(left, right):
    """合併兩個有序陣列"""
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    result.extend(left[i:])
    result.extend(right[j:])
    return result
```

### 快速排序

```python
def quicksort(arr):
    """快速排序"""
    if len(arr) <= 1:
        return arr
    
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    return quicksort(left) + middle + quicksort(right)
```

### 排列組合

```python
def permutations(arr):
    """生成所有排列"""
    if len(arr) <= 1:
        return [arr]
    
    result = []
    for i in range(len(arr)):
        rest = arr[:i] + arr[i+1:]
        for p in permutations(rest):
            result.append([arr[i]] + p)
    return result

def combinations(arr, r):
    """生成所有 r-組合"""
    if r == 0:
        return [[]]
    if len(arr) < r:
        return []
    
    result = []
    for i in range(len(arr)):
        for c in combinations(arr[i+1:], r-1):
            result.append([arr[i]] + c)
    return result
```

## 遞迴 vs 迭代

| 特性 | 遞迴 | 迭代 |
|------|------|------|
| 程式碼簡潔度 | 較簡潔 | 較複雜 |
| 記憶體使用 | 呼叫堆疊 | 固定大小 |
| 時間效率 | 可能有重疊計算 | 通常較好 |
| 適用場景 | 樹結構、分治 | 線性問題 |

## 尾遞迴優化

```python
def factorial_tail_optimized(n, acc=1):
    """尾遞迴（可被優化為迭代）"""
    if n <= 1:
        return acc
    return factorial_tail_optimized(n-1, n * acc)

# Python 不保證尾遞迴優化，但可用迴圈模擬
def factorial_iterative(n):
    result = 1
    for i in range(2, n+1):
        result *= i
    return result
```

## 注意事項

1. 確保有明確的終止條件
2. 注意呼叫堆疊深度（大輸入可能stackoverflow）
3. 可用記憶化避免重複計算
4. 尾遞迴可轉換為迭代省記憶體

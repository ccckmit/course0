# 線性規劃：優化理論的基石

## 概述

線性規劃（Linear Programming, LP）是優化理論中最基本也最廣泛應用的分支，研究在線性約束下線性目標函數的最大化或最小化問題。線性規劃廣泛應用於資源調配、生產計劃、投資組合優化和網路流等領域。1947 年 George Dantzig 發明的單形法（Simplex Method）是解決線性規劃問題的第一個實用演算法。

## 單形法

```python
class SimplexMethod:
    """單形法求解線性規劃"""
    
    def __init__(self, c, A, b):
        """
        Maximize c^T x
        subject to Ax <= b, x >= 0
        """
        self.c = c
        self.A = A
        self.b = b
        self.m = len(b)  # 約束數
        self.n = len(c)  # 變數數
        
        # 鬆弛形式
        self.N = list(range(self.n))  # 非基變數索引
        self.B = list(range(self.n, self.n + self.m))  # 基變數索引
        self.v = 0  # 目標函數常數項
    
    def pivot(self, pivot_row, pivot_col):
        """旋轉操作"""
        # 計算新的 pivot 行
        self.A[pivot_row][pivot_col] = 1.0 / self.A[pivot_row][pivot_col]
        
        for j in self.N:
            if j != pivot_col:
                self.A[pivot_row][j] *= self.A[pivot_row][pivot_col]
        
        self.b[pivot_row] *= self.A[pivot_row][pivot_col]
        
        # 更新其他行
        for i in self.B:
            if i != pivot_row:
                factor = self.A[i][pivot_col]
                self.A[i][pivot_col] = 0
                for j in self.N:
                    if j != pivot_col:
                        self.A[i][j] -= factor * self.A[pivot_row][j]
                self.b[i] -= factor * self.b[pivot_row]
        
        # 更新目標函數
        factor = self.c[pivot_col]
        self.c[pivot_col] = 0
        for j in self.N:
            if j != pivot_col:
                self.c[j] -= factor * self.A[pivot_row][j]
        self.v += factor * self.b[pivot_row]
        
        # 更新基/非基
        self.B[pivot_row], self.N[pivot_col] = self.N[pivot_col], self.B[pivot_row]
    
    def solve(self, max_iterations=1000):
        """
        單形法求解
        返回 (optimal_value, solution) 或 (None, None) 如果無界
        """
        for iteration in range(max_iterations):
            # 檢查是否為最優
            pivot_col = None
            for j in self.N:
                if self.c[j] > 1e-10:
                    pivot_col = j
                    break
            
            if pivot_col is None:
                # 最優解
                x = [0.0] * (self.n + self.m)
                for i in range(self.m):
                    x[self.B[i]] = self.b[i]
                return self.v, x[:self.n]
            
            # 選擇最負約束的列（最小比例規則）
            pivot_row = None
            min_ratio = float('inf')
            
            for i in range(self.m):
                if self.A[i][pivot_col] > 1e-10:
                    ratio = self.b[i] / self.A[i][pivot_col]
                    if ratio < min_ratio:
                        min_ratio = ratio
                        pivot_row = i
            
            if pivot_row is None:
                # 無界
                return float('inf'), None
            
            self.pivot(pivot_row, pivot_col)
        
        return None, None  # 迭代超限

def simplex_tableau(c, A, b):
    """
    表格形式的單形法
    """
    m, n = len(b), len(c)
    
    # 構造表格
    #        |  -c  |  0  |
    # table = |  -A  |  b  |
    
    table = [[0.0] * (n + m + 1) for _ in range(m + 1)]
    
    # 目標函數行
    for j in range(n):
        table[0][j] = -c[j]
    
    # 約束行
    for i in range(m):
        for j in range(n):
            table[i + 1][j] = -A[i][j]
        table[i + 1][n + m] = b[i]
    
    # 基變數
    basis = list(range(n, n + m))
    
    iteration = 0
    max_iterations = 1000
    
    while iteration < max_iterations:
        # 找 pivot 列（最負檢驗數）
        pivot_col = -1
        min_test = 0
        for j in range(n + m):
            if table[0][j] < min_test:
                min_test = table[0][j]
                pivot_col = j
        
        if pivot_col == -1:
            # 最優
            break
        
        # 找 pivot 行（最小比例）
        pivot_row = -1
        min_ratio = float('inf')
        for i in range(1, m + 1):
            if table[i][pivot_col] < -1e-10:
                ratio = table[i][n + m] / table[i][pivot_col]
                if ratio < min_ratio:
                    min_ratio = ratio
                    pivot_row = i
        
        if pivot_row == -1:
            # 無界
            return None, None
        
        # 旋轉
        basis[pivot_row - 1] = pivot_col
        
        pivot_val = table[pivot_row][pivot_col]
        for j in range(n + m + 1):
            table[pivot_row][j] /= pivot_val
        
        for i in range(m + 1):
            if i != pivot_row:
                factor = table[i][pivot_col]
                for j in range(n + m + 1):
                    table[i][j] -= factor * table[pivot_row][j]
        
        iteration += 1
    
    # 提取解
    x = [0.0] * n
    for i in range(m):
        if basis[i] < n:
            x[basis[i]] = table[i + 1][n + m]
    
    optimal_value = -table[0][n + m]
    
    return optimal_value, x
```

## 運輸問題

```python
def transportation_simplex(supply, demand, costs):
    """
    運輸問題的單形法
    最小化 ΣΣ c[i][j] * x[i][j]
    """
    m, n = len(supply), len(demand)
    
    # Northwest Corner Method - 初始基
    def northwest_corner(supply, demand):
        i = j = 0
        allocation = [[0] * n for _ in range(m)]
        supply = supply.copy()
        demand = demand.copy()
        
        while i < m and j < n:
            x = min(supply[i], demand[j])
            allocation[i][j] = x
            supply[i] -= x
            demand[j] -= x
            
            if supply[i] == 0:
                i += 1
            if demand[j] == 0:
                j += 1
        
        return allocation
    
    # Vogel's Approximation Method - 較好的初始解
    def vogel_approximation(supply, demand, costs):
        m, n = len(supply), len(demand)
        allocation = [[0] * n for _ in range(m)]
        supply = supply.copy()
        demand = demand.copy()
        
        while sum(supply) > 0 and sum(demand) > 0:
            # 計算每行每列的罰罰
            row_penalty = []
            for i in range(m):
                if supply[i] > 0:
                    row = [costs[i][j] for j in range(n) if demand[j] > 0]
                    if len(row) >= 2:
                        row.sort()
                        row_penalty.append(row[1] - row[0])
                    else:
                        row_penalty.append(0)
                else:
                    row_penalty.append(-1)
            
            col_penalty = []
            for j in range(n):
                if demand[j] > 0:
                    col = [costs[i][j] for i in range(m) if supply[i] > 0]
                    if len(col) >= 2:
                        col.sort()
                        col_penalty.append(col[1] - col[0])
                    else:
                        col_penalty.append(0)
                else:
                    col_penalty.append(-1)
            
            # 選擇最大罰罰
            max_row_penalty = max(row_penalty)
            max_col_penalty = max(col_penalty)
            
            if max_row_penalty >= max_col_penalty:
                i = row_penalty.index(max_row_penalty)
                j = min((j for j in range(n) if demand[j] > 0),
                       key=lambda x: costs[i][x])
            else:
                j = col_penalty.index(max_col_penalty)
                i = min((i for i in range(m) if supply[i] > 0),
                       key=lambda x: costs[x][j])
            
            x = min(supply[i], demand[j])
            allocation[i][j] = x
            supply[i] -= x
            demand[j] -= x
        
        return allocation
    
    def potential_method(allocation, costs):
        """位勢法檢驗最優性"""
        m, n = len(allocation), len(allocation[0])
        
        u = [0] * m
        v = [float('inf')] * n
        
        # u[0] = 0
        for _ in range(m + n):
            for i in range(m):
                for j in range(n):
                    if allocation[i][j] > 0:
                        if u[i] != float('inf') and v[j] == float('inf'):
                            v[j] = costs[i][j] - u[i]
                        elif u[i] == float('inf') and v[j] != float('inf'):
                            u[i] = costs[i][j] - v[j]
        
        # 計算檢驗數
        for i in range(m):
            for j in range(n):
                if allocation[i][j] == 0:
                    cij = costs[i][j]
                    ui_vj = u[i] + v[j] if v[j] != float('inf') else float('inf')
                    if cij < ui_vj:
                        return False, i, j
        
        return True, -1, -1
    
    # 初始化
    allocation = vogel_approximation(supply, demand, costs)
    
    # 迭代優化
    optimal = False
    while not optimal:
        is_optimal, entering_i, entering_j = potential_method(allocation, costs)
        if is_optimal:
            break
        
        # 找閉環並調整
    
    # 計算總成本
    total_cost = sum(
        allocation[i][j] * costs[i][j]
        for i in range(m) for j in range(n)
    )
    
    return allocation, total_cost
```

## 整數線性規劃

```python
def branch_and_bound(c, A, b):
    """
    分支定界法求解整數線性規劃
    max c^T x
    s.t. Ax <= b
         x >= 0, x integer
    """
    best_value = float('-inf')
    best_solution = None
    
    def solve_lp_relaxation(subproblem):
        """求解 LP 鬆弛"""
        value, solution = simplex_tableau(c, A, b)
        return value, solution
    
    def branch(subproblems, lower_bound, upper_bound):
        """分支"""
        nonlocal best_value, best_solution
        
        if not subproblems:
            return
        
        # 選擇分支
        subproblem = subproblems.pop(0)
        value, solution = solve_lp_relaxation(subproblem)
        
        if value <= lower_bound:
            return  # 剪枝
        
        # 找非整數變數
        fractional_var = None
        for j, xj in enumerate(solution):
            if abs(xj - round(xj)) > 1e-6:
                fractional_var = j
                break
        
        if fractional_var is None:
            # 整數解
            if value > best_value:
                best_value = value
                best_solution = [round(x) for x in solution]
            return
        
        # 分支
        left = subproblem.copy()
        right = subproblem.copy()
        
        # x_j <= floor(x_j)
        left.append((fractional_var, '<=', int(solution[fractional_var])))
        # x_j >= ceil(x_j)
        right.append((fractional_var, '>=', int(solution[fractional_var]) + 1))
        
        branch([left] + subproblems, best_value, upper_bound)
        branch([right] + subproblems, best_value, upper_bound)
    
    branch([{}], best_value, float('inf'))
    
    return best_value, best_solution
```

## 總結

線性規劃是優化理論的基礎。單形法是最经典的求解方法，雖然最壞情況是指數時間，但在實踐中非常高效。對偶理論揭示了原始問題和對偶問題之間的深刻聯繫。對於整數線性規劃，分支定界法是標準的精確求解方法。

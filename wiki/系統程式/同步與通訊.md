# 同步與通訊

同步與通訊是作業系統中最核心的議題之一，決定了多行程或多執行緒環境下如何安全、高效地協作。在現代計算環境中，幾乎所有重要的應用程式都需要處理並發問題，從多核心處理器的程式設計到分散式系統的開發，同步與通訊無處不在。理解這些概念對於開發穩定、高效的軟體系統至關重要。

## 為什麼需要同步？

```
┌─────────────────────────────────────────────────────────────┐
│                  並發帶來的問題                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   競爭條件 (Race Condition)                                 │
│   ┌──────────────────────────────────────────────────────┐ │
│   │ 行程 A: read counter → 10                            │ │
│   │ 行程 B: read counter → 10  (A 還沒寫回)              │ │
│   │ 行程 A: write counter → 11                            │ │
│   │ 行程 B: write counter → 11  (覆蓋 A 的結果)          │ │
│   │ 結果: counter = 11, 應該是 12                         │ │
│   └──────────────────────────────────────────────────────┘ │
│                                                             │
│   死結 (Deadlock)                                           │
│   ┌──────────────────────────────────────────────────────┐ │
│   │ 行程 A: 擁有資源 1, 等待資源 2                        │ │
│   │ 行程 B: 擁有資源 2, 等待資源 1                        │ │
│   │ 結果: 兩個行程永遠等待下去                             │ │
│   └──────────────────────────────────────────────────────┘ │
│                                                             │
│   優先順序反轉 (Priority Inversion)                         │
│   ┌──────────────────────────────────────────────────────┐ │
│   │ 高優先權行程等待低優先權行程持有的資源                 │ │
│   │ 中優先權行程搶佔低優先權行程                          │ │
│   │ 結果: 高優先權行程被阻塞                              │ │
│   └──────────────────────────────────────────────────────┘ │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 臨界區（Critical Section）

臨界區是指存取共享資源的程式碼區域，在任意時刻只能有一個執行緒在執行：

```
┌─────────────────────────────────────────────────────────────┐
│                    臨界區問題                               │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   臨界區的三要素：                                          │
│   1. 互斥存取：一次只有一個執行緒能進入                   │
│   2. 進展：如果沒有執行緒在臨界區，應允許進入              │
│   3. 有限等待：等待時間必須是有限的                       │
│                                                             │
│   進入臨界區的協議：                                       │
│   ┌─────────────┐                                          │
│   │  進入區     │  請求進入許可                           │
│   │─────────────│                                         │
│   │  臨界區     │  獨佔存取共享資源                       │
│   │─────────────│                                         │
│   │  離開區     │  釋放許可                               │
│   │─────────────│                                         │
│   │  剩餘區     │  其他程式碼                              │
│   └─────────────┘                                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 同步機制詳解

### 1. 互斥鎖（Mutex）

```c
#include <pthread.h>
#include <stdio.h>

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
int shared_counter = 0;

void* increment(void* arg) {
    for (int i = 0; i < 1000000; i++) {
        pthread_mutex_lock(&mutex);
        shared_counter++;
        pthread_mutex_unlock(&mutex);
    }
    return NULL;
}

int main() {
    pthread_t t1, t2;
    pthread_create(&t1, NULL, increment, NULL);
    pthread_create(&t2, NULL, increment, NULL);
    
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    
    printf("Final counter: %d\n", shared_counter);
    return 0;
}
```

```python
# Python threading 互斥鎖
import threading

counter = 0
lock = threading.Lock()

def increment():
    global counter
    for _ in range(1000000):
        with lock:
            counter += 1

t1 = threading.Thread(target=increment)
t2 = threading.Thread(target=increment)

t1.start()
t2.start()
t1.join()
t2.join()

print(f"Final counter: {counter}")
```

### 2. 讀寫鎖（Read-Write Lock）

```c
typedef struct {
    pthread_mutex_t mutex;
    pthread_cond_t readers_done;
    int readers;
    int writers_waiting;
    int writer_active;
} RWLock;

void rwlock_init(RWLock* rw) {
    pthread_mutex_init(&rw->mutex, NULL);
    pthread_cond_init(&rw->readers_done, NULL);
    rw->readers = 0;
    rw->writers_waiting = 0;
    rw->writer_active = 0;
}

void read_lock(RWLock* rw) {
    pthread_mutex_lock(&rw->mutex);
    while (rw->writer_active || rw->writers_waiting > 0) {
        pthread_cond_wait(&rw->readers_done, &rw->mutex);
    }
    rw->readers++;
    pthread_mutex_unlock(&rw->mutex);
}

void read_unlock(RWLock* rw) {
    pthread_mutex_lock(&rw->mutex);
    rw->readers--;
    if (rw->readers == 0) {
        pthread_cond_signal(&rw->readers_done);
    }
    pthread_mutex_unlock(&rw->mutex);
}

void write_lock(RWLock* rw) {
    pthread_mutex_lock(&rw->mutex);
    rw->writers_waiting++;
    while (rw->readers > 0 || rw->writer_active) {
        pthread_cond_wait(&rw->readers_done, &rw->mutex);
    }
    rw->writers_waiting--;
    rw->writer_active = 1;
    pthread_mutex_unlock(&rw->mutex);
}

void write_unlock(RWLock* rw) {
    pthread_mutex_lock(&rw->mutex);
    rw->writer_active = 0;
    pthread_cond_broadcast(&rw->readers_done);
    pthread_mutex_unlock(&rw->mutex);
}
```

### 3. 訊號量（Semaphore）

```c
#include <semaphore.h>
#include <pthread.h>
#include <stdio.h>

sem_t semaphore;  // 允許最多 3 個執行緒同時訪問
int shared_resource = 0;

void* access_resource(void* arg) {
    int thread_id = *(int*)arg;
    
    printf("Thread %d waiting for resource\n", thread_id);
    sem_wait(&semaphore);  // P 操作
    
    printf("Thread %d accessing resource\n", thread_id);
    shared_resource++;
    printf("Thread %d: resource value = %d\n", thread_id, shared_resource);
    
    sem_post(&semaphore);  // V 操作
    printf("Thread %d released resource\n", thread_id);
    
    return NULL;
}

int main() {
    sem_init(&semaphore, 0, 3);  // 初始值 3
    
    pthread_t threads[5];
    int ids[5] = {1, 2, 3, 4, 5};
    
    for (int i = 0; i < 5; i++) {
        pthread_create(&threads[i], NULL, access_resource, &ids[i]);
    }
    
    for (int i = 0; i < 5; i++) {
        pthread_join(threads[i], NULL);
    }
    
    sem_destroy(&semaphore);
    return 0;
}
```

### 4. 條件變數（Condition Variable）

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
int ready = 0;
int processed = 0;

void* producer(void* arg) {
    pthread_mutex_lock(&mutex);
    
    // 生產資料
    ready = 1;
    printf("Producer: data ready, signaling consumer\n");
    
    // 通知消費者
    pthread_cond_signal(&cond);
    
    // 等待消費者處理完成
    while (processed == 0) {
        pthread_cond_wait(&cond, &mutex);
    }
    
    printf("Producer: consumer confirmed, continue producing\n");
    processed = 0;
    
    pthread_mutex_unlock(&mutex);
    return NULL;
}

void* consumer(void* arg) {
    pthread_mutex_lock(&mutex);
    
    // 等待資料
    while (ready == 0) {
        pthread_cond_wait(&cond, &mutex);
    }
    
    // 消費資料
    printf("Consumer: processing data\n");
    processed = 1;
    
    // 通知生產者
    pthread_cond_signal(&cond);
    
    pthread_mutex_unlock(&mutex);
    return NULL;
}

int main() {
    pthread_t prod, cons;
    pthread_create(&cons, NULL, consumer, NULL);
    sleep(1);
    pthread_create(&prod, NULL, producer, NULL);
    
    pthread_join(prod, NULL);
    pthread_join(cons, NULL);
    
    return 0;
}
```

### 5. 屏障（Barrier）

```c
#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

#define NUM_THREADS 5
pthread_barrier_t barrier;

void* worker(void* arg) {
    int id = *(int*)arg;
    
    printf("Thread %d: Phase 1\n", id);
    sleep(id + 1);  // 模擬工作
    
    printf("Thread %d: Waiting at barrier\n", id);
    pthread_barrier_wait(&barrier);
    
    printf("Thread %d: Phase 2\n", id);
    return NULL;
}

int main() {
    pthread_t threads[NUM_THREADS];
    int ids[NUM_THREADS];
    
    pthread_barrier_init(&barrier, NULL, NUM_THREADS);
    
    for (int i = 0; i < NUM_THREADS; i++) {
        ids[i] = i;
        pthread_create(&threads[i], NULL, worker, &ids[i]);
    }
    
    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }
    
    pthread_barrier_destroy(&barrier);
    return 0;
}
```

## 程序間通訊（IPC）

### 1. 管道（Pipe）

```c
#include <unistd.h>
#include <stdio.h>
#include <string.h>

int main() {
    int pipefd[2];
    pipe(pipefd);  // 建立管道
    
    pid_t pid = fork();
    
    if (pid == 0) {  // 子行程
        close(pipefd[0]);  // 關閉讀端
        const char* msg = "Hello from child!";
        write(pipefd[1], msg, strlen(msg));
        close(pipefd[1]);
    } else {  // 父行程
        close(pipefd[1]);  // 關閉寫端
        char buf[100];
        read(pipefd[0], buf, sizeof(buf));
        printf("Parent received: %s\n", buf);
        close(pipefd[0]);
    }
    
    return 0;
}
```

```python
# Python 管道
import multiprocessing

def producer(pipe):
```

```c
#include <string.h>

struct msgbuf {
    long mtype;
    char mtext[100];
};

int main() {
    key_t key = ftok("/tmp", 'A');
    int msgid = msgget(key, 0666 | IPC_CREAT);

    struct msgbuf msg;
    msg.mtype = 1;
    strcpy(msg.mtext, "Hello, Message Queue!");
    msgsnd(msgid, &msg, sizeof(msg.mtext), 0);
    
    // 接收
    msgrcv(msgid, &msg, sizeof(msg.mtext), 1, 0);
    printf("Received: %s\n", msg.mtext);
    
    // 清理
    msgctl(msgid, IPC_RMID, NULL);
    return 0;
}
```

### 死結預防

```c
// 方法1: 固定順序獲取鎖
pthread_mutex_t lock1 = PTHREAD_MUTEX_INITIALIZER;
pthread_mutex_t lock2 = PTHREAD_MUTEX_INITIALIZER;

void* thread1_code(void* arg) {
    pthread_mutex_lock(&lock1);
    sleep(1);
    pthread_mutex_lock(&lock2);  // 總是先 lock1 後 lock2
    pthread_mutex_unlock(&lock2);
    pthread_mutex_unlock(&lock1);
    return NULL;
}

void* thread2_code(void* arg) {
    pthread_mutex_lock(&lock1);  // 與 thread1 相同的順序
    sleep(1);
    pthread_mutex_lock(&lock2);
    pthread_mutex_unlock(&lock2);
    pthread_mutex_unlock(&lock1);
    return NULL;
}

// 方法2: 使用 trylock
void* thread_try(void* arg) {
    while (1) {
        pthread_mutex_lock(&lock1);
        if (pthread_mutex_trylock(&lock2) == 0) {
            // 成功獲取兩個鎖
            pthread_mutex_unlock(&lock2);
            break;
        }
        pthread_mutex_unlock(&lock1);  // 釋放並重試
        sleep(1);
    }
    return NULL;
}
```

### 死結檢測

```python
# 死結檢測示例
def detect_deadlock(processes, resources, wait_graph):
    """
    使用圖論方法檢測死結
    processes: 行程列表
    resources: 資源列表
    wait_graph: 等待圖 {process: [needed_resources]}
    """
    in_degree = {p: 0 for p in processes}
    
    for process, needed in wait_graph.items():
        in_degree[process] = len(needed)
    
    # 拓撲排序
    queue = [p for p in processes if in_degree[p] == 0]
    processed = []
    
    while queue:
        process = queue.pop(0)
        processed.append(process)
        
        # 模擬釋放資源
        # (實際需要更複雜的模擬)
    
    if len(processed) < len(processes):
        print("WARNING: Deadlock detected!")
        return False
    return True

# 使用銀行家演算法避免死結
class Banker:
    def __init__(self, max_need, allocated, available):
        self.max_need = max_need       # 每個行程最大需求
        self.allocated = allocated   # 當前分配
        self.available = available   # 可用資源
    
    def is_safe(self):
        work = self.available[:]
        finish = [False] * len(self.allocated)
        
        while True:
            found = False
            for i in range(len(self.allocated)):
                if not finish[i]:
                    # 檢查是否可以滿足需求
                    can_allocate = all(
                        work[j] >= self.max_need[i][j] - self.allocated[i][j]
                        for j in range(len(work))
                    )
                    if can_allocate:
                        # 模擬分配
                        for j in range(len(work)):
                            work[j] += self.allocated[i][j]
                        finish[i] = True
                        found = True
            
            if not found:
                break
        
        return all(finish)
```

## 實現同步的最佳實踐

```python
# 同步設計模式
best_practices = {
    "避免嵌套鎖": "不要在持有一個鎖時請求另一個鎖",
    
    "固定鎖順序": "如果必須多個鎖，確保所有執行緒以相同順序獲取",

    "避免忙等": "使用睡眠而非自旋",
    
    "優先順序繼承": "使用 POSIX priority inheritance"

# 現代 C++ 多執行緒安全設計
import threading

class ThreadSafeCounter:
    def __init__(self):
        self._value = 0
        self._lock = threading.Lock()
    
    def increment(self):
        with self._lock:
            self._value += 1
            return self._value
    
    def get(self):
        with self._lock:
            return self._value
    
    def __enter__(self):
        self._lock.acquire()
        return self
    
    def __exit__(self, *args):
        self._lock.release()

# 使用
counter = ThreadSafeCounter()
with counter:
    current = counter.get()
    counter.increment()
```

# 記憶體管理

記憶體管理是作業系統最核心也是最複雜的功能之一。它的主要職責是在多個行程之間有效地分配和回收實體記憶體，並為每個行程提供虛幻的、獨立的位址空間。好的記憶體管理可以提高系統的記憶體利用率、保護行程之間的隱私、同時為程式設計師提供簡單的程式設計模型。本篇將深入探討記憶體管理的各個層面，從基本的記憶體配置到進階的虛擬記憶體技術。

## 連續記憶體配置

連續記憶體配置是最簡單的記憶體管理方式，它為每個行程分配一塊連續的記憶體區域。

### 配置策略

```c
// 記憶體配置策略的實現

// 1. 首次適應 (First Fit)
void* first_fit(size_t size) {
    Block* block = free_list;
    while (block != NULL) {
        if (block->size >= size) {
            // 找到足夠的空間
            if (block->size > size + BLOCK_MIN) {
                // 分割區塊
                split_block(block, size);
            }
            remove_from_free_list(block);
            return block->data;
        }
        block = block->next;
    }
    return NULL;  // 未找到
}

// 2. 最佳適應 (Best Fit)
void* best_fit(size_t size) {
    Block *best = NULL, *block = free_list;
    while (block != NULL) {
        if (block->size >= size) {
            if (best == NULL || block->size < best->size) {
                best = block;
            }
        }
        block = block->next;
    }
    // 分配最佳區塊
    return allocate_block(best, size);
}

// 3. 最差適應 (Worst Fit)
void* worst_fit(size_t size) {
    Block *worst = NULL, *block = free_list;
    while (block != NULL) {
        if (block->size >= size && 
            (worst == NULL || block->size > worst->size)) {
            worst = block;
        }
        block = block->next;
    }
    return allocate_block(worst, size);
}
```

## 分頁（Paging）

分頁是一種將記憶體劃分為固定大小區塊的記憶體管理技術，它有效地解決了外部碎片問題。

### 分頁原理

```
┌─────────────────────────────────────────────────────────────┐
│                    分頁系統                                  │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   虛擬位址格式：                                            │
│   ┌──────────────────┬────────────────┐                    │
│   │    頁號 (VPN)     │   頁偏移 (Offset)│                   │
│   │   Page Number     │   Page Offset  │                    │
│   └──────────────────┴────────────────┘                    │
│                                                             │
│   例如：32位元虛擬位址，4KB頁面：                            │
│   ┌───────────────┬────────────────┐                        │
│   │  20 bits      │   12 bits     │                        │
│   │  頁號         │   偏移        │                        │
│   └───────────────┴────────────────┘                        │
│                                                             │
│   頁表結構：                                                │
│   ┌─────┬────────┬────────┬────────┐                        │
│   │ VPN │ Valid  │ Frame  │  RWX  │                        │
│   ├─────┼────────┼────────┼────────┤                        │
│   │  0  │   1    │   3    │  R-X  │                        │
│   │  1  │   1    │   7    │  RW-  │                        │
│   │  2  │   0    │   -    │   -   │  (未映射)            │
│   │  3  │   1    │   1    │  R--  │                        │
│   └─────┴────────┴────────┴────────┘                        │
│                                                             │
│   地址轉換：                                                │
│   VPN = 0x1234 >> 12 = 0x123                               │
│   Offset = 0x1234 & 0xFFF = 0x234                          │
│   PFN = page_table[0x123].frame = 3                         │
│   Physical = (3 << 12) | 0x234                             │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 頁面置換演算法

```python
# 1. FIFO (First-In-First-Out)
def fifo_replace(pages, frames):
    queue = []
    page_faults = 0
    frame_contents = []
    
    for page in pages:
        if page not in frame_contents:
            page_faults += 1
            if len(frame_contents) < frames:
                frame_contents.append(page)
            else:
                # 移除最舊的頁面
                frame_contents.pop(0)
                frame_contents.append(page)
    
    return page_faults

# 2. LRU (Least Recently Used)
def lru_replace(pages, frames):
    page_faults = 0
    frame_contents = []
    last_used = {}  # 頁面最近使用時間
    
    for i, page in enumerate(pages):
        if page not in frame_contents:
            page_faults += 1
            if len(frame_contents) >= frames:
                # 找到最久未使用的頁面
                lru_page = min(frame_contents, key=lambda p: last_used.get(p, i))
                frame_contents.remove(lru_page)
            frame_contents.append(page)
        
        last_used[page] = i
    
    return page_faults

# 3. Optimal (OPT) - 未來最久不使用
def optimal_replace(pages, frames):
    page_faults = 0
    frame_contents = []
    
    for i, page in enumerate(pages):
        if page not in frame_contents:
            page_faults += 1
            if len(frame_contents) >= frames:
                # 找到未來最久不使用的頁面
                future_uses = {}
                for p in frame_contents:
                    if p in pages[i+1:]:
                        future_uses[p] = pages[i+1:].index(p)
                    else:
                        future_uses[p] = float('inf')
                
                remove_page = max(frame_contents, key=lambda p: future_uses[p])
                frame_contents.remove(remove_page)
            frame_contents.append(page)
    
    return page_faults

# 4. Clock (Second Chance)
def clock_replace(pages, frames):
    page_faults = 0
    frame_contents = []
    use_bits = {}  # 使用位
    clock_hand = 0
    
    for page in pages:
        if page not in frame_contents:
            page_faults += 1
            if len(frame_contents) >= frames:
                # 循環查找第一個使用位為 0 的頁面
                while use_bits.get(frame_contents[clock_hand], 1) == 1:
                    use_bits[frame_contents[clock_hand]] = 0
                    clock_hand = (clock_hand + 1) % frames
                frame_contents[clock_hand] = page
                clock_hand = (clock_hand + 1) % frames
            else:
                frame_contents.append(page)
        
        use_bits[page] = 1
    
    return page_faults
```

```c
// 1. 記憶體池
typedef struct MemPool {
    void* memory;
    size_t block_size;
    size_t pool_size;
    void* free_list;
} MemPool;

void* mempool_alloc(MemPool* pool) {
    if (pool->free_list == NULL) {
        // 配置新區塊
        void* new_block = malloc(pool->block_size);
        return new_block;
    }
    
    void* block = pool->free_list;
    pool->free_list = *(void**)block;
    return block;
}

void mempool_free(MemPool* pool, void* block) {
    *(void**)block = pool->free_list;
    pool->free_list = block;
}
```

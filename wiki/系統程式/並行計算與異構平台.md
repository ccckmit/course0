# 並行計算與異構平台

並行計算（Parallel Computing）是利用多個計算單元同時執行計算任務以提升效能的範式。隨著摩爾定律放緩，硬體廠商轉向多核、GPU、FPGA 等異構架構提升效能，這使得並行計算和異構平台成為現代系統程式的核心主題。

## 並行與平行的區別

- **並行（Concurrency）**：多個任務在重疊的時間段內執行，不一定同時
- **平行（Parallelism）**：多個任務真正同時執行

```
並行：        平行：
┌──┐ ┌──┐    ┌──┐ ┌──┐
│T1│ │T2│    │T1│ │T2│
└┬─┘ └──┘    └──┘ └──┘
 │  ┌──┐       ↑
 └─→│T2│    同時執行
    └──┘
```

## 硬體平行層級

### 指令級平行（ILP）

一個 CPU 時脈週期內可執行多條指令：
- **管線化（Pipeline）**：將指令執行分為多個階段
- **超純量（Superscalar）**：同時發射多條指令
- **亂序執行（Out-of-Order）**：動態調整指令順序

### 執行緒級平行（TLP）

多個 CPU 核心同時執行不同執行緒：
```
┌─────┬─────┬─────┬─────┐
│ CPU0│ CPU1│ CPU2│ CPU3│  ← 4 核心
└─────┴─────┴─────┴─────┘
  │     │     │     │
  ▼     ▼     ▼     ▼
Thread0 Thread1 Thread2 Thread3
```

### 資料級平行（DLP）

對大量資料執行相同操作：
- SIMD（Single Instruction Multiple Data）
- 向量化（Vectorization）

```
┌─────────────────────────────────┐
│  ADD V0, V1, V2                 │
│  (一次加法，處理 4 組資料)        │
├─────────────────────────────────┤
│ V0 = [a0,a1,a2,a3]             │
│ V1 = [b0,b1,b2,b3]             │
│ V2 = V0 + V1 = [a0+b0, ...]    │
└─────────────────────────────────┘
```

## 多執行緒程式設計

### POSIX Threads

```c
#include <pthread.h>

typedef struct {
    int *array;
    int start;
    int end;
    long long result;
} ThreadData;

void *sum_array(void *arg) {
    ThreadData *data = (ThreadData *)arg;
    long long sum = 0;
    for (int i = data->start; i < data->end; i++) {
        sum += data->array[i];
    }
    data->result = sum;
    return NULL;
}

int parallel_sum(int *array, int size, int num_threads) {
    pthread_t threads[64];
    ThreadData data[64];
    int chunk_size = size / num_threads;
    
    // 建立執行緒
    for (int i = 0; i < num_threads; i++) {
        data[i] = (ThreadData){
            .array = array,
            .start = i * chunk_size,
            .end = (i == num_threads - 1) ? size : (i + 1) * chunk_size
        };
        pthread_create(&threads[i], NULL, sum_array, &data[i]);
    }
    
    // 合併結果
    long long total = 0;
    for (int i = 0; i < num_threads; i++) {
        pthread_join(threads[i], NULL);
        total += data[i].result;
    }
    
    return total;
}
```

### C++ std::thread

```cpp
#include <thread>
#include <vector>
#include <numeric>

int parallel_sum(const std::vector<int>& data, int num_threads) {
    int chunk_size = data.size() / num_threads;
    std::vector<std::thread> threads;
    std::vector<long long> results(num_threads);
    
    for (int i = 0; i < num_threads; i++) {
        int start = i * chunk_size;
        int end = (i == num_threads - 1) ? data.size() : (i + 1) * chunk_size;
        
        threads.emplace_back([&data, start, end, &results, i]() {
            results[i] = std::accumulate(data.begin() + start, 
                                          data.begin() + end, 0LL);
        });
    }
    
    for (auto& t : threads) t.join();
    
    return std::accumulate(results.begin(), results.end(), 0LL);
}
```

### Go Goroutine

```go
package main

import "fmt"
import "sync"

func sumSegment(data []int, result chan<- int, wg *sync.WaitGroup) {
    defer wg.Done()
    sum := 0
    for _, v := range data {
        sum += v
    }
    result <- sum
}

func parallelSum(data []int, numThreads int) int {
    chunkSize := len(data) / numThreads
    result := make(chan int, numThreads)
    var wg sync.WaitGroup
    
    for i := 0; i < numThreads; i++ {
        wg.Add(1)
        start := i * chunkSize
        end := start + chunkSize
        if i == numThreads-1 {
            end = len(data)
        }
        go sumSegment(data[start:end], result, &wg)
    }
    
    go func() {
        wg.Wait()
        close(result)
    }()
    
    total := 0
    for sum := range result {
        total += sum
    }
    return total
}
```

## 執行緒同步

### 互斥鎖（Mutex）

```c
#include <pthread.h>

typedef struct {
    int counter;
    pthread_mutex_t lock;
} Counter;

void counter_init(Counter *c) {
    c->counter = 0;
    pthread_mutex_init(&c->lock, NULL);
}

void counter_increment(Counter *c) {
    pthread_mutex_lock(&c->lock);
    c->counter++;
    pthread_mutex_unlock(&c->lock);
}

int counter_get(Counter *c) {
    pthread_mutex_lock(&c->lock);
    int val = c->counter;
    pthread_mutex_unlock(&c->lock);
    return val;
}
```

### 無鎖程式設計（Lock-Free）

使用原子操作避免鎖的开销：

```cpp
#include <atomic>

class LockFreeStack {
    struct Node {
        int value;
        std::atomic<Node*> next;
    };
    
    std::atomic<Node*> head;
    
public:
    void push(int val) {
        Node* new_node = new Node{val, head.load()};
        while (!head.compare_exchange_weak(new_node->next, new_node)) {
            // 重試
        }
    }
    
    int pop() {
        Node* old_head = head.load();
        while (old_head && 
               !head.compare_exchange_weak(old_head, old_head->next.load())) {
            // 重試
        }
        return old_head ? old_head->value : -1;
    }
};
```

### CUDA 程式設計

```cuda
#include <cuda_runtime.h>

__global__ void vectorAdd(float *a, float *b, float *c, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) {
        c[i] = a[i] + b[i];
    }
}

int main() {
    const int N = 1 << 20;  // 100 萬元素
    const int bytes = N * sizeof(float);
    
    // 主機記憶體
    float *h_a = (float*)malloc(bytes);
    float *h_b = (float*)malloc(bytes);
    float *h_c = (float*)malloc(bytes);
    
    // 設備記憶體
    float *d_a, *d_b, *d_c;
    cudaMalloc(&d_a, bytes);
    cudaMalloc(&d_b, bytes);
    cudaMalloc(&d_c, bytes);
    
    // 複製到設備
    cudaMemcpy(d_a, h_a, bytes, cudaMemcpyHostToDevice);
    cudaMemcpy(d_b, h_b, bytes, cudaMemcpyHostToDevice);
    
    // 啟動核心
    int blockSize = 256;
    int numBlocks = (N + blockSize - 1) / blockSize;
    vectorAdd<<<numBlocks, blockSize>>>(d_a, d_b, d_c, N);
    
    // 複製回主機
    cudaMemcpy(h_c, d_c, bytes, cudaMemcpyDeviceToHost);
    
    // 清理
    cudaFree(d_a); cudaFree(d_b); cudaFree(d_c);
    free(h_a); free(h_b); free(h_c);
    
    return 0;
}
```

### OpenMP

OpenMP 是 CPU 多執行緒的 директорская指示編譯：

```c
#include <omp.h>

void parallel_matrix_multiply(double *A, double *B, double *C, int N) {
    #pragma omp parallel for schedule(dynamic)
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            double sum = 0.0;
            for (int k = 0; k < N; k++) {
                sum += A[i * N + k] * B[k * N + j];
            }
            C[i * N + j] = sum;
        }
    }
}

// 編譯：gcc -fopenmp -O3 matrix.c -o matrix
// 執行：OMP_NUM_THREADS=8 ./matrix
```

### FPGA

現場可程式化閘陣列（FPGA）提供硬體級的平行度：

```verilog
// 簡化 FPGA 流水線範例
module stream_processing #(
    parameter DATA_WIDTH = 32
)(
    input clk,
    input rst_n,
    input [DATA_WIDTH-1:0] data_in,
    input valid_in,
    output reg [DATA_WIDTH-1:0] data_out,
    output reg valid_out
);

    reg [DATA_WIDTH-1:0] pipe0, pipe1, pipe2;
    
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            pipe0 <= 0;
            pipe1 <= 0;
            pipe2 <= 0;
            valid_out <= 0;
        end else begin
            // 三級流水線
            pipe0 <= data_in;
            pipe1 <= pipe0 * 2;  // 乘法
            pipe2 <= pipe1 + 16'h3F800000;  // 加法
            data_out <= pipe2;
            valid_out <= valid_in;
        end
    end
endmodule
```

OpenCL 提供跨平台的異構計算框架：

// OpenCL 核心
const char *kernel_source =

## 效能優化考量

### Amdahl 定律

程式中可平行化的部分決定了最大加速比：

```
最大加速比 = 1 / (S + P/N)

其中：
- S = 程式串列部分比例
- P = 程式平行部分比例
- N = 執行緒/核心數量
```

```
加速比
  8x │        ╭─────────────
     │       ╱
  4x │     ╱
     │   ╱  ← 理論極限 = 4x (當 S=25%)
  2x │ ╱
     │╱
    0──────────────────────
      1    2    4    8   核心數
```

### 通訊开销

- 共享記憶體：需要同步
- 分散式記憶體：訊息傳遞（MPI）
- 記憶體頻寬：GPU 需要高頻寬記憶體（HBM）

### 負載均衡

確保各執行緒工作量均衡：

```cpp
// 動態排程
#pragma omp parallel for schedule(dynamic, 1024)
for (size_t i = 0; i < tasks.size(); i++) {
    process(tasks[i]);  // 任務大小可能不均
}
```

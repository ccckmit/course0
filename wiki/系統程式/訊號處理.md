# 訊號處理

訊號（Signal）是 Unix 系統中用於行程間非同步通訊的機制。當特定事件發生時，核心向行程發送訊號，行程可以選擇忽略、捕捉並處理，或讓系統提供預設行為。

## 基本概念

### 訊號的本質

訊號是一個整數，核心將其傳遞給目標行程：

| 訊號 | 值 | 說明 |
|------|-----|------|
| SIGHUP | 1 | 終端連線掛斷 |
| SIGINT | 2 | 中斷（Ctrl+C） |
| SIGQUIT | 3 | 退出（Ctrl+\） |
| SIGKILL | 9 | 強制終止 |
| SIGSEGV | 11 | 段錯誤 |
| SIGTERM | 15 | 終止請求 |
| SIGCHLD | 17 | 子程序結束 |
| SIGALRM | 14 | 計時器 |
| SIGUSR1 | 10 | 自訂訊號 1 |
| SIGUSR2 | 12 | 自訂訊號 2 |

### 訊號產生時機

- **硬體例外**：除零、段錯誤、無效記憶體存取
- **終端事件**：Ctrl+C、脫離連線
- **軟體條件**：計時器、I/O 就緒、計數已達
- **明示發送**：另一行程呼叫 kill()

## 基本 API

### signal 函數

```c
#include <signal.h>

// 設定訊號處理函數
typedef void (*sighandler_t)(int);

sighandler_t signal(int signum, sighandler_t handler);

// 預設處理
signal(SIGINT, SIG_DFL);

// 忽略訊號
signal(SIGINT, SIG_IGN);

// 自訂處理
void handler(int sig) {
    printf("Caught signal %d\n", sig);
}
signal(SIGINT, handler);
```

### 完整範例

```c
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>

volatile sig_atomic_t running = 1;

void sigint_handler(int sig) {
    // volatile sig_atomic_t 確保原子存取
    running = 0;
}

int main() {
    struct sigaction sa;
    sa.sa_handler = sigint_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = 0;  // 或 SA_RESTART
    
    if (sigaction(SIGINT, &sa, NULL) == -1) {
        perror("sigaction");
        exit(1);
    }
    
    printf("Running... Press Ctrl+C to stop\n");
    while (running) {
        // 主迴圈工作
    }
    
    printf("Clean exit\n");
    return 0;
}
```

## sigaction 結構

```c
struct sigaction {
    void     (*sa_handler)(int);       // 處理函式指標
    void     (*sa_sigaction)(int, siginfo_t *, void *);  // 擴展處理
    sigset_t   sa_mask;                // 阻塞的訊號集合
    int        sa_flags;               // 選項
    void     (*sa_restorer)(void);    // 內部使用
};
```

### sa_flags 選項

| 旗標 | 說明 |
|------|------|
| SA_RESTART | 自動重啟中斷的系統呼叫 |
| SA_NODEFER | 處理時不自動阻塞該訊號 |
| SA_ONSTACK | 使用訊號堆疊 |
| SA_RESETHAND | 處理後恢復為預設行為 |
| SA_SIGINFO | 使用 sa_sigaction |

### 取得訊號資訊

```c
void handler(int sig, siginfo_t *info, void *context) {
    printf("Signal %d from PID %d, UID %d\n",
           sig, info->si_pid, info->si_uid);
    
    // si_code 說明訊號來源
    switch (info->si_code) {
        case SI_USER:      // kill() 或 raise()
            printf("Sent by user process\n");
            break;
        case SI_KERNEL:    // 核心發送
            printf("Sent by kernel\n");
            break;
        case SI_TIMER:     // 計時器到期
            printf("Timer expired\n");
            break;
    }
}

struct sigaction sa;
sa.sa_sigaction = handler;
sa.sa_flags = SA_SIGINFO;  // 重要！
sigemptyset(&sa.sa_mask);
sigaction(SIGUSR1, &sa, NULL);
```

## 訊號集合

### 初始化和操作

```c
#include <signal.h>

sigset_t set;

// 清空集合
sigemptyset(&set);

// 填滿集合
sigfillset(&set);

// 新增訊號
sigaddset(&set, SIGINT);
sigaddset(&set, SIGTERM);

// 刪除訊號
sigdelset(&set, SIGINT);

// 測試成員
if (sigismember(&set, SIGINT)) {
    printf("SIGINT is in set\n");
}
```

### 訊號屏蔽

```c
#include <signal.h>

// 阻塞訊號
sigset_t mask, oldmask;
sigemptyset(&mask);
sigaddset(&mask, SIGINT);

sigprocmask(SIG_BLOCK, &mask, &oldmask);

// 關鍵區段 - SIGINT 不會中斷
// ...

// 恢復原來的屏蔽
sigprocmask(SIG_SETMASK, &oldmask, NULL);
```

### 暫停和等待

```c
// 等待訊號（標準化方式）
sigset_t mask, oldmask, pending;
sigemptyset(&mask);
sigaddset(&mask, SIGTERM);

sigprocmask(SIG_BLOCK, &mask, &oldmask);  // 先阻塞

// 等待 SIGTERM
sigsuspend(&mask);

// 或者使用 pause
pause();  // 阻塞直到收到任何未忽略的訊號
```

## 常見使用模式

### 超時計時

```c
void timeout_handler(int sig) {
    // 計時器到期處理
}

alarm(5);  // 5 秒後產生 SIGALRM

struct sigaction sa;
sa.sa_handler = timeout_handler;
sigemptyset(&sa.sa_mask);
sigaction(SIGALRM, &sa, NULL);

// 執行長時間操作
// 如果超時，會中斷

// 取消計時器
alarm(0);
```

### 可中断的系統呼叫

```c
// 預設行為：系統呼叫被訊號中斷時返回 -1，errno 設為 EINTR
char buffer[1024];
ssize_t n;
while ((n = read(fd, buffer, sizeof(buffer))) < 0) {
    if (errno == EINTR) {
        continue;  // 重試
    }
    perror("read");
    break;
}

// 或使用 SA_RESTART
struct sigaction sa;
sa.sa_handler = handler;
sigemptyset(&sa.sa_mask);
sa.sa_flags = SA_RESTART;  // 自動重啟
sigaction(SIGINT, &sa, NULL);
```

### 一次性訊號

```c
struct sigaction sa;
sa.sa_handler = handler;
sigemptyset(&sa.sa_mask);
sa.sa_flags = SA_RESETHAND;  // 處理後恢復預設
sigaction(SIGUSR1, &sa, NULL);
```

### 訊號驅動的 I/O

```c
#include <fcntl.h>
#include <signal.h>

volatile sig_atomic_t data_ready = 0;

void io_handler(int sig) {
    data_ready = 1;
}

int main() {
    int fd = open("/dev/async", O_RDWR);
    
    // 設定非同步 I/O
    fcntl(fd, F_SETOWN, getpid());
    int flags = fcntl(fd, F_GETFL);
    fcntl(fd, F_SETFL, flags | O_ASYNC);
    
    // 設定訊號處理
    struct sigaction sa;
    sa.sa_handler = io_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = 0;
    sigaction(SIGIO, &sa, NULL);
    
    while (!data_ready) {
        // 做其他工作
    }
    
    // 讀取就緒的資料
    // ...
}
```

## 實用範例

### 優雅的程式終止

```c
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>

volatile sig_atomic_t shutdown_requested = 0;

void shutdown_handler(int sig) {
    shutdown_requested = 1;
}

int main() {
    // 設定處理函式
    struct sigaction sa;
    sa.sa_handler = shutdown_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = SA_RESTART;
    
    sigaction(SIGINT, &sa, NULL);
    sigaction(SIGTERM, &sa, NULL);
    
    printf("Server started (PID %d)\n", getpid());
    
    while (!shutdown_requested) {
        // 處理請求
        printf("Working...\n");
        sleep(1);
    }
    
    // 清理資源
    printf("Cleaning up...\n");
    save_state();
    close_connections();
    
    printf("Shutdown complete\n");
    return 0;
}
```

### 子程序回收

```c
void child_handler(int sig) {
    int status;
    pid_t pid;
    
    // 回收所有已結束的子程序
    while ((pid = waitpid(-1, &status, WNOHANG)) > 0) {
        if (WIFEXITED(status)) {
            printf("Child %d exited with status %d\n",
                   pid, WEXITSTATUS(status));
        } else if (WIFSIGNALED(status)) {
            printf("Child %d killed by signal %d\n",
                   pid, WTERMSIG(status));
        }
    }
}

struct sigaction sa;
sa.sa_handler = child_handler;
sigemptyset(&sa.sa_mask);
sa.sa_flags = SA_RESTART | SA_NOCLDSTOP;
// SA_NOCLDSTOP: 子程序暫停時不產生通知
sigaction(SIGCHLD, &sa, NULL);
```

### 訊號量

```c
// 使用訊號實現簡單的訊號量
static sig_atomic_t sem = 0;

void v(int sig) {
    sem++;
}

void p(int sig, int wait_signal) {
    while (sem == 0) {
        pause();
    }
    sem--;
}

int main() {
    signal(SIGUSR1, v);
    // 使用者發送 SIGUSR1 增加計數
    
    return 0;
}
```

## 執行緒安全

### 訊號處理與執行緒

```c
// 訊號是行程層級的
// 所有執行緒共享相同的訊號處理

// 每個執行緒有自己的訊號屏蔽
// 執行緒可以屏蔽某些訊號

pthread_sigmask(SIG_BLOCK, &mask, NULL);  // 執行緒特定
sigprocmask(SIG_BLOCK, &mask, NULL);      // 整個行程
```

### 異步訊號安全

訊號處理中只能使用非同步訊號安全（async-signal-safe）的函式：

```c
// 安全函式（可選部分）
// _exit, _Exit, _Exit, abort, alarm, cfgetispeed, cfgetospeed,
// cfsetispeed, cfsetospeed, chmod, chown, close, creat, dup, dup2,
// execve, _exit, fcntl, fork, fpathconf, fstat, fsync, getegid,
// geteuid, getgid, getgroups, getpgrp, getpid, getppid, getuid,
// kill, link, lseek, lstat, mkdir, mkfifo, open, pathconf, pause,
// pipe, raise, read, rename, rmdir, setgid, setpgid, setsid,
// setuid, sigaction, sigaddset, sigdelset, sigemptyset,
// sigfillset, sigismember, signal, sigpending, sigprocmask,
// sigsuspend, sleep, stat, stty, symlink, sysconf, tcdrain,
// tcflow, tcflush, tcgetattr, tcgetpgrp, tcsetattr, tcsetpgrp,
// time, times, umask, uname, unlink, ustat, utime, wait, write

// 不安全函式
// printf, malloc, free, exit, longjmp, siglongjmp 等
```

## 高精度計時器

```c
#include <sys/time.h>

void timer_handler(int sig, siginfo_t *info, void *context) {
    // info->si_value 攜帶計時器 ID
    printf("Timer fired!\n");
}

int main() {
    struct sigevent sev;
    sev.sigev_notify = SIGEV_SIGNAL;
    sev.sigev_signo = SIGRTMIN;
    
    timer_t timerid;
    timer_create(CLOCK_REALTIME, &sev, &timerid);
    
    struct itimerspec its;
    its.it_value.tv_sec = 1;     // 首次到期：1 秒
    its.it_value.tv_nsec = 0;
    its.it_interval.tv_sec = 1;  // 間隔：1 秒
    its.it_interval.tv_nsec = 0;
    
    timer_settime(timerid, 0, &its, NULL);
    
    // 等待訊號
    sigset_t mask;
    sigemptyset(&mask);
    sigaddset(&mask, SIGRTMIN);
    sigsuspend(&mask);
}
```

## 移植性考量

```c
// signal() 的行為在不同系統間不一致
// 建議使用 sigaction()

// Linux 行為：
// - 預設會重置為 SIG_DFL
// - 支援 SA_RESTART

// BSD 行為：
// - 預設不重置
// - 支援 SA_RESTART

// 為可移植性總是使用 sigaction()
```

# 組合語言

組合語言是低階的機器碼表示，使用人類可讀的助記符號（mnemonics）代表機器指令。它是連接高階語言與機器硬體的橋樑，讓程式設計師能直接控制硬體資源，理解計算機運作原理，是系統程式設計的基礎。

## 組合語言 vs 機器碼

| 層次 | 說明 | 範例 |
|------|------|------|
| 高階語言 | 人類可讀，抽象程度高 | `a = b + c` |
| 組合語言 | 助記符號，指令導向 | `add a, b, c` |
| 機器碼 | 二進位，硬體直接執行 | `0000 1001 1100 0101` |

## RISC-V 組合語言

RISC-V 是一個精簡指令集架構（Reduced Instruction Set Computer），指令格式簡單，適合教學與實際應用。

### 基本指令格式

```
R-type (Register):   opcode(7) rs2(5) rs1(5) funct3(3) rd(5) funct7(7)
I-type (Immediate):  opcode(7) imm11:0(12) rs1(5) funct3(3) rd(5) funct7(7)
S-type (Store):      opcode(7) imm11:5(7) rs2(5) rs1(5) funct3(3) imm4:0(5) opcode(7)
B-type (Branch):     opcode(7) imm12|imm10:5(6) rs2(5) rs1(5) funct3(3) imm4:1(4) imm11(1) opcode(7)
U-type (Upper):      opcode(7) imm31:12(20) rd(5) opcode(7)
J-type (Jump):       opcode(7) imm20|imm10:1(10) imm11(1) imm19:12(8) rd(5) opcode(7)
```

### 算術指令

```assembly
# 加法
add a0, a1, a2      # a0 = a1 + a2 (溢出拋出異常)
addw a0, a1, a2      # 32位元加法

# 減法
sub a0, a1, a2      # a0 = a1 - a2

# 乘法
mul a0, a1, a2      # a0 = a1 * a2 (低64位)
mulh a0, a1, a2     # a0 = (a1 * a2) >> 64 (高64位)

# 除法
div a0, a1, a2      # a0 = a1 / a2 (整數除法)
rem a0, a1, a2      # a0 = a1 % a2

# 立即數運算
addi a0, a1, 10     # a0 = a1 + 10
subi a0, a1, 10     # a0 = a1 - 10
```

### 邏輯指令

```assembly
# 位元運算
and a0, a1, a2      # a0 = a1 & a2
or a0, a1, a2       # a0 = a1 | a2
xor a0, a1, a2      # a0 = a1 ^ a2

# 立即數版本
andi a0, a1, 0xFF    # a0 = a1 & 0xFF
ori a0, a1, 0xFF     # a0 = a1 | 0xFF
xori a0, a1, 0xFF    # a0 = a1 ^ 0xFF

# 移位
sll a0, a1, a2      # a0 = a1 << a2 (左移)
srl a0, a1, a2      # a0 = a1 >> a2 (邏輯右移)
sra a0, a1, a2      # a0 = a1 >> a2 (算術右移)

# 立即數移位
slli a0, a1, 5      # a0 = a1 << 5
srli a0, a1, 5      # a0 = a1 >> 5
srai a0, a1, 5      # a0 = a1 >> 5 (算術)
```

### 載入/儲存指令

```assembly
# 載入 (Load)
lw a0, 0(sp)        # a0 = mem[sp+0] (32位元)
ld a0, 0(sp)        # a0 = mem[sp+0] (64位元)
lb a0, 0(sp)        # a0 = mem[sp+0] (有號8位元)
lbu a0, 0(sp)       # a0 = mem[sp+0] (無號8位元)
lh a0, 0(sp)        # a0 = mem[sp+0] (有號16位元)
lhu a0, 0(sp)       # a0 = mem[sp+0] (無號16位元)

# 儲存 (Store)
sw a0, 0(sp)        # mem[sp+0] = a0 (32位元)
sd a0, 0(sp)        # mem[sp+0] = a0 (64位元)
sb a0, 0(sp)        # mem[sp+0] = a0 (8位元)
sh a0, 0(sp)        # mem[sp+0] = a0 (16位元)
```

### 分支指令

```assembly
# 相等分支
beq a0, a1, label   # if (a0 == a1) goto label

# 不等分支
bne a0, a1, label   # if (a0 != a1) goto label

# 小於分支 (有號)
blt a0, a1, label   # if (a0 < a1) goto label
bge a0, a1, label   # if (a0 >= a1) goto label

# 小於分支 (無號)
bltu a0, a1, label  # if ((unsigned)a0 < (unsigned)a1) goto label
bgeu a0, a1, label  # if ((unsigned)a0 >= (unsigned)a1) goto label

# 跳轉指令 (無條件)
j label             # goto label
jal label            # goto label, 保存返回位址到 ra
jr a0               # goto a0 指定的位址
jalr a0             # goto a0, 保存返回位址到 ra
```

### 比較指令

```assembly
# 設定小於 (有號)
slt a0, a1, a2      # a0 = (a1 < a2) ? 1 : 0

# 設定小於 (無號)
sltu a0, a1, a2    # a0 = ((unsigned)a1 < (unsigned)a2) ? 1 : 0

# 立即數版本
slti a0, a1, 10     # a0 = (a1 < 10) ? 1 : 0
sltiu a0, a1, 10    # a0 = ((unsigned)a1 < 10) ? 1 : 0
```

## 函式呼叫

### 呼叫約定

RISC-V 呼叫約定（呼叫暫存器）：

| 暫存器 | 用途 | 保存者 |
|--------|------|--------|
| a0-a7 | 函式引數 | 呼叫者 |
| ra | 返回位址 | 呼叫者 |
| sp | 堆疊指標 | 被呼叫者 |
| gp | 全域指標 | 被呼叫者 |
| tp | 執行緒指標 | 被呼叫者 |
| s0-s11 | 保存暫存器 | 被呼叫者 |
| t0-t6 | 臨時暫存器 | 呼叫者 |

### 簡單函式

```assembly
# 呼叫 foo(1, 2)
# a0 = 1, a1 = 2
call foo
# 返回值在 a0

foo:
    addi sp, sp, -16    # 分配堆疊空間
    sd ra, 8(sp)        # 保存返回位址
    sd s0, 0(sp)        # 保存保存暫存器
    
    # 函式本體
    add a0, a0, a1      # a0 = a0 + a1
    
    ld s0, 0(sp)        # 恢復保存暫存器
    ld ra, 8(sp)        # 恢復返回位址
    addi sp, sp, 16     # 釋放堆疊空間
    ret                 # jr ra
```

### 遞迴函式

```assembly
# fac(n):
#   if n <= 1 return 1
#   return n * fac(n-1)

fac:
    # 保存暫存器
    addi sp, sp, -24
    sd ra, 16(sp)
    sd s0, 8(sp)
    sd s1, 0(sp)
    
    # n <= 1 ?
    li s0, 1
    bgt a0, s0, recurse
    
    # return 1
    li a0, 1
    j done
    
recurse:
    # 保存 n
    mv s1, a0
    
    # fac(n-1)
    addi a0, a0, -1
    call fac
    
    # return n * result
    mul a0, s1, a0
    
done:
    # 恢復暫存器
    ld s1, 0(sp)
    ld s0, 8(sp)
    ld ra, 16(sp)
    addi sp, sp, 24
    ret
```

### 結構回傳

```assembly
# 回傳兩個值 (a0, a1)
# 
# C 呼叫: foo(&result)
# a0 = result 指標

foo:
    addi sp, sp, -16
    sd ra, 8(sp)
    
    # 計算結果
    li t0, 100
    li t1, 200
    
    # 儲存到指標
    sd t0, 0(a0)
    sd t1, 8(a0)
    
    ld ra, 8(sp)
    addi sp, sp, 16
    ret
```

## 控制流程

### 迴圈

```assembly
# for (i = 0; i < n; i++)
#     sum += i

loop:
    # i >= n ?
    bge t0, a0, end_loop
    
    # sum += i
    add t1, t1, t0
    
    # i++
    addi t0, t0, 1
    
    j loop
    
end_loop:
    mv a0, t1
```

### 條件分支

```assembly
# if (a > b) max = a; else max = b;

    # a > b ?
    ble a0, a1, else
    mv t0, a0
    j done
    
else:
    mv t0, a1
    
done:
    mv a0, t0
```

### 分頁表

```assembly
# switch (n) {
#     case 1: a = 10; break;
#     case 2: a = 20; break;
#     case 3: a = 30; break;
#     default: a = 0;
# }

    # n - 1
    addi t0, a0, -1
    
    # 範圍檢查
    bgtu t0, a1, default
    
    # 跳轉表
    b Default case1
    b case2
    b case3
    
Default:
    # default
end_switch:
```

## 記憶體管理

### 靜態配置

```assembly
.data
buffer: .space 100      # 配置 100 位元組
message: .asciz "Hello" # 字串
numbers: .word 1,2,3,4,5 # 整數陣列

.text
    la t0, buffer     # 載入緩衝區位址
    la t1, message    # 載入訊息位址
    la t2, numbers    # 載入陣列位址
```

### 堆疊框架

```assembly
# 函式框架佈局
# High Addr
# +----------------+
# |    ra         | <- FP+8
# +----------------+
# |    old FP    | <- FP (Old Base Pointer)
# +----------------+
# |    saved     | <- FP-8
# |    ...      |
# |    saved     | <- FP-24
# +----------------+
# |    local 1   | <- FP-32
# |    ...      |
# +----------------+ <- SP (Stack Pointer)
# Low Addr
```

### 堆積配置

```assembly
# 使用 sbrk (系統調用)
# a0 = request size
li a0, 100         # 配置 100 位元組
ecall               # 呼叫系統
# a0 = 配置的位址

# 釋放 (一般不需要，保留給 GC)
```

## 與 C 語言整合

### 內聯組合語言

```c
// GCC 內聯組語
asm volatile (
    "add %0, %1, %2"
    : "=r" (result)
    : "r" (a), "r" (b)
);
```

### 獨立組語檔案

```assembly
# sum.s
.section .text
.global sum

sum:
    add a0, a0, a1
    ret
```

```c
// main.c
extern int sum(int a, int b);

int main() {
    return sum(1, 2);
}
```

```bash
# 編譯
riscv64-unknown-elf-gcc -o sum main.c sum.s
```

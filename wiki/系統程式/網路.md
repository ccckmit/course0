# 網路

網路是現代資訊科技的基礎設施，它將全球數十億的設備連接在一起，實現資料的傳遞和資訊的共享。從瀏覽網頁、觀看影片到雲端服務，無一不依賴於完善的網路架構和協定。理解網路的基本原理對於軟體工程師、系統管理員和資訊科技從業者來說至關重要。本篇將全面介紹網路的各個層面，從基礎的 OSI 和 TCP/IP 模型到進階的網路安全和管理。

```python
# 計算可用頻寬 (Bandwidth-Delay Product)
def calculate_bdp(bandwidth_mbps, latency_ms):
    """
    計算頻寬延遲乘積 (BDP)
    這是 TCP 視窗大小的理論上限
    """
    bandwidth_bps = bandwidth_mbps * 1_000_000
    latency_seconds = latency_ms / 1000
    
    bdp_bytes = (bandwidth_bps * latency_seconds) / 8
    
    return bdp_bytes

# 範例：100 Mbps, 50ms
bdp = calculate_bdp(100, 50)
print(f"BDP: {bdp / 1024 / 1024:.2f} MB")
# 輸出: BDP: 0.61 MB (需要 ~5 Mbps TCP 視窗)
```

## IP 位址

### IPv4

```python
# IPv4 位址結構
ipv4_info = {
    "格式": "32 位元，4 個 8 位元組",
    "表示": "點分十進制 (192.168.1.1)",
    "範圍": "0.0.0.0 - 255.255.255.255",
    "總數": "2^32 ≈ 43 億個"
}

# 計算網路
def calculate_network(ip_cidr):
    """計算網路位址、廣播位址、主機範圍"""
    ip, prefix = ip_cidr.split('/')
    
    ip_parts = [int(x) for x in ip.split('.')]
    ip_int = (ip_parts[0] << 24) + (ip_parts[1] << 16) + \
             (ip_parts[2] << 8) + ip_parts[3]
    
    prefix = int(prefix)
    mask = (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF
    
    network = ip_int & mask
    broadcast = network | (~mask & 0xFFFFFFFF)
    
    if prefix < 31:
        first_host = network + 1
        last_host = broadcast - 1
        num_hosts = 2 ** (32 - prefix) - 2
    else:
        first_host = last_host = network
        num_hosts = 2
    
    def int_to_ip(n):
        return f"{(n >> 24) & 0xFF}.{(n >> 16) & 0xFF}." \
               f"{(n >> 8) & 0xFF}.{n & 0xFF}"
    
    return {
        "network": int_to_ip(network),
        "broadcast": int_to_ip(broadcast),
        "first_host": int_to_ip(first_host),
        "last_host": int_to_ip(last_host),
        "num_hosts": num_hosts
    }

# 範例
result = calculate_network("192.168.1.0/24")
print(result)
# {'network': '192.168.1.0', 'broadcast': '192.168.1.255', 
#  'first_host': '192.168.1.1', 'last_host': '192.168.1.254', 'num_hosts': 254}
```

## Socket 程式設計

### TCP 伺服器/客戶端

```python
# TCP 伺服器
import socket

def create_tcp_server(host='0.0.0.0', port=8080):
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server_socket.bind((host, port))
    server_socket.listen(5)
    
    print(f"Server listening on {host}:{port}")
    
    while True:
        client_socket, client_address = server_socket.accept()
        print(f"Connection from {client_address}")
        
        # 處理請求
        handle_client(client_socket)
        
        client_socket.close()

def handle_client(client_socket):
    try:
        # 接收資料
        data = client_socket.recv(1024)
        
        if data:
            # 解析 HTTP 請求
            request = data.decode('utf-8')
            print(f"Request: {request[:100]}")
            
            # 發送回應
            response = "HTTP/1.1 200 OK\r\nContent-Type: text/plain\r\n\r\nHello, World!"
            client_socket.send(response.encode('utf-8'))
    
    except Exception as e:
        print(f"Error: {e}")
    finally:
        client_socket.close()

# TCP 客戶端
def create_tcp_client(host='localhost', port=8080):
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    client_socket.connect((host, port))
    
    # 發送請求
    message = "Hello, Server!"
    client_socket.send(message.encode('utf-8'))
    
    # 接收回應
    response = client_socket.recv(1024)
    print(f"Response: {response.decode('utf-8')}")
    
    client_socket.close()
```

### 高效能 Socket

```python
# 使用 select 的 IO 多路複用
import select

def select_server(host='0.0.0.0', port=8080):
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.setblocking(False)
    server_socket.bind((host, port))
    server_socket.listen(5)
    
    sockets = [server_socket]
    
    while True:
        readable, _, _ = select.select(sockets, [], [], 1)
        
        for s in readable:
            if s is server_socket:
                client_socket, _ = server_socket.accept()
                client_socket.setblocking(False)
                sockets.append(client_socket)
            else:
                try:
                    data = s.recv(1024)
                    if data:
                        # 處理資料
                        s.send(b"OK")
                    else:
                        s.close()
                        sockets.remove(s)
                except:
                    s.close()
                    sockets.remove(s)

# 使用 epoll (Linux)
def epoll_server(host='0.0.0.0', port=8080):
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server_socket.bind((host, port))
    server_socket.listen(5)
    server_socket.setblocking(False)
    
    epoll = select.epoll()
    epoll.register(server_socket.fileno(), select.EPOLLIN)
    
    connections = {}
    
    while True:
        events = epoll.poll(1)
        
        for fd, event in events:
            if fd == server_socket.fileno():
                client_socket, _ = server_socket.accept()
                client_socket.setblocking(False)
                epoll.register(client_socket.fileno(), select.EPOLLIN)
                connections[client_socket.fileno()] = client_socket
            
            elif event & select.EPOLLIN:
                client_socket = connections[fd]
                data = client_socket.recv(1024)
                
                if data:
                    client_socket.send(b"OK")
                else:
                    epoll.unregister(fd)
                    client_socket.close()
                    del connections[fd]
```

```python
# Python SSL 伺服器
import ssl
import socket

def create_https_server(certfile, keyfile, port=8443):
    context = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
    context.load_cert_chain(certfile, keyfile)
    
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.bind(('0.0.0.0', port))
    server_socket.listen(5)
    
    wrapped_socket = context.wrap_socket(server_socket, server_side=True)
    
    return wrapped_socket

# Python SSL 客戶端
def create_https_client(host, port):
    context = ssl.create_default_context()
    
    with socket.create_connection((host, port)) as sock:
        with context.wrap_socket(sock, server_hostname=host) as ssock:
            ssock.send(b"GET / HTTP/1.1\r\nHost: {}\r\n\r\n".format(host.encode()))
            response = ssock.recv(4096)
            print(response)
```

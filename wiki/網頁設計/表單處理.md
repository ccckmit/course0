# 表單處理 (Forms)

## 概述

表單是網頁收集使用者輸入資料的基本元件，支援各種類型的輸入欄位和驗證機制。

## HTML 表單結構

```html
<form action="/submit" method="POST" enctype="multipart/form-data">
    <!-- 文字輸入 -->
    <label for="username">使用者名稱：</label>
    <input type="text" id="username" name="username" 
           required minlength="3" maxlength="20" 
           placeholder="輸入使用者名稱">
    
    <!-- 密碼輸入 -->
    <label for="password">密碼：</label>
    <input type="password" id="password" name="password" 
           required pattern="(?=.*\d)(?=.*[a-z]).{8,}">
    
    <!-- 電子郵件 -->
    <label for="email">電子郵件：</label>
    <input type="email" id="email" name="email" 
           required pattern="[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$">
    
    <!-- 數字輸入 -->
    <label for="age">年齡：</label>
    <input type="number" id="age" name="age" 
           min="18" max="120" value="18">
    
    <!-- 日期輸入 -->
    <label for="birthday">生日：</label>
    <input type="date" id="birthday" name="birthday">
    
    <!-- 核取方塊 -->
    <input type="checkbox" id="subscribe" name="subscribe" checked>
    <label for="subscribe">訂閱電子報</label>
    
    <!-- 單選按鈕 -->
    <fieldset>
        <legend>性別</legend>
        <input type="radio" id="male" name="gender" value="male">
        <label for="male">男</label>
        <input type="radio" id="female" name="gender" value="female">
        <label for="female">女</label>
    </fieldset>
    
    <!-- 下拉選單 -->
    <label for="country">國家：</label>
    <select id="country" name="country">
        <option value="">請選擇</option>
        <optgroup label="亞洲">
            <option value="tw">臺灣</option>
            <option value="jp">日本</option>
        </optgroup>
        <optgroup label="歐洲">
            <option value="uk">英國</option>
            <option value="de">德國</option>
        </optgroup>
    </select>
    
    <!-- 多行文字 -->
    <label for="bio">自我介紹：</label>
    <textarea id="bio" name="bio" rows="4" 
              placeholder="簡單介紹自己..."></textarea>
    
    <!-- 檔案上傳 -->
    <label for="avatar">頭像：</label>
    <input type="file" id="avatar" name="avatar" 
           accept="image/*" multiple>
    
    <!-- 隱藏欄位 -->
    <input type="hidden" name="csrf_token" value="abc123">
    
    <!-- 提交按鈕 -->
    <button type="submit">送出</button>
    <button type="reset">重填</button>
</form>
```

## 輸入類型

| 類型 | 說明 | 範例值 |
|------|------|--------|
| `text` | 單行文字 | "John" |
| `password` | 密碼（隱藏） | "********" |
| `email` | 電子郵件 | "a@b.com" |
| `number` | 數字 | "42" |
| `tel` | 電話 | "0912-345-678" |
| `url` | 網址 | "https://..." |
| `search` | 搜尋框 | "查詢..." |
| `range` | 範圍滑桿 | 50 |
| `color` | 顏色選擇器 | "#ff0000" |

## JavaScript 表單處理

### 表單事件

```javascript
const form = document.querySelector('form');

form.addEventListener('submit', (e) => {
    e.preventDefault();  // 阻止預設提交
    
    const formData = new FormData(form);
    
    // 轉換為物件
    const data = Object.fromEntries(formData);
    
    console.log('Form data:', data);
});

form.addEventListener('reset', (e) => {
    console.log('Form reset');
});

// 個別欄位驗證
const input = form.querySelector('input[name="email"]');
input.addEventListener('blur', () => {
    if (!input.validity.valid) {
        showError(input, input.validationMessage);
    }
});

input.addEventListener('input', () => {
    if (input.validity.valid) {
        clearError(input);
    }
});
```

### FormData API

```javascript
// 基本用法
const form = document.querySelector('form');
const formData = new FormData(form);

// 手動新增欄位
formData.append('name', 'Alice');
formData.append('tags', ['tag1', 'tag2']);
formData.set('email', 'alice@example.com');  // 覆蓋

// 取得值
formData.get('name');        // 'Alice'
formData.getAll('tags');     // ['tag1', 'tag2']
formData.has('email');       // true
formData.delete('email');    // 刪除欄位

// 轉換格式
const obj = Object.fromEntries(formData);
const json = JSON.stringify(obj);
const queryString = new URLSearchParams(formData).toString();
```

### 自訂驗證

```javascript
function validateForm(form) {
    let isValid = true;
    const errors = [];
    
    // 檢查必填欄位
    form.querySelectorAll('[required]').forEach(field => {
        if (!field.value.trim()) {
            errors.push(`${field.name} 為必填`);
            isValid = false;
        }
    });
    
    // 自訂驗證：密碼強度
    const password = form.querySelector('#password');
    if (password.value) {
        const hasUpperCase = /[A-Z]/.test(password.value);
        const hasLowerCase = /[a-z]/.test(password.value);
        const hasNumber = /[0-9]/.test(password.value);
        
        if (!(hasUpperCase && hasLowerCase && hasNumber)) {
            errors.push('密碼需包含大小寫字母和數字');
            isValid = false;
        }
    }
    
    // 確認密碼
    const confirm = form.querySelector('#confirm');
    if (confirm.value !== password.value) {
        errors.push('確認密碼不相符');
        isValid = false;
    }
    
    return { isValid, errors };
}
```

## Fetch API 提交

```javascript
async function submitForm(form) {
    const formData = new FormData(form);
    
    try {
        const response = await fetch('/api/submit', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-CSRF-Token': getCsrfToken()
            },
            body: JSON.stringify(Object.fromEntries(formData))
        });
        
        if (!response.ok) {
            const error = await response.json();
            throw new Error(error.message);
        }
        
        const result = await response.json();
        showSuccess(result.message);
        
    } catch (error) {
        showError(error.message);
    }
}
```

## React 表單處理

```jsx
import { useState } from 'react';

function ContactForm() {
    const [formData, setFormData] = useState({
        name: '',
        email: '',
        message: ''
    });
    
    const [errors, setErrors] = useState({});
    
    const handleChange = (e) => {
        const { name, value } = e.target;
        setFormData(prev => ({ ...prev, [name]: value }));
        // 清除錯誤
        if (errors[name]) {
            setErrors(prev => ({ ...prev, [name]: '' }));
        }
    };
    
    const handleSubmit = async (e) => {
        e.preventDefault();
        
        // 驗證
        const newErrors = validate(formData);
        if (Object.keys(newErrors).length > 0) {
            setErrors(newErrors);
            return;
        }
        
        // 提交
        await fetch('/api/contact', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(formData)
        });
    };
    
    return (
        <form onSubmit={handleSubmit}>
            <input
                name="name"
                value={formData.name}
                onChange={handleChange}
            />
            {errors.name && <span>{errors.name}</span>}
            {/* ... */}
        </form>
    );
}
```

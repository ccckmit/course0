## 搜尋演算法

### Minimax 演算法

```python
def minimax(board, depth, is_maximizing, alpha=float('-inf'), beta=float('inf')):
    # 終止條件
    if depth == 0 or game.is_terminal(board):
        return evaluate(board)
    
    if is_maximizing:
        max_eval = float('-inf')
        for move in board.legal_moves:
            eval = minimax(board.make_move(move), depth - 1, False, alpha, beta)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:
                break  # Alpha-Beta 剪枝
        return max_eval
    else:
        min_eval = float('inf')
        for move in board.legal_moves:
            eval = minimax(board.make_move(move), depth - 1, True, alpha, beta)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break  # Alpha-Beta 剪枝
        return min_eval
```

### Alpha-Beta 剪枝

```
搜尋樹示例：

          A (MAX)
         /|\
        B C D
       /|  | /
      5 7 ? ?

MIN(5,7,?) → 若 ? ≤ 5，可直接剪枝
```

## 圍棋 AI

### 蒙地卡羅樹搜尋 (MCTS)

```python
class MonteCarloTreeSearch:
    def __init__(self, root_state):
        self.root = Node(state=root_state)
    
    def search(self, n_simulations):
        for _ in range(n_simulations):
            node = self.select(self.root)
            
            if not node.state.is_terminal():
                node = self.expand(node)
            
            reward = self.simulate(node)
            self.backpropagate(node, reward)
        
        return self.root.best_child().action
    
    def select(self, node):
        """選擇：UCB1 公式"""
        while not node.is_untried() and node.is_expanded():
            node = max(node.children, key=lambda n: UCB1(n))
        return node
    
    def UCB1(self, node):
        """Upper Confidence Bound"""
        return node.wins / node.visits + \
               EXPLORATION * (math.log(node.parent.visits) / node.visits) ** 0.5
    
    def expand(self, node):
        """擴展"""
        for action in node.state.legal_actions():
            node.add_child(action)
        return node.random_child()
    
    def simulate(self, node):
        """模擬"""
        state = node.state.copy()
        while not state.is_terminal():
            state.random_play()
        return state.reward()
    
    def backpropagate(self, node, reward):
        """回傳"""
        while node:
            node.visits += 1
            node.wins += reward
            node = node.parent
```

### 圍棋神經網路

```python
class AlphaGoNetwork(nn.Module):
    def __init__(self):
        super().__init__()
        # 特徵卷積層
        self.conv1 = nn.Conv2d(48, 192, 3, padding=1)
        self.conv2 = nn.Conv2d(192, 192, 3, padding=1)
        self.conv3 = nn.Conv2d(192, 192, 3, padding=1)
        # ... 更多卷積層
        
        # 策略頭（動作機率）
        self.policy_head = nn.Sequential(
            nn.Conv2d(192, 2, 1),
            nn.ReLU(),
            nn.Linear(362, 19*19+1)  # 棋盤 + pass
        )
        
        # 價值頭（勝率）
        self.value_head = nn.Sequential(
            nn.Conv2d(192, 1, 1),
            nn.ReLU(),
            nn.Linear(19*19, 256),
            nn.ReLU(),
            nn.Linear(256, 1),
            nn.Tanh()
        )
    
    def forward(self, x):
        # 殘差網路主體
        x = torch.relu(self.conv1(x))
        # ... 殘差區塊
        
        # 策略和價值輸出
        policy = F.softmax(self.policy_head(x), dim=-1)
        value = self.value_head(x)
        return policy, value
```

## AlphaZero

### 自對弈學習

```python
class AlphaZero:
    def __init__(self, game):
        self.game = game
        self.network = ResNet(game, 20)
        self.mcts = MCTS(game, self.network)
    
    def self_play(self):
        """自對弈產生訓練資料"""
        examples = []
        state = self.game.initial_state()
        
        while not state.is_terminal():
           pi = self.mcts.get_policy(state)
            action = np.random.choice(len(pi), p=pi)
            examples.append((state, pi, state.current_player))
            state = state.make_move(action)
        
        # 最終結果
        winner = state.winner()
        for i, (state, pi, player) in enumerate(examples):
            if player == winner:
                reward = 1
            elif winner is None:
                reward = 0
            else:
                reward = -1
            self.training_data.append((state, pi, reward))
    
    def train(self):
        """訓練網路"""
        for epoch in range(100):
            # 取樣資料
            batch = random.sample(self.training_data, BATCH_SIZE)
            
            states, pis, rewards = zip(*batch)
            
            # 前饋
            policy, value = self.network(states)
            
            # 損失
            policy_loss = F.cross_entropy(policy, pis)
            value_loss = F.mse_loss(value, rewards)
            loss = policy_loss + value_loss
            
            # 反向傳播
            loss.backward()
            optimizer.step()
```

## 評估函數

### 簡單評估

```python
def evaluate_simple(board):
    """西洋棋：棋子價值總和"""
    values = {
        'P': 1, 'N': 3, 'B': 3, 'R': 5, 'Q': 9, 'K': 1000,
        'p': -1, 'n': -3, 'b': -3, 'r': -5, 'q': -9, 'k': -1000
    }
    return sum(values.get(piece, 0) for row in board for piece in row)
```

### 學習評估函數

```python
class LearnedEvaluator(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc = nn.Sequential(
            nn.Linear(64 * 12, 128),  # 64 格 * 12 種棋子
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 1),
            nn.Tanh()
        )
    
    def forward(self, board):
        # 編碼棋盤狀態
        x = self.encode_board(board)
        return self.fc(x)
    
    def encode_board(self, board):
        # 12 通道：白方兵、白方馬...黑方兵...
        encoding = torch.zeros(64 * 12)
        for i, piece in enumerate(board):
            if piece != '.':
                channel = piece_to_channel(piece)
                encoding[i + channel * 64] = 1
        return encoding
```

# 推薦系統：個性化推薦的核心技術

## 概述

推薦系統是解決信息過載問題的關鍵技術，通過分析用戶行為和物品特徵，為用戶提供個性化的內容推薦。從電商平台的商品推薦、視頻網站的影片推薦，到社交媒體的好友推薦，推薦系統已成為現代互聯網服務的核心組件。

推薦系統的三大主流方法包括：**協同過濾**（Collaborative Filtering）基於用戶或物品的相似性；**內容過濾**（Content-Based Filtering）基於物品特徵匹配用戶偏好；**混合方法**結合多種技術提高推薦質量。近年來，深度學習推薦模型（如Wide&Deep、DeepFM）顯著提升了推薦效果。

## 協同過濾

```python
import numpy as np
from scipy.sparse import csr_matrix
from sklearn.metrics.pairwise import cosine_similarity

class UserBasedCF:
    """基於用戶的協同過濾"""
    
    def __init__(self, k=20, similarity='cosine'):
        self.k = k
        self.similarity = similarity
        self.user_similarity = None
        self.ratings = None
        self.user_means = None
    
    def fit(self, ratings_matrix):
        """
        ratings_matrix: (n_users, n_items) 用戶-物品評分矩陣
        """
        self.ratings = ratings_matrix
        
        # 填充缺失值為用戶平均值
        self.user_means = np.nanmean(ratings_matrix, axis=1, keepdims=True)
        ratings_filled = np.where(np.isnan(ratings_matrix), 
                                  self.user_means, ratings_matrix)
        
        # 計算用戶相似度
        self.user_similarity = cosine_similarity(ratings_filled)
        np.fill_diagonal(self.user_similarity, 0)
        
        return self
    
    def predict(self, user_id, item_ids):
        """預測用戶對物品的評分"""
        predictions = []
        
        for item_id in item_ids:
            # 找到評價過該物品的其他用戶
            rated_users = np.where(~np.isnan(self.ratings[:, item_id]))[0]
            
            if len(rated_users) == 0:
                predictions.append(self.user_means[user_id][0])
                continue
            
            # 選擇最相似的 k 個用戶
            sim_scores = self.user_similarity[user_id, rated_users]
            top_k_indices = np.argsort(sim_scores)[-self.k:]
            top_k_users = rated_users[top_k_indices]
            top_k_sims = sim_scores[top_k_indices]
            
            # 加權平均
            top_k_ratings = self.ratings[top_k_users, item_id]
            if np.sum(np.abs(top_k_sims)) > 0:
                pred = np.sum(top_k_sims * (top_k_ratings - self.user_means[top_k_users, 0])) / np.sum(np.abs(top_k_sims))
                pred += self.user_means[user_id][0]
            else:
                pred = self.user_means[user_id][0]
            
            predictions.append(pred)
        
        return np.array(predictions)

class ItemBasedCF:
    """基於物品的協同過濾"""
    
    def __init__(self, k=20):
        self.k = k
        self.item_similarity = None
        self.ratings = None
    
    def fit(self, ratings_matrix):
        self.ratings = ratings_matrix
        
        # 計算物品相似度
        ratings_filled = np.where(np.isnan(ratings_matrix), 0, ratings_matrix)
        self.item_similarity = cosine_similarity(ratings_filled.T)
        np.fill_diagonal(self.item_similarity, 0)
        
        return self
    
    def predict(self, user_id, item_ids):
        predictions = []
        
        for item_id in item_ids:
            # 用戶評分過的物品
            rated_items = np.where(~np.isnan(self.ratings[user_id]))[0]
            
            if len(rated_items) == 0:
                predictions.append(np.nanmean(self.ratings))
                continue
            
            # 最相似的 k 個物品
            sim_scores = self.item_similarity[item_id, rated_items]
            top_k_indices = np.argsort(sim_scores)[-self.k:]
            top_k_items = rated_items[top_k_indices]
            top_k_sims = sim_scores[top_k_indices]
            
            # 加權平均
            top_k_ratings = self.ratings[user_id, top_k_items]
            if np.sum(np.abs(top_k_sims)) > 0:
                pred = np.sum(top_k_sims * top_k_ratings) / np.sum(np.abs(top_k_sims))
            else:
                pred = np.nanmean(top_k_ratings)
            
            predictions.append(pred)
        
        return np.array(predictions)
```

## 矩陣分解

```python
import torch
import torch.nn as nn

class SVD:
    """奇異值分解"""
    
    def __init__(self, n_factors=50):
        self.n_factors = n_factors
        self.user_factors = None
        self.item_factors = None
        self.user_bias = None
        self.item_bias = None
        self.global_mean = None
    
    def fit(self, ratings, n_epochs=20, lr=0.01, reg=0.02):
        """
        使用 SGD 訓練
        """
        n_users, n_items = ratings.shape
        
        # 初始化
        self.global_mean = np.nanmean(ratings)
        self.user_factors = np.random.normal(0, 0.1, (n_users, self.n_factors))
        self.item_factors = np.random.normal(0, 0.1, (n_items, self.n_factors))
        self.user_bias = np.zeros(n_users)
        self.item_bias = np.zeros(n_items)
        
        # 準備訓練數據
        user_ids, item_ids, ratings_list = [], [], []
        for u in range(n_users):
            for i in range(n_items):
                if not np.isnan(ratings[u, i]):
                    user_ids.append(u)
                    item_ids.append(i)
                    ratings_list.append(ratings[u, i])
        
        user_ids = np.array(user_ids)
        item_ids = np.array(item_ids)
        ratings_list = np.array(ratings_list)
        
        for epoch in range(n_epochs):
            # 打亂順序
            indices = np.random.permutation(len(ratings_list))
            
            for idx in indices:
                u, i, r = user_ids[idx], item_ids[idx], ratings_list[idx]
                
                # 預測
                pred = (self.global_mean + 
                       self.user_bias[u] + self.item_bias[i] +
                       np.dot(self.user_factors[u], self.item_factors[i]))
                
                # 誤差
                error = r - pred
                
                # 更新參數
                self.user_bias[u] += lr * (error - reg * self.user_bias[u])
                self.item_bias[i] += lr * (error - reg * self.item_bias[i])
                
                user_factor_old = self.user_factors[u].copy()
                self.user_factors[u] += lr * (error * self.item_factors[i] - reg * self.user_factors[u])
                self.item_factors[i] += lr * (error * user_factor_old - reg * self.item_factors[i])
    
    def predict(self, user_id, item_ids):
        predictions = []
        for item_id in item_ids:
            pred = (self.global_mean + 
                   self.user_bias[user_id] + self.item_bias[item_id] +
                   np.dot(self.user_factors[user_id], self.item_factors[item_id]))
            predictions.append(pred)
        return np.array(predictions)
```

## 深度學習推薦

```python
class NeuMF(nn.Module):
    """神經矩陣分解"""
    
    def __init__(self, n_users, n_items, n_factors=32, layers=[64, 32, 16]):
        super().__init__()
        
        # GMF 部分
        self.gmf_embedding_user = nn.Embedding(n_users, n_factors)
        self.gmf_embedding_item = nn.Embedding(n_items, n_factors)
        
        # MLP 部分
        self.mlp_embedding_user = nn.Embedding(n_users, layers[0] // 2)
        self.mlp_embedding_item = nn.Embedding(n_items, layers[0] // 2)
        
        mlp_layers = []
        input_size = layers[0]
        for layer_size in layers[1:]:
            mlp_layers.append(nn.Linear(input_size, layer_size))
            mlp_layers.append(nn.ReLU())
            mlp_layers.append(nn.Dropout(0.2))
            input_size = layer_size
        self.mlp = nn.Sequential(*mlp_layers)
        
        # 最終預測層
        self.fc = nn.Linear(n_factors + layers[-1], 1)
        self.sigmoid = nn.Sigmoid()
    
    def forward(self, user_ids, item_ids):
        # GMF
        gmf_user = self.gmf_embedding_user(user_ids)
        gmf_item = self.gmf_embedding_item(item_ids)
        gmf_out = gmf_user * gmf_item
        
        # MLP
        mlp_user = self.mlp_embedding_user(user_ids)
        mlp_item = self.mlp_embedding_item(item_ids)
        mlp_input = torch.cat([mlp_user, mlp_item], dim=-1)
        mlp_out = self.mlp(mlp_input)
        
        # 拼接
        concat = torch.cat([gmf_out, mlp_out], dim=-1)
        
        # 預測
        output = self.fc(concat)
        return self.sigmoid(output)

class DeepFM(nn.Module):
    """DeepFM: 結合因子分解機和深度神經網路"""
    
    def __init__(self, n_users, n_items, n_features, embed_dim=4, hidden_dims=[128, 64]):
        super().__init__()
        
        self.n_users = n_users
        self.n_items = n_items
        
        # 一階特徵
        self.first_order_embeddings = nn.Embedding(n_features, 1)
        self.first_order_bias = nn.Parameter(torch.zeros(n_features))
        
        # 二階特徵
        self.second_order_embeddings = nn.Embedding(n_features, embed_dim)
        
        # DNN
        dnn_layers = []
        input_dim = embed_dim
        for hidden_dim in hidden_dims:
            dnn_layers.append(nn.Linear(input_dim, hidden_dim))
            dnn_layers.append(nn.ReLU())
            dnn_layers.append(nn.BatchNorm1d(hidden_dim))
            dnn_layers.append(nn.Dropout(0.5))
            input_dim = hidden_dim
        self.dnn = nn.Sequential(*dnn_layers)
        
        # 最終預測
        self.fc = nn.Linear(hidden_dims[-1] + embed_dim + 1, 1)
        self.sigmoid = nn.Sigmoid()
    
    def forward(self, user_ids, item_ids, features):
        # 一階
        first_order = self.first_order_embeddings(features).sum(dim=1) + \
                     self.first_order_bias[features].sum(dim=1)
        
        # 二階 (FM)
        embeddings = self.second_order_embeddings(features)
        sum_square = embeddings.sum(dim=1) ** 2
        square_sum = (embeddings ** 2).sum(dim=1)
        second_order = 0.5 * (sum_square - square_sum).sum(dim=1)
        
        # DNN
        dnn_out = self.dnn(embeddings.mean(dim=1))
        
        # 拼接預測
        concat = torch.cat([first_order.unsqueeze(1), 
                           second_order.unsqueeze(1), 
                           dnn_out], dim=-1)
        
        output = self.fc(concat)
        return self.sigmoid(output)
```

## 總結

推薦系統通過協同過濾、矩陣分解和深度學習等技術為用戶提供個性化推薦。本頁介紹了基於用戶/物品的協同過濾、奇異值分解、神經矩陣分解和DeepFM等模型。現代推薦系統通常採用混合方法，結合多種技術以獲得更好的推薦效果。

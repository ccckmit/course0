# 自然語言處理：文本理解與生成的核心技術

## 概述

自然語言處理（Natural Language Processing，NLP）是人工智慧和計算語言學的交叉領域，致力於讓計算機能夠理解、解釋和生成人類語言。從早期的基於規則的方法，到統計NLP，再到如今以Transformer為代表的深度學習方法，NLP經歷了革命性的變革。現代NLP系統能夠執行機器翻譯、情感分析、問答系統、文本摘要、對話系統等複雜任務。

理解NLP的基礎技術對於構建有效的語言處理系統至關重要。這些基礎技術包括：文本預處理（分詞、標註）、詞嵌入（Word2Vec、GloVe）、序列模型（RNN、LSTM、GRU）、以及當前主流的Transformer架構。本頁將深入探討這些核心概念，並提供實用的Python和PyTorch實現。

## 文本預處理與分詞

### 分詞技術

```python
import re
import json
from collections import Counter

class Tokenizer:
    """基礎分詞器"""
    
    def __init__(self, vocab=None, unk_token='<UNK>', pad_token='<PAD>'):
        self.unk_token = unk_token
        self.pad_token = pad_token
        self.unk_token_id = 0
        self.pad_token_id = 1
        
        if vocab:
            self.vocab = vocab
            self.token2id = {t: i for i, t in enumerate(vocab)}
            self.id2token = {i: t for t, i in self.token2id.items()}
        else:
            self.vocab = [unk_token, pad_token]
            self.token2id = {unk_token: 0, pad_token: 1}
            self.id2token = {0: unk_token, 1: pad_token}
    
    def fit(self, texts, min_freq=2):
        """從文本構建詞彙表"""
        counter = Counter()
        for text in texts:
            tokens = self._basic_tokenize(text)
            counter.update(tokens)
        
        for token, freq in counter.items():
            if freq >= min_freq:
                idx = len(self.vocab)
                self.vocab.append(token)
                self.token2id[token] = idx
                self.id2token[idx] = token
    
    def _basic_tokenize(self, text):
        """基礎分詞：小寫+標點分開"""
        text = text.lower()
        text = re.sub(r'([.,!?;:])', r' \1 ', text)
        tokens = text.split()
        return tokens
    
    def encode(self, text, max_length=None):
        """文本轉ID序列"""
        tokens = self._basic_tokenize(text)
        ids = [self.token2id.get(t, self.unk_token_id) for t in tokens]
        
        if max_length:
            if len(ids) > max_length:
                ids = ids[:max_length]
            else:
                ids = ids + [self.pad_token_id] * (max_length - len(ids))
        
        return ids
    
    def decode(self, ids):
        """ID序列轉文本"""
        tokens = [self.id2token.get(i, self.unk_token) for i in ids]
        tokens = [t for t in tokens if t not in [self.unk_token, self.pad_token]]
        return ' '.join(tokens)

class CharacterLevelTokenizer:
    """字符級分詞器"""
    
    def __init__(self):
        self.pad_token = '<PAD>'
        self.unk_token = '<UNK>'
    
    def fit(self, texts):
        chars = set()
        for text in texts:
            chars.update(set(text))
        
        self.vocab = [self.pad_token, self.unk_token] + sorted(list(chars))
        self.token2id = {t: i for i, t in enumerate(self.vocab)}
        self.id2token = {i: t for t, i in self.token2id.items()}
    
    def encode(self, text, max_length=None):
        ids = [self.token2id.get(c, self.token2id[self.unk_token]) for c in text]
        
        if max_length:
            if len(ids) > max_length:
                ids = ids[:max_length]
            else:
                ids = ids + [0] * (max_length - len(ids))
        
        return ids
    
    def decode(self, ids):
        return ''.join([self.id2token.get(i, self.unk_token) for i in ids])
```

### 子詞分詞（Byte Pair Encoding）

```python
class BPE:
    """Byte Pair Encoding (BPE) 分詞器"""
    
    def __init__(self, vocab_size=10000):
        self.vocab_size = vocab_size
        self.merges = []
    
    def _get_stats(self, vocab):
        """統計相鄰 token 對的頻率"""
        pairs = Counter()
        for word, freq in vocab.items():
            symbols = word.split()
            for i in range(len(symbols) - 1):
                pairs[(symbols[i], symbols[i+1])] += freq
        return pairs
    
    def _merge_vocab(self, vocab, best_pair):
        """合併最常見的 token 對"""
        out_vocab = {}
        bigram = re.escape(' '.join(best_pair))
        p = re.compile(r'(?<!\S)' + bigram + r'(?!\S)')
        
        for word, freq in vocab.items():
            out_vocab[p.sub(''.join(best_pair), word)] = freq
        
        return out_vocab
    
    def fit(self, texts):
        vocab = Counter()
        for text in texts:
            words = text.lower().split()
            for word in words:
                vocab[' '.join(list(word)) + ' </w>'] += 1
        
        while len(vocab) > self.vocab_size:
            pairs = self._get_stats(vocab)
            if not pairs:
                break
            
            best_pair = max(pairs, key=pairs.get)
            vocab = self._merge_vocab(vocab, best_pair)
            self.merges.append(best_pair)
        
        # 構建詞彙表
        self.vocab = set()
        for word in vocab:
            self.vocab.update(word.split())
        self.vocab = sorted(self.vocab)
    
    def tokenize(self, word):
        """分詞一個單詞"""
        symbols = list(word) + ['</w>']
        
        for merge in self.merges:
            for i in range(len(symbols) - 1):
                if symbols[i] + symbols[i+1] == ''.join(merge):
                    symbols[i:i+2] = [''.join(merge)]
        
        return symbols[:-1]  # 移除 </w>
```

## 詞嵌入

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class Word2VecSkipGram(nn.Module):
    """Word2Vec Skip-Gram 模型"""
    
    def __init__(self, vocab_size, embedding_dim):
        super().__init__()
        self.embeddings = nn.Embedding(vocab_size, embedding_dim)
        self.context_embeddings = nn.Embedding(vocab_size, embedding_dim)
        
        # 初始化
        nn.init.uniform_(self.embeddings.weight, -0.5/embedding_dim, 0.5/embedding_dim)
        nn.init.zeros_(self.context_embeddings.weight)
    
    def forward(self, target, context, neg_samples):
        """
        target: (batch,) 目標詞 ID
        context: (batch,) 上下文詞 ID
        neg_samples: (batch, num_neg) 負採樣 ID
        """
        # 正樣本
        target_emb = self.embeddings(target)  # (batch, dim)
        context_emb = self.context_embeddings(context)  # (batch, dim)
        pos_score = torch.sum(target_emb * context_emb, dim=1)  # (batch,)
        pos_loss = F.logsigmoid(pos_score)
        
        # 負樣本
        neg_emb = self.context_embeddings(neg_samples)  # (batch, num_neg, dim)
        neg_score = torch.bmm(neg_emb, target_emb.unsqueeze(2)).squeeze()  # (batch, num_neg)
        neg_loss = F.logsigmoid(-neg_score).sum(dim=1)
        
        return -(pos_loss + neg_loss).mean()

class GloVeLoss(nn.Module):
    """GloVe 損失函數"""
    
    def __init__(self, vocab_size, embedding_dim, x_max=100, alpha=0.75):
        super().__init__()
        self.target_embeddings = nn.Embedding(vocab_size, embedding_dim)
        self.context_embeddings = nn.Embedding(vocab_size, embedding_dim)
        self.target_bias = nn.Embedding(vocab_size, 1)
        self.context_bias = nn.Embedding(vocab_size, 1)
        self.x_max = x_max
        self.alpha = alpha
    
    def weighting_function(self, x):
        """GloVe 的加權函數"""
        return torch.where(
            x < self.x_max,
            (x / self.x_max) ** self.alpha,
            torch.ones_like(x)
        )
    
    def forward(self, i, j, x_ij):
        """
        i: 目標詞 ID (batch,)
        j: 上下文詞 ID (batch,)
        x_ij: 共現頻率的對數 (batch,)
        """
        w_i = self.target_embeddings(i)
        w_j = self.context_embeddings(j)
        b_i = self.target_bias(i).squeeze()
        b_j = self.context_bias(j).squeeze()
        
        # 預測值
        prediction = torch.sum(w_i * w_j, dim=1) + b_i + b_j
        
        # 加權 MSE
        weight = self.weighting_function(x_ij.exp())
        loss = weight * (prediction - x_ij) ** 2
        
        return loss.mean()

class PositionalEncoding(nn.Module):
    """Transformer 的位置編碼"""
    
    def __init__(self, d_model, max_len=5000, dropout=0.1):
        super().__init__()
        self.dropout = nn.Dropout(p=dropout)
        
        # 創建位置編碼矩陣
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(
            torch.arange(0, d_model, 2).float() * (-torch.log(torch.tensor(10000.0)) / d_model)
        )
        
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0)  # (1, max_len, d_model)
        
        self.register_buffer('pe', pe)
    
    def forward(self, x):
        """x: (batch, seq_len, d_model)"""
        x = x + self.pe[:, :x.size(1), :]
        return self.dropout(x)
```

## 序列模型

```python
class LSTMTextClassifier(nn.Module):
    """基於 LSTM 的文本分類器"""
    
    def __init__(self, vocab_size, embedding_dim, hidden_dim, num_layers, 
                 num_classes, dropout=0.5):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embedding_dim, padding_idx=0)
        self.lstm = nn.LSTM(
            embedding_dim, hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            bidirectional=True,
            dropout=dropout if num_layers > 1 else 0
        )
        
        self.dropout = nn.Dropout(dropout)
        self.fc = nn.Linear(hidden_dim * 2, num_classes)
    
    def forward(self, x):
        """
        x: (batch, seq_len)
        """
        # 嵌入
        embedded = self.embedding(x)  # (batch, seq_len, embed_dim)
        
        # LSTM
        lstm_out, (hidden, cell) = self.lstm(embedded)
        
        # 拼接最後一層的正向和反向隱狀態
        hidden = torch.cat((hidden[-2,:,:], hidden[-1,:,:]), dim=1)
        hidden = self.dropout(hidden)
        
        # 全連接層
        output = self.fc(hidden)
        return output

class Attention(nn.Module):
    """自注意力機制"""
    
    def __init__(self, hidden_dim):
        super().__init__()
        self.attention = nn.Linear(hidden_dim, 1)
    
    def forward(self, encoder_outputs):
        """
        encoder_outputs: (batch, seq_len, hidden_dim)
        """
        # 計算注意力權重
        attention_scores = self.attention(encoder_outputs)  # (batch, seq_len, 1)
        attention_weights = F.softmax(attention_scores, dim=1)  # (batch, seq_len, 1)
        
        # 加權求和
        context = torch.sum(attention_weights * encoder_outputs, dim=1)  # (batch, hidden_dim)
        
        return context, attention_weights
```

## Transformer 模型

```python
class TransformerClassifier(nn.Module):
    """Transformer 文本分類器"""
    
    def __init__(self, vocab_size, d_model=256, nhead=8, 
                 num_layers=6, dim_feedforward=1024, 
                 num_classes=2, dropout=0.1, max_len=512):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, d_model, padding_idx=0)
        self.pos_encoder = PositionalEncoding(d_model, max_len, dropout)
        
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model,
            nhead=nhead,
            dim_feedforward=dim_feedforward,
            dropout=dropout,
            batch_first=True
        )
        self.transformer_encoder = nn.TransformerEncoder(
            encoder_layer,
            num_layers=num_layers
        )
        
        self.fc = nn.Sequential(
            nn.Linear(d_model, d_model // 2),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(d_model // 2, num_classes)
        )
    
    def forward(self, x, mask=None):
        """
        x: (batch, seq_len)
        mask: (batch, seq_len) True for padding
        """
        embedded = self.embedding(x) * np.sqrt(self.embedding.embedding_dim)
        embedded = self.pos_encoder(embedded)
        
        if mask is not None:
            # Transformer 使用 1 表示需要屏蔽的位置
            src_key_padding_mask = mask
        
        encoded = self.transformer_encoder(embedded, src_key_padding_mask=mask)
        
        # 使用 [CLS] 標記或平均池化
        cls_output = encoded[:, 0, :]  # (batch, d_model)
        # 或者使用 mean pooling
        # pooled = encoded.mean(dim=1)
        
        return self.fc(cls_output)

class BERTEmbedding(nn.Module):
    """BERT 的嵌入層"""
    
    def __init__(self, vocab_size, d_model, max_len=512, dropout=0.1):
        super().__init__()
        self.token_embedding = nn.Embedding(vocab_size, d_model, padding_idx=0)
        self.position_embedding = nn.Embedding(max_len, d_model)
        self.segment_embedding = nn.Embedding(2, d_model)
        self.norm = nn.LayerNorm(d_model)
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, token_ids, segment_ids=None):
        """
        token_ids: (batch, seq_len)
        segment_ids: (batch, seq_len) 句子對中的句子標識
        """
        seq_len = token_ids.size(1)
        position_ids = torch.arange(seq_len, dtype=torch.long, device=token_ids.device)
        position_ids = position_ids.unsqueeze(0).expand_as(token_ids)
        
        if segment_ids is None:
            segment_ids = torch.zeros_like(token_ids)
        
        token_emb = self.token_embedding(token_ids)
        position_emb = self.position_embedding(position_ids)
        segment_emb = self.segment_embedding(segment_ids)
        
        embeddings = token_emb + position_emb + segment_emb
        embeddings = self.norm(embeddings)
        embeddings = self.dropout(embeddings)
        
        return embeddings
```

# 損失函數：機器學習優化的核心

## 概述

損失函數（Loss Function）是機器學習中衡量模型預測與實際值之間差距的函數，它是模型訓練過程中被最小化的目標函數。選擇正確的損失函數對於模型能否正確學習至關重要——它定義了學習的目標，直接影響模型的行為和最終性能。不同的任務類型需要不同的損失函數：回歸任務通常使用均方誤差（MSE）或平均絕對誤差（MAE）；分類任務則常用交叉熵（Cross-Entropy）或 hinge loss；更複雜的任務如目標檢測、語義分割則需要組合多個損失函數。

現代深度學習中，損失函數的設計往往決定了模型能否正確收斂。例如，在類別不平衡的分類問題中，使用標準交叉熵會導致模型傾向於預測多數類；這時需要使用加權交叉熵或 focal loss 來解決。近年來，對抗訓練、對比學習和自監督學習等新範式也催生了 hinge loss、triplet loss、InfoNCE loss 等新型損失函數。

## 回歸損失函數

### 均方誤差及其變體

```python
import numpy as np
import torch
import torch.nn as nn
import torch.nn.functional as F

# ===== NumPy 實現 =====

def mse_loss(y_true, y_pred):
    """均方誤差 (Mean Squared Error)"""
    return np.mean((y_true - y_pred) ** 2)

def rmse_loss(y_true, y_pred):
    """均方根誤差 (Root Mean Squared Error)"""
    return np.sqrt(mse_loss(y_true, y_pred))

def mae_loss(y_true, y_pred):
    """平均絕對誤差 (Mean Absolute Error)"""
    return np.mean(np.abs(y_true - y_pred))

def huber_loss(y_true, y_pred, delta=1.0):
    """
    Huber Loss - MSE 和 MAE 的組合
    當誤差小於 delta 時使用 MSE，大於 delta 時使用 MAE
    兼顧了 MSE 的平滑梯度和 MAE 的魯棒性
    """
    error = y_true - y_pred
    abs_error = np.abs(error)
    
    quadratic = 0.5 * error ** 2
    linear = delta * abs_error - 0.5 * delta ** 2
    
    return np.mean(np.where(abs_error <= delta, quadratic, linear))

def smooth_l1_loss(y_true, y_pred, beta=1.0):
    """
    Smooth L1 Loss (也稱為 Huber Loss 的一種)
    在 Fast R-CNN 等物體檢測模型中使用
    """
    error = np.abs(y_true - y_pred)
    quadratic = error ** 2 / (2 * beta)
    linear = error - beta / 2
    
    return np.mean(np.where(error < beta, quadratic, linear))

def quantile_loss(y_true, y_pred, quantile=0.5):
    """
    分位數損失 - 用於分位數回歸
    quantile=0.5 時等價於 MAE
    """
    errors = y_true - y_pred
    return np.mean(np.maximum(quantile * errors, (quantile - 1) * errors))

def log_cosh_loss(y_true, y_pred):
    """
    Log-Cosh Loss - 平滑版本的 MAE
    梯度計算比 Huber Loss 更穩定
    """
    error = y_true - y_pred
    return np.mean(np.log(np.cosh(error)))

# ===== PyTorch 實現 =====

class RegressionLosses:
    """回歸任務的 PyTorch 損失函數集合"""
    
    @staticmethod
    def mse(pred, target):
        """均方誤差"""
        return F.mse_loss(pred, target)
    
    @staticmethod
    def mae(pred, target):
        """平均絕對誤差"""
        return F.l1_loss(pred, target)
    
    @staticmethod
    def huber(pred, target, delta=1.0):
        """Huber Loss"""
        return F.huber_loss(pred, target, delta=delta)
    
    @staticmethod
    def smooth_l1(pred, target, beta=1.0):
        """Smooth L1 (Huber) Loss"""
        return F.smooth_l1_loss(pred, target, beta=beta)
    
    @staticmethod
    def quantile(pred, target, quantile=0.5):
        """分位數損失"""
        errors = target - pred
        return torch.mean(torch.max(quantile * errors, (quantile - 1) * errors))
    
    @staticmethod
    def log_cosh(pred, target):
        """Log-Cosh Loss"""
        error = target - pred
        return torch.mean(torch.log(torch.cosh(error)))
    
    @staticmethod
    def berhu(pred, target, threshold=0.2):
        """
        Reverse Huber (BerHu) Loss
        常用於深度估計和語義分割
        """
        error = torch.abs(target - pred)
        c = threshold * torch.max(error)
        
        mask_small = error <= c
        mask_large = error > c
        
        loss = torch.where(
            mask_small,
            error,
            (error ** 2 + c ** 2) / (2 * c)
        )
        return torch.mean(loss)
```

# ===== NumPy 實現 =====

def binary_cross_entropy(y_true, y_pred, eps=1e-7):
    """
    二元交叉熵 (Binary Cross-Entropy)
    適用於二分類問題
    """
    y_pred = np.clip(y_pred, eps, 1 - eps)
    return -np.mean(
        y_true * np.log(y_pred) + 
        (1 - y_true) * np.log(1 - y_pred)
    )

def categorical_cross_entropy(y_true, y_pred, eps=1e-7):
    """
    類別交叉熵 (Categorical Cross-Entropy)
    適用於多分類問題
    y_true: one-hot 編碼
    y_pred: 機率分佈
    """
    y_pred = np.clip(y_pred, eps, 1 - eps)
    return -np.sum(y_true * np.log(y_pred)) / y_true.shape[0]

def sparse_categorical_cross_entropy(y_true, y_pred, eps=1e-7):
    """
    稀疏類別交叉熵
    y_true: 類別索引（非 one-hot）
    """
    y_pred = np.clip(y_pred, eps, 1 - eps)
    n_samples = len(y_true)
    correct_logprobs = -np.log(y_pred[np.arange(n_samples), y_true])
    return np.mean(correct_logprobs)

def focal_loss(y_true, y_pred, gamma=2.0, alpha=0.25, eps=1e-7):
    """
    Focal Loss - 解決類別不平衡問題
    減少對易分類樣本的關注，增加對難分類樣本的關注
    """
    y_pred = np.clip(y_pred, eps, 1 - eps)
    
    pt = np.where(y_true == 1, y_pred, 1 - y_pred)
    alpha_t = np.where(y_true == 1, alpha, 1 - alpha)
    
    focal_weight = alpha_t * (1 - pt) ** gamma
    bce = -np.log(pt)
    
    return np.mean(focal_weight * bce)

# ===== PyTorch 實現 =====

    @staticmethod
    def focal_loss(pred, target, gamma=2.0, alpha=0.25):
        """Focal Loss"""
        ce_loss = F.cross_entropy(pred, target, reduction='none')
        pt = torch.exp(-ce_loss)
        focal_weight = (1 - pt) ** gamma
        loss = alpha * focal_weight * ce_loss
        return loss.mean()
    
    @staticmethod
    def class_balanced_loss(pred, target, beta=0.9999, num_classes=10):

        Class-Balanced Loss - 基於有效樣本數的加權
        解決類別不平衡問題

        samples_per_class = torch.bincount(target, minlength=num_classes)

        weights = (1.0 - beta) / effective_num
        weights = weights / weights.sum() * num_classes

# 類別加權交叉熵
class WeightedCrossEntropyLoss(nn.Module):
    def __init__(self, weights=None, ignore_index=-100):
        """
        weights: 每個類別的權重
        """
        super().__init__()
        self.weights = weights
        self.ignore_index = ignore_index
    
    def forward(self, pred, target):
        """
        pred: (N, C, ...) 未歸一化的分數
        target: (N, ...) 類別索引
        """
        log_prob = F.log_softmax(pred, dim=1)
        
        if self.weights is not None:
            weights = self.weights.to(pred.device)
            weights = weights.unsqueeze(0).unsqueeze(-1).expand_as(target)
            loss = -weights * log_prob.gather(1, target.unsqueeze(1)).squeeze(1)
        else:
            loss = -log_prob.gather(1, target.unsqueeze(1)).squeeze(1)
        
        if self.ignore_index >= 0:
            mask = target != self.ignore_index
            loss = loss[mask]
        
        return loss.mean()

# 自動計算類別權重
def compute_class_weights(labels, num_classes, scheme='balanced'):

    from sklearn.utils.class_weight import compute_class_weight
    
    class_weights = compute_class_weight(
        class_weight=scheme,
        classes=np.unique(labels),
        y=labels
    )
    return torch.tensor(class_weights, dtype=torch.float32)

class ContrastiveLoss(nn.Module):
    """對比損失 - 用於度量學習"""
    
    def __init__(self, margin=1.0):
        super().__init__()
        self.margin = margin
    
    def forward(self, embeddings1, embeddings2, labels):
        """
        embeddings1, embeddings2: 兩個視圖的嵌入
        labels: 是否為正樣本對 (1: 正, 0: 負)
        """
        distances = F.pairwise_distance(embeddings1, embeddings2)
        
        # 正樣本：距離越小越好
        # 負樣本：距離大於 margin
        losses = labels * distances.pow(2) + \
                 (1 - labels) * torch.clamp(self.margin - distances, min=0).pow(2)
        
        return losses.mean()

class TripletLoss(nn.Module):
    """三元組損失"""
    
    def __init__(self, margin=1.0):
        super().__init__()
        self.margin = margin
    
    def forward(self, anchor, positive, negative):
        """
        anchor: 錨點嵌入
        positive: 正樣本嵌入
        negative: 負樣本嵌入
        """
        pos_dist = F.pairwise_distance(anchor, positive)
        neg_dist = F.pairwise_distance(anchor, negative)
        
        losses = torch.clamp(pos_dist - neg_dist + self.margin, min=0)
        return losses.mean()

class NTXentLoss(nn.Module):
    """NT-Xent Loss - 用於對比學習 (SimCLR)"""
    
    def __init__(self, temperature=0.5):
        super().__init__()
        self.tau = temperature
    
    def forward(self, z_i, z_j):
        """
        z_i, z_j: 兩個增強視圖的嵌入
        """
        batch_size = z_i.shape[0]
        
        # 拼接正樣本對
        z = torch.cat([z_i, z_j], dim=0)
        
        # 計算相似度矩陣
        sim = F.cosine_similarity(z.unsqueeze(1), z.unsqueeze(0), dim=2) / self.tau
        
        # 創建標籤：正樣本在對角線上
        labels = torch.cat([
            torch.arange(batch_size),
            torch.arange(batch_size)
        ]).to(z.device)
        
        # 只計算正樣本對的 loss
        loss = F.cross_entropy(sim, labels)
        return loss

class InfoNCELoss(nn.Module):
    """InfoNCE Loss - 用於自監督學習"""
    
    def __init__(self, temperature=0.07):
        super().__init__()
        self.tau = temperature
    
    def forward(self, query, positive_keys, negative_keys):
        """
        query: 查詢嵌入
        positive_keys: 正樣本鍵
        negative_keys: 負樣本鍵
        """
        all_keys = torch.cat([positive_keys, negative_keys], dim=1)
        
        # 計算 query 和 keys 的相似度
        logits = torch.matmul(query, all_keys.T) / self.tau
        
        # 正樣本在第一列
        labels = torch.zeros(query.size(0), dtype=torch.long).to(query.device)
        
        return F.cross_entropy(logits, labels)

class CenterLoss(nn.Module):
    """Center Loss - 增加類別內聚性"""
    
    def __init__(self, num_classes, feat_dim, size_average=True):
        super().__init__()
        self.centers = nn.Parameter(torch.randn(num_classes, feat_dim))
        self.size_average = size_average
    
    def forward(self, features, labels):
        """
        features: 特徵嵌入
        labels: 類別標籤
        """
        batch_size = features.size(0)
        features_dim = features.size(1)
        
        # 計算每個樣本到其類別中心的距離
        centers_batch = self.centers.index_select(0, labels.long())
        loss = (features - centers_batch).pow(2).sum() / 2.0
        
        if self.size_average:
            loss /= batch_size
        
        return loss

## 多任務損失

```python
class MultiTaskLoss(nn.Module):
    """多任務學習的損失組合"""
    
    def __init__(self, task_weights=None, loss_types=None):
        super().__init__()
        self.task_weights = task_weights or [1.0, 1.0]
        self.loss_types = loss_types or ['cross_entropy', 'mse']
        self.log_vars = nn.Parameter(torch.zeros(len(self.task_weights)))
    
    def forward(self, predictions, targets):
        """
        predictions: 任務預測列表
        targets: 任務目標列表
        """
        losses = []
        for i, (pred, target) in enumerate(zip(predictions, targets)):
            if self.loss_types[i] == 'cross_entropy':
                loss = F.cross_entropy(pred, target)
            elif self.loss_types[i] == 'mse':
                loss = F.mse_loss(pred, target)
            elif self.loss_types[i] == 'l1':
                loss = F.l1_loss(pred, target)
            losses.append(loss)
        
        # 可學習的任務權重
        precision = torch.exp(-self.log_vars)
        loss = sum(
            precision[i] * losses[i] + self.log_vars[i]
            for i in range(len(losses))
        )
        
        return loss, losses

class UncertaintyWeightedLoss(nn.Module):
    """基於任務不確定性的加權"""
    
    def __init__(self, num_tasks):
        super().__init__()
        # 每個任務的 log 精度
        self.log_vars = nn.Parameter(torch.zeros(num_tasks))
    
    def forward(self, losses):
        """
        losses: 各任務的損失列表
        """
        losses = torch.stack(losses)
        precision = torch.exp(-self.log_vars)
        
        weighted_loss = precision * losses + self.log_vars
        return weighted_loss.sum()
```

class IoULoss(nn.Module):
    """IoU Loss"""
    
    def __init__(self, smooth=1e-6):
        super().__init__()
        self.smooth = smooth
    
    def forward(self, pred, target):
        """
        pred: (B, C, H, W) 預測 logits
        target: (B, H, W) 類別索引
        """
        # 軟 IoU
        pred = F.softmax(pred, dim=1)
        
        # 轉換為 one-hot
        target_one_hot = F.one_hot(target, num_classes=pred.size(1))
        target_one_hot = target_one_hot.permute(0, 3, 1, 2).float()
        
        # 計算 IoU
        intersection = (pred * target_one_hot).sum(dim=(2, 3))
        union = pred.sum(dim=(2, 3)) + target_one_hot.sum(dim=(2, 3)) - intersection
        
        iou = (intersection + self.smooth) / (union + self.smooth)
        return 1 - iou.mean()

# 示例：Dice Loss
class DiceLoss(nn.Module):
    """Dice Loss"""
    
    def __init__(self, smooth=1e-6):
        super().__init__()
        self.smooth = smooth
    
    def forward(self, pred, target):
        pred = F.softmax(pred, dim=1)
        target_one_hot = F.one_hot(target, num_classes=pred.size(1))
        target_one_hot = target_one_hot.permute(0, 3, 1, 2).float()
        
        intersection = (pred * target_one_hot).sum(dim=(2, 3))
        dice = (2. * intersection + self.smooth) / (
            pred.sum(dim=(2, 3)) + target_one_hot.sum(dim=(2, 3)) + self.smooth
        )
        return 1 - dice.mean()

## 損失函數選擇指南

```python
# 損失函數選擇決策樹
"""
任務類型 → 損失函數選擇

├── 分類
│   ├── 二分類 → Binary Cross-Entropy / Focal Loss (不平衡)
│   ├── 多分類 → Categorical Cross-Entropy / Label Smoothing
│   └── 多標籤 → BCE with Logits
│
├── 回歸
│   ├── 標準回歸 → MSE / MAE
│   ├── 對異常值魯棒 → Huber / Smooth L1
│   ├── 分位數預測 → Quantile Loss
│   └── 稀疏回歸 → L1 (MAE)
│
├── 度量學習
│   ├── 人臉識別 → Triplet Loss / Contrastive Loss
│   ├── 對比學習 → NT-Xent / InfoNCE
│   └── 聚類 → Center Loss
│
├── 語義分割
│   ├── 通用 → Cross-Entropy + Dice Loss
│   └── 不平衡類別 → Focal Loss + IoU Loss
│
└── 目標檢測
    ├── 檢測框回歸 → Smooth L1 / IoU Loss / GIoU
    ├── 類別分類 → Focal Loss
    └── 置信度回歸 → BCE
"""
```

## 總結

選擇正確的損失函數是機器學習成功的關鍵。損失函數決定了模型學習的目標，直接影響收斂速度和最終性能。本頁介紹了主要的損失函數類型：回歸任務的 MSE、MAE、Huber Loss；分類任務的交叉熵、Focal Loss；度量學習的對比損失；以及多任務學習的損失組合。在實際應用中，經常需要組合多個損失函數，並根據任務特點調整權重。

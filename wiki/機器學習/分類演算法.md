# 分類演算法 (Classification Algorithms)

## 概述

分類是機器學習中最常見的監督式學習任務之一，其目標是根據輸入特徵將樣本分配到預定義的類別中。分類問題廣泛應用於垃圾郵件檢測、疾病診斷、圖像識別、信用風險評估等場景。根據類別數量，分類可分為二分類（Binary Classification）和多分類（Multi-class Classification）；根據類別是否互斥，還可分為單標籤和多標籤分類。

分類演算法的發展經歷了從傳統計數方法到現代深度學習的漫長歷程。1960 年代感知機的出現開啟了線性分類的時代，此後支持向量機、決策樹、集成學習等方法相繼問世，構成了現代機器學習分類的堅實基礎。

## 邏輯回歸

### 原理

邏輯回歸（Logistic Regression）雖然名稱包含「回歸」，但實際上是一個分類演算法。它使用 Sigmoid 函數將線性組合映射到 [0, 1] 區間，表示屬於正類的機率：

$$P(y=1|x) = \frac{1}{1 + e^{-(\mathbf{w}^T \mathbf{x} + b)}}$$

```python
import numpy as np

class LogisticRegression:
    """邏輯回歸實現"""
    
    def __init__(self, learning_rate=0.01, n_iterations=1000):
        self.lr = learning_rate
        self.n_iterations = n_iterations
        self.weights = None
        self.bias = None
    
    def sigmoid(self, z):
        """Sigmoid 函數"""
        return 1 / (1 + np.exp(-np.clip(z, -500, 500)))
    
    def fit(self, X, y):
        n_samples, n_features = X.shape
        
        # 初始化參數
        self.weights = np.zeros(n_features)
        self.bias = 0
        
        # 梯度下降
        for _ in range(self.n_iterations):
            # 線性組合
            linear = np.dot(X, self.weights) + self.bias
            # Sigmoid 激活
            y_pred = self.sigmoid(linear)
            
            # 梯度
            dw = (1 / n_samples) * np.dot(X.T, (y_pred - y))
            db = (1 / n_samples) * np.sum(y_pred - y)
            
            # 更新
            self.weights -= self.lr * dw
            self.bias -= self.lr * db
    
    def predict_proba(self, X):
        """預測機率"""
        linear = np.dot(X, self.weights) + self.bias
        return self.sigmoid(linear)
    
    def predict(self, X, threshold=0.5):
        """預測類別"""
        return (self.predict_proba(X) >= threshold).astype(int)


class MultinomialLogisticRegression:
    """多元邏輯回歸 (Softmax)"""
    
    def __init__(self, learning_rate=0.01, n_iterations=1000):
        self.lr = learning_rate
        self.n_iterations = n_iterations
        self.weights = None
        self.bias = None
        self.num_classes = None
    
    def softmax(self, z):
        """Softmax 函數"""
        exp_z = np.exp(z - np.max(z, axis=1, keepdims=True))
        return exp_z / np.sum(exp_z, axis=1, keepdims=True)
    
    def fit(self, X, y):
        n_samples, n_features = X.shape
        self.num_classes = len(np.unique(y))
        
        # 初始化參數
        self.weights = np.random.randn(n_features, self.num_classes) * 0.01
        self.bias = np.zeros(self.num_classes)
        
        for _ in range(self.n_iterations):
            # Softmax
            z = np.dot(X, self.weights) + self.bias
            probs = self.softmax(z)
            
            # 梯度 (Cross-Entropy)
            one_hot = np.zeros((n_samples, self.num_classes))
            one_hot[np.arange(n_samples), y] = 1
            
            dw = np.dot(X.T, (probs - one_hot)) / n_samples
            db = np.sum(probs - one_hot, axis=0) / n_samples
            
            self.weights -= self.lr * dw
            self.bias -= self.lr * db
    
    def predict(self, X):
        z = np.dot(X, self.weights) + self.bias
        probs = self.softmax(z)
        return np.argmax(probs, axis=1)
```

## 支持向量機 (SVM)

### 原理

支持向量機（Support Vector Machine）找到一個最大化類別間隔的超平面：

```python
class SVM:
    """支持向量機實現 (SMO 簡化版)"""
    
    def __init__(self, C=1.0, kernel='linear', gamma='scale'):
        self.C = C
        self.kernel = kernel
        self.gamma = gamma
        self.alphas = None
        self.b = 0
        self.X_train = None
        self.y_train = None
    
    def kernel_function(self, x1, x2):
        """核函數"""
        if self.kernel == 'linear':
            return np.dot(x1, x2)
        elif self.kernel == 'rbf':
            if self.gamma == 'scale':
                gamma = 1 / (x1.shape[0] * x1.var())
            return np.exp(-gamma * np.sum((x1 - x2) ** 2))
    
    def fit(self, X, y):
        self.X_train = X
        self.y_train = y
        
        n_samples, n_features = X.shape
        self.alphas = np.zeros(n_samples)
        
        # 簡化的 SMO 演算法
        for epoch in range(100):
            for i in range(n_samples):
                # 選擇第二個 alpha
                j = np.random.randint(0, n_samples)
                
                # 計算 Ei, Ej
                def compute_E(idx):
                    return np.sum(self.alphas * self.y_train * 
                                np.array([self.kernel_function(self.X_train[i], self.X_train[j]) 
                                         for j in range(n_samples)])) + self.b - self.y_train[i]
                
                Ei = compute_E(i)
                Ej = compute_E(j)
                
                # 計算新的 alpha
                if self.y_train[i] != self.y_train[j]:
                    L = max(0, self.alphas[j] - self.alphas[i])
                    H = min(self.C, self.C + self.alphas[j] - self.alphas[i])
                else:
                    L = max(0, self.alphas[i] + self.alphas[j] - self.C)
                    H = min(self.C, self.alphas[i] + self.alphas[j])
                
                if L == H:
                    continue
                
                eta = 2 * self.kernel_function(X[i], X[j]) - \
                      self.kernel_function(X[i], X[i]) - \
                      self.kernel_function(X[j], X[j])
                
                if eta >= 0:
                    continue
                
                new_alpha_j = self.alphas[j] - (self.y_train[j] * (Ei - Ej)) / eta
                new_alpha_j = np.clip(new_alpha_j, L, H)
                
                # 更新
                self.alphas[j] = new_alpha_j
                
                # 更新 b
                b1 = self.b - Ei - self.y_train[i] * (self.alphas[i] - self.alphas[j].item()) * \
                     self.kernel_function(X[i], X[i]) - \
                     self.y_train[j] * (self.alphas[j] - self.alphas[j].item()) * \
                     self.kernel_function(X[i], X[j])
                
                b2 = self.b - Ej - self.y_train[i] * (self.alphas[i] - self.alphas[j].item()) * \
                     self.kernel_function(X[i], X[j]) - \
                     self.y_train[j] * (self.alphas[j] - self.alphas[j].item()) * \
                     self.kernel_function(X[j], X[j])
                
                if 0 < self.alphas[j] < self.C:
                    self.b = (b1 + b2) / 2
                else:
                    self.b = b1
    
    def predict(self, X):
        decision = np.zeros(X.shape[0])
        for i in range(X.shape[0]):
            for j in range(len(self.alphas)):
                if self.alphas[j] > 0:
                    decision[i] += self.alphas[j] * self.y_train[j] * \
                                   self.kernel_function(X[i], self.X_train[j])
        decision += self.b
        return np.sign(decision)
```

### 核函數

```python
class KernelFunctions:
    """核函數"""
    
    @staticmethod
    def linear():
        """線性核"""
        return lambda x, y: np.dot(x, y)
    
    @staticmethod
    def polynomial(degree=3, coef0=1):
        """多項式核"""
        return lambda x, y: (np.dot(x, y) + coef0) ** degree
    
    @staticmethod
    def rbf(gamma='scale'):
        """徑向基函數核 (RBF/Gaussian)"""
        return lambda x, y: np.exp(-gamma * np.linalg.norm(x - y) ** 2)
    
    @staticmethod
    def sigmoid(alpha=1, coef0=0):
        """Sigmoid 核"""
        return lambda x, y: np.tanh(alpha * np.dot(x, y) + coef0)
```

## 決策樹

### 原理與實現

```python
class DecisionTreeClassifier:
    """決策樹分類器"""
    
    def __init__(self, max_depth=10, min_samples_split=2, criterion='gini'):
        self.max_depth = max_depth
        self.min_samples_split = min_samples_split
        self.criterion = criterion
        self.tree = None
    
    def gini_impurity(self, y):
        """計算 Gini 不純度"""
        if len(y) == 0:
            return 0
        counts = np.bincount(y)
        probs = counts / len(y)
        return 1 - np.sum(probs ** 2)
    
    def entropy(self, y):
        """計算熵"""
        if len(y) == 0:
            return 0
        counts = np.bincount(y)
        probs = counts / len(y)
        return -np.sum(probs * np.log2(probs + 1e-10))
    
    def find_best_split(self, X, y):
        """找到最佳分割"""
        best_gain = -1
        best_feature = None
        best_threshold = None
        
        n_features = X.shape[1]
        
        for feature in range(n_features):
            values = X[:, feature]
            thresholds = np.unique(values)
            
            for threshold in thresholds:
                left_mask = values <= threshold
                right_mask = ~left_mask
                
                if np.sum(left_mask) < self.min_samples_split or \
                   np.sum(right_mask) < self.min_samples_split:
                    continue
                
                # 計算資訊增益
                if self.criterion == 'gini':
                    parent_impurity = self.gini_impurity(y)
                    left_impurity = self.gini_impurity(y[left_mask])
                    right_impurity = self.gini_impurity(y[right_mask])
                else:
                    parent_impurity = self.entropy(y)
                    left_impurity = self.entropy(y[left_mask])
                    right_impurity = self.entropy(y[right_mask])
                
                n = len(y)
                gain = parent_impurity - \
                       (np.sum(left_mask) / n) * left_impurity - \
                       (np.sum(right_mask) / n) * right_impurity
                
                if gain > best_gain:
                    best_gain = gain
                    best_feature = feature
                    best_threshold = threshold
        
        return best_feature, best_threshold, best_gain
    
    def build_tree(self, X, y, depth=0):
        """遞迴建立樹"""
        n_samples = len(y)
        
        # 停止條件
        if depth >= self.max_depth or n_samples < self.min_samples_split:
            return {'leaf': True, 'class': np.bincount(y).argmax()}
        
        # 找到最佳分割
        feature, threshold, gain = self.find_best_split(X, y)
        
        if feature is None or gain <= 0:
            return {'leaf': True, 'class': np.bincount(y).argmax()}
        
        # 分割資料
        left_mask = X[:, feature] <= threshold
        right_mask = ~left_mask
        
        # 遞迴建立子樹
        left_tree = self.build_tree(X[left_mask], y[left_mask], depth + 1)
        right_tree = self.build_tree(X[right_mask], y[right_mask], depth + 1)
        
        return {
            'leaf': False,
            'feature': feature,
            'threshold': threshold,
            'left': left_tree,
            'right': right_tree
        }
    
    def fit(self, X, y):
        self.tree = self.build_tree(X, y)
        return self
    
    def predict_single(self, x, node):
        """單個樣本預測"""
        if node['leaf']:
            return node['class']
        
        if x[node['feature']] <= node['threshold']:
            return self.predict_single(x, node['left'])
        else:
            return self.predict_single(x, node['right'])
    
    def predict(self, X):
        return np.array([self.predict_single(x, self.tree) for x in X])
```

### 隨機森林

```python
class RandomForestClassifier:
    """隨機森林分類器"""
    
    def __init__(self, n_estimators=100, max_depth=10, 
                 min_samples_split=2, n_features='sqrt'):
        self.n_estimators = n_estimators
        self.max_depth = max_depth
        self.min_samples_split = min_samples_split
        self.n_features = n_features
        self.trees = []
    
    def fit(self, X, y):
        self.trees = []
        n_samples = X.shape[0]
        
        # 自助採樣 (Bootstrap)
        for _ in range(self.n_estimators):
            indices = np.random.choice(n_samples, n_samples, replace=True)
            X_bootstrap = X[indices]
            y_bootstrap = y[indices]
            
            # 隨機選擇特徵
            if self.n_features == 'sqrt':
                n_feats = int(np.sqrt(X.shape[1]))
            else:
                n_feats = X.shape[1]
            
            # 訓練決策樹
            tree = DecisionTreeClassifier(
                max_depth=self.max_depth,
                min_samples_split=self.min_samples_split
            )
            tree.fit(X_bootstrap, y_bootstrap)
            self.trees.append(tree)
        
        return self
    
    def predict(self, X):
        # 投票
        predictions = np.array([tree.predict(X) for tree in self.trees])
        # 多数投票
        return np.apply_along_axis(
            lambda x: np.bincount(x).argmax(), 
            axis=0, 
            arr=predictions
        )
```

### 梯度提升 (Gradient Boosting)

```python
class GradientBoostingClassifier:
    """梯度提升分類器"""
    
    def __init__(self, n_estimators=100, learning_rate=0.1, max_depth=3):
        self.n_estimators = n_estimators
        self.learning_rate = learning_rate
        self.max_depth = max_depth
        self.trees = []
        self.initial_pred = None
    
    def fit(self, X, y):
        # 初始化預測值 (log-odds)
        self.initial_pred = np.log(np.bincount(y) / len(y))
        self.initial_pred = np.array([self.initial_pred[0], self.initial_pred[1]])
        
        # 初始預測
        current_pred = np.full(len(y), self.initial_pred[y])
        
        for _ in range(self.n_estimators):
            # 計算 pseudo-residuals
            prob = 1 / (1 + np.exp(-current_pred))
            residuals = y - prob
            
            # 擬合殘差
            tree = DecisionTreeClassifier(max_depth=self.max_depth)
            tree.fit(X, residuals)
            
            # 更新預測
            tree_pred = tree.predict(X)
            current_pred += self.learning_rate * tree_pred
            
            self.trees.append(tree)
        
        return self
    
    def predict_proba(self, X):
        # 初始預測
        prob = np.full((len(X), 2), self.initial_pred)
        
        # 加上所有樹的預測
        for tree in self.trees:
            prob[:, 1] += self.learning_rate * tree.predict(X)
            prob[:, 0] -= self.learning_rate * tree.predict(X)
        
        # Sigmoid
        prob = 1 / (1 + np.exp(-prob))
        return prob
    
    def predict(self, X):
        prob = self.predict_proba(X)
        return (prob[:, 1] > 0.5).astype(int)
```

## 評估指標

### 混淆矩陣

```python
from sklearn.metrics import confusion_matrix, accuracy_score, precision_score, recall_score, f1_score

def evaluate_classification(y_true, y_pred):
    """分類評估"""
    # 混淆矩陣
    cm = confusion_matrix(y_true, y_pred)
    
    # 準確率
    accuracy = accuracy_score(y_true, y_pred)
    
    # 精確率 (Precision)
    precision = precision_score(y_true, y_pred, average='weighted')
    
    # 召回率 (Recall)
    recall = recall_score(y_true, y_pred, average='weighted')
    
    # F1 分數
    f1 = f1_score(y_true, y_pred, average='weighted')
    
    print(f"混淆矩陣:\n{cm}")
    print(f"準確率: {accuracy:.4f}")
    print(f"精確率: {precision:.4f}")
    print(f"召回率: {recall:.4f}")
    print(f"F1 分數: {f1:.4f}")
    
    return {'accuracy': accuracy, 'precision': precision, 'recall': recall, 'f1': f1}
```

## 演算法比較

| 演算法 | 優點 | 缺點 | 適用場景 |
|--------|------|------|----------|
| 邏輯回歸 | 簡單、快速、可解釋 | 只能處理線性問題 | 基線模型、需要可解釋性 |
| SVM | 高維度表現好、泛化強 | 大資料集訓練慢 | 小中型資料、高維度 |
| 決策樹 | 可解釋、處理非線性 | 容易過擬合 | 特徵重要性分析 |
| 隨機森林 | 抗過擬合、處理大型資料 | 訓練時間長 | 大型資料集、分類/回歸 |
| 梯度提升 | 精度高、處理各類資料 | 訓練時間長、容易過擬合 | Kaggle 競賽、生產環境 |

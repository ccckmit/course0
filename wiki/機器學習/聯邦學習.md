# 聯邦學習：保護隱私的分散式機器學習

## 概述

聯邦學習（Federated Learning）是一種分散式機器學習範式，允許在多個客戶端（如移動設備）上協作訓練模型，同時保持數據的本地化和私密性。在傳統的機器學習中，所有數據需要集中到一個伺服器上進行訓練；而在聯邦學習中，模型被分發到各個客戶端，在本地數據上訓練，只有模型更新（如梯度或權重）被傳回伺服器進行聚合。

聯邦學習的核心挑戰包括：**數據異質性**（各客戶端數據分佈不均）、**通訊效率**（網路頻寬有限）、**隱私保護**（防止梯度洩露）、以及**系統異構性**（設備計算能力差異）。本頁將介紹聯邦學習的基礎架構、主要演算法和Python實現。

## 聯邦平均演算法

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
from collections import OrderedDict

class FederatedAveraging:
    """聯邦平均 (FedAvg) 演算法"""
    
    def __init__(self, model, clients, global_model):
        self.model = model
        self.clients = clients
        self.global_model = global_model
        self.client_weights = [len(client['data']) for client in clients]
    
    def client_update(self, client_model, client_data, local_epochs=5, lr=0.01):
        """在客戶端本地更新模型"""
        client_model.train()
        optimizer = torch.optim.SGD(client_model.parameters(), lr=lr)
        criterion = nn.CrossEntropyLoss()
        
        data_loader = torch.utils.data.DataLoader(
            client_data, batch_size=32, shuffle=True
        )
        
        for epoch in range(local_epochs):
            for batch in data_loader:
                inputs, targets = batch
                optimizer.zero_grad()
                outputs = client_model(inputs)
                loss = criterion(outputs, targets)
                loss.backward()
                optimizer.step()
        
        return client_model.state_dict()
    
    def aggregate(self, client_states):
        """聚合客戶端模型更新"""
        total_weight = sum(self.client_weights)
        
        aggregated_state = OrderedDict()
        for key in client_states[0].keys():
            aggregated_state[key] = sum(
                client_states[i][key] * self.client_weights[i] / total_weight
                for i in range(len(client_states))
            )
        
        self.global_model.load_state_dict(aggregated_state)
        return self.global_model.state_dict()
    
    def train_round(self, num_clients=10, local_epochs=5, lr=0.01):
        """執行一輪聯邦學習"""
        # 選擇客戶端
        selected_indices = np.random.choice(
            len(self.clients), num_clients, replace=False
        )
        
        # 本地訓練
        client_states = []
        for idx in selected_indices:
            client_model = self.copy_model(self.global_model)
            state = self.client_update(
                client_model, 
                self.clients[idx]['data'],
                local_epochs, 
                lr
            )
            client_states.append(state)
        
        # 聚合
        return self.aggregate(client_states)
    
    def copy_model(self, model):
        """複製模型"""
        new_model = type(model)(**model.config)
        new_model.load_state_dict(model.state_dict())
        return new_model
```

## 差分隱私

```python
class DifferentialPrivacy:
    """差分隱私保護"""
    
    def __init__(self, noise_multiplier=1.0, max_grad_norm=1.0):
        self.noise_multiplier = noise_multiplier
        self.max_grad_norm = max_grad_norm
    
    def clip_gradients(self, model):
        """梯度裁剪"""
        total_norm = 0.0
        for p in model.parameters():
            if p.grad is not None:
                param_norm = p.grad.data.norm(2)
                total_norm += param_norm.item() ** 2
        total_norm = total_norm ** 0.5
        
        clip_coef = self.max_grad_norm / (total_norm + 1e-6)
        if clip_coef < 1:
            for p in model.parameters():
                if p.grad is not None:
                    p.grad.data.mul_(clip_coef)
        
        return total_norm
    
    def add_noise(self, model, sample_count):
        """添加高斯噪聲"""
        noise_multiplier = self.noise_multiplier * self.max_grad_norm
        sensitivity = 2 * self.max_grad_norm / sample_count
        
        for p in model.parameters():
            if p.grad is not None:
                noise = torch.randn_like(p.grad) * sensitivity * noise_multiplier
                p.grad.data.add_(noise)
        
        return model
    
    def private_update(self, model, sample_count):
        """帶差分隱私的更新"""
        self.clip_gradients(model)
        self.add_noise(model, sample_count)
        return model
```

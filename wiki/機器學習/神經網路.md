# 神經網路 (Neural Network)

## 概述

神經網路是受人類大腦神經元結構啟發的計算模型，是深度學習的基礎構建模組。1943 年，McCulloch 和 Pitts 提出了第一個人工神經元模型 MCP neuron。1958 年，Rosenblatt 發明了感知機（Perceptron），開創了機器學習的新時代。此後經歷了兩次寒冬，2012 年 AlexNet 在 ImageNet 競賽中取得突破性成績，神經網路再次成為 AI 領域的核心技術。

神經網路的核心思想是透過大量的簡單處理單元（神經元）的連接和學習，來表示複雜的函數關係。每個神經元接收輸入，進行加權求和，通過非線性啟動函數產生輸出。透過堆疊多層神經元形成深度網路，可以學習資料的層次化表示。

## 生物啟發

人類大腦包含約 860 億個神經元，每個神經元與數千個其他神經元相連，形成複雜的神經網絡。神經元透過突觸連接，當訊號超過閾值時會觸發脈衝，將訊號傳遞給下一個神經元。這種並行、分散的資訊處理方式啟發了人工神經網路的設計。

人工神經元模擬了生物神經元的基本功能：接收多個輸入、進行加權組合、通過非線性轉換產生輸出。多個人工神經元連接成網路，形成能夠學習複雜模式的計算系統。

### 前饋神經網路 (FNN)

前饋神經網路是最基本的網路結構，資訊從輸入層向前傳播到輸出層：

```python
import torch
import torch.nn as nn

class FeedForwardNetwork(nn.Module):
    """多層前饋神經網路"""
    
    def __init__(self, input_dim, hidden_dims, output_dim, activation='relu'):
        super().__init__()
        
        layers = []
        prev_dim = input_dim
        
        for hidden_dim in hidden_dims:
            layers.append(nn.Linear(prev_dim, hidden_dim))
            
            if activation == 'relu':
                layers.append(nn.ReLU())
            elif activation == 'sigmoid':
                layers.append(nn.Sigmoid())
            elif activation == 'tanh':
                layers.append(nn.Tanh())
            
            prev_dim = hidden_dim
        
        layers.append(nn.Linear(prev_dim, output_dim))
        
        self.network = nn.Sequential(*layers)
    
    def forward(self, x):
        return self.network(x)

# 使用範例
model = FeedForwardNetwork(784, [512, 256, 128], 10)
x = torch.randn(32, 784)  # batch=32, input=784
output = model(x)  # (32, 10)
```

### 卷積神經網路 (CNN)

CNN 專門處理網格結構資料（圖像），使用卷積運算捕捉空間模式：

```python
class ConvolutionalNetwork(nn.Module):
    """卷積神經網路"""
    
    def __init__(self, num_classes=10):
        super().__init__()
        
        # 卷積層
        self.conv_layers = nn.Sequential(
            # Conv Block 1
            nn.Conv2d(3, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.Conv2d(32, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.MaxPool2d(2),  # 32x32 -> 16x16
            
            # Conv Block 2
            nn.Conv2d(32, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.Conv2d(64, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.MaxPool2d(2),  # 16x16 -> 8x8
            
            # Conv Block 3
            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.Conv2d(128, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.MaxPool2d(2),  # 8x8 -> 4x4
        )
        
        # 全連接層
        self.fc_layers = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128 * 4 * 4, 256),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(256, num_classes)
        )
    
    def forward(self, x):
        x = self.conv_layers(x)
        x = self.fc_layers(x)
        return x
```

### 循環神經網路 (RNN)

RNN 處理序列資料，利用內部狀態記住之前的資訊：

```python
class RecurrentNetwork(nn.Module):
    """循環神經網路"""
    
    def __init__(self, input_size, hidden_size, num_layers, num_classes):
        super().__init__()
        
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        
        self.rnn = nn.RNN(
            input_size, 
            hidden_size, 
            num_layers, 
            batch_first=True,
            dropout=0.3
        )
        
        self.fc = nn.Linear(hidden_size, num_classes)
    
    def forward(self, x):
        # x: (batch, seq_len, input_size)
        
        # RNN 前向
        output, hidden = self.rnn(x)
        # output: (batch, seq_len, hidden_size)
        # hidden: (num_layers, batch, hidden_size)
        
        # 取最後時刻的輸出
        last_output = output[:, -1, :]  # (batch, hidden_size)
        
        # 分類
        logits = self.fc(last_output)
        return logits


class LSTMCell(nn.Module):
    """LSTM 單元"""
    
    def __init__(self, input_size, hidden_size):
        super().__init__()
        
        self.hidden_size = hidden_size
        
        # 四個門的線性層
        self gates = nn.Linear(input_size + hidden_size, 4 * hidden_size)
    
    def forward(self, x, hidden):
        h, c = hidden
        
        # 拼接輸入和隱藏狀態
        combined = torch.cat([x, h], dim=1)
        
        # 計算四個門
        gates = self.gates(combined)
        i, f, g, o = gates.chunk(4, dim=1)
        
        # LSTM 公式
        i = torch.sigmoid(i)
        f = torch.sigmoid(f)
        g = torch.tanh(g)
        o = torch.sigmoid(o)
        
        c_new = f * c + i * g
        h_new = o * torch.tanh(c_new)
        
        return h_new, c_new
```

### 神經元模型

```python
class Neuron(nn.Module):
    """單個神經元"""
    
    def __init__(self, input_dim):
        super().__init__()
        self.linear = nn.Linear(input_dim, 1)
        self.activation = nn.Sigmoid()
    
    def forward(self, x):
        return self.activation(self.linear(x))


class DenseLayer(nn.Module):
    """全連接層"""
    
    def __init__(self, input_dim, output_dim, activation=None):
        super().__init__()
        
        self.linear = nn.Linear(input_dim, output_dim)
        
        if activation == 'relu':
            self.activation = nn.ReLU()
        elif activation == 'sigmoid':
            self.activation = nn.Sigmoid()
        elif activation == 'tanh':
            self.activation = nn.Tanh()
        elif activation == 'softmax':
            self.activation = nn.Softmax(dim=-1)
        else:
            self.activation = None
    
    def forward(self, x):
        x = self.linear(x)
        if self.activation is not None:
            x = self.activation(x)
        return x
```

### 損失函數

```python
class LossFunctions:
    """常用損失函數"""
    
    @staticmethod
    def mse_loss(pred, target):
        """均方誤差"""
        return torch.mean((pred - target) ** 2)
    
    @staticmethod
    def cross_entropy_loss(pred, target):
        """交叉熵損失"""
        return torch.nn.functional.cross_entropy(pred, target)
    
    @staticmethod
    def binary_cross_entropy(pred, target):
        """二元交叉熵"""
        return torch.nn.functional.binary_cross_entropy(
            torch.sigmoid(pred), target
        )
    
    @staticmethod
    def contrastive_loss(anchor, positive, negative, margin=1.0):
        """對比損失"""
        pos_dist = torch.sum((anchor - positive) ** 2, dim=1)
        neg_dist = torch.sum((anchor - negative) ** 2, dim=1)
        return torch.relu(pos_dist - neg_dist + margin).mean()
```

## 訓練過程

### 前向傳播

```python
def forward_pass(model, x, y):
    """前向傳播"""
    # 清除梯度
    model.zero_grad()
    
    # 前向計算
    output = model(x)
    
    # 計算損失
    loss = nn.functional.cross_entropy(output, y)
    
    return output, loss
```

### 反向傳播

```python
def backward_pass(model, loss):
    """反向傳播"""
    loss.backward()
    
    # 更新參數
    learning_rate = 0.001
    with torch.no_grad():
        for param in model.parameters():
            param -= learning_rate * param.grad
```

### 完整訓練循環

```python
def train_neural_network(model, train_loader, num_epochs, learning_rate):
    """完整訓練循環"""
    
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=learning_rate)
    
    for epoch in range(num_epochs):
        total_loss = 0
        correct = 0
        total = 0
        
        for batch_x, batch_y in train_loader:
            # 前向傳播
            outputs = model(batch_x)
            loss = criterion(outputs, batch_y)
            
            # 反向傳播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
            _, predicted = outputs.max(1)
            total += batch_y.size(0)
            correct += predicted.eq(batch_y).sum().item()
        
        avg_loss = total_loss / len(train_loader)
        accuracy = 100 * correct / total
        
        if (epoch + 1) % 10 == 0:
            print(f'Epoch [{epoch+1}/{num_epochs}], Loss: {avg_loss:.4f}, Accuracy: {accuracy:.2f}%')
    
    return model
```

```python
class Regularization:
    """正則化技術"""
    
    # 1. L2 正則化 (Weight Decay)
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001, weight_decay=0.01)
    
    # 2. Dropout
    model = nn.Sequential(
        nn.Linear(784, 256),
        nn.ReLU(),
        nn.Dropout(0.5),  # 訓練時隨機丟棄 50% 神經元
        nn.Linear(256, 10)
    )
    
    # 3. Batch Normalization
    model = nn.Sequential(
        nn.Linear(784, 256),
        nn.BatchNorm1d(256),  # 批標準化
        nn.ReLU(),
        nn.Linear(256, 10)
    )
    
    # 4. Early Stopping
    class EarlyStopping:
        def __init__(self, patience=10, min_delta=0):
            self.patience = patience
            self.min_delta = min_delta
            self.counter = 0
            self.best_loss = None
        
        def __call__(self, val_loss):
            if self.best_loss is None:
                self.best_loss = val_loss
            elif val_loss > self.best_loss - self.min_delta:
                self.counter += 1
                if self.counter >= self.patience:
                    return True
            else:
                self.best_loss = val_loss
                self.counter = 0
            return False
```

## 網路初始化

```python
def initialize_weights(model, init_type='xavier'):
    """權重初始化"""
    
    for m in model.modules():
        if isinstance(m, nn.Linear):
            if init_type == 'xavier':
                nn.init.xavier_uniform_(m.weight)
            elif init_type == 'kaiming':
                nn.init.kaiming_uniform_(m.weight, nonlinearity='relu')
            elif init_type == 'normal':
                nn.init.normal_(m.weight, mean=0, std=0.01)
            
            if m.bias is not None:
                nn.init.zeros_(m.bias)
        
        elif isinstance(m, nn.Conv2d):
            nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
            if m.bias is not None:
                nn.init.zeros_(m.bias)
```

## 常見問題與解決

| 問題 | 原因 | 解決方法 |
|------|------|----------|
| 梯度消失 | 激活函數導數過小 | 使用 ReLU、殘差連接 |
| 梯度爆炸 | 權重過大 | 梯度裁剪、歸一化 |
| 過擬合 | 模型複雜度過高 | Dropout、正則化、資料增強 |
| 欠擬合 | 模型複雜度不足 | 增加層數、正則化減弱 |

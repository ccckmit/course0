# 語音處理與識別 (Speech Processing and Recognition)

## 概述

語音處理與識別是人工智慧領域中重要且具有挑戰性的研究方向，涵蓋語音識別（Speech Recognition）、語音合成（Speech Synthesis）、語音增強（Speech Enhancement）等多個子領域。隨著深度學習技術的快速發展，語音技術的準確率和自然度達到了前所未有的水平，廣泛應用於智慧助理、語音翻譯、字幕生成、聲紋識別等場景。

## 語音處理的基礎

### 聲學信號特性

語音信號是一種時間連續的類比信號，其特性包括：

```
時域特性：
- 振幅：聲音的大小
- 頻率：聲音的高低（Hz）
- 基頻：聲帶振動的基本頻率
- 週期性：清音 vs. 濁音

頻域特性：
- 頻譜：信號在不同頻率的能量分布
- 共振峰：聲道共振形成的頻率峰值
- 梅爾頻率：符合人類聽覺特性的頻率尺度
```

### 語音信號數位化

```python
import numpy as np
import librosa

def preprocess_audio(audio_path, sample_rate=16000):
    """語音預處理：採樣、歸一化、分幀"""
    
    # 載入音頻
    y, sr = librosa.load(audio_path, sr=sample_rate)
    
    # 預強調：補償高頻衰減
    y_filtered = librosa.effects.preemphasis(y)
    
    # 分幀：將連續信號分成小片段
    frame_length = 25  # 25ms
    frame_shift = 10   # 10ms
    frames = librosa.util.frame(
        y_filtered, 
        frame_length=int(sample_rate * frame_length / 1000),
        hop_length=int(sample_rate * frame_shift / 1000)
    )
    
    return frames, sample_rate


def extract_features(audio, sample_rate):
    """特徵提取：MFCC、FBANK、Log Mel"""
    
    # MFCC（梅爾頻率倒譜系數）
    mfcc = librosa.feature.mfcc(
        y=audio, 
        sr=sample_rate,
        n_mfcc=13,
        n_fft=400,
        hop_length=160
    )
    
    # FBANK 特徵
    fbank = librosa.feature.melspectrogram(
        y=audio,
        sr=sample_rate,
        n_mels=80,
        n_fft=400,
        hop_length=160
    )
    fbank = np.log(fbank + 1e-10)
    
    # 語音活動檢測（VAD）
    is_speech = librosa.effects.split(
        audio, 
        top_db=20,
        frame_length=400,
        hop_length=160
    )
    
    return mfcc, fbank, is_speech
```

## 語音識別 (ASR)

### 傳統方法

基於隱馬爾可夫模型（HMM）的傳統語音識別系統：

```
輸入語音 → 特徵提取 → 聲學模型（HMM） → 語言模型 → 解碼 → 輸出文字
```

```python
# 傳統 GMM-HMM 語音識別概念
class GMMHMMASR:
    """高斯混合模型-隱馬爾可夫模型語音識別"""
    
    def __init__(self, num_states, num_mixtures):
        self.num_states = num_states
        self.num_mixtures = num_mixtures
        self.transition_matrix = np.random.rand(num_states, num_states)
        self.emission_probs = []  # 每個狀態的 GMM
        self.initial_probs = np.random.rand(num_states)
        
    def forward_algorithm(self, observations):
        """前向算法：計算觀測序列的機率"""
        alpha = np.zeros((len(observations), self.num_states))
        
        # 初始化
        alpha[0] = self.initial_probs * self.emission_probs[0].score(observations[0])
        
        # 遞推
        for t in range(1, len(observations)):
            for j in range(self.num_states):
                alpha[t, j] = np.sum(
                    alpha[t-1] * self.transition_matrix[:, j]
                ) * self.emission_probs[j].score(observations[t])
        
        return np.sum(alpha[-1])
    
    def viterbi_decoding(self, observations):
        """Viterbi 解碼：找最可能的状态序列"""
        delta = np.zeros((len(observations), self.num_states))
        psi = np.zeros((len(observations), self.num_states), dtype=int)
        
        # 初始化
        delta[0] = self.initial_probs * self.emission_probs[0].score(observations[0])
        
        # 遞推
        for t in range(1, len(observations)):
            for j in range(self.num_states):
                delta[t, j] = np.max(
                    delta[t-1] * self.transition_matrix[:, j]
                ) * self.emission_probs[j].score(observations[t])
                psi[t, j] = np.argmax(
                    delta[t-1] * self.transition_matrix[:, j]
                )
        
        # 回溯
        path = [np.argmax(delta[-1])]
        for t in range(len(observations) - 1, 0, -1):
            path.append(psi[t, path[-1]])
        
        return list(reversed(path))
```

### 深度學習方法

#### 基於 RNN 的 CTC 模型

```python
import torch
import torch.nn as nn

class CTCASR(nn.Module):
    """CTC 語音識別模型"""
    
    def __init__(self, input_dim, hidden_dim, num_classes):
        super().__init__()
        
        # 雙向 LSTM 編碼器
        self.encoder = nn.LSTM(
            input_dim,
            hidden_dim,
            num_layers=3,
            batch_first=True,
            bidirectional=True,
            dropout=0.1
        )
        
        # 投影層
        self.projection = nn.Linear(hidden_dim * 2, num_classes)
        
    def forward(self, x, input_lengths):
        # 編碼
        x, _ = self.encoder(x)
        
        # 投影到字符
        logits = self.projection(x)
        
        # CTC loss
        log_probs = torch.log_softmax(logits, dim=-1)
        
        return log_probs


class CTCLoss(nn.Module):
    """Connectionist Temporal Classification Loss"""
    
    def __init__(self, blank=0):
        super().__init__()
        self.blank = blank
    
    def forward(self, log_probs, targets, input_lengths, target_lengths):
        # 扁平化目標
        targets_flat = []
        for t, l in zip(targets, target_lengths):
            targets_flat.extend(t[:l].tolist())
        targets_flat = torch.tensor(targets_flat, device=log_probs.device)
        
        # 使用 CTC loss
        loss = torch.nn.functional.ctc_loss(
            log_probs.transpose(0, 1),
            targets_flat,
            input_lengths,
            target_lengths,
            blank=self.blank,
            reduction='mean'
        )
        
        return loss
```

#### Transformer 模型

```python
class TransformerASR(nn.Module):
    """Transformer 語音識別"""
    
    def __init__(self, input_dim, d_model, nhead, num_layers, num_classes):
        super().__init__()
        
        # 輸入投影
        self.input_projection = nn.Linear(input_dim, d_model)
        
        # 位置編碼
        self.pos_encoder = PositionalEncoding(d_model)
        
        # Transformer 編碼器
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model,
            nhead=nhead,
            dim_feedforward=d_model * 4,
            dropout=0.1,
            activation='gelu'
        )
        self.transformer_encoder = nn.TransformerEncoder(
            encoder_layer,
            num_layers=num_layers
        )
        
        # 輸出層
        self.output_projection = nn.Linear(d_model, num_classes)
        
    def forward(self, x, src_key_padding_mask=None):
        # 輸入投影 + 位置編碼
        x = self.input_projection(x)
        x = self.pos_encoder(x)
        
        # Transformer 編碼
        x = self.transformer_encoder(
            x, 
            src_key_padding_mask=src_key_padding_mask
        )
        
        # 輸出
        log_probs = torch.log_softmax(
            self.output_projection(x),
            dim=-1
        )
        
        return log_probs


class PositionalEncoding(nn.Module):
    """位置編碼"""
    
    def __init__(self, d_model, max_len=5000):
        super().__init__()
        
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len).unsqueeze(1).float()
        div_term = torch.exp(
            torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model)
        )
        
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0)
        
        self.register_buffer('pe', pe)
    
    def forward(self, x):
        return x + self.pe[:, :x.size(1)]
```

### 端到端模型

#### Listen-Attend-Spell (LAS)

```python
class ListenAttendSpell(nn.Module):
    """LAS 端到端語音識別"""
    
    def __init__(self, input_dim, d_model, num_classes):
        super().__init__()
        
        # Listener：金字塔 RNN
        self.listener = nn.Sequential(
            nn.Linear(input_dim, d_model),
            nn.ReLU(),
            nn.LSTM(d_model, d_model, num_layers=2, batch_first=True, dropout=0.1)
        )
        
        # Speller：注意力解碼器
        self.attend = nn.MultiheadAttention(d_model, 4)
        self.speller = nn.LSTMCell(d_model, d_model)
        
        self.char_prob = nn.Linear(d_model, num_classes)
        
    def forward(self, audio_features, target=None, target_lengths=None):
        # Listen：編碼音頻
        encoder_output, _ = self.listener(audio_features)
        
        # 初始化解碼器狀態
        decoder_state = None
        max_len = target.size(1) if target is not None else 100
        
        outputs = []
        for t in range(max_len):
            # Attend：計算注意力
            context, _ = self.attend(
                decoder_output[-1:] if decoder_state else encoder_output[:1],
                encoder_output,
                encoder_output
            )
            
            # Spell：解碼
            if decoder_state is None:
                decoder_state = self.speller(context.squeeze(0))
            else:
                h, c = decoder_state
                decoder_output, (h, c) = self.speller(context.squeeze(0), (h, c))
                decoder_state = (h, c)
            
            # 輸出機率
            char_prob = self.char_prob(decoder_output[-1])
            outputs.append(char_prob)
        
        return torch.stack(outputs, dim=1)
```

```python
class Tacotron(nn.Module):
    """Tacotron 端到端語音合成"""
    
    def __init__(self, vocab_size, embed_dim, encoder_dim, decoder_dim):
        super().__init__()
        
        # 嵌入層
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        
        # 編碼器：CBHG
        self.encoder = CBHG(embed_dim, encoder_dim)
        
        # 解碼器
        self.decoder = Decoder(encoder_dim, decoder_dim)
        
        # 注意力機制
        self.attention = Attention(decoder_dim, encoder_dim)
        
        # 梅爾頻譜輸出
        self.mel_projection = nn.Linear(decoder_dim, 80)  # 80 維梅爾頻譜
        
        # 線性頻譜輸出
        self.linear_projection = nn.Linear(decoder_dim, 1025)
        
    def forward(self, text, target_mel=None):
        # 編碼
        encoder_output = self.encoder(self.embedding(text))
        
        # 解碼
        mel_outputs = []
        decoder_state = None
        
        max_steps = target_mel.size(1) if target_mel is not None else 200
        
        for t in range(max_steps):
            # 注意力
            context, attention_weights = self.attention(
                decoder_state[0] if decoder_state else None,
                encoder_output
            )
            
            # 解碼
            decoder_output, decoder_state = self.decoder(
                target_mel[:, t-1] if t > 0 and target_mel is not None else None,
                decoder_state,
                context
            )
            
            # 輸出
            mel = self.mel_projection(decoder_output)
            mel_outputs.append(mel)
        
        mel_outputs = torch.stack(mel_outputs, dim=1)
        
        # 轉換為線性頻譜
        linear = self.linear_projection(mel_outputs)
        
        return mel_outputs, linear


class CBHG(nn.Module):
    """卷積bank + Highway + GRU"""
    
    def __init__(self, input_dim, output_dim):
        super().__init__()
        
        # 卷積 bank
        self.conv1d_banks = nn.ModuleList([
            nn.Conv1d(input_dim, output_dim, kernel_size=k, padding=k//2)
            for k in range(1, 17)
        ])
        
        # 池化
        self.max_pool = nn.MaxPool1d(2, stride=1)
        
        # Highway 網路
        self.highway = nn.Linear(output_dim, output_dim)
        
        # BiGRU
        self.gru = nn.GRU(output_dim, output_dim, batch_first=True, bidirectional=True)
    
    def forward(self, x):
        # 卷積 bank
        conv_outputs = []
        for conv in self.conv1d_banks:
            c = torch.relu(conv(x.transpose(1, 2)))
            conv_outputs.append(c)
        
        x = torch.cat(conv_outputs, dim=1)
        x = self.max_pool(x)[:, :, :-1]  # 移除最後一個
        
        # 殘差連接
        x = x.transpose(1, 2) + nn.Linear(x.size(1), x.size(2))(x).transpose(1, 2)
        
        # GRU
        gru_out, _ = self.gru(x)
        
        return gru_out
```

#### Neural Vocoder

```python
class WaveNet(nn.Module):
    """WaveNet 神經聲碼器"""
    
    def __init__(self, num_mels, residual_channels, num_layers):
        super().__init__()
        
        self.num_mels = num_mels
        self.embedding = nn.Embedding(256, residual_channels)
        
        # Input conv
        self.input_conv = nn.Conv1d(num_mels, residual_channels, 1)
        
        # Dilated conv layers
        self.dilated_convs = nn.ModuleList([
            nn.Conv1d(
                residual_channels,
                residual_channels,
                kernel_size=3,
                dilation=2**i,
                padding=2**i
            )
            for i in range(num_layers)
        ])
        
        # Output
        self.output_conv = nn.Conv1d(residual_channels, 256, 1)
        
    def forward(self, mel_spectrogram):
        # 條件輸入
        x = self.input_conv(mel_spectrogram)
        
        # 殘差連接
        for conv in self.dilated_convs:
            residual = x
            x = torch.relu(conv(x))
            x = x + residual
        
        # 輸出
        x = self.output_conv(x)
        
        return x
```

### 說話人識別

```python
class SpeakerRecognition(nn.Module):
    """說話人識別：驗證或辨識"""
    
    def __init__(self, num_speakers):
        super().__init__()
        
        # TDNN 骨幹網路
        self.backbone = nn.Sequential(
            nn.Conv1d(80, 512, 5, 1, 2),
            nn.ReLU(),
            nn.BatchNorm1d(512),
            nn.Conv1d(512, 512, 3, 1, 1),
            nn.ReLU(),
            nn.BatchNorm1d(512),
            nn.Conv1d(512, 512, 3, 1, 2),
            nn.ReLU(),
            nn.BatchNorm1d(512),
        )
        
        # Attention pooling
        self.attention = nn.Linear(512, 1)
        
        # 分類/嵌入
        self.fc = nn.Linear(512, num_speakers)
        
    def forward(self, mel_spectrogram):
        # 提取特徵
        x = self.backbone(mel_spectrogram)
        
        # Attention pooling
        a = torch.softmax(self.attention(x.transpose(1, 2)), dim=1)
        x = (x * a).sum(dim=2)
        
        # 嵌入向量（用於驗證）
        embedding = nn.functional.normalize(x, dim=1)
        
        # 分類（用於辨識）
        logits = self.fc(x)
        
        return embedding, logits
```

## 評估指標

```python
def evaluate_asr(predictions, references):
    """語音識別評估指標"""
    
    # 字錯誤率 (WER)
    def word_error_rate(pred, ref):
        return edit_distance(pred.split(), ref.split()) / len(ref.split())
    
    # 字元錯誤率 (CER)
    def char_error_rate(pred, ref):
        return edit_distance(pred, ref) / len(ref)
    
    # 計算平均錯誤率
    wer = sum(wer(p, r) for p, r in zip(predictions, references)) / len(predictions)
    cer = sum(cer(p, r) for p, r in zip(predictions, references)) / len(predictions)
    
    return {
        'WER': wer,
        'CER': cer,
        'Accuracy': 1 - wer
    }
```

# 多模態模型 (Multimodal Models)

## 概述

多模態模型是指能夠同時處理和理解多種資訊類型（如文字、圖像、音頻、影片等）的深度學習模型。隨著大型語言模型（LLM）的發展，多模態學習成為人工智慧研究的前沿領域，實現了更自然、更強大的人機交互方式。從 GPT-4V 到 Gemini，從 CLIP 到 LLaVA，多模態模型正在重新定義 AI 的能力邊界。

## 多模態學習的基礎

### 什麼是模態？

```
視覺模態：圖像、影片、深度圖
語言模態：文字、語音、字幕
音頻模態：語音、音樂、噪音
觸覺模態：觸覺感測、力回饋
傳感器：雷達、LiDAR、IMU
```

### 核心挑戰

1. **表示對齊**：不同模態的資料表示方式不同，需要找到共同的表示空間
2. **跨模態遷移**：一個模態學到的知識能否遷移到另一個模態
3. **模態融合**：如何有效結合不同模態的資訊
4. **缺失模態**：當某些模態缺失時如何處理

## 視覺-語言模型

### CLIP（Contrastive Language-Image Pre-Training）

CLIP 是 OpenAI 提出的里程碑式模型，透過對比學習將文字和圖像映射到同一表示空間。

```python
class CLIP(nn.Module):
    """CLIP 模型：文字-圖像對比學習"""
    
    def __init__(self, config):
        super().__init__()
        
        # 圖像編碼器（ViT）
        self.image_encoder = ViT(
            image_size=config.image_size,
            patch_size=config.patch_size,
            d_model=config.d_model,
            num_heads=config.num_heads,
            num_layers=config.num_layers
        )
        
        # 文字編碼器（Transformer）
        self.text_encoder = Transformer(
            vocab_size=config.vocab_size,
            d_model=config.d_model,
            num_heads=config.num_heads,
            num_layers=config.num_layers
        )
        
        # 投影層
        self.image_projection = nn.Linear(config.d_model, config.projection_dim)
        self.text_projection = nn.Linear(config.d_model, config.projection_dim)
        
        # 溫度參數
        self.logit_scale = nn.Parameter(torch.ones([]) * np.log(1 / 0.07))
    
    def forward(self, images, text_tokens):
        # 編碼
        image_features = self.image_encoder(images)
        text_features = self.text_encoder(text_tokens)
        
        # 投影
        image_embedding = self.image_projection(image_features)
        text_embedding = self.text_projection(text_features)
        
        # 歸一化
        image_embedding = torch.nn.functional.normalize(image_embedding, dim=-1)
        text_embedding = torch.nn.functional.normalize(text_embedding, dim=-1)
        
        # 對比損失
        logits = torch.matmul(image_embedding, text_embedding.t()) * self.logit_scale.exp()
        
        return logits, image_embedding, text_embedding


def clip_contrastive_loss(logits, batch_size):
    """CLIP 對比損失"""
    labels = torch.arange(batch_size, device=logits.device)
    
    loss_i = torch.nn.functional.cross_entropy(logits, labels)
    loss_t = torch.nn.functional.cross_entropy(logits.t(), labels)
    
    return (loss_i + loss_t) / 2


class ViT(nn.Module):
    """Vision Transformer"""
    
    def __init__(self, image_size=224, patch_size=16, d_model=512, num_heads=8, num_layers=12):
        super().__init__()
        
        num_patches = (image_size // patch_size) ** 2
        patch_dim = 3 * patch_size ** 2
        
        # Patch 嵌入
        self.patch_embed = nn.Linear(patch_dim, d_model)
        
        # 位置編碼
        self.pos_embed = nn.Parameter(torch.zeros(1, num_patches + 1, d_model))
        self.cls_token = nn.Parameter(torch.zeros(1, 1, d_model))
        
        # Transformer  encoder
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model,
            nhead=num_heads,
            dim_feedforward=d_model * 4,
            dropout=0.1,
            activation='gelu',
            batch_first=True
        )
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers)
        
        # 輸出
        self.norm = nn.LayerNorm(d_model)
    
    def forward(self, x):
        B = x.size(0)
        
        # 分塊
        x = x.unfold(2, 16, 16).unfold(3, 16, 16)
        x = x.contiguous().view(B, -1, 16 * 16 * 3)
        
        # 嵌入
        x = self.patch_embed(x)
        
        # 添加 CLS token
        cls_tokens = self.cls_token.expand(B, -1, -1)
        x = torch.cat([cls_tokens, x], dim=1)
        
        # 添加位置編碼
        x = x + self.pos_embed
        
        # Transformer
        x = self.transformer(x)
        x = self.norm(x)
        
        return x
```

### BLIP（Bootstrapped Language-Image Pre-training）

BLIP 是一個靈活的多模態框架，支持理解和生成任務。

```python
class BLIP(nn.Module):
    """BLIP 模型：理解 + 生成"""
    
    def __init__(self, config):
        super().__init__()
        
        # 視覺編碼器
        self.visual_encoder = ViT(config)
        
        # 文字編碼器
        self.text_encoder = Transformer(config)
        
        # 文字解碼器（用於生成）
        self.text_decoder = Transformer(config, is_decoder=True)
        
        # 融合層
        self.Qformer = QFormer(config)
        
        # 投影層
        self.vision_proj = nn.Linear(config.d_model, config.embed_dim)
        self.text_proj = nn.Linear(config.d_model, config.embed_dim)
        
        # 語言模型頭
        self.lm_head = nn.Linear(config.d_model, config.vocab_size)
    
    def forward(self, image, caption=None, mode='pretrain'):
        # 視覺特徵
        image_embeds = self.visual_encoder(image)
        image_embeds = self.Qformer(image_embeds)
        
        if mode == 'pretrain':
            # ITM: 圖文匹配
            image_feat = torch.nn.functional.normalize(
                self.vision_proj(image_embeds), dim=-1
            )
            
            # 文字編碼
            text_tokens = self.text_encoder(caption)
            
            # 對比學習
            text_feat = torch.nn.functional.normalize(
                self.text_proj(text_tokens.last_hidden_state), dim=-1
            )
            
            return image_feat, text_feat, image_embeds
        
        elif mode == 'caption':
            # 標題生成
            return self.text_decoder(caption, image_embeds)


class QFormer(nn.Module):
    """Query Transformer：學習查詢向量"""
    
    def __init__(self, config):
        super().__init__()
        
        self.query_tokens = nn.Parameter(
            torch.zeros(1, config.num_queries, config.d_model)
        )
        
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=config.d_model,
            nhead=config.num_heads,
            dim_feedforward=config.d_model * 4,
            batch_first=True
        )
        self.encoder = nn.TransformerEncoder(encoder_layer, config.num_layers)
        
        self.cross_attention = nn.MultiheadAttention(
            config.d_model, config.num_heads, batch_first=True
        )
    
    def forward(self, image_embeds):
        # 初始化查詢
        query_tokens = self.query_tokens.expand(image_embeds.size(0), -1, -1)
        
        # 自注意力和交叉注意力
        query_tokens = self.encoder(query_tokens)
        query_tokens, _ = self.cross_attention(
            query_tokens, image_embeds, image_embeds
        )
        
        return query_tokens
```

### LLaVA（Large Language and Vision Assistant）

LLaVA 是首個將 LLM 與視覺模型結合的開放源碼助教手。

```python
class LLaVA(nn.Module):
    """LLaVA：大語言模型 + 視覺理解"""
    
    def __init__(self, config):
        super().__init__()
        
        # 視覺編碼器
        self.vision_encoder = CLIPVisionEncoder(config)
        
        # 投影層
        self.vision_projection = nn.Linear(
            config.vision_dim, config.llm_dim
        )
        
        # LLM（Vicuna 或 LLaMA）
        self.llm = load_language_model(config)
        
        # 分詞器
        self.tokenizer = load_tokenizer(config)
    
    def forward(self, image, text):
        # 視覺特徵
        vision_features = self.vision_encoder(image)
        vision_features = self.vision_projection(vision_features)
        
        # 文字嵌入
        text_tokens = self.tokenizer(text)
        
        # 拼接
        # [IMG] token + 視覺特徵 + [SEP] token + 文字
        combined_input = self._combine_modalities(vision_features, text_tokens)
        
        # LLM 生成
        output = self.llm(combined_input)
        
        return output
    
    def chat(self, image, user_message):
        """對話模式"""
        # 編碼圖像
        vision_features = self.vision_encoder(image)
        vision_features = self.vision_projection(vision_features)
        
        # 構建 prompt
        prompt = f"USER: <image>\n{user_message}\nASSISTANT:"
        
        # 生成回覆
        response = self.generate(prompt, vision_features)
        
        return response
```

## 大型多模態模型

### GPT-4V 與多模態能力

```python
class MultimodalGPT(nn.Module):
    """多模態 GPT 概念架構"""
    
    def __init__(self, config):
        super().__init__()
        
        # 視覺編碼器
        self.vision_encoder = VisionEncoder(config)
        
        # 音頻編碼器
        self.audio_encoder = AudioEncoder(config)
        
        # 統一投影
        self.projection = nn.ModuleDict({
            'vision': nn.Linear(config.vision_dim, config.llm_dim),
            'audio': nn.Linear(config.audio_dim, config.llm_dim),
            'text': nn.Linear(config.text_dim, config.llm_dim)
        })
        
        # LLM Transformer
        self.transformer = TransformerDecoder(config)
        
        # 分詞器
        self.tokenizer = AutoTokenizer.from_pretrained(config.llm_name)
    
    def forward(self, inputs):
        # 處理各模態輸入
        embeddings = []
        
        if 'image' in inputs:
            img_emb = self.projection['vision'](self.vision_encoder(inputs['image']))
            embeddings.append(img_emb)
        
        if 'audio' in inputs:
            aud_emb = self.projection['audio'](self.audio_encoder(inputs['audio']))
            embeddings.append(aud_emb)
        
        if 'text' in inputs:
            txt_emb = self.projection['text'](self.text_encoder(inputs['text']))
            embeddings.append(txt_emb)
        
        # 拼接並排序位置
        combined = torch.cat(embeddings, dim=1)
        
        # LLM 處理
        output = self.transformer(combined)
        
        return output
    
    def generate(self, inputs, max_length=100):
        """自迴歸生成"""
        output = self.forward(inputs)
        
        for _ in range(max_length):
            next_token = self._sample(output)
            output = torch.cat([output, next_token], dim=1)
            
            if next_token == self.tokenizer.eos_token:
                break
        
        return self.tokenizer.decode(output[0])
```

### Gemini 原生多模態

Google 的 Gemini 設計為原生多模態，從訓練開始就處理多種模態。

```python
class GeminiArchitecture(nn.Module):
    """Gemini 原生多模態架構"""
    
    def __init__(self, config):
        super().__init__()
        
        # 統一輸入處理
        self.input_processor = InputProcessor(config)
        
        # 專家混合（MoE）
        self.moe_layers = nn.ModuleList([
            MixtureOfExperts(config) for _ in range(config.num_layers)
        ])
        
        # 輸出生成
        self.output_generator = OutputGenerator(config)
        
        # 跨模態注意力
        self.cross_modal_attention = CrossModalAttention(config)
    
    def process_inputs(self, inputs):
        """統一處理文字、圖像、音頻、影片"""
        processed = {}
        
        for modality, data in inputs.items():
            if modality == 'text':
                processed['text'] = self.input_processor.text(data)
            elif modality == 'image':
                processed['image'] = self.input_processor.image(data)
            elif modality == 'audio':
                processed['audio'] = self.input_processor.audio(data)
            elif modality == 'video':
                processed['video'] = self.input_processor.video(data)
        
        return processed
    
    def forward(self, inputs):
        # 統一處理
        x = self.process_inputs(inputs)
        
        # 通過 MoE 層
        for moe in self.moe_layers:
            x = moe(x)
            x = self.cross_modal_attention(x)
        
        # 生成輸出
        output = self.output_generator(x)
        
        return output
```

### Whisper 語音識別

```python
class WhisperModel(nn.Module):
    """Whisper 語音識別模型"""
    
    def __init__(self, config):
        super().__init__()
        
        # 音頻編碼器
        self.encoder = WhisperEncoder(config)
        
        # 文字解碼器
        self.decoder = WhisperDecoder(config)
        
        # 位置編碼
        self.pos_embed = PositionalEmbedding(config.d_model)
        
    def forward(self, mel_spectrogram, target_tokens=None):
        # 編碼
        audio_features = self.encoder(mel_spectrogram)
        audio_features = self.pos_embed(audio_features)
        
        if target_tokens is not None:
            # 訓練模式
            return self.decoder(target_tokens, audio_features)
        else:
            # 推論模式
            return self._decode(audio_features)
    
    def _decode(self, audio_features):
        """貪心解碼"""
        max_length = 448
        generated = [self.decoder.sos_token]
        
        for _ in range(max_length):
            tokens = torch.tensor([generated], device=audio_features.device)
            logits = self.decoder(tokens, audio_features)
            next_token = logits[0, -1].argmax()
            generated.append(next_token.item())
            
            if next_token == self.decoder.eos_token:
                break
        
        return self.decoder.tokenizer.decode(generated)
```

```python
class VideoLanguageModel(nn.Module):
    """影片-語言模型"""
    
    def __init__(self, config):
        super().__init__()
        
        # 影片編碼器
        self.video_encoder = VideoEncoder(config)
        
        # 時間注意力
        self.temporal_attention = TemporalAttention(config)
        
        # 空間注意力
        self.spatial_attention = SpatialAttention(config)
        
        # 融合 LLM
        self.llm = TransformerLM(config)
    
    def forward(self, video_frames, text):
        # 影片編碼
        video_features = self.video_encoder(video_frames)
        
        # 時空注意力
        video_features = self.temporal_attention(video_features)
        video_features = self.spatial_attention(video_features)
        
        # 與文字融合
        combined = self._fuse(video_features, text)
        
        # 生成
        output = self.llm(combined)
        
        return output


class VideoEncoder(nn.Module):
    """影片編碼器：3D CNN + Transformer"""
    
    def __init__(self, config):
        super().__init__()
        
        # 3D 卷積：用於時序特徵
        self.conv3d = nn.Sequential(
            nn.Conv3d(3, 64, kernel_size=(3, 7, 7), stride=(1, 2, 2), padding=(1, 3, 3)),
            nn.BatchNorm3d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool3d(kernel_size=(2, 3, 3), stride=(2, 2, 2))
        )
        
        # ViT 編碼
        self.vit = ViT(
            d_model=config.d_model,
            num_layers=config.num_layers,
            num_heads=config.num_heads
        )
    
    def forward(self, x):
        # x: [B, T, C, H, W]
        B, T, C, H, W = x.shape
        
        # 合併時間維度
        x = x.view(B, C, T, H, W)
        
        # 3D 卷積
        x = self.conv3d(x)
        
        # 重新排列：特徵 -> 時間
        x = x.permute(0, 2, 1, 3, 4)  # [B, T, C', H', W']
        x = x.view(B, T, -1)
        
        # ViT
        x = self.vit(x)
        
        return x
```

## 多模態訓練策略

### 對比預訓練

```python
def contrastive_pretraining(model, dataloader, config):
    """多模態對比預訓練"""
    
    optimizer = torch.optim.AdamW(model.parameters(), lr=config.lr)
    
    for epoch in range(config.epochs):
        for batch in dataloader:
            images = batch['image']
            texts = batch['text']
            
            # 前向
            image_logits, text_logits = model(images, texts)
            
            # 對比損失
            loss = symmetric_contrastive_loss(image_logits, text_logits)
            
            # 反向傳播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()


def symmetric_contrastive_loss(image_to_text_logits, text_to_image_logits):
    """對稱對比損失"""
    batch_size = image_to_text_logits.size(0)
    labels = torch.arange(batch_size, device=image_to_text_logits.device)
    
    loss_i2t = torch.nn.functional.cross_entropy(image_to_text_logits, labels)
    loss_t2i = torch.nn.functional.cross_entropy(text_to_image_logits, labels)
    
    return (loss_i2t + loss_t2i) / 2
```

```python
class LoraConfig:
    def __init__(self):
        self.r = 8  # 低秩維度
        self.alpha = 16
        self.dropout = 0.05
        self.target_modules = ['q_proj', 'v_proj', 'k_proj', 'o_proj']


class LoRALayer(nn.Module):
    """LoRA 低秩適配層"""
    
    def __init__(self, config, in_features, out_features):
        super().__init__()
        
        self.lora_A = nn.Parameter(torch.randn(in_features, config.r))
        self.lora_B = nn.Parameter(torch.zeros(config.r, out_features))
        self.dropout = nn.Dropout(config.dropout)
        
        # 初始化
        nn.init.kaiming_uniform_(self.lora_A, a=np.sqrt(5))
    
    def forward(self, x):
        return self.dropout(x @ self.lora_A @ self.lora_B)
```

## 挑戰與未來方向

### 當前挑戰

1. **計算資源**：訓練大型多模態模型需要大量 GPU/TPU
2. **資料品質**：需要高質量、多樣性的配對資料
3. **評估困難**：缺乏標準化的多模態基準
4. **安全問題**：可能生成有害內容

### 未來發展

# 影像處理與識別 (Image Processing and Recognition)

## 概述

影像處理與識別是電腦視覺（Computer Vision）領域的核心課題，涵蓋圖像增強、特徵提取、物體偵測、語義分割、影像生成等多個研究方向。隨著深度學習技術的突破，影像識別的準確率已达到甚至超越人類水平，廣泛應用於自動駕駛、醫學影像診斷、人臉識別、工業檢測等領域。

### 數位影像表示

```python
import numpy as np
import cv2

class ImageRepresentation:
    """數位影像表示"""
    
    @staticmethod
    def to_grayscale(color_image):
        """彩色轉灰度：RGB -> YUV -> Y"""
        # 加權平均：0.299R + 0.587G + 0.114B
        gray = 0.299 * color_image[:, :, 0] + \
               0.587 * color_image[:, :, 1] + \
               0.114 * color_image[:, :, 2]
        return gray.astype(np.uint8)
    
    @staticmethod
    def to_binary(gray_image, threshold=128):
        """灰度轉二值化"""
        return (gray_image > threshold).astype(np.uint8) * 255
    
    @staticmethod
    def rgb_to_hsv(rgb_image):
        """RGB 轉 HSV"""
        r, g, b = rgb_image[:, :, 0] / 255.0, rgb_image[:, :, 1] / 255.0, rgb_image[:, :, 2] / 255.0
        maxc = np.maximum(np.maximum(r, g), b)
        minc = np.minimum(np.minimum(r, g), b)
        v = maxc
        deltac = maxc - minc
        
        # Saturation
        s = np.where(maxc != 0, deltac / maxc, 0)
        
        # Hue
        h = np.zeros_like(v)
        mask = deltac != 0
        mask_r = mask & (maxc == r)
        mask_g = mask & (maxc == g) & ~mask_r
        mask_b = mask & (maxc == b) & ~mask_r & ~mask_g
        
        h[mask_r] = (60 * ((g[mask_r] - b[mask_r]) / deltac[mask_r]) + 360) % 360
        h[mask_g] = (60 * ((b[mask_g] - r[mask_g]) / deltac[mask_g]) + 120) % 360
        h[mask_b] = (60 * ((r[mask_b] - g[mask_b]) / deltac[mask_b]) + 240) % 360
        
        return np.stack([h / 360.0, s, v], axis=-1)
```

### 影像增強

```python
class ImageAugmentation:
    """影像資料增強"""
    
    @staticmethod
    def random_flip(image, p=0.5):
        """隨機水平翻轉"""
        if np.random.rand() < p:
            return np.fliplr(image)
        return image
    
    @staticmethod
    def random_crop(image, crop_size):
        """隨機裁剪"""
        h, w = image.shape[:2]
        crop_h, crop_w = crop_size
        
        if h <= crop_h or w <= crop_w:
            return image
        
        top = np.random.randint(0, h - crop_h + 1)
        left = np.random.randint(0, w - crop_w + 1)
        
        return image[top:top+crop_h, left:left+crop_w]
    
    @staticmethod
    def random_rotation(image, max_angle=15):
        """隨機旋轉"""
        angle = np.random.uniform(-max_angle, max_angle)
        h, w = image.shape[:2]
        center = (w // 2, h // 2)
        
        M = cv2.getRotationMatrix2D(center, angle, 1.0)
        rotated = cv2.warpAffine(image, M, (w, h))
        
        return rotated
    
    @staticmethod
    def random_brightness(image, factor_range=(0.8, 1.2)):
        """隨機亮度調整"""
        factor = np.random.uniform(*factor_range)
        return np.clip(image * factor, 0, 255).astype(np.uint8)
    
    @staticmethod
    def random_contrast(image, factor_range=(0.8, 1.2)):
        """隨機對比度調整"""
        factor = np.random.uniform(*factor_range)
        mean = image.mean()
        return np.clip((image - mean) * factor + mean, 0, 255).astype(np.uint8)
    
    @staticmethod
    def add_gaussian_noise(image, sigma=10):
        """高斯雜訊"""
        noise = np.random.normal(0, sigma, image.shape)
        noisy = image + noise
        return np.clip(noisy, 0, 255).astype(np.uint8)
    
    @staticmethod
    def gauss_blur(image, kernel_size=5):
        """高斯模糊"""
        return cv2.GaussianBlur(image, (kernel_size, kernel_size), 0)
```

### 傳統影像處理

```python
class ClassicalImageProcessing:
    """傳統影像處理方法"""
    
    @staticmethod
    def sobel_edge(image):
        """Sobel 邊緣偵測"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY) if len(image.shape) == 3 else image
        
        sobel_x = cv2.Sobel(gray, cv2.CV_64F, 1, 0, ksize=3)
        sobel_y = cv2.Sobel(gray, cv2.CV_64F, 0, 1, ksize=3)
        
        magnitude = np.sqrt(sobel_x**2 + sobel_y**2)
        direction = np.arctan2(sobel_y, sobel_x)
        
        return magnitude, direction
    
    @staticmethod
    def canny_edge(image, low_threshold=50, high_threshold=150):
        """Canny 邊緣偵測"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY) if len(image.shape) == 3 else image
        edges = cv2.Canny(gray, low_threshold, high_threshold)
        return edges
    
    @staticmethod
    def histogram_equalization(image):
        """直方圖均衡化"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY) if len(image.shape) == 3 else image
        equalized = cv2.equalizeHist(gray)
        return equalized
    
    @staticmethod
    def morphological_op(image, operation='erosion', kernel_size=5):
        """形態學操作"""
        kernel = np.ones((kernel_size, kernel_size), np.uint8)
        
        if operation == 'erosion':
            return cv2.erode(image, kernel)
        elif operation == 'dilation':
            return cv2.dilate(image, kernel)
        elif operation == 'opening':
            return cv2.morphologyEx(image, cv2.MORPH_OPEN, kernel)
        elif operation == 'closing':
            return cv2.morphologyEx(image, cv2.MORPH_CLOSE, kernel)
    
    @staticmethod
    def hough_transform(image):
        """霍夫直線轉換"""
        edges = ClassicalImageProcessing.canny_edge(image)
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 100, minLineLength=50, maxLineGap=10)
        return lines
```

```python
import torch
import torch.nn as nn

class ImageClassifier(nn.Module):
    """影像分類模型"""
    
    def __init__(self, num_classes):
        super().__init__()
        
        # 特徵提取網路
        self.features = nn.Sequential(
            # Block 1
            nn.Conv2d(3, 64, kernel_size=3, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(64, 64, kernel_size=3, padding=1),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),
            nn.BatchNorm2d(64),
            nn.Dropout(0.25),
            
            # Block 2
            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(128, 128, kernel_size=3, padding=1),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),
            nn.BatchNorm2d(128),
            nn.Dropout(0.25),
            
            # Block 3
            nn.Conv2d(128, 256, kernel_size=3, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(256, 256, kernel_size=3, padding=1),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),
            nn.BatchNorm2d(256),
        )
        
        # 分類器
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(256 * 4 * 4, 512),
            nn.ReLU(inplace=True),
            nn.Dropout(0.5),
            nn.Linear(512, num_classes)
        )
    
    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)
        return x


def train_classifier(model, train_loader, val_loader, epochs):
    """訓練影像分類器"""
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    model = model.to(device)
    
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=10, gamma=0.5)
    
    for epoch in range(epochs):
        model.train()
        train_loss = 0
        train_correct = 0
        train_total = 0
        
        for images, labels in train_loader:
            images, labels = images.to(device), labels.to(device)
            
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            
            train_loss += loss.item()
            _, predicted = outputs.max(1)
            train_total += labels.size(0)
            train_correct += predicted.eq(labels).sum().item()
        
        scheduler.step()
        
        # 驗證
        val_loss = 0
        val_correct = 0
        val_total = 0
        
        model.eval()
        with torch.no_grad():
            for images, labels in val_loader:
                images, labels = images.to(device), labels.to(device)
                outputs = model(images)
                loss = criterion(outputs, labels)
                
                val_loss += loss.item()
                _, predicted = outputs.max(1)
                val_total += labels.size(0)
                val_correct += predicted.eq(labels).sum().item()
        
        print(f'Epoch {epoch+1}: Train Acc: {100*train_correct/train_total:.2f}%, Val Acc: {100*val_correct/val_total:.2f}%')
```

```python
class RPN(nn.Module):
    """Region Proposal Network"""
    
    def __init__(self, num_anchors):
        super().__init__()
        
        self.conv = nn.Conv2d(256, 512, 3, padding=1)
        self.cls_layer = nn.Conv2d(512, num_anchors, 1)
        self.reg_layer = nn.Conv2d(512, num_anchors * 4, 1)
    
    def forward(self, x):
        x = torch.relu(self.conv(x))
        cls = self.cls_layer(x)
        reg = self.reg_layer(x)
        return cls, reg
```

```python
class UNet(nn.Module):
    """U-Net 語義分割網路"""
    
    def __init__(self, in_channels, num_classes):
        super().__init__()
        
        # Encoder
        self.enc1 = self._conv_block(in_channels, 64)
        self.enc2 = self._conv_block(64, 128)
        self.enc3 = self._conv_block(128, 256)
        self.enc4 = self._conv_block(256, 512)
        self.enc5 = self._conv_block(512, 1024)
        
        self.pool = nn.MaxPool2d(2)
        
        # Decoder
        self.up4 = nn.ConvTranspose2d(1024, 512, 2, stride=2)
        self.dec4 = self._conv_block(1024, 512)
        
        self.up3 = nn.ConvTranspose2d(512, 256, 2, stride=2)
        self.dec3 = self._conv_block(512, 256)
        
        self.up2 = nn.ConvTranspose2d(256, 128, 2, stride=2)
        self.dec2 = self._conv_block(256, 128)
        
        self.up1 = nn.ConvTranspose2d(128, 64, 2, stride=2)
        self.dec1 = self._conv_block(128, 64)
        
        # 輸出
        self.final = nn.Conv2d(64, num_classes, 1)
    
    def _conv_block(self, in_ch, out_ch):
        return nn.Sequential(
            nn.Conv2d(in_ch, out_ch, 3, padding=1),
            nn.ReLU(inplace=True),
            nn.BatchNorm2d(out_ch),
            nn.Conv2d(out_ch, out_ch, 3, padding=1),
            nn.ReLU(inplace=True),
            nn.BatchNorm2d(out_ch)
        )
    
    def forward(self, x):
        # Encoder
        e1 = self.enc1(x)
        e2 = self.enc2(self.pool(e1))
        e3 = self.enc3(self.pool(e2))
        e4 = self.enc4(self.pool(e3))
        e5 = self.enc5(self.pool(e4))
        
        # Decoder
        d4 = self.up4(e5)
        d4 = torch.cat([d4, e4], dim=1)
        d4 = self.dec4(d4)
        
        d3 = self.up3(d4)
        d3 = torch.cat([d3, e3], dim=1)
        d3 = self.dec3(d3)
        
        d2 = self.up2(d3)
        d2 = torch.cat([d2, e2], dim=1)
        d2 = self.dec2(d2)
        
        d1 = self.up1(d2)
        d1 = torch.cat([d1, e1], dim=1)
        d1 = self.dec1(d1)
        
        return self.final(d1)
```

```python
            'box_reg': box_reg,
            'mask_logits': mask_logits
```

```python
class FaceRecognition(nn.Module):
    """人臉識別：特徵嵌入 + 比對"""
    
    def __init__(self, embedding_dim=512):
        super().__init__()
        
        #backbone
        self.backbone = nn.Sequential(
            nn.Conv2d(3, 64, 3, stride=2, padding=1),  # 112x112 -> 56x56
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            
            nn.Conv2d(64, 64, 3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),
            
            # ... 更多卷積層
            ResidualBlock(64, 128),
            ResidualBlock(128, 256),
            ResidualBlock(256, 512),
        )
        
        # 嵌入層
        self.embedding = nn.Linear(512 * 7 * 7, embedding_dim)
        
        # 歸一化
        self.bn = nn.BatchNorm1d(embedding_dim)
    
    def forward(self, x):
        x = self.backbone(x)
        x = x.view(x.size(0), -1)
        x = self.embedding(x)
        x = self.bn(x)
        
        # L2 歸一化（用於餘弦相似度）
        x = torch.nn.functional.normalize(x, p=2, dim=1)
        
        return x


class ArcFaceLoss(nn.Module):
    """ArcFace 損失函數"""
    
    def __init__(self, embedding_dim, num_classes):
        super().__init__()
        self.weight = nn.Parameter(torch.FloatTensor(num_classes, embedding_dim))
        nn.init.xavier_uniform_(self.weight)
        
        self.s = 30.0  # scale
        self.m = 0.5   # margin
    
    def forward(self, embeddings, labels):
        # 餘弦相似度
        cos_theta = torch.mm(embeddings, self.weight.t())
        cos_theta = torch.clamp(cos_theta, -1.0, 1.0)
        
        # Add margin
        theta = torch.acos(cos_theta)
        target_logits = torch.cos(theta + self.m)
        
        # One-hot 編碼
        one_hot = torch.zeros_like(cos_theta)
        one_hot.scatter_(1, labels.view(-1, 1), 1)
        
        # 組合 logits
        logits = cos_theta * (1 - one_hot) + target_logits * one_hot
        logits *= self.s
        
        return torch.nn.functional.cross_entropy(logits, labels)
```

### ResNet

```python
class ResidualBlock(nn.Module):
    """ResNet 殘差塊"""
    
    def __init__(self, in_channels, out_channels, stride=1):
        super().__init__()
        
        self.conv1 = nn.Conv2d(in_channels, out_channels, 3, stride, 1)
        self.bn1 = nn.BatchNorm2d(out_channels)
        self.relu = nn.ReLU(inplace=True)
        
        self.conv2 = nn.Conv2d(out_channels, out_channels, 3, 1, 1)
        self.bn2 = nn.BatchNorm2d(out_channels)
        
        # 捷徑連接
        self.shortcut = nn.Sequential()
        if stride != 1 or in_channels != out_channels:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, 1, stride),
                nn.BatchNorm2d(out_channels)
            )
    
    def forward(self, x):
        residual = self.shortcut(x)
        
        out = self.conv1(x)
        out = self.bn1(out)
        out = self.relu(out)
        
        out = self.conv2(out)
        out = self.bn2(out)
        
        out += residual
        out = self.relu(out)
        
        return out
```

```python
def evaluate_detection(predictions, ground_truths, iou_threshold=0.5):
    """物體偵測評估：mAP"""
    
    # 計算每張圖的 IoU
    all_predictions = []
    
    for pred, gt in zip(predictions, ground_truths):
        for p in pred:
            matched = False
            best_iou = 0
            
            for g in gt:
                iou = calculate_iou(p['bbox'], g['bbox'])
                if iou > iou_threshold and iou > best_iou:
                    best_iou = iou
                    matched = True
            
            all_predictions.append({
                'confidence': p['confidence'],
                'matched': matched,
                'class': p['class']
            })
    
    # 計算 AP
    all_predictions.sort(key=lambda x: x['confidence'], reverse=True)
    
    tp = 0
    fp = 0
    precision_recall = []
    
    for pred in all_predictions:
        if pred['matched']:
            tp += 1
        else:
            fp += 1
        
        precision = tp / (tp + fp)
        recall = tp / len(ground_truths)
        precision_recall.append((precision, recall))
    
    # 計算 mAP
    ap = calculate_ap(precision_recall)
    return ap


def calculate_iou(box1, box2):
    """計算 IoU"""
    x1 = max(box1[0], box2[0])
    y1 = max(box1[1], box2[1])
    x2 = min(box1[2], box2[2])
    y2 = min(box1[3], box2[3])
    
    intersection = max(0, x2 - x1) * max(0, y2 - y1)
    
    area1 = (box1[2] - box1[0]) * (box1[3] - box1[1])
    area2 = (box2[2] - box2[0]) * (box2[3] - box2[1])
    union = area1 + area2 - intersection
    
    return intersection / union if union > 0 else 0


def calculate_segmentation_iou(pred_mask, gt_mask):
    """語義分割 IoU"""
    intersection = np.logical_and(pred_mask, gt_mask).sum()
    union = np.logical_or(pred_mask, gt_mask).sum()
    return intersection / union if union > 0 else 0
```

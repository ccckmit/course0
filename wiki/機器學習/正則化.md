# 正則化：防止過擬合的核心技術

## 概述

正則化（Regularization）是機器學習和深度學習中防止過擬合（Overfitting）的核心技術。當模型在訓練數據上表現良好但在測試數據上表現較差時，我們稱模型過擬合了。過擬合的根源是模型複雜度過高，學習了訓練數據中的噪聲和細節，而非潜在的規律。正則化通過在損失函數中添加額外的約束或懲罰項，鼓勵模型學習更簡單、更通用的表示。

正則化技術可以分為兩大類：顯式正則化和隱式正則化。顯式正則化直接在目標函數中添加正則項，如 L1 正則化（Lasso）、L2 正則化（Ridge）；隱式正則化則通過優化過程的動態特性間接實現，如 Dropout、早停（Early Stopping）和資料增強。理解正則化的原理對於設計高效的機器學習系統至關重要。

## L1 和 L2 正則化

### 理論基礎

```python
import numpy as np
import torch
import torch.nn as nn
import torch.nn.functional as F

# ===== 正則化的數學形式 =====

"""
標準最小二乘:
    min ||y - Xw||²

L2 正則化 (Ridge):
    min ||y - Xw||² + λ||w||²
    懲罰權重的平方和，傾向於使權重趨於零但不完全為零

L1 正則化 (Lasso):
    min ||y - Xw||² + λ||w||₁
    懲罰權重的絕對值和，傾向於使權重恰好為零（稀疏性）

Elastic Net:
    min ||y - Xw||² + λ₁||w||₁ + λ₂||w||²
    結合 L1 和 L2 的優點
"""

# ===== NumPy 實現 =====

def ridge_regression(X, y, alpha=1.0):
    """
    Ridge 迴歸 (L2 正則化)
    閉式解: w = (XᵀX + λI)⁻¹Xᵀy
    """
    n_features = X.shape[1]
    I = np.eye(n_features)
    return np.linalg.inv(X.T @ X + alpha * I) @ X.T @ y

def lasso_regression(X, y, alpha=1.0, max_iter=1000, tol=1e-4):
    """
    Lasso 迴歸 (L1 正則化)
    使用座標下降法求解
    """
    n_samples, n_features = X.shape
    
    # 標準化特徵
    X_mean = X.mean(axis=0)
    X_std = X.std(axis=0)
    X_std[X_std == 0] = 1
    X_norm = (X - X_mean) / X_std
    
    # 初始化權重
    w = np.zeros(n_features)
    
    for _ in range(max_iter):
        w_old = w.copy()
        
        for j in range(n_features):
            # 計算殘差
            residual = y - X_norm @ w + w[j] * X_norm[:, j]
            
            # 軟閾值操作
            rho_j = X_norm[:, j] @ residual
            w[j] = soft_threshold(rho_j, alpha) / (X_norm[:, j] @ X_norm[:, j])
        
        # 收斂檢查
        if np.max(np.abs(w - w_old)) < tol:
            break
    
    # 反標準化
    w = w / X_std
    return w

def soft_threshold(x, threshold):
    """軟閾值函數"""
    return np.sign(x) * np.maximum(np.abs(x) - threshold, 0)

def elastic_net_regression(X, y, alpha=1.0, l1_ratio=0.5, max_iter=1000):
    """
    Elastic Net 迴歸
    結合 L1 和 L2 正則化
    """
    n_samples, n_features = X.shape
    
    # 標準化
    X_mean = X.mean(axis=0)
    X_std = X.std(axis=0)
    X_std[X_std == 0] = 1
    X_norm = (X - X_mean) / X_std
    
    w = np.zeros(n_features)
    
    for _ in range(max_iter):
        for j in range(n_features):
            residual = y - X_norm @ w + w[j] * X_norm[:, j]
            rho_j = X_norm[:, j] @ residual
            
            # Elastic Net 的軟閾值
            if rho_j < alpha * l1_ratio - 1e-6:
                w[j] = (rho_j + alpha * l1_ratio) / (X_norm[:, j] @ X_norm[:, j])
            elif rho_j > alpha * l1_ratio + 1e-6:
                w[j] = (rho_j - alpha * l1_ratio) / (X_norm[:, j] @ X_norm[:, j])
            else:
                w[j] = 0
    
    w = w / X_std
    return w
```

```python
# ===== PyTorch L2 正則化 =====

class L2Regularization:
    """手動實現 L2 正則化"""
    
    def __init__(self, model, lambda_l2=0.01):
        self.model = model
        self.lambda_l2 = lambda_l2
    
    def penalty(self):
        """計算 L2 懲罰項"""
        l2_loss = 0.0
        for param in self.model.parameters():
            l2_loss += torch.sum(param ** 2)
        return self.lambda_l2 * l2_loss
    
    def loss(self, output, target):
        """總損失 = 任務損失 + L2 懲罰"""
        task_loss = F.cross_entropy(output, target)
        return task_loss + self.penalty()

class L1Regularization:
    """L1 正則化"""
    
    def __init__(self, model, lambda_l1=0.01):
        self.model = model
        self.lambda_l1 = lambda_l1
    
    def penalty(self):
        """計算 L1 懲罰項"""
        l1_loss = 0.0
        for param in self.model.parameters():
            l1_loss += torch.sum(torch.abs(param))
        return self.lambda_l1 * l1_loss

class CombinedL1L2:
    """組合 L1 和 L2 正則化"""
    
    def __init__(self, model, lambda_l1=0.01, lambda_l2=0.01):
        self.model = model
        self.lambda_l1 = lambda_l1
        self.lambda_l2 = lambda_l2
    
    def penalty(self):
        l1_loss = sum(torch.sum(torch.abs(p)) for p in self.model.parameters())
        l2_loss = sum(torch.sum(p ** 2) for p in self.model.parameters())
        return self.lambda_l1 * l1_loss + self.lambda_l2 * l2_loss
```

## Dropout 和隨機化正則化

```python
class Dropout(nn.Module):
    """標準 Dropout"""
    
    def __init__(self, p=0.5):
        super().__init__()
        self.p = p  # 丟棄概率
    
    def forward(self, x):
        if not self.training:
            return x
        
        # Bernoulli 分佈采樣
        mask = torch.rand_like(x) > self.p
        return mask * x / (1 - self.p)  # 訓練時保持期望值不變

class Dropout2d(nn.Module):
    """Spatial Dropout (2D) - 丟棄整個通道"""
    
    def __init__(self, p=0.5):
        super().__init__()
        self.p = p
    
    def forward(self, x):
        # x shape: (N, C, H, W)
        if not self.training:
            return x
        
        # 對通道維度應用 Dropout
        mask = torch.rand(x.size(0), x.size(1), 1, 1, device=x.device) > self.p
        return x * mask / (1 - self.p)

class DropBlock(nn.Module):
    """
    DropBlock - 丟棄連續區域而非單個神經元
    在 CNN 中比標準 Dropout 更有效
    """
    
    def __init__(self, block_size=7, p=0.1):
        super().__init__()
        self.block_size = block_size
        self.p = p
    
    def forward(self, x):
        if not self.training:
            return x
        
        N, C, H, W = x.shape
        
        # 計算 gamma
        gamma = self.p * H * W / (
            self.block_size ** 2 * 
            (H - self.block_size + 1) * 
            (W - self.block_size + 1)
        )
        
        # 創建 drop mask
        mask = (torch.rand(N, C, H - self.block_size + 1, W - self.block_size + 1) 
                > gamma).float()
        
        # 擴展 mask
        mask = F.pad(mask, (self.block_size // 2,) * 4, value=1)
        
        return x * mask

class Stochastic Depth(nn.Module):
    """
    Stochastic Depth - 隨機跳過殘差塊
    訓練時隨機丟棄某些殘差路徑
    """
    
    def __init__(self, block, survival_prob=0.8):
        super().__init__()
        self.block = block
        self.survival_prob = survival_prob
    
    def forward(self, x):
        if self.training:
            if torch.rand(1).item() < self.survival_prob:
                return self.block(x) + x
            else:
                return x
        else:
            # 測試時按比例縮放
            return self.survival_prob * self.block(x) + x

class DropConnect(nn.Module):
    """DropConnect - 連接級別的 Dropout"""
    
    def __init__(self, p=0.5):
        super().__init__()
        self.p = p
    
    def forward(self, x, weight):
        if not self.training:
            return F.linear(x, weight)
        
        mask = (torch.rand(weight.size(0), device=weight.device) > self.p).unsqueeze(1)
        masked_weight = weight * mask / (1 - self.p)
        return F.linear(x, masked_weight)
```

```python
class EarlyStopping:
    """早停策略"""
    
    def __init__(self, patience=10, min_delta=0.001, mode='min'):
        self.patience = patience
        self.min_delta = min_delta
        self.mode = mode
        self.counter = 0
        self.best_score = None
        self.early_stop = False
        self.best_state = None
    
    def __call__(self, score, model):
        if self.best_score is None:
            self.best_score = score
            self.best_state = model.state_dict().copy()
        elif self._is_improvement(score):
            self.best_score = score
            self.best_state = model.state_dict().copy()
            self.counter = 0
        else:
            self.counter += 1
            if self.counter >= self.patience:
                self.early_stop = True
    
    def _is_improvement(self, score):
        if self.mode == 'min':
            return score < self.best_score - self.min_delta
        else:
            return score > self.best_score + self.min_delta
    
    def restore(self, model):
        model.load_state_dict(self.best_state)

class Mixup:
    """Mixup 資料增強"""
    
    def __init__(self, alpha=1.0):
        self.alpha = alpha
    
    def __call__(self, x, y):
        batch_size = x.size(0)
        indices = torch.randperm(batch_size)
        
        lam = np.random.beta(self.alpha, self.alpha)
        
        mixed_x = lam * x + (1 - lam) * x[indices]
        y_a, y_b = y, y[indices]
        
        return mixed_x, y_a, y_b, lam

def mixup_criterion(criterion, pred, y_a, y_b, lam):
    """Mixup 的損失計算"""
    return lam * criterion(pred, y_a) + (1 - lam) * criterion(pred, y_b)

class CutMix:
    """CutMix 資料增強"""
    
    def __init__(self, alpha=1.0):
        self.alpha = alpha
    
    def __call__(self, x, y):
        batch_size = x.size(0)
        indices = torch.randperm(batch_size)
        
        lam = np.random.beta(self.alpha, self.alpha)
        
        bbx1, bby1, bbx2, bby2 = self._rand_bbox(x.size(), lam)
        
        x[:, :, bbx1:bbx2, bby1:bby2] = x[indices, :, bbx1:bbx2, bby1:bby2]
        
        lam = 1 - ((bbx2 - bbx1) * (bby2 - bby1) / (x.size()[-1] * x.size()[-2]))
        
        return x, y, y[indices], lam
    
    def _rand_bbox(self, size, lam):
        W = size[2]
        H = size[3]
        cut_rat = np.sqrt(1. - lam)
        cut_w = int(W * cut_rat)
        cut_h = int(H * cut_rat)
        
        cx = np.random.randint(W)
        cy = np.random.randint(H)
        
        bbx1 = np.clip(cx - cut_w // 2, 0, W)
        bby1 = np.clip(cy - cut_h // 2, 0, H)
        bbx2 = np.clip(cx + cut_w // 2, 0, W)
        bby2 = np.clip(cy + cut_h // 2, 0, H)
        
        return bbx1, bby1, bbx2, bby2
```

## Batch Normalization 和權重初始化

```python
class BatchNorm2d(nn.Module):
    """Batch Normalization - 隱式正則化"""
    
    def __init__(self, num_features, eps=1e-5, momentum=0.1):
        super().__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        
        # 可學習參數
        self.gamma = nn.Parameter(torch.ones(num_features))
        self.beta = nn.Parameter(torch.zeros(num_features))
        
        # 移動平均（用於推理）
        self.register_buffer('running_mean', torch.zeros(num_features))
        self.register_buffer('running_var', torch.ones(num_features))
        self.running_mean.zero_()
        self.running_var.fill_(1.)
    
    def forward(self, x):
        if self.training:
            # 訓練時：使用 batch 統計
            mean = x.mean(dim=(0, 2, 3), keepdim=True)
            var = x.var(dim=(0, 2, 3), keepdim=True, unbiased=False)
            
            # 更新移動平均
            self.running_mean = (1 - self.momentum) * self.running_mean + \
                               self.momentum * mean.squeeze()
            self.running_var = (1 - self.momentum) * self.running_var + \
                              self.momentum * var.squeeze()
        else:
            # 推理時：使用移動平均
            mean = self.running_mean.view(1, -1, 1, 1)
            var = self.running_var.view(1, -1, 1, 1)
        
        # 標準化
        x_hat = (x - mean) / torch.sqrt(var + self.eps)
        
        # 縮放和平移
        return self.gamma.view(1, -1, 1, 1) * x_hat + \
               self.beta.view(1, -1, 1, 1)

class WeightInitialization:
    """權重初始化"""
    
    @staticmethod
    def kaiming_normal(model, mode='fan_out', nonlinearity='relu'):
        """Kaiming 正態初始化"""
        for m in model.modules():
            if isinstance(m, (nn.Conv2d, nn.Linear)):
                nn.init.kaiming_normal_(
                    m.weight, 
                    mode=mode, 
                    nonlinearity=nonlinearity
                )
                if m.bias is not None:
                    nn.init.constant_(m.bias, 0)
            elif isinstance(m, (nn.BatchNorm2d, nn.BatchNorm1d)):
                nn.init.constant_(m.weight, 1)
                nn.init.constant_(m.bias, 0)
    
    @staticmethod
    def xavier_uniform(model):
        """Xavier 均勻初始化"""
        for m in model.modules():
            if isinstance(m, (nn.Conv2d, nn.Linear)):
                nn.init.xavier_uniform_(m.weight)
                if m.bias is not None:
                    nn.init.constant_(m.bias, 0)
    
    @staticmethod
    def orthogonal(model):
        """正交初始化"""
        for m in model.modules():
            if isinstance(m, (nn.Conv2d, nn.Linear)):
                nn.init.orthogonal_(m.weight)
                if m.bias is not None:
                    nn.init.constant_(m.bias, 0)
```

```python
class LabelSmoothingLoss(nn.Module):
    """標籤平滑"""
    
    def __init__(self, num_classes, smoothing=0.1):
        super().__init__()
        self.num_classes = num_classes
        self.smoothing = smoothing
        self.confidence = 1.0 - smoothing
    
    def forward(self, pred, target):
        pred = pred.log_softmax(dim=-1)
        
        with torch.no_grad():
            true_dist = torch.zeros_like(pred)
            true_dist.fill_(self.smoothing / (self.num_classes - 1))
            true_dist.scatter_(1, target.unsqueeze(1), self.confidence)
        
        return torch.mean(torch.sum(-true_dist * pred, dim=-1))

class VirtualBatchNormalization(nn.Module):
    """虛擬批次正規化 - 減少 BatchNorm 的 batch 依賴"""
    
    def __init__(self, num_features, eps=1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps
        
        self.gamma = nn.Parameter(torch.ones(num_features))
        self.beta = nn.Parameter(torch.zeros(num_features))
        
        self.register_buffer('mean', torch.zeros(num_features))
        self.register_buffer('var', torch.ones(num_features))
    
    def forward(self, x, ref_mean, ref_var):
        """使用參考 batch 的統計量"""
        x_hat = (x - ref_mean.view(1, -1, 1, 1)) / \
                torch.sqrt(ref_var.view(1, -1, 1, 1) + self.eps)
        return self.gamma.view(1, -1, 1, 1) * x_hat + \
               self.beta.view(1, -1, 1, 1)

class MixUpBatchNorm(nn.Module):
    """支持 Mixup 的 BatchNorm"""
    
    def __init__(self, num_features):
        super().__init__()
        self.bn = nn.BatchNorm2d(num_features)
        self.num_features = num_features
    
    def forward(self, x):
        if self.training:
            # 分別計算每個樣本的統計量
            mean = x.mean(dim=(0, 2, 3))
            var = x.var(dim=(0, 2, 3), unbiased=False)
            
            # 獨立標準化每個樣本
            x_norm = (x - mean.unsqueeze(0).unsqueeze(-1).unsqueeze(-1)) / \
                     torch.sqrt(var.unsqueeze(0).unsqueeze(-1).unsqueeze(-1) + 1e-5)
            
            return self.bn.weight.view(1, -1, 1, 1) * x_norm + \
                   self.bn.bias.view(1, -1, 1, 1)
        else:
            return self.bn(x)

class GradientPenalty(nn.Module):
    """梯度懲罰 - 用於 GAN"""
    
    def forward(self, discriminator, real_data, fake_data):
        batch_size = real_data.size(0)
        
        alpha = torch.rand(batch_size, 1, 1, 1, device=real_data.device)
        interpolates = alpha * real_data + (1 - alpha) * fake_data
        
        interpolates = interpolates.requires_grad_(True)
        
        disc_interpolates = discriminator(interpolates)
        
        gradients = torch.autograd.grad(
            outputs=disc_interpolates,
            inputs=interpolates,
            grad_outputs=torch.ones_like(disc_interpolates),
            create_graph=True,
            retain_graph=True
        )[0]
        
        gradients = gradients.view(batch_size, -1)
        gradient_norm = gradients.norm(2, dim=1)
        
        penalty = ((gradient_norm - 1) ** 2).mean()
        return penalty
```

## 正則化策略選擇

```python
# 正則化策略選擇指南

"""
選擇正則化方法時考慮的因素:

1. 數據集大小
   - 小數據集: 更強的正則化 (L1+L2, Dropout)
   - 大數據集: 較弱的正則化或無正則化

2. 模型複雜度
   - 簡單模型: 較少正則化
   - 深度模型: 需要更多正則化 (Dropout, BatchNorm)

3. 任務類型
   - 圖像分類: Dropout, Data Augmentation, Label Smoothing
   - 語義分割: Dropout, Auxiliary Losses
   - 目標檢測: DropPath (Stochastic Depth)

4. 類別不平衡
   - 類別加權
   - Focal Loss
   - 過採樣/欠採樣

5. 推薦的正則化組合:

   圖像分類 (CNN):
   - Data Augmentation (翻轉、旋轉、裁剪、顏色變換)
   - Dropout (p=0.5)
   - Weight Decay (L2, λ=1e-4)
   - Batch Normalization
   - Label Smoothing (ε=0.1)
   - Mixup / CutMix

   語言模型:
   - Dropout (p=0.1-0.3)
   - Weight Decay (λ=0.01, AdamW)
   - Embedding Dropout
   - Label Smoothing

   Transformer:
   - Dropout (p=0.1)
   - DropPath (Stochastic Depth, p=0.1)
   - Weight Decay
   - LayerNorm
"""
```

## 總結

正則化是防止機器學習模型過擬合的核心技術。本頁介紹了主要的正則化方法：顯式正則化（L1、L2、Elastic Net）、隨機化正則化（Dropout、Stochastic Depth）、資料增強（Mixup、CutMix）、以及批歸一化和權重初始化等隱式正則化技術。在實際應用中，通常需要組合多種正則化方法，並根據具體任務和數據特點進行調整。記住：正則化的強度需要根據數據集大小、模型複雜度和任務難度來權衡。

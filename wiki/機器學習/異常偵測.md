# 異常偵測：識別偏離正常模式的數據

## 概述

異常偵測（Anomaly Detection）是識別偏離預期行為或模式的數據點的技術。這些異常點可能表示欺詐行為、設備故障、網路入侵或疾病徵兆。異常偵測的挑戰在於：異常樣本往往很少，缺乏標註；正常和異常的邊界可能不明顯；數據可能來自多種分佈。

異常偵測方法可分為三大類：**有監督方法**使用標註數據訓練分類器；**半監督方法**僅使用正常數據訓練；**無監督方法**在無標籤數據中發現異常。本頁介紹主要的異常偵測技術和Python實現。

## 統計方法

```python
import numpy as np
from scipy import stats

class ZScoreAnomalyDetector:
    """Z-Score 異常偵測"""
    
    def __init__(self, threshold=3):
        self.threshold = threshold
        self.mean = None
        self.std = None
    
    def fit(self, X):
        self.mean = np.mean(X, axis=0)
        self.std = np.std(X, axis=0)
        return self
    
    def predict(self, X):
        z_scores = np.abs((X - self.mean) / (self.std + 1e-8))
        return (z_scores > self.threshold).any(axis=1)

class IQRAnomalyDetector:
    """IQR（四分位距）異常偵測"""
    
    def __init__(self, factor=1.5):
        self.factor = factor
    
    def fit(self, X):
        self.q1 = np.percentile(X, 25, axis=0)
        self.q3 = np.percentile(X, 75, axis=0)
        self.iqr = self.q3 - self.q1
        self.lower_bound = self.q1 - self.factor * self.iqr
        self.upper_bound = self.q3 + self.factor * self.iqr
        return self
    
    def predict(self, X):
        return ((X < self.lower_bound) | (X > self.upper_bound)).any(axis=1)

class MahalanobisDetector:
    """馬氏距離異常偵測"""
    
    def __init__(self, threshold=None):
        self.threshold = threshold
        self.mean = None
        self.cov_inv = None
    
    def fit(self, X):
        self.mean = np.mean(X, axis=0)
        cov = np.cov(X, rowvar=False)
        self.cov_inv = np.linalg.pinv(cov)
        return self
    
    def mahalanobis_distance(self, X):
        diff = X - self.mean
        return np.sqrt(np.sum(diff @ self.cov_inv * diff, axis=1))
    
    def predict(self, X):
        distances = self.mahalanobis_distance(X)
        if self.threshold is None:
            self.threshold = np.percentile(distances, 95)
        return distances > self.threshold
```

## 基於重建的方法

```python
import torch
import torch.nn as nn

class AutoencoderAnomalyDetector(nn.Module):
    """自編碼器異常偵測"""
    
    def __init__(self, input_dim, encoding_dim=32):
        super().__init__()
        
        # Encoder
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, 128),
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, encoding_dim),
            nn.ReLU()
        )
        
        # Decoder
        self.decoder = nn.Sequential(
            nn.Linear(encoding_dim, 64),
            nn.ReLU(),
            nn.Linear(64, 128),
            nn.ReLU(),
            nn.Linear(128, input_dim)
        )
    
    def forward(self, x):
        encoded = self.encoder(x)
        decoded = self.decoder(encoded)
        return decoded
    
    def anomaly_score(self, x):
        """計算重構誤差作為異常分數"""
        with torch.no_grad():
            reconstructed = self.forward(x)
            mse = torch.mean((x - reconstructed) ** 2, dim=1)
        return mse

def train_autoencoder(X_train, encoding_dim=32, epochs=100, batch_size=256):
    """訓練自編碼器"""
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    
    input_dim = X_train.shape[1]
    model = AutoencoderAnomalyDetector(input_dim, encoding_dim).to(device)
    optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
    criterion = nn.MSELoss()
    
    dataset = torch.tensor(X_train, dtype=torch.float32)
    dataloader = torch.utils.data.DataLoader(dataset, batch_size=batch_size, shuffle=True)
    
    model.train()
    for epoch in range(epochs):
        total_loss = 0
        for batch in dataloader:
            batch = batch.to(device)
            optimizer.zero_grad()
            
            reconstructed = model(batch)
            loss = criterion(reconstructed, batch)
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
        
        if (epoch + 1) % 10 == 0:
            print(f'Epoch {epoch+1}, Loss: {total_loss/len(dataloader):.4f}')
    
    return model

class VariationalAutoencoder(nn.Module):
    """變分自編碼器用於異常偵測"""
    
    def __init__(self, input_dim, latent_dim=16):
        super().__init__()
        
        self.latent_dim = latent_dim
        
        # Encoder
        self.fc1 = nn.Linear(input_dim, 256)
        self.fc21 = nn.Linear(256, latent_dim)  # mean
        self.fc22 = nn.Linear(256, latent_dim)  # logvar
        
        # Decoder
        self.fc3 = nn.Linear(latent_dim, 256)
        self.fc4 = nn.Linear(256, input_dim)
    
    def encode(self, x):
        h = torch.relu(self.fc1(x))
        return self.fc21(h), self.fc22(h)
    
    def reparameterize(self, mu, logvar):
        std = torch.exp(0.5 * logvar)
        eps = torch.randn_like(std)
        return mu + eps * std
    
    def decode(self, z):
        h = torch.relu(self.fc3(z))
        return self.fc4(h)
    
    def forward(self, x):
        mu, logvar = self.encode(x)
        z = self.reparameterize(mu, logvar)
        return self.decode(z), mu, logvar
    
    def anomaly_score(self, x):
        """使用重構概率作為異常分數"""
        with torch.no_grad():
            reconstructed, mu, logvar = self.forward(x)
            
            # 重構誤差
            recon_error = torch.mean((x - reconstructed) ** 2, dim=1)
            
            # KL 散度
            kl = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp(), dim=1)
            
            # 組合分數
            return recon_error + 0.1 * kl
```

## 集成方法

```python
from sklearn.ensemble import IsolationForest
from sklearn.svm import OneClassSVM
from sklearn.neighbors import LocalOutlierFactor

class IsolationForestDetector:
    """隔離森林"""
    
    def __init__(self, n_estimators=100, contamination=0.1, random_state=42):
        self.model = IsolationForest(
            n_estimators=n_estimators,
            contamination=contamination,
            random_state=random_state
        )
    
    def fit_predict(self, X):
        """擬合並預測"""
        return self.model.fit_predict(X)
    
    def anomaly_scores(self, X):
        """返回異常分數（越小越異常）"""
        return self.model.decision_function(X)

class OneClassSVMDetector:
    """One-Class SVM"""
    
    def __init__(self, kernel='rbf', nu=0.1, gamma='scale'):
        self.model = OneClassSVM(kernel=kernel, nu=nu, gamma=gamma)
    
    def fit_predict(self, X):
        predictions = self.model.fit_predict(X)
        return predictions == -1  # -1 表示異常

class LocalOutlierFactorDetector:
    """局部離群因子"""
    
    def __init__(self, n_neighbors=20, contamination=0.1):
        self.model = LocalOutlierFactor(
            n_neighbors=n_neighbors,
            contamination=contamination,
            novelty=True  # 用於新樣本預測
        )
    
    def fit_predict(self, X):
        return self.model.fit_predict(X) == -1
    
    def anomaly_scores(self, X):
        return -self.model.decision_function(X)
```

```python
class DeepSVDD(nn.Module):
    """Deep SVDD (Support Vector Data Description)"""
    
    def __init__(self, input_dim, hidden_dims=[128, 64, 32]):
        super().__init__()
        
        layers = []
        prev_dim = input_dim
        
        for hidden_dim in hidden_dims:
            layers.extend([
                nn.Linear(prev_dim, hidden_dim),
                nn.BatchNorm1d(hidden_dim),
                nn.ReLU()
            ])
            prev_dim = hidden_dim
        
        self.encoder = nn.Sequential(*layers)
        self.radius = nn.Parameter(torch.tensor(0.1))
    
    def forward(self, x):
        return self.encoder(x)
    
    def anomaly_score(self, x, c=None):
        """
        異常分數 = ||f(x) - c||²
        c: 中心點（可學習或固定）
        """
        features = self.forward(x)
        if c is None:
            c = torch.zeros_like(features[0])
        distances = torch.sum((features - c) ** 2, dim=1)
        return distances

class GANomaly(nn.Module):
    """GANomaly: 基於 GAN 的異常偵測"""
    
    def __init__(self, input_dim):
        super().__init__()
        
        # Encoder E
        self.enc1 = nn.Linear(input_dim, 512)
        self.enc2 = nn.Linear(512, 256)
        self.enc3 = nn.Linear(256, 128)
        
        # Decoder G
        self.dec1 = nn.Linear(128, 256)
        self.dec2 = nn.Linear(256, 512)
        self.dec3 = nn.Linear(512, input_dim)
        
        # Discriminator
        self.disc = nn.Sequential(
            nn.Linear(input_dim, 256),
            nn.LeakyReLU(0.2),
            nn.Linear(256, 128),
            nn.LeakyReLU(0.2),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
    
    def encode(self, x):
        h = torch.relu(self.enc1(x))
        h = torch.relu(self.enc2(h))
        return torch.relu(self.enc3(h))
    
    def decode(self, z):
        h = torch.relu(self.dec1(z))
        h = torch.relu(self.dec2(h))
        return self.dec3(h)
    
    def forward(self, x):
        z = self.encode(x)
        x_hat = self.decode(z)
        z_hat = self.encode(x_hat)
        return x_hat, z, z_hat
    
    def anomaly_score(self, x):
        with torch.no_grad():
            x_hat, z, z_hat = self.forward(x)
            
            # 多尺度異常分數
            recon_error = torch.mean((x - x_hat) ** 2, dim=1)
            latent_error = torch.mean((z - z_hat) ** 2, dim=1)
            
            return recon_error + latent_error
```

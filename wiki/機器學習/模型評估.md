# 模型評估：機器學習效能度量

## 概述

模型評估是機器學習工作流程中至關重要的環節，它決定了我們如何客觀地衡量模型在未見數據上的表現。一個訓練誤差很低的模型可能在測試數據上表現很差，這稱為過擬合（Overfitting）；反之，如果模型連訓練數據都無法擬合，則是欠擬合（Underfitting）。正確的評估方法能幫助我們選擇最佳模型、診斷問題並指導超參數調優。

評估指標的選擇應該與任務目標一致。對於分類問題，我們關心準確率、精確率、召回率、F1 分數和 AUC-ROC；对于回归问题，则关注 MSE、RMSE、MAE 和 R²；而對於排序任務，則使用 NDCG 和 MAP。正確的驗證策略（如交叉驗證）同樣重要，它能確保評估結果的穩定性和可靠性。

### 基礎指標

```python
import numpy as np
from sklearn.metrics import (
    accuracy_score, precision_score, recall_score, f1_score,
    confusion_matrix, classification_report, roc_auc_score,
    precision_recall_curve, average_precision_score
)
from sklearn.model_selection import cross_val_score, StratifiedKFold
import matplotlib.pyplot as plt

# 混淆矩陣的組成
"""
                預測為正例        預測為負例
實際為正例         TP               FN
實際為負例         FP               TN

真正例 (TP): 正確預測為正
真負例 (TN): 正確預測為負
假正例 (FP): 錯誤預測為正 (Type I Error)
假負例 (FN): 錯誤預測為負 (Type II Error)
"""

def calculate_classification_metrics(y_true, y_pred, y_prob=None):
    """計算完整的分類評估指標"""
    
    # 基礎指標
    accuracy = accuracy_score(y_true, y_pred)
    precision = precision_score(y_true, y_pred, average='weighted')
    recall = recall_score(y_true, y_pred, average='weighted')
    f1 = f1_score(y_true, y_pred, average='weighted')
    
    # 混淆矩陣
    cm = confusion_matrix(y_true, y_pred)
    
    # 每個類別的指標
    precision_per_class = precision_score(y_true, y_pred, average=None)
    recall_per_class = recall_score(y_true, y_pred, average=None)
    f1_per_class = f1_score(y_true, y_pred, average=None)
    
    results = {
        'accuracy': accuracy,
        'precision': precision,
        'recall': recall,
        'f1': f1,
        'confusion_matrix': cm,
        'precision_per_class': precision_per_class,
        'recall_per_class': recall_per_class,
        'f1_per_class': f1_per_class
    }
    
    # 如果有概率輸出，計算 AUC
    if y_prob is not None:
        if len(y_prob.shape) > 1 and y_prob.shape[1] > 1:
            # 多類別：使用 One-vs-Rest
            auc = roc_auc_score(y_true, y_prob, multi_class='ovr')
        else:
            auc = roc_auc_score(y_true, y_prob)
        results['auc'] = auc
    
    return results

# 詳細的分類報告
def detailed_classification_report(y_true, y_pred, class_names=None):
    """生成詳細的分類報告"""
    report = classification_report(
        y_true, y_pred, 
        target_names=class_names,
        digits=4,
        output_dict=True
    )
    return report

# 混淆矩陣可視化
def plot_confusion_matrix(cm, class_names, normalize=False, save_path=None):
    """繪製混淆矩陣熱圖"""
    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]
    
    fig, ax = plt.subplots(figsize=(10, 8))
    im = ax.imshow(cm, interpolation='nearest', cmap=plt.cm.Blues)
    ax.figure.colorbar(im, ax=ax)
    
    ax.set(xticks=np.arange(cm.shape[1]),
           yticks=np.arange(cm.shape[0]),
           xticklabels=class_names,
           yticklabels=class_names,
           title='Confusion Matrix' + (' (Normalized)' if normalize else ''),
           ylabel='True label',
           xlabel='Predicted label')
    
    plt.setp(ax.get_xticklabels(), rotation=45, ha="right", rotation_mode="anchor")
    
    fmt = '.2f' if normalize else 'd'
    thresh = cm.max() / 2.
    for i in range(cm.shape[0]):
        for j in range(cm.shape[1]):
            ax.text(j, i, format(cm[i, j], fmt),
                   ha="center", va="center",
                   color="white" if cm[i, j] > thresh else "black")
    
    fig.tight_layout()
    
    if save_path:
        plt.savefig(save_path, dpi=300, bbox_inches='tight')
    return fig
```

## 回歸評估指標

```python
from sklearn.metrics import (
    mean_squared_error, mean_absolute_error, r2_score,
    mean_absolute_percentage_error, explained_variance_score
)
import numpy as np

def calculate_regression_metrics(y_true, y_pred):
    """計算回歸評估指標"""
    
    # 均方誤差 (MSE)
    mse = mean_squared_error(y_true, y_pred)
    
    # 均方根誤差 (RMSE)
    rmse = np.sqrt(mse)
    
    # 平均絕對誤差 (MAE)
    mae = mean_absolute_error(y_true, y_pred)
    
    # R² 決定係數
    r2 = r2_score(y_true, y_pred)
    
    # 解釋方差
    explained_var = explained_variance_score(y_true, y_pred)
    
    # 平均絕對百分比誤差 (MAPE)
    # 注意：當 y_true 有零值時需要特殊處理
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    
    # 對稱 MAPE (避免除零問題)
    smape = np.mean(2 * np.abs(y_true - y_pred) / 
                   (np.abs(y_true) + np.abs(y_pred))) * 100
    
    return {
        'MSE': mse,
        'RMSE': rmse,
        'MAE': mae,
        'R2': r2,
        'Explained_Variance': explained_var,
        'MAPE': mape,
        'SMAPE': smape
    }

def plot_regression_diagnostics(y_true, y_pred, save_path=None):
    """回歸診斷圖：預測值 vs 實際值、殘差圖"""
    residuals = y_true - y_pred
    
    fig, axes = plt.subplots(2, 2, figsize=(12, 10))
    
    # 1. 預測值 vs 實際值
    ax1 = axes[0, 0]
    ax1.scatter(y_true, y_pred, alpha=0.5)
    max_val = max(y_true.max(), y_pred.max())
    ax1.plot([0, max_val], [0, max_val], 'r--', linewidth=2)
    ax1.set_xlabel('Actual Values')
    ax1.set_ylabel('Predicted Values')
    ax1.set_title('Predicted vs Actual')
    
    # 2. 殘差分佈
    ax2 = axes[0, 1]
    ax2.hist(residuals, bins=50, edgecolor='black', alpha=0.7)
    ax2.axvline(x=0, color='r', linestyle='--')
    ax2.set_xlabel('Residuals')
    ax2.set_ylabel('Frequency')
    ax2.set_title('Residual Distribution')
    
    # 3. 殘差 vs 預測值（檢查異方差性）
    ax3 = axes[1, 0]
    ax3.scatter(y_pred, residuals, alpha=0.5)
    ax3.axhline(y=0, color='r', linestyle='--')
    ax3.set_xlabel('Predicted Values')
    ax3.set_ylabel('Residuals')
    ax3.set_title('Residuals vs Predicted')
    
    # 4. Q-Q 圖（檢查正態性）
    ax4 = axes[1, 1]
    from scipy import stats
    stats.probplot(residuals, dist="norm", plot=ax4)
    ax4.set_title('Q-Q Plot of Residuals')
    
    plt.tight_layout()
    
    if save_path:
        plt.savefig(save_path, dpi=300, bbox_inches='tight')
    return fig
```

```python
from sklearn.model_selection import (
    KFold, StratifiedKFold, LeaveOneOut, cross_val_score,
    cross_val_predict, learning_curve, validation_curve
)
import numpy as np

# K 折交叉驗證
def k_fold_cross_validation(model, X, y, k=5, scoring='accuracy'):
    """標準 K 折交叉驗證"""
    kfold = KFold(n_splits=k, shuffle=True, random_state=42)
    
    scores = cross_val_score(model, X, y, cv=kfold, scoring=scoring)
    
    print(f"K-Fold CV Results (k={k}):")
    print(f"  Scores: {scores}")
    print(f"  Mean: {scores.mean():.4f}")
    print(f"  Std: {scores.std():.4f}")
    print(f"  95% CI: [{scores.mean() - 1.96*scores.std():.4f}, "
          f"{scores.mean() + 1.96*scores.std():.4f}]")
    
    return scores

# 分層 K 折交叉驗證（保持類別比例）
def stratified_k_fold_cv(model, X, y, k=5, scoring='accuracy'):
    """分層 K 折交叉驗證"""
    skfold = StratifiedKFold(n_splits=k, shuffle=True, random_state=42)
    
    scores = cross_val_score(model, X, y, cv=skfold, scoring=scoring)
    return scores

# 留一法交叉驗證 (LOO-CV)
def leave_one_out_cv(model, X, y, scoring='accuracy'):
    """留一法交叉驗證（適用於小數據集）"""
    loo = LeaveOneOut()
    scores = cross_val_score(model, X, y, cv=loo, scoring=scoring)
    
    print(f"LOO-CV Results:")
    print(f"  Number of folds: {len(scores)}")
    print(f"  Mean: {scores.mean():.4f}")
    print(f"  Std: {scores.std():.4f}")
    
    return scores

# 學習曲線
def plot_learning_curve(model, X, y, train_sizes=np.linspace(0.1, 1.0, 10),
                       cv=5, scoring='accuracy', save_path=None):
    """繪製學習曲線"""

        train_sizes=train_sizes,

    train_std = train_scores.std(axis=1)

    ax.plot(train_sizes_abs, train_mean, 'o-', color='blue',
            label='Training score')
    ax.plot(train_sizes_abs, test_mean, 'o-', color='orange',
            label='Cross-validation score')
    
    ax.set_xlabel('Training Examples')
    ax.set_ylabel('Score')
    ax.set_title('Learning Curve')
    ax.legend(loc='best')
    ax.grid(True, alpha=0.3)
    
    if save_path:

def bootstrap_confidence_interval(model, X, y, n_iterations=1000, 
                                  ci=0.95, scoring='accuracy'):
    """使用自助法計算置信區間"""
    scores = []
    n_samples = len(y)
    
    for _ in range(n_iterations):
        indices = np.random.randint(0, n_samples, n_samples)
        X_boot, y_boot = X[indices], y[indices]
        
        model.fit(X_boot, y_boot)
        y_pred = model.predict(X_test)
        score = accuracy_score(y_test, y_pred)
        scores.append(score)

    upper = np.percentile(scores, (1 - alpha) * 100)
    
    print(f"{ci*100:.0f}% Confidence Interval: [{lower:.4f}, {upper:.4f}]")
```

```python
def mcnemar_test(y_true, y_pred1, y_pred2):
    """McNemar 檢驗"""
    # b: model1 錯誤但 model2 正確
    # c: model1 正確但 model2 錯誤
    b = np.sum((y_pred1 != y_true) & (y_pred2 == y_true))
    c = np.sum((y_pred1 == y_true) & (y_pred2 != y_true))
    
    stat = (abs(b - c) - 1) ** 2 / (b + c)
    p_value = 1 - stats.chi2.cdf(stat, df=1)
    
    return stat, p_value, b, c
```

### 評估流程

```python
def comprehensive_model_evaluation(model, X, y, test_size=0.2, 
                                   random_state=42, save_dir=None):
    """完整的模型評估流程"""
    from sklearn.model_selection import train_test_split
    from sklearn.preprocessing import StandardScaler
    
    # 分割數據
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=test_size, random_state=random_state,
        stratify=y if len(np.unique(y)) > 1 else None
    )
    
    # 標準化（如果需要）
    scaler = StandardScaler()
    X_train_scaled = scaler.fit_transform(X_train)
    X_test_scaled = scaler.transform(X_test)
    
    # 訓練模型
    model.fit(X_train_scaled, y_train)
    
    # 預測
    y_train_pred = model.predict(X_train_scaled)
    y_test_pred = model.predict(X_test_scaled)
    
    if hasattr(model, 'predict_proba'):
        y_train_prob = model.predict_proba(X_train_scaled)
        y_test_prob = model.predict_proba(X_test_scaled)
    else:
        y_train_prob = None
        y_test_prob = None
    
    # 計算指標
    results = {
        'train': calculate_classification_metrics(y_train, y_train_pred, y_train_prob),
        'test': calculate_classification_metrics(y_test, y_test_pred, y_test_prob)
    }
    
    # 交叉驗證
    cv_scores = cross_val_score(model, X_train_scaled, y_train, cv=5)
    results['cv_scores'] = cv_scores
    results['cv_mean'] = cv_scores.mean()
    results['cv_std'] = cv_scores.std()
    
    # 診斷過擬合
    results['overfitting'] = {
        'train_accuracy': results['train']['accuracy'],
        'test_accuracy': results['test']['accuracy'],
        'gap': results['train']['accuracy'] - results['test']['accuracy']
    }
    
    return results

# 類別不平衡處理
def handle_imbalanced_evaluation(y_true, y_pred, y_prob=None):
    """處理類別不平衡的評估"""
    from sklearn.metrics import balanced_accuracy_score
    
    # 平衡準確率
    balanced_acc = balanced_accuracy_score(y_true, y_pred)
    
    # 每個類別的召回率（真正例率）
    report = classification_report(y_true, y_pred, output_dict=True)
    
    # 類別分佈
    unique, counts = np.unique(y_true, return_counts=True)
    class_distribution = dict(zip(unique, counts))
    
    return {
        'balanced_accuracy': balanced_acc,
        'classification_report': report,
        'class_distribution': class_distribution
    }
```

## 總結

模型評估是機器學習中不可或缺的一環。正確選擇評估指標、采用恰當的驗證策略，並進行統計顯著性檢驗，能確保我們的結論可靠。對於不同類型的任務，應選擇相應的指標：分類任務關注 AUC、Precision、Recall；回歸任務關注 RMSE、MAE、R²；不平衡數據集則需使用 Balanced Accuracy 和 Macro-F1。同時需要注意過擬合診斷，通過學習曲線和交叉驗證確保模型的泛化能力。

# 循環神經網路

循環神經網路（Recurrent Neural Network，簡稱 RNN）是一類專門處理序列資料的神經網路架構。與傳統的前饋神經網路不同，RNN 具有內部狀態（hidden state），能夠記住過去的資訊，這使它特別適合處理具有時間依賴性的資料，如自然語言、音訊、時間序列等。RNN 的設計靈感來自於人類理解語言的方式——我們閱讀時會記住前面的內容來理解當前的句子，這種「記憶」機制正是 RNN 所擅長的。

## RNN 的核心概念

### RNN 的基本結構

```
┌─────────────────────────────────────────────────────────────┐
│                    RNN 結構                                │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   標準 RNN 表示：                                           │
│                                                             │
│   x₀ ──┐                                                       │
│        │                                                       │
│   x₁ ──┼──→ ┌───┐  h₀ ──┐                                    │
│        │    │   │      │                                       │
│   x₂ ──┼──→ │   │  h₁ ─┼──→ h₂ → ...                        │
│        │    │   │      │                                       │
│   x₃ ──┘    └───┘      │                                       │
│                  └──────┘                                   │
│                                                             │
│   展開形式（沿時間）：                                       │
│                                                             │
│   x₀ → h₀ → x₁ → h₁ → x₂ → h₂ → x₃ → h₃ → ...              │
│        ↓       ↓       ↓       ↓                            │
│       h₀      h₁      h₂      h₃                             │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 時間展開

```
┌─────────────────────────────────────────────────────────────┐
│                  RNN 時間展開                               │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   輸入: x₀, x₁, x₂, x₃, x₄                                 │
│                                                             │
│   ┌───┐    ┌───┐    ┌───┐    ┌───┐    ┌───┐               │
│   │x₀ │    │x₁ │    │x₂ │    │x₃ │    │x₄ │               │
│   └───┘    └───┘    └───┘    └───┘    └───┘                │
│    │        │        │        │        │                    │
│    ▼        ▼        ▼        ▼        ▼                   │
│   ┌───┐    ┌───┐    ┌───┐    ┌───┐    ┌───┐               │
│   │h₀ │───→│h₁ │───→│h₂ │───→│h₃ │───→│h₄ │               │
│   └───┘    └───┘    └───┘    └───┘    └───┘                │
│    │        │        │        │        │                    │
│    ▼        ▼        ▼        ▼        ▼                   │
│   y₀       y₁       y₂       y₃       y₄                   │
│                                                             │
│   每個時間步的計算：                                         │
│   h_t = tanh(W_xh · x_t + W_hh · h_{t-1} + b_h)           │
│   y_t = W_hy · h_t + b_y                                    │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## RNN 數學推導

```python
import numpy as np

class SimpleRNN:
    """基本 RNN 實現"""
    
    def __init__(self, input_size, hidden_size, output_size):
        self.input_size = input_size
        self.hidden_size = hidden_size
        
        # 權重初始化
        scale = 0.1
        self.W_xh = np.random.randn(hidden_size, input_size) * scale  # 輸入到隱藏
        self.W_hh = np.random.randn(hidden_size, hidden_size) * scale  # 隱藏到隱藏
        self.W_hy = np.random.randn(output_size, hidden_size) * scale  # 隱藏到輸出
        
        self.b_h = np.zeros((hidden_size, 1))
        self.b_y = np.zeros((output_size, 1))
    
    def forward(self, x_sequence):
        """
        前向傳播
        x_sequence: (seq_len, input_size) 的列表
        """
        seq_len = len(x_sequence)
        h = np.zeros((self.hidden_size, 1))  # 初始隱藏狀態
        outputs = []
        
        for t in range(seq_len):
            x = x_sequence[t].reshape(-1, 1)
            
            # RNN 核心公式
            h = np.tanh(self.W_xh @ x + self.W_hh @ h + self.b_h)
            
            # 輸出
            y = self.W_hy @ h + self.b_y
            outputs.append(y)
        
        return outputs
    
    def backward(self, x_sequence, y_sequence, predictions):
        """
        反向傳播 Through Time (BPTT)
        """
        seq_len = len(x_sequence)
        
        # 初始化梯度
        dW_xh = np.zeros_like(self.W_xh)
        dW_hh = np.zeros_like(self.W_hh)
        dW_hy = np.zeros_like(self.W_hy)
        db_h = np.zeros_like(self.b_h)
        db_y = np.zeros_like(self.b_y)
        
        dh_next = np.zeros((self.hidden_size, 1))
        
        for t in reversed(range(seq_len)):
            x = x_sequence[t].reshape(-1, 1)
            y_true = y_sequence[t].reshape(-1, 1)
            y_pred = predictions[t]
            
            # 輸出梯度
            dy = y_pred - y_true
            dW_hy += dy @ h.T
            db_y += dy
            
            # 隱藏狀態梯度
            dh = self.W_hy.T @ dy + dh_next
            
            # tanh 的梯度
            dh_raw = dh * (1 - h ** 2)
            
            # 各權重梯度
            dW_xh += dh_raw @ x.T
            dW_hh += dh_raw @ h_prev.T if t > 0 else 0
            db_h += dh_raw
            
            # 傳遞給上一個時間步
            dh_next = self.W_hh.T @ dh_raw
        
        return dW_xh, dW_hh, dW_hy, db_h, db_y
```

## 標準 RNN 的梯度問題

### 梯度消失與爆炸

```python
# 梯度問題演示
def analyze_gradient_flow(W_hh, seq_length):
    """
    分析梯度沿時間的變化
    """
    eigenvalues = np.linalg.eigvals(W_hh)
    max_eigenvalue = np.max(np.abs(eigenvalues))
    
    print(f"最大特徵值: {max_eigenvalue:.4f}")
    
    if max_eigenvalue > 1:
        print(f"梯度會爆炸！(第 {seq_length} 步後 ~{max_eigenvalue ** seq_length:.2e}x)")
    else:
        print(f"梯度會消失！(第 {seq_length} 步後 ~{max_eigenvalue ** seq_length:.2e}x)")

# 示例
W_hh = np.array([[0.9, 0.2], [0.1, 0.8]])
analyze_gradient_flow(W_hh, seq_length=20)
```

```
梯度消失的後果：
- 早期時間步的梯度無法傳播到後期
- 難以學習長期依賴
- 模型只能記住最近的資訊

梯度爆炸的後果：
- 梯度過大導致參數更新失控
- 訓練不穩定
- 損失震蕩或發散
```

## LSTM（長短期記憶網路）

LSTM 由 Sepp Hochreiter 和 Jürgen Schmidhuber 於 1997 年提出，是解決梯度問題的經典方案。

### LSTM 核心概念

```
┌─────────────────────────────────────────────────────────────┐
│                    LSTM 結構                               │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   ┌─────────────────────────────────────────────────────┐ │
│   │                   LSTM Cell                          │ │
│   │                                                      │ │
│   │   x_t ──┬──→ ┌────────┐                             │ │
│   │         │    │ Forget  │                             │ │
│   │   h_{t-1}──→ │  Gate   │→ ╳                           │ │
│   │         │    └────────┘                             │ │
│   │         │    ┌────────┐                             │ │
│   │         ├──→ │  Input  │→ ┼─→ C_t                   │ │
│   │         │    │  Gate   │                             │ │
│   │         │    └────────┘                             │ │
│   │         │    ┌────────┐                             │ │
│   │         ├──→ │  Output│→ ──→ h_t                    │ │
│   │         │    │  Gate   │                             │ │
│   │         │    └────────┘                             │ │
│   │                                                      │ │
│   └─────────────────────────────────────────────────────┘ │
│                                                             │
│   公式：                                                    │
│   f_t = σ(W_f · [h_{t-1}, x_t] + b_f)    # 遺忘門         │
│   i_t = σ(W_i · [h_{t-1}, x_t] + b_i)    # 輸入門         │
│   C̃_t = tanh(W_c · [h_{t-1}, x_t] + b_c) # 候選值         │
│   C_t = f_t * C_{t-1} + i_t * C̃_t         # 細胞狀態更新  │
│   o_t = σ(W_o · [h_{t-1}, x_t] + b_o)     # 輸出門         │
│   h_t = o_t * tanh(C_t)                   # 隱藏狀態       │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### PyTorch LSTM 實現

```python
import torch
import torch.nn as nn

class LSTMModel(nn.Module):
    """LSTM 模型"""
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers, dropout=0.2):
        super(LSTMModel, self).__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.lstm = nn.LSTM(
            input_size=embed_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            bidirectional=False,
            dropout=dropout if num_layers > 1 else 0
        )
        self.dropout = nn.Dropout(dropout)
        self.fc = nn.Linear(hidden_size, vocab_size)
    
    def forward(self, x, hidden=None):
        # x: (batch_size, seq_len)
        
        # 嵌入層
        x = self.embedding(x)  # (batch_size, seq_len, embed_size)
        
        # LSTM
        if hidden is None:
            output, (h_n, c_n) = self.lstm(x)
        else:
            output, (h_n, c_n) = self.lstm(x, hidden)
        
        # output: (batch_size, seq_len, hidden_size)
        
        # 取最後一個時間步的輸出
        last_output = output[:, -1, :]  # (batch_size, hidden_size)
        
        # Dropout 和全連接
        last_output = self.dropout(last_output)
        logits = self.fc(last_output)  # (batch_size, vocab_size)
        
        return logits, (h_n, c_n)

# 使用範例
model = LSTMModel(
    vocab_size=10000,
    embed_size=256,
    hidden_size=512,
    num_layers=2
)

# 輸入
batch_size = 32
seq_len = 50
x = torch.randint(0, 10000, (batch_size, seq_len))

# 前向傳播
logits, (h_n, c_n) = model(x)
print(f"Output shape: {logits.shape}")  # torch.Size([32, 10000])
```

### LSTM 門控機制詳解

```python
class LSTMCell(nn.Module):
    """LSTM 單元詳細實現"""
    
    def __init__(self, input_size, hidden_size):
        super(LSTMCell, self).__init__()
        
        self.hidden_size = hidden_size
        
        # 四個門的權重
        self.W_ih = nn.Linear(input_size, hidden_size * 4)
        self.W_hh = nn.Linear(hidden_size, hidden_size * 4)
        self.b_ih = nn.Parameter(torch.zeros(hidden_size * 4))
        self.b_hh = nn.Parameter(torch.zeros(hidden_size * 4))
    
    def forward(self, x, state):
        h, c = state
        
        # 計算四個門
        gates = self.W_ih(x) + self.W_hh(h) + self.b_ih + self.b_hh
        
        # 分割成四個門
        f_gate, i_gate, g_gate, o_gate = gates.chunk(4, dim=-1)
        
        # Sigmoid 激活
        f_gate = torch.sigmoid(f_gate)  # 遺忘門
        i_gate = torch.sigmoid(i_gate)  # 輸入門
        g_gate = torch.tanh(g_gate)     # 候選值
        o_gate = torch.sigmoid(o_gate)  # 輸出門
        
        # 細胞狀態更新
        c_new = f_gate * c + i_gate * g_gate
        
        # 隱藏狀態更新
        h_new = o_gate * torch.tanh(c_new)
        
        return h_new, c_new
```

## GRU（Gated Recurrent Unit）

GRU 是 LSTM 的簡化版本，計算更高效：

```python
class GRUModel(nn.Module):
    """GRU 模型"""
    def __init__(self, input_size, hidden_size, num_layers, dropout=0.2):
        super(GRUModel, self).__init__()
        
        self.gru = nn.GRU(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0
        )
        self.dropout = nn.Dropout(dropout)
        self.fc = nn.Linear(hidden_size, input_size)
    
    def forward(self, x):
        output, hidden = self.gru(x)
        output = self.dropout(output)
        output = self.fc(output)
        return output

# GRU 結構比 LSTM 少一個門
print("LSTM gates: Forget, Input, Output, Cell State")
print("GRU gates: Update, Reset (2 個門)")
```

## 序列到序列模型（Seq2Seq）

```python
class Seq2SeqLSTM(nn.Module):
    """Sequence to Sequence 模型"""
    
    def __init__(self, input_vocab, output_vocab, embed_size, hidden_size):
        super(Seq2SeqLSTM, self).__init__()
        
        self.encoder = nn.LSTM(embed_size, hidden_size, batch_first=True)
        self.decoder = nn.LSTM(embed_size, hidden_size, batch_first=True)
        
        self.encoder_embedding = nn.Embedding(input_vocab, embed_size)
        self.decoder_embedding = nn.Embedding(output_vocab, embed_size)
        
        self.fc = nn.Linear(hidden_size, output_vocab)
    
    def forward(self, src, tgt, teacher_forcing_ratio=0.5):
        # 編碼
        src_embed = self.encoder_embedding(src)
        _, (h, c) = self.encoder(src_embed)
        
        # 解碼
        tgt_embed = self.decoder_embedding(tgt)
        output, _ = self.decoder(tgt_embed, (h, c))
        
        # 輸出層
        output = self.fc(output)
        
        return output
    
    def greedy_decode(self, src, max_len, start_token, end_token):
        """貪心解碼"""
        # 編碼
        src_embed = self.encoder_embedding(src)
        _, (h, c) = self.encoder(src_embed)
        
        # 解碼
        outputs = []
        current_token = start_token
        
        for _ in range(max_len):
            embed = self.decoder_embedding(current_token)
            out, (h, c) = self.decoder(embed.unsqueeze(1), (h, c))
            
            predicted = self.fc(out.squeeze(1))
            current_token = predicted.argmax(dim=-1)
            
            outputs.append(current_token.item())
            
            if current_token == end_token:
                break
        
        return outputs
```

## 注意力機制

```python
class Attention(nn.Module):
    """Bahdanau Attention"""
    def __init__(self, hidden_size):
        super(Attention, self).__init__()
        
        self.W_a = nn.Linear(hidden_size * 2, hidden_size)
        self.W_b = nn.Linear(hidden_size * 2, hidden_size)
        self.V = nn.Linear(hidden_size, 1)
    
    def forward(self, encoder_outputs, decoder_hidden):
        # encoder_outputs: (batch, src_len, hidden*2)
        # decoder_hidden: (batch, hidden)
        
        # 擴展 decoder hidden
        src_len = encoder_outputs.shape[1]
        decoder_hidden = decoder_hidden.unsqueeze(1).repeat(1, src_len, 1)
        
        # 計算 attention score
        energy = self.V(torch.tanh(self.W_a(encoder_outputs) + self.W_b(decoder_hidden)))
        attention_weights = torch.softmax(energy, dim=1)
        
        # 上下文向量
        context = torch.sum(attention_weights * encoder_outputs, dim=1)
        
        return context, attention_weights
```

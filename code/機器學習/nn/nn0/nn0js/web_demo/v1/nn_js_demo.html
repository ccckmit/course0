Value 節點 — 自動微分核心

每個 Value 節點儲存：前向值 data、梯度 grad、子節點與局部梯度。 

class Value:
    # __slots__ 省記憶體，固定屬性名稱
    __slots__ = ('data', 'grad', '_children', '_local_grads')

    def __mul__(self, other):
        # z = x * y  →  ∂z/∂x = y,  ∂z/∂y = x
        return Value(
            self.data * other.data,
            children=(self, other),
            local_grads=(other.data, self.data)  # 局部梯度
        )

    def backward(self):
        # 1. 拓樸排序建立計算圖
        topo = []; visited = set()
        def build_topo(v):
            if v not in visited:
                visited.add(v)
                for child in v._children: build_topo(child)
                topo.append(v)
        build_topo(self)
        # 2. 反向傳播：鏈式法則
        self.grad = 1
        for v in reversed(topo):
            for child, lg in zip(v._children, v._local_grads):
                child.grad += lg * v.grad  # 梯度累積

class Adam:
    def step(self, lr_override=None):
        self.step_count += 1
        for i, p in enumerate(self.params):
            # 一階矩（動量）：平滑梯度方向
            self.m[i] = β1 * self.m[i] + (1-β1) * p.grad

            # 二階矩（速度）：自適應步長縮放
            self.v[i] = β2 * self.v[i] + (1-β2) * p.grad**2

            v_hat = self.v[i] / (1 - β2**t)

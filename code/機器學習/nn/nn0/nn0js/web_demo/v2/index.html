Value 節點 — 自動微分核心

每個 Value 節點儲存： 前向值 data、 梯度 grad、子節點與局部梯度。 

class Value:
    # __slots__ 省記憶體，固定屬性名稱
    __slots__ = ('data', 'grad', '_children', '_local_grads')

    def __mul__(self, other):
        # z = x * y  →  ∂z/∂x = y,  ∂z/∂y = x
        return Value(
            self.data * other.data,
            children=(self, other),
            local_grads=(other.data, self.data)  # 局部梯度
        )

    def backward(self):
        # 1. 拓樸排序建立計算圖
        topo = []; visited = set()
        def build_topo(v):
            if v not in visited:
                visited.add(v)
                for child in v._children: build_topo(child)
                topo.append(v)
        build_topo(self)
        # 2. 反向傳播：鏈式法則
        self.grad = 1
        for v in reversed(topo):
            for child, lg in zip(v._children, v._local_grads):
                child.grad += lg * v.grad  # 梯度累積
class Value {
  constructor(data, children=[], localGrads=[]) {
    this.data = data;
    this.grad = 0;
    this._children = children;     // 依賴的子節點
    this._localGrads = localGrads;  // 對應的局部梯度
  }

  mul(other) {
    // z = x * y  →  ∂z/∂x = y,  ∂z/∂y = x
    const out = new Value(
      this.data * other.data,
      [this, other],
      [other.data, this.data]  // 局部梯度
    );
    out._op = '×'; return out;
  }

  backward() {
    const topo = [], visited = new Set();
    const build = v => {
      if (!visited.has(v)) {
        visited.add(v);
        v._children.forEach(build);
        topo.push(v);
      }
    };
    build(this);
    this.grad = 1;
    // 反向傳播：鏈式法則累積梯度
    for (const v of [...topo].reverse()) {
      v._children.forEach((child, i) => {
        child.grad += v._localGrads[i] * v.grad;
      });
    }
  }
}
Adam 優化器 — 動量 + 自適應學習率
class Adam:
    def step(self, lr_override=None):
        self.step_count += 1
        for i, p in enumerate(self.params):
            # 一階矩（動量）：平滑梯度方向
            self.m[i] = β1 * self.m[i] + (1-β1) * p.grad

            # 二階矩（速度）：自適應步長縮放
            self.v[i] = β2 * self.v[i] + (1-β2) * p.grad**2

            # 偏差校正（早期步驟補償）
            m_hat = self.m[i] / (1 - β1**t)
            v_hat = self.v[i] / (1 - β2**t)

            # 參數更新（學習率線性衰減）
            p.data -= lr * m_hat / (v_hat**0.5 + ε)
class AdamOpt {
  constructor(params, lr=0.01, b1=0.85, b2=0.99, eps=1e-8) {
    this.params = params; this.lr = lr;
    this.b1 = b1; this.b2 = b2; this.eps = eps;
    this.m = new Array(params.length).fill(0);
    this.v = new Array(params.length).fill(0);
    this.t = 0;
  }

  step(lrOvr) {
    this.t++;
    const lr = lrOvr ?? this.lr;
    this.params.forEach((p, i) => {
      // 一階矩（動量）
      this.m[i] = this.b1*this.m[i] + (1-this.b1)*p.grad;
      // 二階矩（速度）
      this.v[i] = this.b2*this.v[i] + (1-this.b2)*p.grad**2;
      // 偏差校正
      const mh = this.m[i]/(1-this.b1**this.t);
      const vh = this.v[i]/(1-this.b2**this.t);
      // 參數更新
      p.data -= lr * mh / (Math.sqrt(vh) + this.eps);
      p.grad = 0;
    });
  }
}
